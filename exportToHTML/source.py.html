14   class Source(object):

22       def isUrl(self):
23           return 'url' in self.type
24   
25       def isFile(self, path=None):
26           # dirty hack to check where file is opened with codecs module
27           # (because it returns 'instance' type when encoding is specified
28           if path:
29               return isinstance(path, io.IOBase) or path.__class__.__name__ == 'StreamReaderWriter'
30           else:
31               return 'file' in self.type
32   
33       def checkFiles(self):
34           if isinstance(self.source, list):
35               for path in self.source:
36                   if not os.path.exists(path):
37                       raise IOError('No such file: %s' % path)
38           else:
39               if not hasattr(self.source, 'read') and not os.path.exists(self.source):
40                   raise IOError('No such file: %s' % self.source)
41   
42       def isString(self):
43           return 'string' in self.type
44   
45       def isFileObj(self):
46           return hasattr(self.source, 'read')
47   
48       def to_s(self):
49           # String should be in unicode(python2)/str(python3) type since we will
50           # later encode it to utf-8 bytes array to pipe into subprocess
51           # With some charachters on python 2 it sets to str type (bytes) which is wrong
52           # and cant later encode properly, this is a workaround for this.
53           # See issue #42
54           if isinstance(self.source, unicode):
55               return self.source
56           else:
57               return unicode(self.source, 'utf-8')
