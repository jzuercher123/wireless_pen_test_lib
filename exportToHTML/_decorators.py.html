158  @dataclass  # can't use slots here since we set attributes on `__post_init__`
159  class PydanticDescriptorProxy(Generic[ReturnType]):
160      """Wrap a classmethod, staticmethod, property or unbound function 
161      and act as a descriptor that allows us to detect decorated items 
162      from the class' attributes. 
163   
164      This class' __get__ returns the wrapped item's __get__ result, 
165      which makes it transparent for classmethods and staticmethods. 
166   
167      Attributes: 
168          wrapped: The decorator that has to be wrapped. 
169          decorator_info: The decorator info. 
170          shim: A wrapper function to wrap V1 style function. 
171      """
172  
173      wrapped: DecoratedType[ReturnType]
174      decorator_info: DecoratorInfo
175      shim: Callable[[Callable[..., Any]], Callable[..., Any]] | None = None
176  
177      def __post_init__(self):
178          for attr in 'setter', 'deleter':
179              if hasattr(self.wrapped, attr):
180                  f = partial(self._call_wrapped_attr, name=attr)
181                  setattr(self, attr, f)
182  
183      def _call_wrapped_attr(self, func: Callable[[Any], None], *, name: str) -> PydanticDescriptorProxy[ReturnType]:
184          self.wrapped = getattr(self.wrapped, name)(func)
185          if isinstance(self.wrapped, property):
186              # update ComputedFieldInfo.wrapped_property
187              from ..fields import ComputedFieldInfo
188  
189              if isinstance(self.decorator_info, ComputedFieldInfo):
190                  self.decorator_info.wrapped_property = self.wrapped
191          return self
192  
193      def __get__(self, obj: object | None, obj_type: type[object] | None = None) -> PydanticDescriptorProxy[ReturnType]:
194          try:
195              return self.wrapped.__get__(obj, obj_type)
196          except AttributeError:
197              # not a descriptor, e.g. a partial object
198              return self.wrapped  # type: ignore[return-value]
199  
200      def __set_name__(self, instance: Any, name: str) -> None:
201          if hasattr(self.wrapped, '__set_name__'):
202              self.wrapped.__set_name__(instance, name)  # pyright: ignore[reportFunctionMemberAccess]
203  
204      def __getattr__(self, __name: str) -> Any:
205          """Forward checks for __isabstractmethod__ and such."""
206          return getattr(self.wrapped, __name)

212  @dataclass(**slots_true)
213  class Decorator(Generic[DecoratorInfoType]):
214      """A generic container class to join together the decorator metadata 
215      (metadata from decorator itself, which we have when the 
216      decorator is called but not when we are building the core-schema) 
217      and the bound function (which we have after the class itself is created). 
218   
219      Attributes: 
220          cls_ref: The class ref. 
221          cls_var_name: The decorated function name. 
222          func: The decorated function. 
223          shim: A wrapper function to wrap V1 style function. 
224          info: The decorator info. 
225      """
226  
227      cls_ref: str
228      cls_var_name: str
229      func: Callable[..., Any]
230      shim: Callable[[Any], Any] | None
231      info: DecoratorInfoType
232  
233      @staticmethod
234      def build(
235          cls_: Any,
236          *,
237          cls_var_name: str,
238          shim: Callable[[Any], Any] | None,
239          info: DecoratorInfoType,
240      ) -> Decorator[DecoratorInfoType]:
241          """Build a new decorator. 
242   
243          Args: 
244              cls_: The class. 
245              cls_var_name: The decorated function name. 
246              shim: A wrapper function to wrap V1 style function. 
247              info: The decorator info. 
248   
249          Returns: 
250              The new decorator instance. 
251          """
252          func = get_attribute_from_bases(cls_, cls_var_name)
253          if shim is not None:
254              func = shim(func)
255          func = unwrap_wrapped_function(func, unwrap_partial=False)
256          if not callable(func):
257              # This branch will get hit for classmethod properties
258              attribute = get_attribute_from_base_dicts(cls_, cls_var_name)  # prevents the binding call to `__get__`
259              if isinstance(attribute, PydanticDescriptorProxy):
260                  func = unwrap_wrapped_function(attribute.wrapped)
261          return Decorator(
262              cls_ref=get_type_ref(cls_),
263              cls_var_name=cls_var_name,
264              func=func,
265              shim=shim,
266              info=info,
267          )
268  
269      def bind_to_cls(self, cls: Any) -> Decorator[DecoratorInfoType]:
270          """Bind the decorator to a class. 
271   
272          Args: 
273              cls: the class. 
274   
275          Returns: 
276              The new decorator instance. 
277          """
278          return self.build(
279              cls,
280              cls_var_name=self.cls_var_name,
281              shim=self.shim,
282              info=self.info,
283          )

286  def get_bases(tp: type[Any]) -> tuple[type[Any], ...]:
287      """Get the base classes of a class or typeddict. 
288   
289      Args: 
290          tp: The type or class to get the bases. 
291   
292      Returns: 
293          The base classes. 
294      """
295      if is_typeddict(tp):
296          return tp.__orig_bases__  # type: ignore
297      try:
298          return tp.__bases__
299      except AttributeError:
300          return ()

303  def mro(tp: type[Any]) -> tuple[type[Any], ...]:
304      """Calculate the Method Resolution Order of bases using the C3 algorithm. 
305   
306      See https://www.python.org/download/releases/2.3/mro/ 
307      """
308      # try to use the existing mro, for performance mainly
309      # but also because it helps verify the implementation below
310      if not is_typeddict(tp):
311          try:
312              return tp.__mro__
313          except AttributeError:
314              # GenericAlias and some other cases
315              pass
316  
317      bases = get_bases(tp)
318      return (tp,) + mro_for_bases(bases)
319  
320  
321  def mro_for_bases(bases: tuple[type[Any], ...]) -> tuple[type[Any], ...]:
322      def merge_seqs(seqs: list[deque[type[Any]]]) -> Iterable[type[Any]]:
323          while True:
324              non_empty = [seq for seq in seqs if seq]
325              if not non_empty:
326                  # Nothing left to process, we're done.
327                  return
328              candidate: type[Any] | None = None
329              for seq in non_empty:  # Find merge candidates among seq heads.
330                  candidate = seq[0]
331                  not_head = [s for s in non_empty if candidate in islice(s, 1, None)]
332                  if not_head:
333                      # Reject the candidate.
334                      candidate = None
335                  else:
336                      break
337              if not candidate:
338                  raise TypeError('Inconsistent hierarchy, no C3 MRO is possible')
339              yield candidate
340              for seq in non_empty:
341                  # Remove candidate.
342                  if seq[0] == candidate:
343                      seq.popleft()
344  
345      seqs = [deque(mro(base)) for base in bases] + [deque(bases)]
346      return tuple(merge_seqs(seqs))

352  def get_attribute_from_bases(tp: type[Any] | tuple[type[Any], ...], name: str) -> Any:
353      """Get the attribute from the next class in the MRO that has it, 
354      aiming to simulate calling the method on the actual class. 
355   
356      The reason for iterating over the mro instead of just getting 
357      the attribute (which would do that for us) is to support TypedDict, 
358      which lacks a real __mro__, but can have a virtual one constructed 
359      from its bases (as done here). 
360   
361      Args: 
362          tp: The type or class to search for the attribute. If a tuple, this is treated as a set of base classes. 
363          name: The name of the attribute to retrieve. 
364   
365      Returns: 
366          Any: The attribute value, if found. 
367   
368      Raises: 
369          AttributeError: If the attribute is not found in any class in the MRO. 
370      """
371      if isinstance(tp, tuple):
372          for base in mro_for_bases(tp):
373              attribute = base.__dict__.get(name, _sentinel)
374              if attribute is not _sentinel:
375                  attribute_get = getattr(attribute, '__get__', None)
376                  if attribute_get is not None:
377                      return attribute_get(None, tp)
378                  return attribute
379          raise AttributeError(f'{name} not found in {tp}')
380      else:
381          try:
382              return getattr(tp, name)
383          except AttributeError:
384              return get_attribute_from_bases(mro(tp), name)

387  def get_attribute_from_base_dicts(tp: type[Any], name: str) -> Any:
388      """Get an attribute out of the `__dict__` following the MRO. 
389      This prevents the call to `__get__` on the descriptor, and allows 
390      us to get the original function for classmethod properties. 
391   
392      Args: 
393          tp: The type or class to search for the attribute. 
394          name: The name of the attribute to retrieve. 
395   
396      Returns: 
397          Any: The attribute value, if found. 
398   
399      Raises: 
400          KeyError: If the attribute is not found in any class's `__dict__` in the MRO. 
401      """
402      for base in reversed(mro(tp)):
403          if name in base.__dict__:
404              return base.__dict__[name]
405      return tp.__dict__[name]  # raise the error

408  @dataclass(**slots_true)
409  class DecoratorInfos:
410      """Mapping of name in the class namespace to decorator info. 
411   
412      note that the name in the class namespace is the function or attribute name 
413      not the field name! 
414      """
415  
416      validators: dict[str, Decorator[ValidatorDecoratorInfo]] = field(default_factory=dict)
417      field_validators: dict[str, Decorator[FieldValidatorDecoratorInfo]] = field(default_factory=dict)
418      root_validators: dict[str, Decorator[RootValidatorDecoratorInfo]] = field(default_factory=dict)
419      field_serializers: dict[str, Decorator[FieldSerializerDecoratorInfo]] = field(default_factory=dict)
420      model_serializers: dict[str, Decorator[ModelSerializerDecoratorInfo]] = field(default_factory=dict)
421      model_validators: dict[str, Decorator[ModelValidatorDecoratorInfo]] = field(default_factory=dict)
422      computed_fields: dict[str, Decorator[ComputedFieldInfo]] = field(default_factory=dict)
423  
424      @staticmethod
425      def build(model_dc: type[Any]) -> DecoratorInfos:  # noqa: C901 (ignore complexity)
426          """We want to collect all DecFunc instances that exist as 
427          attributes in the namespace of the class (a BaseModel or dataclass) 
428          that called us 
429          But we want to collect these in the order of the bases 
430          So instead of getting them all from the leaf class (the class that called us), 
431          we traverse the bases from root (the oldest ancestor class) to leaf 
432          and collect all of the instances as we go, taking care to replace 
433          any duplicate ones with the last one we see to mimic how function overriding 
434          works with inheritance. 
435          If we do replace any functions we put the replacement into the position 
436          the replaced function was in; that is, we maintain the order. 
437          """
438          # reminder: dicts are ordered and replacement does not alter the order
439          res = DecoratorInfos()
440          for base in reversed(mro(model_dc)[1:]):
441              existing: DecoratorInfos | None = base.__dict__.get('__pydantic_decorators__')
442              if existing is None:
443                  existing = DecoratorInfos.build(base)
444              res.validators.update({k: v.bind_to_cls(model_dc) for k, v in existing.validators.items()})
445              res.field_validators.update({k: v.bind_to_cls(model_dc) for k, v in existing.field_validators.items()})
446              res.root_validators.update({k: v.bind_to_cls(model_dc) for k, v in existing.root_validators.items()})
447              res.field_serializers.update({k: v.bind_to_cls(model_dc) for k, v in existing.field_serializers.items()})
448              res.model_serializers.update({k: v.bind_to_cls(model_dc) for k, v in existing.model_serializers.items()})
449              res.model_validators.update({k: v.bind_to_cls(model_dc) for k, v in existing.model_validators.items()})
450              res.computed_fields.update({k: v.bind_to_cls(model_dc) for k, v in existing.computed_fields.items()})
451  
452          to_replace: list[tuple[str, Any]] = []
453  
454          for var_name, var_value in vars(model_dc).items():
455              if isinstance(var_value, PydanticDescriptorProxy):
456                  info = var_value.decorator_info
457                  if isinstance(info, ValidatorDecoratorInfo):
458                      res.validators[var_name] = Decorator.build(
459                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
460                      )
461                  elif isinstance(info, FieldValidatorDecoratorInfo):
462                      res.field_validators[var_name] = Decorator.build(
463                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
464                      )
465                  elif isinstance(info, RootValidatorDecoratorInfo):
466                      res.root_validators[var_name] = Decorator.build(
467                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
468                      )
469                  elif isinstance(info, FieldSerializerDecoratorInfo):
470                      # check whether a serializer function is already registered for fields
471                      for field_serializer_decorator in res.field_serializers.values():
472                          # check that each field has at most one serializer function.
473                          # serializer functions for the same field in subclasses are allowed,
474                          # and are treated as overrides
475                          if field_serializer_decorator.cls_var_name == var_name:
476                              continue
477                          for f in info.fields:
478                              if f in field_serializer_decorator.info.fields:
479                                  raise PydanticUserError(
480                                      'Multiple field serializer functions were defined '
481                                      f'for field {f!r}, this is not allowed.',
482                                      code='multiple-field-serializers',
483                                  )
484                      res.field_serializers[var_name] = Decorator.build(
485                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
486                      )
487                  elif isinstance(info, ModelValidatorDecoratorInfo):
488                      res.model_validators[var_name] = Decorator.build(
489                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
490                      )
491                  elif isinstance(info, ModelSerializerDecoratorInfo):
492                      res.model_serializers[var_name] = Decorator.build(
493                          model_dc, cls_var_name=var_name, shim=var_value.shim, info=info
494                      )
495                  else:
496                      from ..fields import ComputedFieldInfo
497  
498                      isinstance(var_value, ComputedFieldInfo)
499                      res.computed_fields[var_name] = Decorator.build(
500                          model_dc, cls_var_name=var_name, shim=None, info=info
501                      )
502                  to_replace.append((var_name, var_value.wrapped))
503          if to_replace:
504              # If we can save `__pydantic_decorators__` on the class we'll be able to check for it above
505              # so then we don't need to re-process the type, which means we can discard our descriptor wrappers
506              # and replace them with the thing they are wrapping (see the other setattr call below)
507              # which allows validator class methods to also function as regular class methods
508              model_dc.__pydantic_decorators__ = res
509              for name, value in to_replace:
510                  setattr(model_dc, name, value)
511          return res

514  def inspect_validator(validator: Callable[..., Any], mode: FieldValidatorModes) -> bool:
515      """Look at a field or model validator function and determine whether it takes an info argument. 
516   
517      An error is raised if the function has an invalid signature. 
518   
519      Args: 
520          validator: The validator function to inspect. 
521          mode: The proposed validator mode. 
522   
523      Returns: 
524          Whether the validator takes an info argument. 
525      """
526      try:
527          sig = signature(validator)
528      except (ValueError, TypeError):
529          # `inspect.signature` might not be able to infer a signature, e.g. with C objects.
530          # In this case, we assume no info argument is present:
531          return False
532      n_positional = count_positional_required_params(sig)
533      if mode == 'wrap':
534          if n_positional == 3:
535              return True
536          elif n_positional == 2:
537              return False
538      else:
539          assert mode in {'before', 'after', 'plain'}, f"invalid mode: {mode!r}, expected 'before', 'after' or 'plain"
540          if n_positional == 2:
541              return True
542          elif n_positional == 1:
543              return False
544  
545      raise PydanticUserError(
546          f'Unrecognized field_validator function signature for {validator} with `mode={mode}`:{sig}',
547          code='validator-signature',
548      )

551  def inspect_field_serializer(
552      serializer: Callable[..., Any], mode: Literal['plain', 'wrap'], computed_field: bool = False
553  ) -> tuple[bool, bool]:
554      """Look at a field serializer function and determine if it is a field serializer, 
555      and whether it takes an info argument. 
556   
557      An error is raised if the function has an invalid signature. 
558   
559      Args: 
560          serializer: The serializer function to inspect. 
561          mode: The serializer mode, either 'plain' or 'wrap'. 
562          computed_field: When serializer is applied on computed_field. It doesn't require 
563              info signature. 
564   
565      Returns: 
566          Tuple of (is_field_serializer, info_arg). 
567      """
568      try:
569          sig = signature(serializer)
570      except (ValueError, TypeError):
571          # `inspect.signature` might not be able to infer a signature, e.g. with C objects.
572          # In this case, we assume no info argument is present and this is not a method:
573          return (False, False)
574  
575      first = next(iter(sig.parameters.values()), None)
576      is_field_serializer = first is not None and first.name == 'self'
577  
578      n_positional = count_positional_required_params(sig)
579      if is_field_serializer:
580          # -1 to correct for self parameter
581          info_arg = _serializer_info_arg(mode, n_positional - 1)
582      else:
583          info_arg = _serializer_info_arg(mode, n_positional)
584  
585      if info_arg is None:
586          raise PydanticUserError(
587              f'Unrecognized field_serializer function signature for {serializer} with `mode={mode}`:{sig}',
588              code='field-serializer-signature',
589          )
590      if info_arg and computed_field:
591          raise PydanticUserError(
592              'field_serializer on computed_field does not use info signature', code='field-serializer-signature'
593          )
594  
595      else:
596          return is_field_serializer, info_arg

600      """Look at a serializer function used via `Annotated` and determine whether it takes an info argument. 
601   
602      An error is raised if the function has an invalid signature. 
603   
604      Args:

608      Returns: 
609          info_arg 
610      """
611      try:
612          sig = signature(serializer)
613      except (ValueError, TypeError):
614          # `inspect.signature` might not be able to infer a signature, e.g. with C objects.
615          # In this case, we assume no info argument is present:
616          return False
617      info_arg = _serializer_info_arg(mode, count_positional_required_params(sig))
618      if info_arg is None:
619          raise PydanticUserError(
620              f'Unrecognized field_serializer function signature for {serializer} with `mode={mode}`:{sig}',
621              code='field-serializer-signature',
622          )
623      else:
624          return info_arg
625  
626  
627  def inspect_model_serializer(serializer: Callable[..., Any], mode: Literal['plain', 'wrap']) -> bool:
628      """Look at a model serializer function and determine whether it takes an info argument. 
629   
630      An error is raised if the function has an invalid signature. 
631   
632      Args: 
633          serializer: The serializer function to check. 
634          mode: The serializer mode, either 'plain' or 'wrap'. 
635   
636      Returns: 
637          `info_arg` - whether the function expects an info argument. 
638      """
639      if isinstance(serializer, (staticmethod, classmethod)) or not is_instance_method_from_sig(serializer):
640          raise PydanticUserError(
641              '`@model_serializer` must be applied to instance methods', code='model-serializer-instance-method'
642          )
643  
644      sig = signature(serializer)
645      info_arg = _serializer_info_arg(mode, count_positional_required_params(sig))
646      if info_arg is None:
647          raise PydanticUserError(
648              f'Unrecognized model_serializer function signature for {serializer} with `mode={mode}`:{sig}',
649              code='model-serializer-signature',
650          )
651      else:
652          return info_arg
653  
654  
655  def _serializer_info_arg(mode: Literal['plain', 'wrap'], n_positional: int) -> bool | None:
656      if mode == 'plain':
657          if n_positional == 1:
658              # (input_value: Any, /) -> Any
659              return False
660          elif n_positional == 2:
661              # (model: Any, input_value: Any, /) -> Any
662              return True
663      else:
664          assert mode == 'wrap', f"invalid mode: {mode!r}, expected 'plain' or 'wrap'"
665          if n_positional == 2:
666              # (input_value: Any, serializer: SerializerFunctionWrapHandler, /) -> Any
667              return False
668          elif n_positional == 3:
669              # (input_value: Any, serializer: SerializerFunctionWrapHandler, info: SerializationInfo, /) -> Any
670              return True
671  
672      return None

680  def is_instance_method_from_sig(function: AnyDecoratorCallable) -> bool:
681      """Whether the function is an instance method. 
682   
683      It will consider a function as instance method if the first parameter of 
684      function is `self`. 
685   
686      Args: 
687          function: The function to check. 
688   
689      Returns: 
690          `True` if the function is an instance method, `False` otherwise. 
691      """
692      sig = signature(unwrap_wrapped_function(function))
693      first = next(iter(sig.parameters.values()), None)
694      if first and first.name == 'self':
695          return True
696      return False

699  def ensure_classmethod_based_on_signature(function: AnyDecoratorCallable) -> Any:
700      """Apply the `@classmethod` decorator on the function. 
701   
702      Args: 
703          function: The function to apply the decorator on. 
704   
705      Return: 
706          The `@classmethod` decorator applied function. 
707      """
708      if not isinstance(
709          unwrap_wrapped_function(function, unwrap_class_static_method=False), classmethod
710      ) and _is_classmethod_from_sig(function):
711          return classmethod(function)  # type: ignore[arg-type]
712      return function
713  
714  
715  def _is_classmethod_from_sig(function: AnyDecoratorCallable) -> bool:
716      sig = signature(unwrap_wrapped_function(function))
717      first = next(iter(sig.parameters.values()), None)
718      if first and first.name == 'cls':
719          return True
720      return False

723  def unwrap_wrapped_function(
724      func: Any,
725      *,
726      unwrap_partial: bool = True,
727      unwrap_class_static_method: bool = True,
728  ) -> Any:
729      """Recursively unwraps a wrapped function until the underlying function is reached. 
730      This handles property, functools.partial, functools.partialmethod, staticmethod, and classmethod. 
731   
732      Args: 
733          func: The function to unwrap. 
734          unwrap_partial: If True (default), unwrap partial and partialmethod decorators. 
735          unwrap_class_static_method: If True (default), also unwrap classmethod and staticmethod 
736              decorators. If False, only unwrap partial and partialmethod decorators. 
737   
738      Returns: 
739          The underlying function of the wrapped function. 
740      """
741      # Define the types we want to check against as a single tuple.
742      unwrap_types = (
743          (property, cached_property)
744          + ((partial, partialmethod) if unwrap_partial else ())
745          + ((staticmethod, classmethod) if unwrap_class_static_method else ())
746      )
747  
748      while isinstance(func, unwrap_types):
749          if unwrap_class_static_method and isinstance(func, (classmethod, staticmethod)):
750              func = func.__func__
751          elif isinstance(func, (partial, partialmethod)):
752              func = func.func
753          elif isinstance(func, property):
754              func = func.fget  # arbitrary choice, convenient for computed fields
755          else:
756              # Make coverage happy as it can only get here in the last possible case
757              assert isinstance(func, cached_property)
758              func = func.func  # type: ignore
759  
760      return func

763  def get_function_return_type(
764      func: Any, explicit_return_type: Any, types_namespace: dict[str, Any] | None = None
765  ) -> Any:
766      """Get the function return type. 
767   
768      It gets the return type from the type annotation if `explicit_return_type` is `None`. 
769      Otherwise, it returns `explicit_return_type`. 
770   
771      Args: 
772          func: The function to get its return type. 
773          explicit_return_type: The explicit return type. 
774          types_namespace: The types namespace, defaults to `None`. 
775   
776      Returns: 
777          The function return type. 
778      """
779      if explicit_return_type is PydanticUndefined:
780          # try to get it from the type annotation
781          hints = get_function_type_hints(
782              unwrap_wrapped_function(func), include_keys={'return'}, types_namespace=types_namespace
783          )
784          return hints.get('return', PydanticUndefined)
785      else:
786          return explicit_return_type

789  def count_positional_required_params(sig: Signature) -> int:
790      """Get the number of positional (required) arguments of a signature. 
791   
792      This function should only be used to inspect signatures of validation and serialization functions. 
793      The first argument (the value being serialized or validated) is counted as a required argument 
794      even if a default value exists. 
795   
796      Returns: 
797          The number of positional arguments of a signature. 
798      """
799      parameters = list(sig.parameters.values())
800      return sum(
801          1
802          for param in parameters
803          if can_be_positional(param)
804          # First argument is the value being validated/serialized, and can have a default value
805          # (e.g. `float`, which has signature `(x=0, /)`). We assume other parameters (the info arg
806          # for instance) should be required, and thus without any default value.
807          and (param.default is Parameter.empty or param == parameters[0])
808      )
809  
810  
811  def can_be_positional(param: Parameter) -> bool:
812      return param.kind in (Parameter.POSITIONAL_ONLY, Parameter.POSITIONAL_OR_KEYWORD)
813

815  def ensure_property(f: Any) -> Any:
816      """Ensure that a function is a `property` or `cached_property`, or is a valid descriptor. 
817   
818      Args: 
819          f: The function to check. 
820   
821      Returns: 
822          The function, or a `property` or `cached_property` instance wrapping the function. 
823      """
824      if ismethoddescriptor(f) or isdatadescriptor(f):
825          return f
826      else:
827          return property(f)
