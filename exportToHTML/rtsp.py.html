30   RTSP_REQ_HEADERS = [
31       "Accept",
32       "Accept-Encoding",
33       "Accept-Language",
34       "Authorization",
35       "From",
36       "If-Modified-Since",
37       "Range",

41   RTSP_RESP_HEADERS = [
42       "Location",
43       "Proxy-Authenticate",
44       "Public",
45       "Retry-After",
46       "Server",
47       "Vary",
48       "WWW-Authenticate",
49   ]

52   class RTSPRequest(_HTTPContent):
53       name = "RTSP Request"
54       fields_desc = (
55           [
56               # First line
57               _HTTPHeaderField("Method", "DESCRIBE"),
58               _HTTPHeaderField("Request_Uri", "*"),
59               _HTTPHeaderField("Version", "RTSP/1.0"),
60               # Headers
61           ]
62           + (
63               _generate_headers(
64                   RTSP_REQ_HEADERS,
65               )
66           )
67           + [
68               _HTTPHeaderField("Unknown-Headers", None),
69           ]
70       )
71   
72       def do_dissect(self, s):
73           first_line, body = _dissect_headers(self, s)
74           try:
75               method, uri, version = re.split(rb"\s+", first_line, maxsplit=2)
76               self.setfieldval("Method", method)
77               self.setfieldval("Request_Uri", uri)
78               self.setfieldval("Version", version)
79           except ValueError:
80               pass
81           if body:
82               self.raw_packet_cache = s[: -len(body)]
83           else:
84               self.raw_packet_cache = s
85           return body
86   
87       def mysummary(self):
88           return self.sprintf(
89               "%RTSPRequest.Method% %RTSPRequest.Request_Uri% " "%RTSPRequest.Version%"
90           )

93   class RTSPResponse(_HTTPContent):
94       name = "RTSP Response"
95       fields_desc = (
96           [
97               # First line
98               _HTTPHeaderField("Version", "RTSP/1.1"),
99               _HTTPHeaderField("Status_Code", "200"),
100              _HTTPHeaderField("Reason_Phrase", "OK"),
101              # Headers
102          ]
103          + (
104              _generate_headers(
105                  RTSP_RESP_HEADERS,
106              )
107          )
108          + [
109              _HTTPHeaderField("Unknown-Headers", None),
110          ]
111      )
112  
113      def answers(self, other):
114          return RTSPRequest in other
115  
116      def do_dissect(self, s):
117          first_line, body = _dissect_headers(self, s)
118          try:
119              Version, Status, Reason = re.split(rb"\s+", first_line, maxsplit=2)
120              self.setfieldval("Version", Version)
121              self.setfieldval("Status_Code", Status)
122              self.setfieldval("Reason_Phrase", Reason)
123          except ValueError:
124              pass
125          if body:
126              self.raw_packet_cache = s[: -len(body)]
127          else:
128              self.raw_packet_cache = s
129          return body
130  
131      def mysummary(self):
132          return self.sprintf(
133              "%RTSPResponse.Version% %RTSPResponse.Status_Code% "
134              "%RTSPResponse.Reason_Phrase%"
135          )

138  class RTSP(HTTP):
139      name = "RTSP"
140      clsreq = RTSPRequest
141      clsresp = RTSPResponse
142      hdr = b"RTSP"
143      reqmethods = b"|".join(
144          [
145              b"DESCRIBE",
146              b"ANNOUNCE",
147              b"GET_PARAMETER",
148              b"OPTIONS",
149              b"PAUSE",
150              b"PLAY",
151              b"RECORD",
152              b"REDIRECT",
153              b"SETUP",
154              b"SET_PARAMETER",
155              b"TEARDOWN",
156          ]
157      )
158  
159      @classmethod
160      def dispatch_hook(cls, _pkt=None, *args, **kargs):
161          return cls
