20   def ARC4_encrypt(key, data, skip=0):
21       """Encrypt data @data with key @key, skipping @skip first bytes of the 
22       keystream"""
23   
24       algorithm = algorithms.ARC4(key)
25       cipher = Cipher(algorithm, mode=None, backend=default_backend())
26       encryptor = cipher.encryptor()
27       if skip:
28           encryptor.update(b"\x00" * skip)
29       return encryptor.update(data)
30   
31   
32   def ARC4_decrypt(key, data, skip=0):
33       """Decrypt data @data with key @key, skipping @skip first bytes of the 
34       keystream"""
35       return ARC4_encrypt(key, data, skip)

40   def customPRF512(key, amac, smac, anonce, snonce):
41       """Source https://stackoverflow.com/questions/12018920/"""
42       A = b"Pairwise key expansion"
43       B = b"".join(sorted([amac, smac]) + sorted([anonce, snonce]))
44   
45       blen = 64
46       i = 0
47       R = b''
48       while i <= ((blen * 8 + 159) // 160):
49           hmacsha1 = hmac.new(key, A + chb(0x00) + B + chb(i), hashlib.sha1)
50           i += 1
51           R = R + hmacsha1.digest()
52       return R[:blen]

134  def _MK16(b1, b2):
135      return (b1 << 8) | b2
136  
137  
138  def _SBOX16(index):
139      return _SBOXS[0][index & 0xff] ^ _SBOXS[1][(index >> 8)]
140

142  def _CAST16(value):
143      return value & 0xffff
144

146  def _RotR1(value):
147      return ((value >> 1) & 0x7fff) | (value << 15)

150  def gen_TKIP_RC4_key(TSC, TA, TK):
151      """Implement TKIP WEPSeed generation 
152      TSC: packet IV 
153      TA: target addr bytes 
154      TK: temporal key 
155      """
156  
157      assert len(TSC) == 6
158      assert len(TA) == 6
159      assert len(TK) == 16
160      assert all(isinstance(x, int) for x in TSC + TA + TK)
161  
162      # Phase 1
163      # 802.11i p.54
164  
165      # Phase 1 - Step 1
166      TTAK = []
167      TTAK.append(_MK16(TSC[3], TSC[2]))
168      TTAK.append(_MK16(TSC[5], TSC[4]))
169      TTAK.append(_MK16(TA[1], TA[0]))
170      TTAK.append(_MK16(TA[3], TA[2]))
171      TTAK.append(_MK16(TA[5], TA[4]))
172  
173      # Phase 1 - Step 2
174      for i in range(PHASE1_LOOP_CNT):
175          j = 2 * (i & 1)
176          TTAK[0] = _CAST16(TTAK[0] + _SBOX16(TTAK[4] ^ _MK16(TK[1 + j], TK[0 + j])))  # noqa: E501
177          TTAK[1] = _CAST16(TTAK[1] + _SBOX16(TTAK[0] ^ _MK16(TK[5 + j], TK[4 + j])))  # noqa: E501
178          TTAK[2] = _CAST16(TTAK[2] + _SBOX16(TTAK[1] ^ _MK16(TK[9 + j], TK[8 + j])))  # noqa: E501
179          TTAK[3] = _CAST16(TTAK[3] + _SBOX16(TTAK[2] ^ _MK16(TK[13 + j], TK[12 + j])))  # noqa: E501
180          TTAK[4] = _CAST16(TTAK[4] + _SBOX16(TTAK[3] ^ _MK16(TK[1 + j], TK[0 + j])) + i)  # noqa: E501
181  
182      # Phase 2
183      # 802.11i p.56
184  
185      # Phase 2 - Step 1
186      PPK = list(TTAK)
187      PPK.append(_CAST16(TTAK[4] + _MK16(TSC[1], TSC[0])))
188  
189      # Phase 2 - Step 2
190      PPK[0] = _CAST16(PPK[0] + _SBOX16(PPK[5] ^ _MK16(TK[1], TK[0])))
191      PPK[1] = _CAST16(PPK[1] + _SBOX16(PPK[0] ^ _MK16(TK[3], TK[2])))
192      PPK[2] = _CAST16(PPK[2] + _SBOX16(PPK[1] ^ _MK16(TK[5], TK[4])))
193      PPK[3] = _CAST16(PPK[3] + _SBOX16(PPK[2] ^ _MK16(TK[7], TK[6])))
194      PPK[4] = _CAST16(PPK[4] + _SBOX16(PPK[3] ^ _MK16(TK[9], TK[8])))
195      PPK[5] = _CAST16(PPK[5] + _SBOX16(PPK[4] ^ _MK16(TK[11], TK[10])))
196  
197      PPK[0] = _CAST16(PPK[0] + _RotR1(PPK[5] ^ _MK16(TK[13], TK[12])))
198      PPK[1] = _CAST16(PPK[1] + _RotR1(PPK[0] ^ _MK16(TK[15], TK[14])))
199      PPK[2] = _CAST16(PPK[2] + _RotR1(PPK[1]))
200      PPK[3] = _CAST16(PPK[3] + _RotR1(PPK[2]))
201      PPK[4] = _CAST16(PPK[4] + _RotR1(PPK[3]))
202      PPK[5] = _CAST16(PPK[5] + _RotR1(PPK[4]))
203  
204      # Phase 2 - Step 3
205      WEPSeed = []
206      WEPSeed.append(TSC[1])
207      WEPSeed.append((TSC[1] | 0x20) & 0x7f)
208      WEPSeed.append(TSC[0])
209      WEPSeed.append(((PPK[5] ^ _MK16(TK[1], TK[0])) >> 1) & 0xFF)
210      for i in range(6):
211          WEPSeed.append(PPK[i] & 0xFF)
212          WEPSeed.append(PPK[i] >> 8)
213  
214      assert len(WEPSeed) == 16
215  
216      return b"".join(chb(x) for x in WEPSeed)

222  def _rotate_right32(value, shift):
223      return (value >> (shift % 32) | value << ((32 - shift) % 32)) & 0xFFFFFFFF
224  
225  
226  def _rotate_left32(value, shift):
227      return (value << (shift % 32) | value >> ((32 - shift) % 32)) & 0xFFFFFFFF
228  
229  
230  def _XSWAP(value):
231      """Swap 2 least significant bytes of @value"""
232      return ((value & 0xFF00FF00) >> 8) | ((value & 0x00FF00FF) << 8)
233

235  def _michael_b(m_l, m_r):
236      """Defined in 802.11i p.49"""
237      m_r = m_r ^ _rotate_left32(m_l, 17)
238      m_l = (m_l + m_r) % 2**32
239      m_r = m_r ^ _XSWAP(m_l)
240      m_l = (m_l + m_r) % 2**32
241      m_r = m_r ^ _rotate_left32(m_l, 3)
242      m_l = (m_l + m_r) % 2**32
243      m_r = m_r ^ _rotate_right32(m_l, 2)
244      m_l = (m_l + m_r) % 2**32
245      return m_l, m_r

248  def michael(key, to_hash):
249      """Defined in 802.11i p.48"""
250  
251      # Block size: 4
252      nb_block, nb_extra_bytes = divmod(len(to_hash), 4)
253      # Add padding
254      data = to_hash + chb(0x5a) + b"\x00" * (7 - nb_extra_bytes)
255  
256      # Hash
257      m_l, m_r = unpack('<II', key)
258      for i in range(nb_block + 2):
259          # Convert i-th block to int
260          block_i = unpack('<I', data[i * 4:i * 4 + 4])[0]
261          m_l ^= block_i
262          m_l, m_r = _michael_b(m_l, m_r)
263      return pack('<II', m_l, m_r)

268  def parse_TKIP_hdr(pkt):
269      """Extract TSCs, TA and encoded-data from a packet @pkt"""
270      # Note: FCS bit is not handled
271      assert pkt.FCfield.protected
272  
273      # 802.11i - 8.3.2.2
274      tkip_layer = pkt[Dot11TKIP]
275      payload = tkip_layer.data
276  
277      # IV
278      if not tkip_layer.ext_iv:
279          # 802.11i p. 46
280          raise ValueError("Extended IV must be set for TKIP")
281      TSC0 = tkip_layer.TSC0
282      TSC1 = tkip_layer.TSC1
283      WEPseed = tkip_layer.WEPSeed
284  
285      # Extended IV
286      TSC2 = tkip_layer.TSC2
287      TSC3 = tkip_layer.TSC3
288      TSC4 = tkip_layer.TSC4
289      TSC5 = tkip_layer.TSC5
290  
291      # 802.11i p. 46
292      assert (TSC1 | 0x20) & 0x7f == WEPseed
293  
294      TA = [orb(e) for e in mac2str(pkt.addr2)]
295      TSC = [TSC0, TSC1, TSC2, TSC3, TSC4, TSC5]
296  
297      return TSC, TA, payload

300  def build_TKIP_payload(data, iv, mac, tk):
301      """Build a TKIP header for IV @iv and mac @mac, and encrypt @data 
302      based on temporal key @tk 
303      """
304      TSC5, TSC4, TSC3, TSC2, TSC1, TSC0 = (
305          (iv >> 40) & 0xFF,
306          (iv >> 32) & 0xFF,
307          (iv >> 24) & 0xFF,
308          (iv >> 16) & 0xFF,
309          (iv >> 8) & 0xFF,
310          iv & 0xFF
311      )
312      bitfield = 1 << 5  # Extended IV
313      TKIP_hdr = chb(TSC1) + chb((TSC1 | 0x20) & 0x7f) + chb(TSC0) + chb(bitfield)  # noqa: E501
314      TKIP_hdr += chb(TSC2) + chb(TSC3) + chb(TSC4) + chb(TSC5)
315  
316      TA = [orb(e) for e in mac2str(mac)]
317      TSC = [TSC0, TSC1, TSC2, TSC3, TSC4, TSC5]
318      TK = [orb(x) for x in tk]
319  
320      rc4_key = gen_TKIP_RC4_key(TSC, TA, TK)
321      return TKIP_hdr + ARC4_encrypt(rc4_key, data)

324  def parse_data_pkt(pkt, tk):
325      """Extract data from a WPA packet @pkt with temporal key @tk"""
326      TSC, TA, data = parse_TKIP_hdr(pkt)
327      TK = [orb(x) for x in tk]
328  
329      rc4_key = gen_TKIP_RC4_key(TSC, TA, TK)
330      return ARC4_decrypt(rc4_key, data)

343  def check_MIC_ICV(data, mic_key, source, dest):
344      """Check MIC, ICV & return the data from a decrypted TKIP packet"""
345      assert len(data) > 12
346  
347      # DATA - MIC(DA - SA - Priority=0 - 0 - 0 - 0 - DATA) - ICV
348      # 802.11i p.47
349  
350      ICV = data[-4:]
351      MIC = data[-12:-4]
352      data_clear = data[:-12]
353  
354      expected_ICV = pack("<I", crc32(data_clear + MIC) & 0xFFFFFFFF)
355      if expected_ICV != ICV:
356          raise ICVError()
357  
358      sa = mac2str(source)  # Source MAC
359      da = mac2str(dest)  # Dest MAC
360  
361      expected_MIC = michael(mic_key, da + sa + b"\x00" * 4 + data_clear)
362      if expected_MIC != MIC:
363          raise MICError()
364  
365      return data_clear

368  def build_MIC_ICV(data, mic_key, source, dest):
369      """Compute and return the data with its MIC and ICV"""
370      # DATA - MIC(DA - SA - Priority=0 - 0 - 0 - 0 - DATA) - ICV
371      # 802.11i p.47
372  
373      sa = mac2str(source)  # Source MAC
374      da = mac2str(dest)  # Dest MAC
375      MIC = michael(mic_key, da + sa + b"\x00" + b"\x00" * 3 + data)
376      ICV = pack("<I", crc32(data + MIC) & 0xFFFFFFFF)
377  
378      return data + MIC + ICV
