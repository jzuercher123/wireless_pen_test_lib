34       # type: (bool) -> None
35       print('''usage:\tisotpscanner [-i interface] [-c channel]

38                   [-s start] [-e end]\n
39       Scan for open ISOTP-Sockets.\n
40       required arguments: 
41       -c, --channel         python-can channel or Linux SocketCAN interface name 
42       -s, --start           Start scan at this identifier (hex) 
43       -e, --end             End scan at this identifier (hex)\n
44       additional required arguments for WINDOWS or Python 2: 
45       -i, --interface       python-can interface for the scan.

76   def create_socket(python_can_args, interface, channel):
77       # type: (Optional[str], Optional[str], str) -> Tuple[CANSocket, str]
78   
79       if PYTHON_CAN:
80           if python_can_args:
81               interface_string = "CANSocket(bustype=" \
82                                  "'%s', channel='%s', %s)" % \
83                                  (interface, channel, python_can_args)
84               arg_dict = dict((k, literal_eval(v)) for k, v in
85                               (pair.split('=') for pair in
86                                re.split(', | |,', python_can_args)))
87               sock = CANSocket(bustype=interface, channel=channel,
88                                **arg_dict)
89           else:
90               interface_string = "CANSocket(bustype=" \
91                                  "'%s', channel='%s')" % \
92                                  (interface, channel)
93               sock = CANSocket(bustype=interface, channel=channel)
94       else:
95           sock = CANSocket(channel=channel)
96           interface_string = "\"%s\"" % channel
97   
98       return sock, interface_string

101  def main():
102      # type: () -> None
103      extended = False
104      piso = False
105      verbose = False
106      extended_can_id = False
107      sniff_time = 100
108      noise_listen_time = 2
109      start = None
110      end = None
111      channel = None
112      interface = None
113      python_can_args = None
114      conf.verb = -1
115  
116      options = getopt.getopt(
117          sys.argv[1:],
118          'vxCt:n:i:c:a:s:e:h:w',
119          ['verbose', 'noise_listen_time=', 'sniff_time=', 'interface=', 'piso',
120           'channel=', 'python-can_args=', 'start=', 'end=', 'help', 'extended',
121           'extended_can_id'])
122  
123      try:
124          for opt, arg in options[0]:
125              if opt in ('-v', '--verbose'):
126                  verbose = True
127              elif opt in ('-x', '--extended'):
128                  extended = True
129              elif opt in ('-C', '--piso'):
130                  piso = True
131              elif opt in ('-h', '--help'):
132                  usage(False)
133                  sys.exit(0)
134              elif opt in ('-t', '--sniff_time'):
135                  sniff_time = int(arg)
136              elif opt in ('-n', '--noise_listen_time'):
137                  noise_listen_time = int(arg)
138              elif opt in ('-i', '--interface'):
139                  interface = arg
140              elif opt in ('-c', '--channel'):
141                  channel = arg
142              elif opt in ('-a', '--python-can_args'):
143                  python_can_args = arg
144              elif opt in ('-s', '--start'):
145                  start = int(arg, 16)
146              elif opt in ('-e', '--end'):
147                  end = int(arg, 16)
148              elif opt in '--extended_can_id':
149                  extended_can_id = True
150      except getopt.GetoptError as msg:
151          usage(True)
152          print("ERROR:", msg, file=sys.stderr)
153          raise SystemExit
154  
155      if start is None or \
156              end is None or \
157              channel is None or \
158              (PYTHON_CAN and interface is None):
159          usage(True)
160          print("\nPlease provide all required arguments.\n", file=sys.stderr)
161          sys.exit(1)
162  
163      if end >= 2**29 or start >= 2**29:
164          print("Argument 'start' and 'end' must be < " + hex(2**29),
165                file=sys.stderr)
166          sys.exit(1)
167  
168      if not extended_can_id and (end >= 0x800 or start >= 0x800):
169          print("Standard can identifiers must be < 0x800.\n"
170                "Use --extended_can_id option to scan with "
171                "extended CAN identifiers.",
172                file=sys.stderr)
173          sys.exit(1)
174  
175      if end < start:
176          print("start must be equal or smaller than end.", file=sys.stderr)
177          sys.exit(1)
178  
179      try:
180          sock, interface_string = \
181              create_socket(python_can_args, interface, channel)
182  
183          if verbose:
184              print("Start scan (%s - %s)" % (hex(start), hex(end)))
185  
186          stop_event = threading.Event()
187  
188          def signal_handler(*args):
189              # type: (Any) -> None
190              print('Interrupting scan!')
191              stop_event.set()
192  
193          signal.signal(signal.SIGINT, signal_handler)
194          signal.signal(signal.SIGTERM, signal_handler)
195  
196          result = isotp_scan(sock,
197                              range(start, end + 1),
198                              extended_addressing=extended,
199                              noise_listen_time=noise_listen_time,
200                              sniff_time=float(sniff_time) / 1000,
201                              output_format="code" if piso else "text",
202                              can_interface=interface_string,
203                              extended_can_id=extended_can_id,
204                              verbose=verbose,
205                              stop_event=stop_event)
206  
207          print("Scan: \n%s" % result)
208  
209      except Exception as e:
210          usage(True)
211          print("\nSocket couldn't be created. Check your arguments.\n",
212                file=sys.stderr)
213          print(e, file=sys.stderr)
214          sys.exit(1)
215  
216      finally:
217          if sock is not None and not sock.closed:
218              sock.close()
219  
220  
221  if __name__ == '__main__':
222      main()
