57   class ConnectPositiveResponse(Packet):
58       fields_desc = [
59           FlagsField("resource", 0, 8,
60                      ["cal_pag", "x1", "daq", "stim", "pgm", "x5", "x6", "x7"]),
61           FlagsField("comm_mode_basic", 0, 8,
62                      ["byte_order", "address_granularity_0",
63                       "address_granularity_1", "x3", "x4", "x5",
64                       "slave_block_mode", "optional"]),
65           ByteField("max_cto", 0),
66           ConditionalField(ShortField("max_dto", 0),
67                            lambda p: p.comm_mode_basic.byte_order),
68           ConditionalField(LEShortField("max_dto_le", 0),
69                            lambda p: not p.comm_mode_basic.byte_order),
70           ByteField("xcp_protocol_layer_version_number_msb", 1),
71           ByteField("xcp_transport_layer_version_number_msb", 1)
72       ]
73   
74       def post_dissection(self, pkt):
75           if conf.contribs["XCP"]["allow_byte_order_change"]:
76               new_value = int(self.comm_mode_basic.byte_order)
77               if new_value != conf.contribs["XCP"]["byte_order"]:
78                   conf.contribs["XCP"]["byte_order"] = new_value
79   
80                   desc = "Big Endian" if new_value else "Little Endian"
81                   log_automotive.warning("Byte order changed to {0} because of received "
82                                          "positive connect packet".format(desc))
83   
84           if conf.contribs["XCP"]["allow_ag_change"]:
85               conf.contribs["XCP"][
86                   "Address_Granularity_Byte"] = self.get_address_granularity()
87   
88           if conf.contribs["XCP"]["allow_cto_and_dto_change"]:
89               conf.contribs["XCP"]["MAX_CTO"] = self.max_cto
90               conf.contribs["XCP"]["MAX_DTO"] = self.max_dto or self.max_dto_le
91   
92       def get_address_granularity(self):
93           comm_mode_basic = self.comm_mode_basic
94           if not comm_mode_basic.address_granularity_0 and \
95                   not comm_mode_basic.address_granularity_1:
96               return 1
97           if comm_mode_basic.address_granularity_0 and \
98                   not comm_mode_basic.address_granularity_1:
99               return 2
100          if not comm_mode_basic.address_granularity_0 and \
101                  comm_mode_basic.address_granularity_1:
102              return 4
103          else:
104              log_automotive.warning(
105                  "Getting address granularity from packet failed:"
106                  "both flags are 1")

185          0x01: "XCP_ADD_11",
186          0x02: "XCP_ADD_12",
187          0x03: "XCP_ADD_14",
188          0x04: "XCP_ADD_22",
189          0x05: "XCP_ADD_24",

194          0xFF: "XCP_USER_DEFINED"
195      }
196      fields_desc = [
197          ByteEnumField("checksum_type", 0, checksum_type_dict),
198          # specification says: position 2,3 type byte (not WORD) The example(
199          # Part 5 Example Communication Sequences) shows 2 bytes for

201          # http://read.pudn.com/downloads192/doc/comm/903802/XCP%20-Part%205-%20Example%20Communication%20Sequences%20-1.0.pdf # noqa: E501
202          # --> 2 bytes
203          XCPEndiannessField(ShortField("reserved", 0)),
204          XCPEndiannessField(XIntField("checksum", 0)),
205      ]

359  class DAQResolutionInfoPositiveResponse(Packet):
360      fields_desc = [
361          ByteField("granularity_odt_entry_size_daq", 0),
362          ByteField("max_odt_entry_size_daq", 0),
363          ByteField("granularity_odt_entry_size_stim", 0),
364          ByteField("max_odt_entry_size_stim", 0),
365          FlagsField("timestamp_mode", 0, 8,
366                     ["size_0", "size_1", "size_2", "timestamp_fixed", "unit_0",
367                      "unit_1", "unit_2", "unit_3"]),
368          XCPEndiannessField(ShortField("timestamp_ticks", 0)),
369      ]
370  
371      def get_timestamp_size(self):
372          size_0 = bool(self.timestamp_mode.size_0)
373          size_1 = bool(self.timestamp_mode.size_1)
374          size_2 = bool(self.timestamp_mode.size_2)
375  
376          if not size_2 and not size_1 == 0 and size_0:
377              return 1
378          if not size_2 and size_1 and not size_0:
379              return 2
380          if size_2 and not size_1 and not size_0:
381              return 4
382          return 0
383  
384      def write_timestamp_size_to_config(self):
385          conf.contribs["XCP"]["timestamp_size"] = self.get_timestamp_size()
386  
387      def post_dissection(self, pkt):
388          self.write_timestamp_size_to_config()
