46   class EigrpIPField(StrField, IPField):
47       """ 
48       This is a special field type for handling ip addresses of destination networks in internal and 
49       external route updates. 
50    
51       EIGRP removes zeros from the host portion of the ip address if the netmask is 8, 16 or 24 bits. 
52       """
53   
54       __slots__ = ["length_from"]
55   
56       def __init__(self, name, default, length=None, length_from=None):
57           StrField.__init__(self, name, default)
58           self.length_from = length_from
59           if length is not None:
60               self.length_from = lambda pkt, length=length: length
61   
62       def h2i(self, pkt, x):
63           return IPField.h2i(self, pkt, x)
64   
65       def i2m(self, pkt, x):
66           x = inet_aton(x)
67           tmp_len = self.length_from(pkt)
68   
69           if tmp_len <= 8:
70               return x[:1]
71           elif tmp_len <= 16:
72               return x[:2]
73           elif tmp_len <= 24:
74               return x[:3]
75           else:
76               return x
77   
78       def m2i(self, pkt, x):
79           tmp_len = self.length_from(pkt)
80   
81           if tmp_len <= 8:
82               x += b"\x00\x00\x00"
83           elif tmp_len <= 16:
84               x += b"\x00\x00"
85           elif tmp_len <= 24:
86               x += b"\x00"
87   
88           return inet_ntoa(x)
89   
90       def prefixlen_to_bytelen(self, tmp_len):
91           if tmp_len <= 8:
92               tmp_len = 1
93           elif tmp_len <= 16:
94               tmp_len = 2
95           elif tmp_len <= 24:
96               tmp_len = 3
97           else:
98               tmp_len = 4
99   
100          return tmp_len
101  
102      def i2len(self, pkt, x):
103          tmp_len = self.length_from(pkt)
104          tmp_len = self.prefixlen_to_bytelen(tmp_len)
105          return tmp_len
106  
107      def getfield(self, pkt, s):
108          tmp_len = self.length_from(pkt)
109          tmp_len = self.prefixlen_to_bytelen(tmp_len)
110          return s[tmp_len:], self.m2i(pkt, s[:tmp_len])
111  
112      def randval(self):
113          return IPField.randval(self)

116  class EigrpIP6Field(StrField, IP6Field):
117      """ 
118      This is a special field type for handling ip addresses of destination networks in internal and 
119      external route updates. 
120   
121      """
122  
123      __slots__ = ["length_from"]
124  
125      def __init__(self, name, default, length=None, length_from=None):
126          StrField.__init__(self, name, default)
127          self.length_from = length_from
128          if length is not None:
129              self.length_from = lambda pkt, length=length: length
130  
131      def any2i(self, pkt, x):
132          return IP6Field.any2i(self, pkt, x)
133  
134      def i2repr(self, pkt, x):
135          return IP6Field.i2repr(self, pkt, x)
136  
137      def h2i(self, pkt, x):
138          return IP6Field.h2i(self, pkt, x)
139  
140      def i2m(self, pkt, x):
141          x = inet_pton(socket.AF_INET6, x)
142          tmp_len = self.length_from(pkt)
143          tmp_len = self.prefixlen_to_bytelen(tmp_len)
144  
145          return x[:tmp_len]
146  
147      def m2i(self, pkt, x):
148          tmp_len = self.length_from(pkt)
149  
150          prefixlen = self.prefixlen_to_bytelen(tmp_len)
151          if tmp_len > 128:
152              warning("EigrpIP6Field: Prefix length is > 128. Dissection of this packet will fail")  # noqa: E501
153          else:
154              pad = b"\x00" * (16 - prefixlen)
155              x += pad
156  
157          return inet_ntop(socket.AF_INET6, x)
158  
159      def prefixlen_to_bytelen(self, plen):
160          plen = plen // 8
161  
162          if plen < 16:
163              plen += 1
164  
165          return plen
166  
167      def i2len(self, pkt, x):
168          tmp_len = self.length_from(pkt)
169          tmp_len = self.prefixlen_to_bytelen(tmp_len)
170          return tmp_len
171  
172      def getfield(self, pkt, s):
173          tmp_len = self.length_from(pkt)
174          tmp_len = self.prefixlen_to_bytelen(tmp_len)
175          return s[tmp_len:], self.m2i(pkt, s[:tmp_len])
176  
177      def randval(self):
178          return IP6Field.randval(self)

210  class EIGRPAuthData(EIGRPGeneric):
211      name = "EIGRP Authentication Data"
212      fields_desc = [XShortField("type", 0x0002),
213                     FieldLenField("len", None, "authdata", "!H", adjust=lambda pkt, x: x + 24),  # noqa: E501
214                     ShortEnumField("authtype", 2, {2: "MD5"}),
215                     ShortField("keysize", None),
216                     IntField("keyid", 1),
217                     StrFixedLenField("nullpad", b"\x00" * 12, 12),
218                     StrLenField("authdata", RandString(16), length_from=lambda pkt: pkt.keysize)  # noqa: E501
219                     ]
220  
221      def post_build(self, p, pay):
222          p += pay
223  
224          if self.keysize is None:
225              keysize = len(self.authdata)
226              p = p[:6] + chb((keysize >> 8) & 0xff) + chb(keysize & 0xff) + p[8:]  # noqa: E501
227  
228          return p

232      name = "EIGRP Sequence"
233      fields_desc = [XShortField("type", 0x0003),
234                     ShortField("len", None),
235                     ByteField("addrlen", 4),
236                     ConditionalField(IPField("ipaddr", "192.168.0.1"),
237                                      lambda pkt:pkt.addrlen == 4),
238                     ConditionalField(IP6Field("ip6addr", "2001::"),
239                                      lambda pkt:pkt.addrlen == 16)
240                     ]
241  
242      def post_build(self, p, pay):
243          p += pay
244  
245          if self.len is None:
246              tmp_len = len(p)
247              tmp_p = p[:2] + chb((tmp_len >> 8) & 0xff)
248              p = tmp_p + chb(tmp_len & 0xff) + p[4:]
249  
250          return p

253  class ShortVersionField(ShortField):
254      def i2repr(self, pkt, x):
255          try:
256              minor = x & 0xff
257              major = (x >> 8) & 0xff
258          except TypeError:
259              return "unknown"
260          else:
261              # We print a leading 'v' so that these values don't look like floats  # noqa: E501
262              return "v%s.%s" % (major, minor)
263  
264      def h2i(self, pkt, x):
265          """The field accepts string values like v12.1, v1.1 or integer values. 
266             String values have to start with a "v" followed by a 
267             floating point number. Valid numbers are between 0 and 255. 
268   
269          """
270  
271          if isinstance(x, str) and x.startswith("v") and len(x) <= 8:
272              major = int(x.split(".")[0][1:])
273              minor = int(x.split(".")[1])
274  
275              return (major << 8) | minor
276  
277          elif isinstance(x, int) and 0 <= x <= 65535:
278              return x
279          else:
280              if not hasattr(self, "default"):
281                  return x
282              if self.default is not None:
283                  warning("set value to default. Format of %r is invalid", x)
284                  return self.default
285              else:
286                  raise Scapy_Exception("Format of value is invalid")
287  
288      def randval(self):
289          return RandShort()

436  def _EIGRPGuessPayloadClass(p, **kargs):
437      cls = conf.raw_layer
438      if len(p) >= 2:
439          t = struct.unpack("!H", p[:2])[0]
440          clsname = _eigrp_tlv_cls.get(t, "EIGRPGeneric")
441          cls = globals()[clsname]
442      return cls(p, **kargs)

460  class EIGRP(Packet):
461      name = "EIGRP"
462      fields_desc = [ByteField("ver", 2),
463                     ByteEnumField("opcode", 5, _EIGRP_OPCODES),
464                     XShortField("chksum", None),
465                     FlagsField("flags", 0, 32, _EIGRP_FLAGS),
466                     IntField("seq", 0),
467                     IntField("ack", 0),
468                     IntField("asn", 100),
469                     PacketListField("tlvlist", [], _EIGRPGuessPayloadClass)
470                     ]
471  
472      def post_build(self, p, pay):
473          p += pay
474          if self.chksum is None:
475              c = checksum(p)
476              p = p[:2] + chb((c >> 8) & 0xff) + chb(c & 0xff) + p[4:]
477          return p
478  
479      def mysummary(self):
480          summarystr = "EIGRP (AS=%EIGRP.asn% Opcode=%EIGRP.opcode%"
481          if self.opcode == 5 and self.ack != 0:
482              summarystr += " (ACK)"
483          if self.flags != 0:
484              summarystr += " Flags=%EIGRP.flags%"
485  
486          return self.sprintf(summarystr + ")")
