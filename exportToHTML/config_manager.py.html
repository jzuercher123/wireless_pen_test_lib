55   class ConfigManager:
56       def __init__(self, config_dir: str = "config"):
57           self.logger = logging.getLogger(self.__class__.__name__)
58           self.config_dir = config_dir
59           self.default_config_path = os.path.join(self.config_dir, "config_defaults.yaml")
60           self.user_config_path = os.path.join(self.config_dir, "config.yaml")
61   
62           # Load environment variables from .env file if it exists
63           dotenv_path = os.path.join(os.getcwd(), '.env')
64           if os.path.exists(dotenv_path):
65               load_dotenv(dotenv_path)
66               self.logger.debug("Loaded environment variables from .env file.")

68           self.config: Optional[Config] = None
69           self.load_config()

73       def load_yaml(self, path: str) -> Dict:
74           if not os.path.exists(path):
75               self.logger.warning(f"Configuration file {path} does not exist.")
76               return {}
77           with open(path, 'r') as f:
78               try:
79                   data = yaml.safe_load(f) or {}
80                   self.logger.debug(f"Loaded configuration from {path}")
81                   return data
82               except yaml.YAMLError as e:
83                   self.logger.error(f"Error parsing YAML file {path}: {e}")
84                   return {}
85   
86       def merge_configs(self, default: Dict, override: Dict) -> Dict:
87           """ 
88           Recursively merge two dictionaries. 
89           """
90           for key, value in override.items():
91               if key in default and isinstance(default[key], dict) and isinstance(value, dict):
92                   default[key] = self.merge_configs(default[key], value)
93               else:
94                   default[key] = value
95           return default
96   
97       def load_config(self):
98           """ 
99           Load and validate configuration from YAML files and environment variables. 
100          """
101          # Load default and user configurations
102          default_config = self.load_yaml(self.default_config_path)
103          user_config = self.load_yaml(self.user_config_path)
104          merged_config = self.merge_configs(default_config, user_config)
105  
106          # Override with environment variables if set
107          merged_config = self.override_with_env(merged_config)
108  
109          # Validate the merged configuration
110          try:
111              self.config = Config(**merged_config)
112              self.logger.info("Configuration loaded and validated successfully.")
113          except ValidationError as e:
114              self.logger.error(f"Configuration validation error: {e}")
115              raise e

117      def override_with_env(self, config: Dict) -> Dict:
118          """ 
119          Override configuration with environment variables if they exist. 
120          The environment variables should be in uppercase and use underscores to separate sections. 
121          Example: GENERAL_INTERFACE, SCANNERS_ENCRYPTION_SCANNER_SCAN_DURATION 
122          """
123          for section in config:
124              for key in config[section]:
125                  env_var = f"{section.upper()}_{key.upper()}"
126                  if isinstance(config[section][key], dict):
127                      for sub_key in config[section][key]:
128                          sub_env_var = f"{env_var}_{sub_key.upper()}"
129                          if sub_env_var in os.environ:
130                              config[section][key][sub_key] = self.parse_env_value(os.environ[sub_env_var])
131                  else:
132                      if env_var in os.environ:
133                          config[section][key] = self.parse_env_value(os.environ[env_var])
134          return config
135  
136      def parse_env_value(self, value: str):
137          """ 
138          Parse environment variable string to appropriate type. 
139          """
140          # Attempt to parse integers
141          if value.isdigit():
142              return int(value)
143          # Attempt to parse booleans
144          if value.lower() in ['true', 'false']:
145              return value.lower() == 'true'
146          # Return as string
147          return value

149      def get_config(self) -> Config:
150          if not self.config:
151              self.logger.error("Configuration not loaded.")
152              raise Exception("Configuration not loaded.")
153          return self.config

155      def update_user_config(self, updates: Dict):
156          """ 
157          Update the user configuration file with the provided updates. 
158          """
159          user_config = self.load_yaml(self.user_config_path)
160          merged_config = self.merge_configs(user_config, updates)
161          with open(self.user_config_path, 'w') as f:
162              yaml.dump(merged_config, f)
163          self.logger.info(f"User configuration updated: {updates}")
164          self.load_config()
