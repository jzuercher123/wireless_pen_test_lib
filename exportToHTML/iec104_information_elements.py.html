44   def _generate_attributes_and_dicts(cls):
45       """ 
46       create class attributes and dict entries for range-based attributes 
47    
48       class attributes will take the form: cls.<attribute_name_prefix>_<index> 
49    
50       dictionary entries will be generated as: 
51    
52         the_dict[index] = "<dict_entry_prefix> (<index>)" 
53    
54       expects a GENERATED_ATTRIBUTES attribute within the class that contains a 
55       list of the specification for the attributes and dictionary entries to be 
56       generated. each list entry must have this format: 
57    
58           (attribute_name_prefix, dict_entry_prefix, dictionary, first_index, 
59            last_index) 
60    
61       with 
62           <attribute_name_prefix> - the prefix of the attribute name 
63           first_index - index of the first attribute to be generated 
64           last_index - index of the last attribute to be generated 
65       :param cls: the class the attributes should be added to 
66       :return: cls extended by generated attributes 
67       """
68   
69       for attribute_name_prefix, dict_entry_prefix, the_dict, first_index, \
70           last_index \
71               in cls.GENERATED_ATTRIBUTES:
72   
73           for index in range(first_index, last_index + 1):
74               the_dict[index] = '{} ({})'.format(dict_entry_prefix, index)
75   
76               setattr(cls, '{}_{}'.format(attribute_name_prefix, index), index)
77   
78       return cls
