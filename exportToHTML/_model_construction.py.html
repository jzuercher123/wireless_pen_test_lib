58   class _ModelNamespaceDict(dict):
59       """A dictionary subclass that intercepts attribute setting on model classes and 
60       warns about overriding of decorators. 
61       """
62   
63       def __setitem__(self, k: str, v: object) -> None:
64           existing: Any = self.get(k, None)
65           if existing and v is not existing and isinstance(existing, PydanticDescriptorProxy):
66               warnings.warn(f'`{k}` overrides an existing Pydantic `{existing.decorator_info.decorator_repr}` decorator')
67   
68           return super().__setitem__(k, v)

81   @dataclass_transform(kw_only_default=True, field_specifiers=(PydanticModelField, PydanticModelPrivateAttr, NoInitField))
82   class ModelMetaclass(ABCMeta):
83       def __new__(
84           mcs,
85           cls_name: str,
86           bases: tuple[type[Any], ...],
87           namespace: dict[str, Any],
88           __pydantic_generic_metadata__: PydanticGenericMetadata | None = None,
89           __pydantic_reset_parent_namespace__: bool = True,
90           _create_model_module: str | None = None,
91           **kwargs: Any,
92       ) -> type:
93           """Metaclass for creating Pydantic models. 
94    
95           Args: 
96               cls_name: The name of the class to be created. 
97               bases: The base classes of the class to be created. 
98               namespace: The attribute dictionary of the class to be created. 
99               __pydantic_generic_metadata__: Metadata for generic models. 
100              __pydantic_reset_parent_namespace__: Reset parent namespace. 
101              _create_model_module: The module of the class to be created, if created by `create_model`. 
102              **kwargs: Catch-all for any other keyword arguments. 
103   
104          Returns: 
105              The new class created by the metaclass. 
106          """
107          # Note `ModelMetaclass` refers to `BaseModel`, but is also used to *create* `BaseModel`, so we rely on the fact
108          # that `BaseModel` itself won't have any bases, but any subclass of it will, to determine whether the `__new__`
109          # call we're in the middle of is for the `BaseModel` class.
110          if bases:
111              base_field_names, class_vars, base_private_attributes = mcs._collect_bases_data(bases)
112  
113              config_wrapper = ConfigWrapper.for_model(bases, namespace, kwargs)
114              namespace['model_config'] = config_wrapper.config_dict
115              private_attributes = inspect_namespace(
116                  namespace, config_wrapper.ignored_types, class_vars, base_field_names
117              )
118              if private_attributes or base_private_attributes:
119                  original_model_post_init = get_model_post_init(namespace, bases)
120                  if original_model_post_init is not None:
121                      # if there are private_attributes and a model_post_init function, we handle both
122  
123                      def wrapped_model_post_init(self: BaseModel, context: Any, /) -> None:
124                          """We need to both initialize private attributes and call the user-defined model_post_init 
125                          method. 
126                          """
127                          init_private_attributes(self, context)
128                          original_model_post_init(self, context)
129  
130                      namespace['model_post_init'] = wrapped_model_post_init
131                  else:
132                      namespace['model_post_init'] = init_private_attributes
133  
134              namespace['__class_vars__'] = class_vars
135              namespace['__private_attributes__'] = {**base_private_attributes, **private_attributes}
136  
137              cls: type[BaseModel] = super().__new__(mcs, cls_name, bases, namespace, **kwargs)  # type: ignore
138  
139              BaseModel = import_cached_base_model()
140  
141              mro = cls.__mro__
142              if Generic in mro and mro.index(Generic) < mro.index(BaseModel):
143                  warnings.warn(
144                      GenericBeforeBaseModelWarning(
145                          'Classes should inherit from `BaseModel` before generic classes (e.g. `typing.Generic[T]`) '
146                          'for pydantic generics to work properly.'
147                      ),
148                      stacklevel=2,
149                  )
150  
151              cls.__pydantic_custom_init__ = not getattr(cls.__init__, '__pydantic_base_init__', False)
152              cls.__pydantic_post_init__ = None if cls.model_post_init is BaseModel.model_post_init else 'model_post_init'
153  
154              cls.__pydantic_decorators__ = DecoratorInfos.build(cls)
155  
156              # Use the getattr below to grab the __parameters__ from the `typing.Generic` parent class
157              if __pydantic_generic_metadata__:
158                  cls.__pydantic_generic_metadata__ = __pydantic_generic_metadata__
159              else:
160                  parent_parameters = getattr(cls, '__pydantic_generic_metadata__', {}).get('parameters', ())
161                  parameters = getattr(cls, '__parameters__', None) or parent_parameters
162                  if parameters and parent_parameters and not all(x in parameters for x in parent_parameters):
163                      from ..root_model import RootModelRootType
164  
165                      missing_parameters = tuple(x for x in parameters if x not in parent_parameters)
166                      if RootModelRootType in parent_parameters and RootModelRootType not in parameters:
167                          # This is a special case where the user has subclassed `RootModel`, but has not parametrized
168                          # RootModel with the generic type identifiers being used. Ex:
169                          # class MyModel(RootModel, Generic[T]):
170                          #    root: T
171                          # Should instead just be:
172                          # class MyModel(RootModel[T]):
173                          #   root: T
174                          parameters_str = ', '.join([x.__name__ for x in missing_parameters])
175                          error_message = (
176                              f'{cls.__name__} is a subclass of `RootModel`, but does not include the generic type identifier(s) '
177                              f'{parameters_str} in its parameters. '
178                              f'You should parametrize RootModel directly, e.g., `class {cls.__name__}(RootModel[{parameters_str}]): ...`.'
179                          )
180                      else:
181                          combined_parameters = parent_parameters + missing_parameters
182                          parameters_str = ', '.join([str(x) for x in combined_parameters])
183                          generic_type_label = f'typing.Generic[{parameters_str}]'
184                          error_message = (
185                              f'All parameters must be present on typing.Generic;'
186                              f' you should inherit from {generic_type_label}.'
187                          )
188                          if Generic not in bases:  # pragma: no cover
189                              # We raise an error here not because it is desirable, but because some cases are mishandled.
190                              # It would be nice to remove this error and still have things behave as expected, it's just
191                              # challenging because we are using a custom `__class_getitem__` to parametrize generic models,
192                              # and not returning a typing._GenericAlias from it.
193                              bases_str = ', '.join([x.__name__ for x in bases] + [generic_type_label])
194                              error_message += (
195                                  f' Note: `typing.Generic` must go last: `class {cls.__name__}({bases_str}): ...`)'
196                              )
197                      raise TypeError(error_message)
198  
199                  cls.__pydantic_generic_metadata__ = {
200                      'origin': None,
201                      'args': (),
202                      'parameters': parameters,
203                  }
204  
205              cls.__pydantic_complete__ = False  # Ensure this specific class gets completed
206  
207              # preserve `__set_name__` protocol defined in https://peps.python.org/pep-0487
208              # for attributes not in `new_namespace` (e.g. private attributes)
209              for name, obj in private_attributes.items():
210                  obj.__set_name__(cls, name)
211  
212              if __pydantic_reset_parent_namespace__:
213                  cls.__pydantic_parent_namespace__ = build_lenient_weakvaluedict(parent_frame_namespace())
214              parent_namespace = getattr(cls, '__pydantic_parent_namespace__', None)
215              if isinstance(parent_namespace, dict):
216                  parent_namespace = unpack_lenient_weakvaluedict(parent_namespace)
217  
218              types_namespace = merge_cls_and_parent_ns(cls, parent_namespace)
219              set_model_fields(cls, bases, config_wrapper, types_namespace)
220  
221              if config_wrapper.frozen and '__hash__' not in namespace:
222                  set_default_hash_func(cls, bases)
223  
224              complete_model_class(
225                  cls,
226                  cls_name,
227                  config_wrapper,
228                  raise_errors=False,
229                  types_namespace=types_namespace,
230                  create_model_module=_create_model_module,
231              )
232  
233              # If this is placed before the complete_model_class call above,
234              # the generic computed fields return type is set to PydanticUndefined
235              cls.model_computed_fields = {k: v.info for k, v in cls.__pydantic_decorators__.computed_fields.items()}
236  
237              set_deprecated_descriptors(cls)
238  
239              # using super(cls, cls) on the next line ensures we only call the parent class's __pydantic_init_subclass__
240              # I believe the `type: ignore` is only necessary because mypy doesn't realize that this code branch is
241              # only hit for _proper_ subclasses of BaseModel
242              super(cls, cls).__pydantic_init_subclass__(**kwargs)  # type: ignore[misc]
243              return cls
244          else:
245              # These are instance variables, but have been assigned to `NoInitField` to trick the type checker.
246              for instance_slot in '__pydantic_fields_set__', '__pydantic_extra__', '__pydantic_private__':
247                  namespace.pop(
248                      instance_slot,
249                      None,  # In case the metaclass is used with a class other than `BaseModel`.
250                  )
251              namespace.get('__annotations__', {}).clear()
252              return super().__new__(mcs, cls_name, bases, namespace, **kwargs)
253  
254      if not typing.TYPE_CHECKING:  # pragma: no branch
255          # We put `__getattr__` in a non-TYPE_CHECKING block because otherwise, mypy allows arbitrary attribute access
256  
257          def __getattr__(self, item: str) -> Any:
258              """This is necessary to keep attribute access working for class attribute access."""
259              private_attributes = self.__dict__.get('__private_attributes__')
260              if private_attributes and item in private_attributes:
261                  return private_attributes[item]
262              raise AttributeError(item)
263  
264      @classmethod
265      def __prepare__(cls, *args: Any, **kwargs: Any) -> dict[str, object]:
266          return _ModelNamespaceDict()
267  
268      def __instancecheck__(self, instance: Any) -> bool:
269          """Avoid calling ABC _abc_subclasscheck unless we're pretty sure. 
270   
271          See #3829 and python/cpython#92810 
272          """
273          return hasattr(instance, '__pydantic_validator__') and super().__instancecheck__(instance)

275      @staticmethod
276      def _collect_bases_data(bases: tuple[type[Any], ...]) -> tuple[set[str], set[str], dict[str, ModelPrivateAttr]]:
277          BaseModel = import_cached_base_model()
278  
279          field_names: set[str] = set()
280          class_vars: set[str] = set()
281          private_attributes: dict[str, ModelPrivateAttr] = {}
282          for base in bases:
283              if issubclass(base, BaseModel) and base is not BaseModel:
284                  # model_fields might not be defined yet in the case of generics, so we use getattr here:
285                  field_names.update(getattr(base, 'model_fields', {}).keys())
286                  class_vars.update(base.__class_vars__)
287                  private_attributes.update(base.__private_attributes__)
288          return field_names, class_vars, private_attributes
289  
290      @property
291      @deprecated('The `__fields__` attribute is deprecated, use `model_fields` instead.', category=None)
292      def __fields__(self) -> dict[str, FieldInfo]:
293          warnings.warn(
294              'The `__fields__` attribute is deprecated, use `model_fields` instead.', PydanticDeprecatedSince20
295          )
296          return self.model_fields  # type: ignore
297  
298      def __dir__(self) -> list[str]:
299          attributes = list(super().__dir__())
300          if '__fields__' in attributes:
301              attributes.remove('__fields__')
302          return attributes

305  def init_private_attributes(self: BaseModel, context: Any, /) -> None:
306      """This function is meant to behave like a BaseModel method to initialise private attributes. 
307   
308      It takes context as an argument since that's what pydantic-core passes when calling it. 
309   
310      Args: 
311          self: The BaseModel instance. 
312          context: The context. 
313      """
314      if getattr(self, '__pydantic_private__', None) is None:
315          pydantic_private = {}
316          for name, private_attr in self.__private_attributes__.items():
317              default = private_attr.get_default()
318              if default is not PydanticUndefined:
319                  pydantic_private[name] = default
320          object_setattr(self, '__pydantic_private__', pydantic_private)

335  def inspect_namespace(  # noqa C901
336      namespace: dict[str, Any],
337      ignored_types: tuple[type[Any], ...],
338      base_class_vars: set[str],
339      base_class_fields: set[str],
340  ) -> dict[str, ModelPrivateAttr]:
341      """Iterate over the namespace and: 
342      * gather private attributes 
343      * check for items which look like fields but are not (e.g. have no annotation) and warn. 
344   
345      Args: 
346          namespace: The attribute dictionary of the class to be created. 
347          ignored_types: A tuple of ignore types. 
348          base_class_vars: A set of base class class variables. 
349          base_class_fields: A set of base class fields. 
350   
351      Returns: 
352          A dict contains private attributes info. 
353   
354      Raises: 
355          TypeError: If there is a `__root__` field in model. 
356          NameError: If private attribute name is invalid. 
357          PydanticUserError: 
358              - If a field does not have a type annotation. 
359              - If a field on base class was overridden by a non-annotated attribute. 
360      """
361      from ..fields import ModelPrivateAttr, PrivateAttr
362  
363      FieldInfo = import_cached_field_info()
364  
365      all_ignored_types = ignored_types + default_ignored_types()
366  
367      private_attributes: dict[str, ModelPrivateAttr] = {}
368      raw_annotations = namespace.get('__annotations__', {})
369  
370      if '__root__' in raw_annotations or '__root__' in namespace:
371          raise TypeError("To define root models, use `pydantic.RootModel` rather than a field called '__root__'")
372  
373      ignored_names: set[str] = set()
374      for var_name, value in list(namespace.items()):
375          if var_name == 'model_config' or var_name == '__pydantic_extra__':
376              continue
377          elif (
378              isinstance(value, type)
379              and value.__module__ == namespace['__module__']
380              and '__qualname__' in namespace
381              and value.__qualname__.startswith(namespace['__qualname__'])
382          ):
383              # `value` is a nested type defined in this namespace; don't error
384              continue
385          elif isinstance(value, all_ignored_types) or value.__class__.__module__ == 'functools':
386              ignored_names.add(var_name)
387              continue
388          elif isinstance(value, ModelPrivateAttr):
389              if var_name.startswith('__'):
390                  raise NameError(
391                      'Private attributes must not use dunder names;'
392                      f' use a single underscore prefix instead of {var_name!r}.'
393                  )
394              elif is_valid_field_name(var_name):
395                  raise NameError(
396                      'Private attributes must not use valid field names;'
397                      f' use sunder names, e.g. {"_" + var_name!r} instead of {var_name!r}.'
398                  )
399              private_attributes[var_name] = value
400              del namespace[var_name]
401          elif isinstance(value, FieldInfo) and not is_valid_field_name(var_name):
402              suggested_name = var_name.lstrip('_') or 'my_field'  # don't suggest '' for all-underscore name
403              raise NameError(
404                  f'Fields must not use names with leading underscores;'
405                  f' e.g., use {suggested_name!r} instead of {var_name!r}.'
406              )
407  
408          elif var_name.startswith('__'):
409              continue
410          elif is_valid_privateattr_name(var_name):
411              if var_name not in raw_annotations or not is_classvar(raw_annotations[var_name]):
412                  private_attributes[var_name] = PrivateAttr(default=value)
413                  del namespace[var_name]
414          elif var_name in base_class_vars:
415              continue
416          elif var_name not in raw_annotations:
417              if var_name in base_class_fields:
418                  raise PydanticUserError(
419                      f'Field {var_name!r} defined on a base class was overridden by a non-annotated attribute. '
420                      f'All field definitions, including overrides, require a type annotation.',
421                      code='model-field-overridden',
422                  )
423              elif isinstance(value, FieldInfo):
424                  raise PydanticUserError(
425                      f'Field {var_name!r} requires a type annotation', code='model-field-missing-annotation'
426                  )
427              else:
428                  raise PydanticUserError(
429                      f'A non-annotated attribute was detected: `{var_name} = {value!r}`. All model fields require a '
430                      f'type annotation; if `{var_name}` is not meant to be a field, you may be able to resolve this '
431                      f"error by annotating it as a `ClassVar` or updating `model_config['ignored_types']`.",
432                      code='model-field-missing-annotation',
433                  )
434  
435      for ann_name, ann_type in raw_annotations.items():
436          if (
437              is_valid_privateattr_name(ann_name)
438              and ann_name not in private_attributes
439              and ann_name not in ignored_names
440              # This condition is a false negative when `ann_type` is stringified,
441              # but it is handled in `set_model_fields`:
442              and not is_classvar(ann_type)
443              and ann_type not in all_ignored_types
444              and getattr(ann_type, '__module__', None) != 'functools'
445          ):
446              if isinstance(ann_type, str):
447                  # Walking up the frames to get the module namespace where the model is defined
448                  # (as the model class wasn't created yet, we unfortunately can't use `cls.__module__`):
449                  frame = sys._getframe(2)
450                  if frame is not None:
451                      try:
452                          ann_type = eval_type_backport(
453                              _make_forward_ref(ann_type, is_argument=False, is_class=True),
454                              globalns=frame.f_globals,
455                              localns=frame.f_locals,
456                          )
457                      except (NameError, TypeError):
458                          pass
459  
460              if is_annotated(ann_type):
461                  _, *metadata = typing_extensions.get_args(ann_type)
462                  private_attr = next((v for v in metadata if isinstance(v, ModelPrivateAttr)), None)
463                  if private_attr is not None:
464                      private_attributes[ann_name] = private_attr
465                      continue
466              private_attributes[ann_name] = PrivateAttr()
467  
468      return private_attributes

471  def set_default_hash_func(cls: type[BaseModel], bases: tuple[type[Any], ...]) -> None:
472      base_hash_func = get_attribute_from_bases(bases, '__hash__')
473      new_hash_func = make_hash_func(cls)
474      if base_hash_func in {None, object.__hash__} or getattr(base_hash_func, '__code__', None) == new_hash_func.__code__:
475          # If `__hash__` is some default, we generate a hash function.
476          # It will be `None` if not overridden from BaseModel.
477          # It may be `object.__hash__` if there is another
478          # parent class earlier in the bases which doesn't override `__hash__` (e.g. `typing.Generic`).
479          # It may be a value set by `set_default_hash_func` if `cls` is a subclass of another frozen model.
480          # In the last case we still need a new hash function to account for new `model_fields`.
481          cls.__hash__ = new_hash_func
482  
483  
484  def make_hash_func(cls: type[BaseModel]) -> Any:
485      getter = operator.itemgetter(*cls.model_fields.keys()) if cls.model_fields else lambda _: 0
486  
487      def hash_func(self: Any) -> int:
488          try:
489              return hash(getter(self.__dict__))
490          except KeyError:
491              # In rare cases (such as when using the deprecated copy method), the __dict__ may not contain
492              # all model fields, which is how we can get here.
493              # getter(self.__dict__) is much faster than any 'safe' method that accounts for missing keys,
494              # and wrapping it in a `try` doesn't slow things down much in the common case.
495              return hash(getter(SafeGetItemProxy(self.__dict__)))
496  
497      return hash_func

500  def set_model_fields(
501      cls: type[BaseModel], bases: tuple[type[Any], ...], config_wrapper: ConfigWrapper, types_namespace: dict[str, Any]
502  ) -> None:
503      """Collect and set `cls.model_fields` and `cls.__class_vars__`. 
504   
505      Args: 
506          cls: BaseModel or dataclass. 
507          bases: Parents of the class, generally `cls.__bases__`. 
508          config_wrapper: The config wrapper instance. 
509          types_namespace: Optional extra namespace to look for types in. 
510      """
511      typevars_map = get_model_typevars_map(cls)
512      fields, class_vars = collect_model_fields(cls, bases, config_wrapper, types_namespace, typevars_map=typevars_map)
513  
514      cls.model_fields = fields
515      cls.__class_vars__.update(class_vars)
516  
517      for k in class_vars:
518          # Class vars should not be private attributes
519          #     We remove them _here_ and not earlier because we rely on inspecting the class to determine its classvars,
520          #     but private attributes are determined by inspecting the namespace _prior_ to class creation.
521          #     In the case that a classvar with a leading-'_' is defined via a ForwardRef (e.g., when using
522          #     `__future__.annotations`), we want to remove the private attribute which was detected _before_ we knew it
523          #     evaluated to a classvar
524  
525          value = cls.__private_attributes__.pop(k, None)
526          if value is not None and value.default is not PydanticUndefined:
527              setattr(cls, k, value.default)

530  def complete_model_class(
531      cls: type[BaseModel],
532      cls_name: str,
533      config_wrapper: ConfigWrapper,
534      *,
535      raise_errors: bool = True,
536      types_namespace: dict[str, Any] | None,
537      create_model_module: str | None = None,
538  ) -> bool:
539      """Finish building a model class. 
540   
541      This logic must be called after class has been created since validation functions must be bound 
542      and `get_type_hints` requires a class object. 
543   
544      Args: 
545          cls: BaseModel or dataclass. 
546          cls_name: The model or dataclass name. 
547          config_wrapper: The config wrapper instance. 
548          raise_errors: Whether to raise errors. 
549          types_namespace: Optional extra namespace to look for types in. 
550          create_model_module: The module of the class to be created, if created by `create_model`. 
551   
552      Returns: 
553          `True` if the model is successfully completed, else `False`. 
554   
555      Raises: 
556          PydanticUndefinedAnnotation: If `PydanticUndefinedAnnotation` occurs in`__get_pydantic_core_schema__` 
557              and `raise_errors=True`. 
558      """
559      typevars_map = get_model_typevars_map(cls)
560      gen_schema = GenerateSchema(
561          config_wrapper,
562          types_namespace,
563          typevars_map,
564      )
565  
566      handler = CallbackGetCoreSchemaHandler(
567          partial(gen_schema.generate_schema, from_dunder_get_core_schema=False),
568          gen_schema,
569          ref_mode='unpack',
570      )
571  
572      if config_wrapper.defer_build and 'model' in config_wrapper.experimental_defer_build_mode:
573          set_model_mocks(cls, cls_name)
574          return False
575  
576      try:
577          schema = cls.__get_pydantic_core_schema__(cls, handler)
578      except PydanticUndefinedAnnotation as e:
579          if raise_errors:
580              raise
581          set_model_mocks(cls, cls_name, f'`{e.name}`')
582          return False
583  
584      core_config = config_wrapper.core_config(cls)
585  
586      try:
587          schema = gen_schema.clean_schema(schema)
588      except gen_schema.CollectedInvalid:
589          set_model_mocks(cls, cls_name)
590          return False
591  
592      # debug(schema)
593      cls.__pydantic_core_schema__ = schema
594  
595      cls.__pydantic_validator__ = create_schema_validator(
596          schema,
597          cls,
598          create_model_module or cls.__module__,
599          cls.__qualname__,
600          'create_model' if create_model_module else 'BaseModel',
601          core_config,
602          config_wrapper.plugin_settings,
603      )
604      cls.__pydantic_serializer__ = SchemaSerializer(schema, core_config)
605      cls.__pydantic_complete__ = True
606  
607      # set __signature__ attr only for model class, but not for its instances
608      cls.__signature__ = ClassAttribute(
609          '__signature__',
610          generate_pydantic_signature(init=cls.__init__, fields=cls.model_fields, config_wrapper=config_wrapper),
611      )
612      return True

615  def set_deprecated_descriptors(cls: type[BaseModel]) -> None:
616      """Set data descriptors on the class for deprecated fields."""
617      for field, field_info in cls.model_fields.items():
618          if (msg := field_info.deprecation_message) is not None:
619              desc = _DeprecatedFieldDescriptor(msg)
620              desc.__set_name__(cls, field)
621              setattr(cls, field, desc)
622  
623      for field, computed_field_info in cls.model_computed_fields.items():
624          if (
625              (msg := computed_field_info.deprecation_message) is not None
626              # Avoid having two warnings emitted:
627              and not hasattr(unwrap_wrapped_function(computed_field_info.wrapped_property), '__deprecated__')
628          ):
629              desc = _DeprecatedFieldDescriptor(msg, computed_field_info.wrapped_property)
630              desc.__set_name__(cls, field)
631              setattr(cls, field, desc)

634  class _DeprecatedFieldDescriptor:
635      """Data descriptor used to emit a runtime deprecation warning before accessing a deprecated field. 
636   
637      Attributes: 
638          msg: The deprecation message to be emitted. 
639          wrapped_property: The property instance if the deprecated field is a computed field, or `None`. 
640          field_name: The name of the field being deprecated. 
641      """
642  
643      field_name: str
644  
645      def __init__(self, msg: str, wrapped_property: property | None = None) -> None:
646          self.msg = msg
647          self.wrapped_property = wrapped_property
648  
649      def __set_name__(self, cls: type[BaseModel], name: str) -> None:
650          self.field_name = name
651  
652      def __get__(self, obj: BaseModel | None, obj_type: type[BaseModel] | None = None) -> Any:
653          if obj is None:
654              raise AttributeError(self.field_name)
655  
656          warnings.warn(self.msg, builtins.DeprecationWarning, stacklevel=2)
657  
658          if self.wrapped_property is not None:
659              return self.wrapped_property.__get__(obj, obj_type)
660          return obj.__dict__[self.field_name]
661  
662      # Defined to take precedence over the instance's dictionary
663      # Note that it will not be called when setting a value on a model instance
664      # as `BaseModel.__setattr__` is defined and takes priority.
665      def __set__(self, obj: Any, value: Any) -> NoReturn:
666          raise AttributeError(self.field_name)

669  class _PydanticWeakRef:
670      """Wrapper for `weakref.ref` that enables `pickle` serialization. 
671   
672      Cloudpickle fails to serialize `weakref.ref` objects due to an arcane error related 
673      to abstract base classes (`abc.ABC`). This class works around the issue by wrapping 
674      `weakref.ref` instead of subclassing it. 
675   
676      See https://github.com/pydantic/pydantic/issues/6763 for context. 
677   
678      Semantics: 
679          - If not pickled, behaves the same as a `weakref.ref`. 
680          - If pickled along with the referenced object, the same `weakref.ref` behavior 
681            will be maintained between them after unpickling. 
682          - If pickled without the referenced object, after unpickling the underlying 
683            reference will be cleared (`__call__` will always return `None`). 
684      """
685  
686      def __init__(self, obj: Any):
687          if obj is None:
688              # The object will be `None` upon deserialization if the serialized weakref
689              # had lost its underlying object.
690              self._wr = None
691          else:
692              self._wr = weakref.ref(obj)
693  
694      def __call__(self) -> Any:
695          if self._wr is None:
696              return None
697          else:
698              return self._wr()
699  
700      def __reduce__(self) -> tuple[Callable, tuple[weakref.ReferenceType | None]]:
701          return _PydanticWeakRef, (self(),)

704  def build_lenient_weakvaluedict(d: dict[str, Any] | None) -> dict[str, Any] | None:
705      """Takes an input dictionary, and produces a new value that (invertibly) replaces the values with weakrefs. 
706   
707      We can't just use a WeakValueDictionary because many types (including int, str, etc.) can't be stored as values 
708      in a WeakValueDictionary. 
709   
710      The `unpack_lenient_weakvaluedict` function can be used to reverse this operation. 
711      """
712      if d is None:
713          return None
714      result = {}
715      for k, v in d.items():
716          try:
717              proxy = _PydanticWeakRef(v)
718          except TypeError:
719              proxy = v
720          result[k] = proxy
721      return result
722  
723  
724  def unpack_lenient_weakvaluedict(d: dict[str, Any] | None) -> dict[str, Any] | None:
725      """Inverts the transform performed by `build_lenient_weakvaluedict`."""
726      if d is None:
727          return None
728  
729      result = {}
730      for k, v in d.items():
731          if isinstance(v, _PydanticWeakRef):
732              v = v()
733              if v is not None:
734                  result[k] = v
735          else:
736              result[k] = v
737      return result
