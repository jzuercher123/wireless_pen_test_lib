53   class HPackMagicBitField(fields.BitField):
54       """ HPackMagicBitField is a BitField variant that cannot be assigned another 
55       value than the default one. This field must not be used where there is 
56       potential for fuzzing. OTOH, this field makes sense (for instance, if the 
57       magic bits are used by a dispatcher to select the payload class) 
58       """
59   
60       __slots__ = ['_magic']
61   
62       def __init__(self, name, default, size):
63           # type: (str, int, int) -> None
64           """ 
65           :param str name: this field instance name. 
66           :param int default: this field only valid value. 
67           :param int size: this bitfield bitlength. 
68           :return: None 
69           :raises: AssertionError 
70           """
71           assert default >= 0
72           # size can be negative if encoding is little-endian (see rev property of bitfields)  # noqa: E501
73           assert size != 0
74           self._magic = default
75           super(HPackMagicBitField, self).__init__(name, default, size)
76   
77       def addfield(self, pkt, s, val):
78           # type: (Optional[packet.Packet], Union[str, Tuple[str, int, int]], int) -> Union[str, Tuple[str, int, int]]  # noqa: E501
79           """ 
80           :param packet.Packet|None pkt: the packet instance containing this field instance; probably unused.  # noqa: E501 
81           :param str|(str, int, long) s: either a str if 0 == size%8 or a tuple with the string to add this field to, the  # noqa: E501 
82             number of bits already generated and the generated value so far. 
83           :param int val: unused; must be equal to default value 
84           :return: str|(str, int, long): the s string extended with this field machine representation  # noqa: E501 
85           :raises: AssertionError 
86           """
87           assert val == self._magic, 'val parameter must value {}; received: {}'.format(self._magic, val)  # noqa: E501
88           return super(HPackMagicBitField, self).addfield(pkt, s, self._magic)
89   
90       def getfield(self, pkt, s):
91           # type: (Optional[packet.Packet], Union[str, Tuple[str, int]]) -> Tuple[Union[Tuple[str, int], str], int]  # noqa: E501
92           """ 
93           :param packet.Packet|None pkt: the packet instance containing this field instance; probably unused.  # noqa: E501 
94           :param str|(str, int) s: either a str if size%8==0 or a tuple with the string to parse from and the number of  # noqa: E501 
95             bits already consumed by previous bitfield-compatible fields. 
96           :return: (str|(str, int), int): Returns the remaining string and the parsed value. May return a tuple if there  # noqa: E501 
97             are remaining bits to parse in the first byte. Returned value is equal to default value  # noqa: E501 
98           :raises: AssertionError 
99           """
100          r = super(HPackMagicBitField, self).getfield(pkt, s)
101          assert (
102              isinstance(r, tuple) and

105          ), 'Second element of BitField.getfield return value expected to be an int or a long; API change detected'  # noqa: E501
106          assert r[1] == self._magic, 'Invalid value parsed from s; error in class guessing detected!'  # noqa: E501
107          return r

170  class AbstractUVarIntField(fields.Field):
171      """AbstractUVarIntField represents an integer as defined in RFC7541 
172      """
173  
174      __slots__ = ['_max_value', 'size', 'rev']
175      """ 
176      :var int size: the bit length of the prefix of this AbstractUVarIntField. It  # noqa: E501 
177          represents the complement of the number of MSB that are used in the 
178          current byte for other purposes by some other BitFields 
179      :var int _max_value: the maximum value that can be stored in the 
180          sole prefix. If the integer equals or exceeds this value, the max prefix 
181          value is assigned to the size first bits and the multibyte representation 
182          is used 
183      :var bool rev: is a fake property, also emulated for the sake of 
184          compatibility with Bitfields 
185      """
186  
187      def __init__(self, name, default, size):
188          # type: (str, Optional[int], int) -> None
189          """ 
190          :param str name: the name of this field instance 
191          :param int|None default: positive, null or None default value for this field instance.  # noqa: E501 
192          :param int size: the number of bits to consider in the first byte. Valid range is ]0;8]  # noqa: E501 
193          :return: None 
194          :raises: AssertionError 
195          """
196          assert default is None or (isinstance(default, int) and default >= 0)
197          assert 0 < size <= 8
198          super(AbstractUVarIntField, self).__init__(name, default)
199          self.size = size
200          self._max_value = (1 << self.size) - 1
201  
202          # Configuring the fake property that is useless for this class
203          # but that is expected from BitFields
204          self.rev = False

206      def h2i(self, pkt, x):
207          # type: (Optional[packet.Packet], Optional[int]) -> Optional[int]
208          """ 
209          :param packet.Packet|None pkt: unused. 
210          :param int|None x: the value to convert. 
211          :return: int|None: the converted value. 
212          :raises: AssertionError 
213          """
214          assert not isinstance(x, int) or x >= 0
215          return x

221          :param int|None x: the value to convert. 
222          :return:: int|None: the converted value.

226      def _detect_multi_byte(self, fb):
227          # type: (str) -> bool
228          """ _detect_multi_byte returns whether the AbstractUVarIntField is represented on  # noqa: E501 
229            multiple bytes or not. 
230   
231            A multibyte representation is indicated by all of the first size bits being set  # noqa: E501 
232   
233          :param str fb: first byte, as a character. 
234          :return: bool: True if multibyte repr detected, else False. 
235          :raises: AssertionError 
236          """
237          assert isinstance(fb, int) or len(fb) == 1
238          return (orb(fb) & self._max_value) == self._max_value
239  
240      def _parse_multi_byte(self, s):
241          # type: (str) -> int
242          """ _parse_multi_byte parses x as a multibyte representation to get the 
243            int value of this AbstractUVarIntField. 
244   
245          :param str s: the multibyte string to parse. 
246          :return: int: The parsed int value represented by this AbstractUVarIntField.  # noqa: E501 
247          :raises:: AssertionError 
248          :raises:: Scapy_Exception if the input value encodes an integer larger than 1<<64  # noqa: E501 
249          """
250  
251          assert len(s) >= 2
252  
253          tmp_len = len(s)
254  
255          value = 0
256          i = 1
257          byte = orb(s[i])
258          # For CPU sake, stops at an arbitrary large number!
259          max_value = 1 << 64
260          # As long as the MSG is set, an another byte must be read
261          while byte & 0x80:
262              value += (byte ^ 0x80) << (7 * (i - 1))
263              if value > max_value:
264                  raise error.Scapy_Exception(
265                      'out-of-bound value: the string encodes a value that is too large (>2^{{64}}): {}'.format(value)  # noqa: E501
266                  )
267              i += 1
268              assert i < tmp_len, 'EINVAL: x: out-of-bound read: the string ends before the AbstractUVarIntField!'  # noqa: E501
269              byte = orb(s[i])
270          value += byte << (7 * (i - 1))
271          value += self._max_value
272  
273          assert value >= 0
274          return value
275  
276      def m2i(self, pkt, x):
277          # type: (Optional[packet.Packet], Union[str, Tuple[str, int]]) -> int
278          """ 
279            A tuple is expected for the "x" param only if "size" is different than 8. If a tuple is received, some bits  # noqa: E501 
280            were consumed by another field. This field consumes the remaining bits, therefore the int of the tuple must  # noqa: E501 
281            equal "size". 
282   
283          :param packet.Packet|None pkt: unused. 
284          :param str|(str, int) x: the string to convert. If bits were consumed by a previous bitfield-compatible field.  # noqa: E501 
285          :raises: AssertionError 
286          """
287          assert isinstance(x, bytes) or (isinstance(x, tuple) and x[1] >= 0)
288  
289          if isinstance(x, tuple):
290              assert (8 - x[1]) == self.size, 'EINVAL: x: not enough bits remaining in current byte to read the prefix'  # noqa: E501
291              val = x[0]
292          else:
293              assert isinstance(x, bytes) and self.size == 8, 'EINVAL: x: tuple expected when prefix_len is not a full byte'  # noqa: E501
294              val = x
295  
296          if self._detect_multi_byte(val[0]):
297              ret = self._parse_multi_byte(val)
298          else:
299              ret = orb(val[0]) & self._max_value
300  
301          assert ret >= 0
302          return ret

304      def i2m(self, pkt, x):
305          # type: (Optional[packet.Packet], int) -> str
306          """ 
307          :param packet.Packet|None pkt: unused. 
308          :param int x: the value to convert. 
309          :return: str: the converted value. 
310          :raises: AssertionError 
311          """
312          assert x >= 0
313  
314          if x < self._max_value:
315              return chb(x)
316          else:
317              # The sl list join is a performance trick, because string
318              # concatenation is not efficient with Python immutable strings
319              sl = [chb(self._max_value)]
320              x -= self._max_value
321              while x >= 0x80:
322                  sl.append(chb(0x80 | (x & 0x7F)))
323                  x >>= 7
324              sl.append(chb(x))
325              return b''.join(sl)

327      def any2i(self, pkt, x):
328          # type: (Optional[packet.Packet], Union[None, str, int]) -> Optional[int]  # noqa: E501
329          """ 
330            A "x" value as a string is parsed as a binary encoding of a UVarInt. An int is considered an internal value.  # noqa: E501 
331            None is returned as is. 
332   
333          :param packet.Packet|None pkt: the packet containing this field; probably unused.  # noqa: E501 
334          :param str|int|None x: the value to convert. 
335          :return: int|None: the converted value. 
336          :raises: AssertionError 
337          """
338          if isinstance(x, type(None)):
339              return x
340          if isinstance(x, int):
341              assert x >= 0
342              ret = self.h2i(pkt, x)
343              assert isinstance(ret, int) and ret >= 0
344              return ret
345          elif isinstance(x, bytes):
346              ret = self.m2i(pkt, x)
347              assert (isinstance(ret, int) and ret >= 0)
348              return ret
349          assert False, 'EINVAL: x: No idea what the parameter format is'

360      def addfield(self, pkt, s, val):
361          # type: (Optional[packet.Packet], Union[str, Tuple[str, int, int]], int) -> str  # noqa: E501
362          """ 
363            An AbstractUVarIntField prefix always consumes the remaining bits 
364            of a BitField;if no current BitField is in use (no tuple in 
365            entry) then the prefix length is 8 bits and the whole byte is to 
366            be consumed 
367   
368          :param packet.Packet|None pkt: the packet containing this field. 
369            Probably unused. 
370          :param str|(str, int, long) s: the string to append this field to. 
371            A tuple indicates that some bits were already generated by another 
372            bitfield-compatible field. This MUST be the case if "size" is not 8. 
373            The int is the number of bits already generated in the first byte of 
374            the str. The long is the value that was generated by the previous 
375            bitfield-compatible fields. 
376          :param int val: the positive or null value to be added. 
377          :return: str: s concatenated with the machine representation of this 
378            field. 
379          :raises: AssertionError 
380          """
381          assert val >= 0
382          if isinstance(s, bytes):
383              assert self.size == 8, 'EINVAL: s: tuple expected when prefix_len is not a full byte'  # noqa: E501
384              return s + self.i2m(pkt, val)
385  
386          # s is a tuple
387          # assert s[1] >= 0
388          # assert s[2] >= 0
389          # assert (8 - s[1]) == self.size, 'EINVAL: s: not enough bits remaining in current byte to read the prefix'  # noqa: E501
390  
391          if val >= self._max_value:
392              return s[0] + chb((s[2] << self.size) + self._max_value) + self.i2m(pkt, val)[1:]  # noqa: E501
393          # This AbstractUVarIntField is only one byte long; setting the prefix value  # noqa: E501
394          # and appending the resulting byte to the string
395          return s[0] + chb((s[2] << self.size) + orb(self.i2m(pkt, val)))

397      @staticmethod
398      def _detect_bytelen_from_str(s):
399          # type: (str) -> int
400          """ _detect_bytelen_from_str returns the length of the machine 
401            representation of an AbstractUVarIntField starting at the beginning 
402            of s and which is assumed to expand over multiple bytes 
403            (value > _max_prefix_value). 
404   
405          :param str s: the string to parse. It is assumed that it is a multibyte int.  # noqa: E501 
406          :return: The bytelength of the AbstractUVarIntField. 
407          :raises: AssertionError 
408          """
409          assert len(s) >= 2
410          tmp_len = len(s)
411  
412          i = 1
413          while orb(s[i]) & 0x80 > 0:
414              i += 1
415              assert i < tmp_len, 'EINVAL: s: out-of-bound read: unfinished AbstractUVarIntField detected'  # noqa: E501
416          ret = i + 1
417  
418          assert ret >= 0
419          return ret

421      def i2len(self, pkt, x):
422          # type: (Optional[packet.Packet], int) -> int
423          """ 
424          :param packet.Packet|None pkt: unused. 
425          :param int x: the positive or null value whose binary size if requested.  # noqa: E501 
426          :raises: AssertionError 
427          """
428          assert x >= 0
429          if x < self._max_value:
430              return 1
431  
432          # x is expressed over multiple bytes
433          x -= self._max_value
434          i = 1
435          if x == 0:
436              i += 1
437          while x > 0:
438              x >>= 7
439              i += 1
440  
441          ret = i
442          assert ret >= 0
443          return ret

445      def getfield(self, pkt, s):
446          # type: (Optional[packet.Packet], Union[str, Tuple[str, int]]) -> Tuple[str, int]  # noqa: E501
447          """ 
448          :param packet.Packet|None pkt: the packet instance containing this 
449            field; probably unused. 
450          :param str|(str, int) s: the input value to get this field value from. 
451            If size is 8, s is a string, else it is a tuple containing the value 
452            and an int indicating the number of bits already consumed in the 
453            first byte of the str. The number of remaining bits to consume in the 
454            first byte must be equal to "size". 
455          :return: (str, int): the remaining bytes of s and the parsed value. 
456          :raises: AssertionError 
457          """
458          if isinstance(s, tuple):
459              assert len(s) == 2
460              temp = s  # type: Tuple[str, int]
461              ts, ti = temp
462              assert ti >= 0
463              assert 8 - ti == self.size, 'EINVAL: s: not enough bits remaining in current byte to read the prefix'  # noqa: E501
464              val = ts
465          else:
466              assert isinstance(s, bytes) and self.size == 8, 'EINVAL: s: tuple expected when prefix_len is not a full byte'  # noqa: E501
467              val = s
468  
469          if self._detect_multi_byte(val[0]):
470              tmp_len = self._detect_bytelen_from_str(val)
471          else:
472              tmp_len = 1
473  
474          ret = val[tmp_len:], self.m2i(pkt, s)
475          assert ret[1] >= 0
476          return ret

555  class FieldUVarLenField(AbstractUVarIntField):
556      __slots__ = ['_length_of', '_adjust']
557  
558      def __init__(self, name, default, size, length_of, adjust=lambda x: x):
559          # type: (str, Optional[int], int, str, Callable[[int], int]) -> None
560          """ Initializes a FieldUVarLenField 
561   
562          :param str name: The name of this field instance. 
563          :param int|None default: the default value of this field instance. 
564          :param int size: the number of bits that are occupied by this field in the first byte of a binary string.  # noqa: E501 
565            size must be in the range ]0;8]. 
566          :param str length_of: The name of the field this field value is measuring/representing.  # noqa: E501 
567          :param callable adjust: A function that modifies the value computed from the "length_of" field.  # noqa: E501 
568   
569          adjust can be used for instance to add a constant to the length_of field  # noqa: E501 
570           length. For instance, let's say that i2len of the length_of field 
571           returns 2. If adjust is lambda x: x+1 In that case, this field will 
572           value 3 at build time. 
573          :return: None 
574          :raises: AssertionError

1016     @classmethod
1017     def huffman_encode(cls, s):
1018         # type: (str) -> Tuple[int, int]
1019         """ huffman_encode returns the bitstring and the bitlength of the 
1020         bitstring representing the string provided as a parameter 
1021  
1022         :param str s: the string to encode 
1023         :return: (int, int): the bitstring of s and its bitlength 
1024         :raises: AssertionError 
1025         """
1026         i = 0
1027         ibl = 0
1028         for c in s:
1029             val, bl = cls._huffman_encode_char(c)
1030             i = (i << bl) + val
1031             ibl += bl
1032 
1033         padlen = 8 - (ibl % 8)
1034         if padlen != 8:
1035             val, bl = cls._huffman_encode_char(EOS())
1036             i = (i << padlen) + (val >> (bl - padlen))
1037             ibl += padlen

1041         assert (ret[1] >= 0)

1044     @classmethod
1045     def huffman_decode(cls, i, ibl):
1046         # type: (int, int) -> str
1047         """ huffman_decode decodes the bitstring provided as parameters. 
1048  
1049         :param int i: the bitstring to decode 
1050         :param int ibl: the bitlength of i 
1051         :return: str: the string decoded from the bitstring 
1052         :raises: AssertionError, InvalidEncodingException 
1053         """
1054         assert i >= 0
1055         assert ibl >= 0
1056 
1057         if isinstance(cls.static_huffman_tree, type(None)):
1058             cls.huffman_compute_decode_tree()
1059         assert not isinstance(cls.static_huffman_tree, type(None))
1060 
1061         s = []
1062         j = 0
1063         interrupted = False
1064         cur = cls.static_huffman_tree
1065         cur_sym = 0
1066         cur_sym_bl = 0
1067         while j < ibl:
1068             b = (i >> (ibl - j - 1)) & 1
1069             cur_sym = (cur_sym << 1) + b
1070             cur_sym_bl += 1
1071             elmt = cur[b]
1072 
1073             if isinstance(elmt, HuffmanNode):
1074                 interrupted = True
1075                 cur = elmt
1076                 if isinstance(cur, type(None)):
1077                     raise AssertionError()
1078             elif isinstance(elmt, EOS):
1079                 raise InvalidEncodingException('Huffman decoder met the full EOS symbol')  # noqa: E501
1080             elif isinstance(elmt, bytes):
1081                 interrupted = False
1082                 s.append(elmt)
1083                 cur = cls.static_huffman_tree
1084                 cur_sym = 0
1085                 cur_sym_bl = 0
1086             else:
1087                 raise InvalidEncodingException('Should never happen, so incidentally it will')  # noqa: E501
1088             j += 1
1089 
1090         if interrupted:
1091             # Interrupted values true if the bitstring ends in the middle of a
1092             # symbol; this symbol must be, according to RFC7541 par5.2 the MSB
1093             # of the EOS symbol
1094             if cur_sym_bl > 7:
1095                 raise InvalidEncodingException('Huffman decoder is detecting padding longer than 7 bits')  # noqa: E501
1096             eos_symbol = cls.static_huffman_code[-1]
1097             eos_msb = eos_symbol[0] >> (eos_symbol[1] - cur_sym_bl)
1098             if eos_msb != cur_sym:
1099                 raise InvalidEncodingException('Huffman decoder is detecting unexpected padding format')  # noqa: E501
1100         return b''.join(s)

1103     def huffman_conv2str(cls, bit_str, bit_len):
1104         # type: (int, int) -> str
1105         """ huffman_conv2str converts a bitstring of bit_len bitlength into a 
1106         binary string. It DOES NOT compress/decompress the bitstring! 
1107  
1108         :param int bit_str: the bitstring to convert. 
1109         :param int bit_len: the bitlength of bit_str. 
1110         :return: str: the converted bitstring as a bytestring. 
1111         :raises: AssertionError 
1112         """
1113         assert bit_str >= 0
1114         assert bit_len >= 0
1115 
1116         byte_len = bit_len // 8
1117         rem_bit = bit_len % 8
1118         if rem_bit != 0:
1119             bit_str <<= 8 - rem_bit
1120             byte_len += 1
1121 
1122         # As usual the list/join tricks is a performance trick to build
1123         # efficiently a Python string
1124         s = []  # type: List[str]
1125         i = 0
1126         while i < byte_len:
1127             s.insert(0, chb((bit_str >> (i * 8)) & 0xFF))
1128             i += 1
1129         return b''.join(s)

1154         # type: () -> None
1155         """ huffman_compute_decode_tree initializes/builds the static_huffman_tree 
1156  
1157         :return: None 
1158         :raises: InvalidEncodingException if there is an encoding problem 
1159         """
1160         cls.static_huffman_tree = HuffmanNode(None, None)
1161         i = 0
1162         for entry in cls.static_huffman_code:
1163             parent = cls.static_huffman_tree
1164             for idx in range(entry[1] - 1, -1, -1):
1165                 b = (entry[0] >> idx) & 1

1167                     raise InvalidEncodingException('Huffman unique prefix violation :/')  # noqa: E501
1168                 if idx == 0:
1169                     parent[b] = chb(i) if i < 256 else EOS()
1170                 elif parent[b] is None:
1171                     parent[b] = HuffmanNode(None, None)
1172                 parent = parent[b]

1214         """

1218         :raises: InvalidEncodingException

1230         :return: (str, HPackStringsInterface): the remaining string after this field was carved out & the extracted  # noqa: E501 
1231           value.

1233         :raises: InvalidEncodingException 
1234         """
1235         tmp_len = self._length_from(pkt)
1236         t = pkt.getfieldval(self._type_from) == 1

1332 class HPackHeaders(packet.Packet):
1333     """HPackHeaders uses the "dispatch_hook" trick of Packet_metaclass to select 
1334     the correct HPack header packet type. For this, the first byte of the string  # noqa: E501 
1335     to dissect is snooped on.

1337     @classmethod
1338     def dispatch_hook(cls, s=None, *_args, **_kwds):
1339         # type: (Optional[str], *Any, **Any) -> Packet_metaclass
1340         """dispatch_hook returns the subclass of HPackHeaders that must be used 
1341         to dissect the string.

1347             return HPackIndexedHdr
1348         if fb & 0x40 != 0:
1349             return HPackLitHdrFldWithIncrIndexing
1350         if fb & 0x20 != 0:
1351             return HPackDynamicSizeUpdate
1352         return HPackLitHdrFldWithoutIndexing

2344         :param int new_entry_size: if called before adding a new entry, the size of the new entry in bytes (following  # noqa: E501 
2345         the RFC7541 definition of the size of an entry) 
2346         :raises: AssertionError 
2347         """
2348         assert new_entry_size >= 0
2349         cur_sz = len(self)
2350         dyn_tbl_sz = len(self._dynamic_table)
2351         while dyn_tbl_sz > 0 and cur_sz + new_entry_size > self._dynamic_table_max_size:  # noqa: E501
2352             last_elmt_sz = len(self._dynamic_table[-1])
2353             self._dynamic_table.pop()
2354             dyn_tbl_sz -= 1
2355             cur_sz -= last_elmt_sz

2357     def register(self, hdrs):
2358         # type: (Union[HPackLitHdrFldWithIncrIndexing, H2Frame, List[HPackHeaders]]) -> None  # noqa: E501
2359         """register adds to this table the instances of 
2360         HPackLitHdrFldWithIncrIndexing provided as parameters. 
2361  
2362         A H2Frame with a H2HeadersFrame payload can be provided, as much as a 
2363         python list of HPackHeaders or a single HPackLitHdrFldWithIncrIndexing 
2364         instance. 
2365         :param HPackLitHdrFldWithIncrIndexing|H2Frame|list of HPackHeaders hdrs: the header(s) to register  # noqa: E501 
2366         :raises: AssertionError 
2367         """
2368         if isinstance(hdrs, H2Frame):
2369             hdrs = [hdr for hdr in hdrs.payload.hdrs if isinstance(hdr, HPackLitHdrFldWithIncrIndexing)]  # noqa: E501
2370         elif isinstance(hdrs, HPackLitHdrFldWithIncrIndexing):
2371             hdrs = [hdrs]
2372         else:
2373             hdrs = [hdr for hdr in hdrs if isinstance(hdr, HPackLitHdrFldWithIncrIndexing)]  # noqa: E501
2374 
2375         for hdr in hdrs:
2376             if hdr.index == 0:
2377                 hdr_name = hdr.hdr_name.getfieldval('data').origin()
2378             else:
2379                 idx = int(hdr.index)
2380                 hdr_name = self[idx].name()
2381             hdr_value = hdr.hdr_value.getfieldval('data').origin()
2382 
2383             # Note: we do not delete any existing hdrentry with the same names
2384             # and values, as dictated by RFC 7541 par2.3.2
2385 
2386             entry = HPackHdrEntry(hdr_name, hdr_value)
2387             # According to RFC7541 par4.4, "Before a new entry is added to
2388             # the dynamic table, entries are evicted
2389             # from the end of the dynamic table until the size of the dynamic
2390             # table is less than or equal to (maximum size - new entry size)
2391             # or until the table is empty"
2392             # Also, "It is not an error to attempt to add an entry that is
2393             # larger than the maximum size; an attempt to add an entry larger
2394             # than the maximum size causes the table to be emptied of all
2395             # existing entries and results in an empty table"
2396             # For this reason, we first call the _reduce_dynamic_table and
2397             # then throw an assertion error if the new entry does not fit in
2398             new_entry_len = len(entry)
2399             self._reduce_dynamic_table(new_entry_len)
2400             assert new_entry_len <= self._dynamic_table_max_size
2401             self._dynamic_table.insert(0, entry)

2403     def get_idx_by_name(self, name):
2404         # type: (str) -> Optional[int]
2405         """ get_idx_by_name returns the index of a matching registered header 
2406  
2407         This implementation will prefer returning a static entry index whenever 
2408         possible. If multiple matching header name are found in the static 
2409         table, there is insurance that the first entry (lowest index number) 
2410         will be returned. 
2411         If no matching header is found, this method returns None. 
2412         """
2413         name = name.lower()
2414         for key, val in type(self)._static_entries.items():
2415             if val.name() == name:
2416                 return key
2417         for idx, val in enumerate(self._dynamic_table):
2418             if val.name() == name:
2419                 return type(self)._static_entries_last_idx + idx + 1
2420         return None

2422     def get_idx_by_name_and_value(self, name, value):
2423         # type: (str, str) -> Optional[int]
2424         """ get_idx_by_name_and_value returns the index of a matching registered 
2425         header 
2426  
2427         This implementation will prefer returning a static entry index whenever 
2428         possible. If multiple matching headers are found in the dynamic table, 
2429         the lowest index is returned 
2430         If no matching header is found, this method returns None. 
2431         """
2432         name = name.lower()
2433         for key, val in type(self)._static_entries.items():
2434             if val.name() == name and val.value() == value:
2435                 return key
2436         for idx, val in enumerate(self._dynamic_table):
2437             if val.name() == name and val.value() == value:
2438                 return type(self)._static_entries_last_idx + idx + 1
2439         return None

2507     def _convert_a_header_to_a_h2_header(self, hdr_name, hdr_value, is_sensitive, should_index):  # noqa: E501
2508         # type: (str, str, Callable[[str, str], bool], Callable[[str], bool]) -> Tuple[HPackHeaders, int]  # noqa: E501
2509         """ _convert_a_header_to_a_h2_header builds a HPackHeaders from a header 
2510         name and a value. It returns a HPackIndexedHdr whenever possible. If not,  # noqa: E501 
2511         it returns a HPackLitHdrFldWithoutIndexing or a 
2512         HPackLitHdrFldWithIncrIndexing, based on the should_index callback. 
2513         HPackLitHdrFldWithoutIndexing is forced if the is_sensitive callback 
2514         returns True and its never_index bit is set. 
2515         """
2516 
2517         # If both name and value are already indexed
2518         idx = self.get_idx_by_name_and_value(hdr_name, hdr_value)
2519         if idx is not None:
2520             return HPackIndexedHdr(index=idx), len(self[idx])
2521 
2522         # The value is not indexed for this headers
2523 
2524         _hdr_value = self._optimize_header_length_and_packetify(hdr_value)
2525 
2526         # Searching if the header name is indexed
2527         idx = self.get_idx_by_name(hdr_name)
2528         if idx is not None:
2529             if is_sensitive(
2530                 hdr_name,
2531                 _hdr_value.getfieldval('data').origin()
2532             ):
2533                 return HPackLitHdrFldWithoutIndexing(
2534                     never_index=1,
2535                     index=idx,
2536                     hdr_value=_hdr_value
2537                 ), len(
2538                     HPackHdrEntry(
2539                         self[idx].name(),
2540                         _hdr_value.getfieldval('data').origin()
2541                     )
2542                 )
2543             if should_index(hdr_name):
2544                 return HPackLitHdrFldWithIncrIndexing(
2545                     index=idx,
2546                     hdr_value=_hdr_value
2547                 ), len(
2548                     HPackHdrEntry(
2549                         self[idx].name(),
2550                         _hdr_value.getfieldval('data').origin()
2551                     )
2552                 )
2553             return HPackLitHdrFldWithoutIndexing(
2554                 index=idx,
2555                 hdr_value=_hdr_value
2556             ), len(
2557                 HPackHdrEntry(
2558                     self[idx].name(),
2559                     _hdr_value.getfieldval('data').origin()
2560                 )
2561             )
2562 
2563         _hdr_name = self._optimize_header_length_and_packetify(hdr_name)
2564 
2565         if is_sensitive(
2566             _hdr_name.getfieldval('data').origin(),
2567             _hdr_value.getfieldval('data').origin()
2568         ):
2569             return HPackLitHdrFldWithoutIndexing(
2570                 never_index=1,
2571                 index=0,
2572                 hdr_name=_hdr_name,
2573                 hdr_value=_hdr_value
2574             ), len(
2575                 HPackHdrEntry(
2576                     _hdr_name.getfieldval('data').origin(),
2577                     _hdr_value.getfieldval('data').origin()
2578                 )
2579             )
2580         if should_index(_hdr_name.getfieldval('data').origin()):

2582                 index=0,
2583                 hdr_name=_hdr_name,
2584                 hdr_value=_hdr_value
2585             ), len(
2586                 HPackHdrEntry(
2587                     _hdr_name.getfieldval('data').origin(),
2588                     _hdr_value.getfieldval('data').origin()
2589                 )
2590             )
2591         return HPackLitHdrFldWithoutIndexing(
2592             index=0,
2593             hdr_name=_hdr_name,
2594             hdr_value=_hdr_value
2595         ), len(

2620     def parse_txt_hdrs(self,
2621                        s,  # type: Union[bytes, str]
2622                        stream_id=1,  # type: int
2623                        body=None,  # type: Optional[str]
2624                        max_frm_sz=4096,  # type: int
2625                        max_hdr_lst_sz=0,  # type: int
2626                        is_sensitive=lambda n, v: False,  # type: Callable[[str, str], bool]  # noqa: E501
2627                        should_index=lambda x: False,  # type: Callable[[str], bool]  # noqa: E501
2628                        register=True,  # type: bool
2629                        ):
2630         # type: (...) -> H2Seq
2631         """ 
2632         parse_txt_hdrs parses headers expressed in text and converts them 
2633         into a series of H2Frames with the "correct" flags. A body can be 
2634         provided in which case, the data frames are added, bearing the End 
2635         Stream flag, instead of the H2HeadersFrame/H2ContinuationFrame. 
2636         The generated frames may respect max_frm_sz (SETTINGS_MAX_FRAME_SIZE) 
2637         and max_hdr_lst_sz (SETTINGS_MAX_HEADER_LIST_SIZE) if provided. 
2638         The headers are split into multiple headers fragment (and H2Frames) 
2639         to respect these limits. Also, a callback can be provided to tell if 
2640         a header should be never indexed (sensitive headers, such as cookies), 
2641         and another callback say if the header should be registered into the 
2642         index table at all. 
2643         For an header to be registered, the is_sensitive callback must return 
2644         False AND the should_index callback should return True. This is the 
2645         default behavior. 
2646  
2647         :param str s: the string to parse for headers 
2648         :param int stream_id: the stream id to use in the generated H2Frames 
2649         :param str/None body: the eventual body of the request, that is added 
2650           to the generated frames 
2651         :param int max_frm_sz: the maximum frame size. This is used to split 
2652           the headers and data frames according to the maximum frame size 
2653           negotiated for this connection. 
2654         :param int max_hdr_lst_sz: the maximum size of a "header fragment" as 
2655           defined in RFC7540 
2656         :param callable is_sensitive: callback that returns True if the 
2657           provided header is sensible and must be stored in a header packet 
2658           requesting this header never to be indexed 
2659         :param callable should_index: callback that returns True if the 
2660           provided header should be stored in a header packet requesting 
2661           indexation in the dynamic header table. 
2662         :param bool register: whether to register new headers with incremental 
2663           indexing as we parse them 
2664         :raises: Exception 
2665         """
2666 
2667         sio = BytesIO(s.encode() if isinstance(s, str) else s)
2668 
2669         base_frm_len = len(raw(H2Frame()))
2670 
2671         ret = H2Seq()
2672         cur_frm = H2HeadersFrame()  # type: Union[H2HeadersFrame, H2ContinuationFrame]  # noqa: E501
2673         cur_hdr_sz = 0
2674 
2675         # For each line in the headers str to parse
2676         for hdr_line in sio:
2677             hdr_name, hdr_value = self._parse_header_line(hdr_line)
2678             if hdr_name is None:
2679                 continue
2680 
2681             new_hdr, new_hdr_len = self._convert_a_header_to_a_h2_header(
2682                 hdr_name, hdr_value, is_sensitive, should_index
2683             )
2684             new_hdr_bin_len = len(raw(new_hdr))
2685 
2686             if register and isinstance(new_hdr, HPackLitHdrFldWithIncrIndexing):  # noqa: E501
2687                 self.register(new_hdr)
2688 
2689             # The new header binary length (+ base frame size) must not exceed
2690             # the maximum frame size or it will just never fit. Also, the
2691             # header entry length (as specified in RFC7540 par6.5.2) must not
2692             # exceed the maximum length of a header fragment or it will just
2693             # never fit
2694             if (new_hdr_bin_len + base_frm_len > max_frm_sz or
2695                     (max_hdr_lst_sz != 0 and new_hdr_len > max_hdr_lst_sz)):
2696                 raise Exception('Header too long: {}'.format(hdr_name))
2697 
2698             if (max_frm_sz < len(raw(cur_frm)) + base_frm_len + new_hdr_len or
2699                 (
2700                     max_hdr_lst_sz != 0 and
2701                     max_hdr_lst_sz < cur_hdr_sz + new_hdr_len
2702             )
2703             ):
2704                 flags = set()
2705                 if isinstance(cur_frm, H2HeadersFrame) and not body:
2706                     flags.add('ES')
2707                 ret.frames.append(H2Frame(stream_id=stream_id, flags=flags) / cur_frm)  # noqa: E501
2708                 cur_frm = H2ContinuationFrame()
2709                 cur_hdr_sz = 0
2710 
2711             hdr_list = cur_frm.hdrs
2712             hdr_list += new_hdr
2713             cur_hdr_sz += new_hdr_len
2714 
2715         flags = {'EH'}
2716         if isinstance(cur_frm, H2HeadersFrame) and not body:
2717             flags.add('ES')
2718         ret.frames.append(H2Frame(stream_id=stream_id, flags=flags) / cur_frm)
2719 
2720         if body:
2721             base_data_frm_len = len(raw(H2DataFrame()))
2722             sio = BytesIO(body)
2723             frgmt = sio.read(max_frm_sz - base_data_frm_len - base_frm_len)
2724             while frgmt:
2725                 nxt_frgmt = sio.read(max_frm_sz - base_data_frm_len - base_frm_len)  # noqa: E501
2726                 flags = set()
2727                 if len(nxt_frgmt) == 0:
2728                     flags.add('ES')
2729                 ret.frames.append(
2730                     H2Frame(stream_id=stream_id, flags=flags) / H2DataFrame(data=frgmt)  # noqa: E501
2731                 )
2732                 frgmt = nxt_frgmt
2733         return ret
