99   def add_js_update_schema(s: cs.CoreSchema, f: Callable[[], dict[str, Any]]) -> None:
100      def update_js_schema(s: cs.CoreSchema, handler: GetJsonSchemaHandler) -> dict[str, Any]:
101          js_schema = handler(s)
102          js_schema.update(f())
103          return js_schema
104  
105      if 'metadata' in s:
106          metadata = s['metadata']
107          if 'pydantic_js_functions' in s:
108              metadata['pydantic_js_functions'].append(update_js_schema)
109          else:
110              metadata['pydantic_js_functions'] = [update_js_schema]
111      else:
112          s['metadata'] = {'pydantic_js_functions': [update_js_schema]}

115  def as_jsonable_value(v: Any) -> Any:
116      if type(v) not in (int, str, float, bytes, bool, type(None)):
117          return to_jsonable_python(v)
118      return v

121  def expand_grouped_metadata(annotations: Iterable[Any]) -> Iterable[Any]:
122      """Expand the annotations. 
123   
124      Args: 
125          annotations: An iterable of annotations. 
126   
127      Returns: 
128          An iterable of expanded annotations. 
129   
130      Example: 
131          ```py 
132          from annotated_types import Ge, Len 
133   
134          from pydantic._internal._known_annotated_metadata import expand_grouped_metadata 
135   
136          print(list(expand_grouped_metadata([Ge(4), Len(5)]))) 
137          #> [Ge(ge=4), MinLen(min_length=5)] 
138          ``` 
139      """
140      import annotated_types as at
141  
142      FieldInfo = import_cached_field_info()
143  
144      for annotation in annotations:
145          if isinstance(annotation, at.GroupedMetadata):
146              yield from annotation
147          elif isinstance(annotation, FieldInfo):
148              yield from annotation.metadata
149              # this is a bit problematic in that it results in duplicate metadata
150              # all of our "consumers" can handle it, but it is not ideal
151              # we probably should split up FieldInfo into:
152              # - annotated types metadata
153              # - individual metadata known only to Pydantic
154              annotation = copy(annotation)
155              annotation.metadata = []
156              yield annotation
157          else:
158              yield annotation

162  def _get_at_to_constraint_map() -> dict[type, str]:
163      """Return a mapping of annotated types to constraints. 
164   
165      Normally, we would define a mapping like this in the module scope, but we can't do that 
166      because we don't permit module level imports of `annotated_types`, in an attempt to speed up 
167      the import time of `pydantic`. We still only want to have this dictionary defined in one place, 
168      so we use this function to cache the result. 
169      """
170      import annotated_types as at
171  
172      return {
173          at.Gt: 'gt',
174          at.Ge: 'ge',
175          at.Lt: 'lt',
176          at.Le: 'le',
177          at.MultipleOf: 'multiple_of',
178          at.MinLen: 'min_length',
179          at.MaxLen: 'max_length',
180      }

183  def apply_known_metadata(annotation: Any, schema: CoreSchema) -> CoreSchema | None:  # noqa: C901
184      """Apply `annotation` to `schema` if it is an annotation we know about (Gt, Le, etc.). 
185      Otherwise return `None`. 
186   
187      This does not handle all known annotations. If / when it does, it can always 
188      return a CoreSchema and return the unmodified schema if the annotation should be ignored. 
189   
190      Assumes that GroupedMetadata has already been expanded via `expand_grouped_metadata`. 
191   
192      Args: 
193          annotation: The annotation. 
194          schema: The schema. 
195   
196      Returns: 
197          An updated schema with annotation if it is an annotation we know about, `None` otherwise. 
198   
199      Raises: 
200          PydanticCustomError: If `Predicate` fails. 
201      """
202      import annotated_types as at
203  
204      from ._validators import forbid_inf_nan_check, get_constraint_validator
205  
206      schema = schema.copy()
207      schema_update, other_metadata = collect_known_metadata([annotation])
208      schema_type = schema['type']
209  
210      chain_schema_constraints: set[str] = {
211          'pattern',
212          'strip_whitespace',
213          'to_lower',
214          'to_upper',
215          'coerce_numbers_to_str',
216      }
217      chain_schema_steps: list[CoreSchema] = []
218  
219      for constraint, value in schema_update.items():
220          if constraint not in CONSTRAINTS_TO_ALLOWED_SCHEMAS:
221              raise ValueError(f'Unknown constraint {constraint}')
222          allowed_schemas = CONSTRAINTS_TO_ALLOWED_SCHEMAS[constraint]
223  
224          # if it becomes necessary to handle more than one constraint
225          # in this recursive case with function-after or function-wrap, we should refactor
226          # this is a bit challenging because we sometimes want to apply constraints to the inner schema,
227          # whereas other times we want to wrap the existing schema with a new one that enforces a new constraint.
228          if schema_type in {'function-before', 'function-wrap', 'function-after'} and constraint == 'strict':
229              schema['schema'] = apply_known_metadata(annotation, schema['schema'])  # type: ignore  # schema is function schema
230              return schema
231  
232          # if we're allowed to apply constraint directly to the schema, like le to int, do that
233          if schema_type in allowed_schemas:
234              if constraint == 'union_mode' and schema_type == 'union':
235                  schema['mode'] = value  # type: ignore  # schema is UnionSchema
236              else:
237                  schema[constraint] = value
238              continue
239  
240          #  else, apply a function after validator to the schema to enforce the corresponding constraint
241          if constraint in chain_schema_constraints:
242  
243              def _apply_constraint_with_incompatibility_info(
244                  value: Any, handler: cs.ValidatorFunctionWrapHandler
245              ) -> Any:
246                  try:
247                      x = handler(value)
248                  except ValidationError as ve:
249                      # if the error is about the type, it's likely that the constraint is incompatible the type of the field
250                      # for example, the following invalid schema wouldn't be caught during schema build, but rather at this point
251                      # with a cryptic 'string_type' error coming from the string validator,
252                      # that we'd rather express as a constraint incompatibility error (TypeError)
253                      # Annotated[list[int], Field(pattern='abc')]
254                      if 'type' in ve.errors()[0]['type']:
255                          raise TypeError(
256                              f"Unable to apply constraint '{constraint}' to supplied value {value} for schema of type '{schema_type}'"  # noqa: B023
257                          )
258                      raise ve
259                  return x
260  
261              chain_schema_steps.append(
262                  cs.no_info_wrap_validator_function(
263                      _apply_constraint_with_incompatibility_info, cs.str_schema(**{constraint: value})
264                  )
265              )
266          elif constraint in {*NUMERIC_CONSTRAINTS, *LENGTH_CONSTRAINTS}:
267              if constraint in NUMERIC_CONSTRAINTS:
268                  json_schema_constraint = constraint
269              elif constraint in LENGTH_CONSTRAINTS:
270                  inner_schema = schema
271                  while inner_schema['type'] in {'function-before', 'function-wrap', 'function-after'}:
272                      inner_schema = inner_schema['schema']  # type: ignore
273                  inner_schema_type = inner_schema['type']
274                  if inner_schema_type == 'list' or (
275                      inner_schema_type == 'json-or-python' and inner_schema['json_schema']['type'] == 'list'  # type: ignore
276                  ):
277                      json_schema_constraint = 'minItems' if constraint == 'min_length' else 'maxItems'
278                  else:
279                      json_schema_constraint = 'minLength' if constraint == 'min_length' else 'maxLength'
280  
281              schema = cs.no_info_after_validator_function(
282                  partial(get_constraint_validator(constraint), **{'constraint_value': value}), schema
283              )
284              add_js_update_schema(schema, lambda: {json_schema_constraint: as_jsonable_value(value)})  # noqa: B023
285          elif constraint == 'allow_inf_nan' and value is False:
286              schema = cs.no_info_after_validator_function(
287                  forbid_inf_nan_check,
288                  schema,
289              )
290          else:
291              # It's rare that we'd get here, but it's possible if we add a new constraint and forget to handle it
292              # Most constraint errors are caught at runtime during attempted application
293              raise RuntimeError(f"Unable to apply constraint '{constraint}' to schema of type '{schema_type}'")
294  
295      for annotation in other_metadata:
296          if (annotation_type := type(annotation)) in (at_to_constraint_map := _get_at_to_constraint_map()):
297              constraint = at_to_constraint_map[annotation_type]
298              schema = cs.no_info_after_validator_function(
299                  partial(get_constraint_validator(constraint), {constraint: getattr(annotation, constraint)}), schema
300              )
301              continue
302          elif isinstance(annotation, (at.Predicate, at.Not)):
303              predicate_name = f'{annotation.func.__qualname__}' if hasattr(annotation.func, '__qualname__') else ''
304  
305              def val_func(v: Any) -> Any:
306                  predicate_satisfied = annotation.func(v)  # noqa: B023
307  
308                  # annotation.func may also raise an exception, let it pass through
309                  if isinstance(annotation, at.Predicate):  # noqa: B023
310                      if not predicate_satisfied:
311                          raise PydanticCustomError(
312                              'predicate_failed',
313                              f'Predicate {predicate_name} failed',  # type: ignore  # noqa: B023
314                          )
315                  else:
316                      if predicate_satisfied:
317                          raise PydanticCustomError(
318                              'not_operation_failed',
319                              f'Not of {predicate_name} failed',  # type: ignore  # noqa: B023
320                          )
321  
322                  return v
323  
324              schema = cs.no_info_after_validator_function(val_func, schema)
325          else:
326              # ignore any other unknown metadata
327              return None
328  
329      if chain_schema_steps:
330          chain_schema_steps = [schema] + chain_schema_steps
331          return cs.chain_schema(chain_schema_steps)
332  
333      return schema

336  def collect_known_metadata(annotations: Iterable[Any]) -> tuple[dict[str, Any], list[Any]]:
337      """Split `annotations` into known metadata and unknown annotations. 
338   
339      Args: 
340          annotations: An iterable of annotations. 
341   
342      Returns: 
343          A tuple contains a dict of known metadata and a list of unknown annotations. 
344   
345      Example: 
346          ```py 
347          from annotated_types import Gt, Len 
348   
349          from pydantic._internal._known_annotated_metadata import collect_known_metadata 
350   
351          print(collect_known_metadata([Gt(1), Len(42), ...])) 
352          #> ({'gt': 1, 'min_length': 42}, [Ellipsis]) 
353          ``` 
354      """
355      annotations = expand_grouped_metadata(annotations)
356  
357      res: dict[str, Any] = {}
358      remaining: list[Any] = []
359  
360      for annotation in annotations:
361          # isinstance(annotation, PydanticMetadata) also covers ._fields:_PydanticGeneralMetadata
362          if isinstance(annotation, PydanticMetadata):
363              res.update(annotation.__dict__)
364          # we don't use dataclasses.asdict because that recursively calls asdict on the field values
365          elif (annotation_type := type(annotation)) in (at_to_constraint_map := _get_at_to_constraint_map()):
366              constraint = at_to_constraint_map[annotation_type]
367              res[constraint] = getattr(annotation, constraint)
368          elif isinstance(annotation, type) and issubclass(annotation, PydanticMetadata):
369              # also support PydanticMetadata classes being used without initialisation,
370              # e.g. `Annotated[int, Strict]` as well as `Annotated[int, Strict()]`
371              res.update({k: v for k, v in vars(annotation).items() if not k.startswith('_')})
372          else:
373              remaining.append(annotation)
374      # Nones can sneak in but pydantic-core will reject them
375      # it'd be nice to clean things up so we don't put in None (we probably don't _need_ to, it was just easier)
376      # but this is simple enough to kick that can down the road
377      res = {k: v for k, v in res.items() if v is not None}
378      return res, remaining

381  def check_metadata(metadata: dict[str, Any], allowed: Iterable[str], source_type: Any) -> None:
382      """A small utility function to validate that the given metadata can be applied to the target. 
383      More than saving lines of code, this gives us a consistent error message for all of our internal implementations. 
384   
385      Args: 
386          metadata: A dict of metadata. 
387          allowed: An iterable of allowed metadata. 
388          source_type: The source type. 
389   
390      Raises: 
391          TypeError: If there is metadatas that can't be applied on source type. 
392      """
393      unknown = metadata.keys() - set(allowed)
394      if unknown:
395          raise TypeError(
396              f'The following constraints cannot be applied to {source_type!r}: {", ".join([f"{k!r}" for k in unknown])}'
397          )
