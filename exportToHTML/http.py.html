8    """ 
9    HTTP 1.0 layer. 
10    
11   Load using:: 
12    
13       from scapy.layers.http import * 
14    
15   Or (console only):: 
16    
17       >>> load_layer("http") 
18    
19   Note that this layer ISN'T loaded by default, as quite experimental for now. 
20    
21   To follow HTTP packets streams = group packets together to get the 
22   whole request/answer, use ``TCPSession`` as:: 
23    
24       >>> sniff(session=TCPSession)  # Live on-the-flow session 
25       >>> sniff(offline="./http_chunk.pcap", session=TCPSession)  # pcap 
26    
27   This will decode HTTP packets using ``Content_Length`` or chunks, 
28   and will also decompress the packets when needed. 
29   Note: on failure, decompression will be ignored. 
30    
31   You can turn auto-decompression/auto-compression off with:: 
32    
33       >>> conf.contribs["http"]["auto_compression"] = False 
34    
35   (Defaults to True) 
36    
37   You can also turn auto-chunking/dechunking off with:: 
38    
39       >>> conf.contribs["http"]["auto_chunk"] = False 
40    
41   (Defaults to True) 
42   """

228  def _strip_header_name(name):
229      """Takes a header key (i.e., "Host" in "Host: www.google.com", 
230      and returns a stripped representation of it 
231      """
232      return plain_str(name.strip()).replace("-", "_")

235  def _header_line(name, val):
236      """Creates a HTTP header line"""
237      # Python 3.4 doesn't support % on bytes
238      return bytes_encode(name) + b": " + bytes_encode(val)

241  def _parse_headers(s):
242      headers = s.split(b"\r\n")
243      headers_found = {}
244      for header_line in headers:
245          try:
246              key, value = header_line.split(b':', 1)
247          except ValueError:
248              continue
249          header_key = _strip_header_name(key).lower()
250          headers_found[header_key] = (key, value.strip())
251      return headers_found

254  def _parse_headers_and_body(s):
255      ''' Takes a HTTP packet, and returns a tuple containing: 
256        _ the first line (e.g., "GET ...") 
257        _ the headers in a dictionary 
258        _ the body 
259      '''
260      crlfcrlf = b"\r\n\r\n"
261      crlfcrlfIndex = s.find(crlfcrlf)
262      if crlfcrlfIndex != -1:
263          headers = s[:crlfcrlfIndex + len(crlfcrlf)]
264          body = s[crlfcrlfIndex + len(crlfcrlf):]
265      else:
266          headers = s
267          body = b''
268      first_line, headers = headers.split(b"\r\n", 1)
269      return first_line.strip(), _parse_headers(headers), body

272  def _dissect_headers(obj, s):
273      """Takes a HTTP packet as the string s, and populates the scapy layer obj 
274      (either HTTPResponse or HTTPRequest). Returns the first line of the 
275      HTTP packet, and the body 
276      """
277      first_line, headers, body = _parse_headers_and_body(s)
278      for f in obj.fields_desc:
279          # We want to still parse wrongly capitalized fields
280          stripped_name = _strip_header_name(f.name).lower()
281          try:
282              _, value = headers.pop(stripped_name)
283          except KeyError:
284              continue
285          obj.setfieldval(f.name, value)
286      if headers:
287          headers = dict(headers.values())
288          obj.setfieldval('Unknown_Headers', headers)
289      return first_line, body

292  class _HTTPContent(Packet):
293      __slots__ = ["_original_len"]

295      # https://developer.mozilla.org/fr/docs/Web/HTTP/Headers/Transfer-Encoding
296      def _get_encodings(self):
297          encodings = []
298          if isinstance(self, HTTPResponse):
299              if self.Transfer_Encoding:
300                  encodings += [plain_str(x).strip().lower() for x in
301                                plain_str(self.Transfer_Encoding).split(",")]
302              if self.Content_Encoding:
303                  encodings += [plain_str(x).strip().lower() for x in
304                                plain_str(self.Content_Encoding).split(",")]
305          return encodings

307      def hashret(self):
308          return b"HTTP1"

310      def post_dissect(self, s):
311          self._original_len = len(s)
312          encodings = self._get_encodings()
313          # Un-chunkify
314          if conf.contribs["http"]["auto_chunk"] and "chunked" in encodings:
315              data = b""
316              while s:
317                  length, _, body = s.partition(b"\r\n")
318                  try:
319                      length = int(length, 16)
320                  except ValueError:
321                      # Not a valid chunk. Ignore
322                      break
323                  else:
324                      load = body[:length]
325                      if body[length:length + 2] != b"\r\n":
326                          # Invalid chunk. Ignore
327                          break
328                      s = body[length + 2:]
329                      data += load
330              if not s:
331                  s = data
332          if not conf.contribs["http"]["auto_compression"]:
333              return s
334          # Decompress
335          try:
336              if "deflate" in encodings:
337                  import zlib
338                  s = zlib.decompress(s)
339              elif "gzip" in encodings:
340                  s = gzip.decompress(s)
341              elif "compress" in encodings:
342                  if _is_lzw_available:
343                      s = lzw.decompress(s)
344                  else:
345                      log_loading.info(
346                          "Can't import lzw. compress decompression "
347                          "will be ignored !"
348                      )
349              elif "br" in encodings:
350                  if _is_brotli_available:
351                      s = brotli.decompress(s)
352                  else:
353                      log_loading.info(
354                          "Can't import brotli. brotli decompression "
355                          "will be ignored !"
356                      )
357              elif "zstd" in encodings:
358                  if _is_zstd_available:
359                      # Using its streaming API since its simple API could handle
360                      # only cases where there is content size data embedded in
361                      # the frame
362                      bio = io.BytesIO(s)
363                      reader = zstandard.ZstdDecompressor().stream_reader(bio)
364                      s = reader.read()
365                  else:
366                      log_loading.info(
367                          "Can't import zstandard. zstd decompression "
368                          "will be ignored !"
369                      )
370          except Exception:
371              # Cannot decompress - probably incomplete data
372              pass
373          return s

375      def post_build(self, pkt, pay):
376          encodings = self._get_encodings()
377          if conf.contribs["http"]["auto_compression"]:
378              # Compress
379              if "deflate" in encodings:
380                  import zlib
381                  pay = zlib.compress(pay)
382              elif "gzip" in encodings:
383                  pay = gzip.compress(pay)
384              elif "compress" in encodings:
385                  if _is_lzw_available:
386                      pay = lzw.compress(pay)
387                  else:
388                      log_loading.info(
389                          "Can't import lzw. compress compression "
390                          "will be ignored !"
391                      )
392              elif "br" in encodings:
393                  if _is_brotli_available:
394                      pay = brotli.compress(pay)
395                  else:
396                      log_loading.info(
397                          "Can't import brotli. brotli compression will "
398                          "be ignored !"
399                      )
400              elif "zstd" in encodings:
401                  if _is_zstd_available:
402                      pay = zstandard.ZstdCompressor().compress(pay)
403                  else:
404                      log_loading.info(
405                          "Can't import zstandard. zstd compression will "
406                          "be ignored !"
407                      )
408          # Chunkify
409          if conf.contribs["http"]["auto_chunk"] and "chunked" in encodings:
410              # Dumb: 1 single chunk.
411              pay = (b"%X" % len(pay)) + b"\r\n" + pay + b"\r\n0\r\n\r\n"
412          return pkt + pay

414      def self_build(self, **kwargs):
415          ''' Takes an HTTPRequest or HTTPResponse object, and creates its 
416          string representation.'''
417          if not isinstance(self.underlayer, HTTP):
418              warning(
419                  "An HTTPResponse/HTTPRequest should always be below an HTTP"
420              )
421          # Check for cache
422          if self.raw_packet_cache is not None:
423              return self.raw_packet_cache
424          p = b""
425          encodings = self._get_encodings()
426          # Walk all the fields, in order
427          for i, f in enumerate(self.fields_desc):
428              if f.name == "Unknown_Headers":
429                  continue
430              # Get the field value
431              val = self.getfieldval(f.name)
432              if not val:
433                  if f.name == "Content_Length" and "chunked" not in encodings:
434                      # Add Content-Length anyways
435                      val = str(len(self.payload or b""))
436                  elif f.name == "Date" and isinstance(self, HTTPResponse):
437                      val = datetime.datetime.now(datetime.timezone.utc).strftime(
438                          '%a, %d %b %Y %H:%M:%S GMT'
439                      )
440                  else:
441                      # Not specified. Skip
442                      continue
443  
444              if i >= 3:
445                  val = _header_line(f.real_name, val)
446              # Fields used in the first line have a space as a separator,
447              # whereas headers are terminated by a new line
448              if i <= 1:
449                  separator = b' '
450              else:
451                  separator = b'\r\n'
452              # Add the field into the packet
453              p = f.addfield(self, p, val + separator)
454          # Handle Unknown_Headers
455          if self.Unknown_Headers:
456              headers_text = b""
457              for name, value in self.Unknown_Headers.items():
458                  headers_text += _header_line(name, value) + b"\r\n"
459              p = self.get_field("Unknown_Headers").addfield(
460                  self, p, headers_text
461              )
462          # The packet might be empty, and in that case it should stay empty.
463          if p:
464              # Add an additional line after the last header
465              p = f.addfield(self, p, b'\r\n')
466          return p

494  def _generate_headers(*args):
495      """Generate the header fields based on their name"""
496      # Order headers
497      all_headers = []
498      for headers in args:
499          all_headers += headers
500      # Generate header fields
501      results = []
502      for h in sorted(all_headers):
503          results.append(_HTTPHeaderField(h, None))
504      return results

529          """From the HTTP packet string, populate the scapy object"""
530          first_line, body = _dissect_headers(self, s)

533              self.setfieldval('Method', Method)

536          except ValueError:
537              pass
538          if body:
539              self.raw_packet_cache = s[:-len(body)]
540          else:
541              self.raw_packet_cache = s
542          return body

596  class HTTP(Packet):
597      name = "HTTP 1"
598      fields_desc = []
599      show_indent = 0
600      clsreq = HTTPRequest
601      clsresp = HTTPResponse
602      hdr = b"HTTP"
603      reqmethods = b"|".join([
604          b"OPTIONS",
605          b"GET",
606          b"HEAD",
607          b"POST",
608          b"PUT",
609          b"DELETE",
610          b"TRACE",
611          b"CONNECT",
612      ])
613  
614      @classmethod
615      def dispatch_hook(cls, _pkt=None, *args, **kargs):
616          if _pkt and len(_pkt) >= 9:
617              from scapy.contrib.http2 import _HTTP2_types, H2Frame
618              # To detect a valid HTTP2, we check that the type is correct
619              # that the Reserved bit is set and length makes sense.
620              while _pkt:
621                  if len(_pkt) < 9:
622                      # Invalid total length
623                      return cls
624                  if ord(_pkt[3:4]) not in _HTTP2_types:
625                      # Invalid type
626                      return cls
627                  length = struct.unpack("!I", b"\0" + _pkt[:3])[0] + 9
628                  if length > len(_pkt):
629                      # Invalid length
630                      return cls
631                  sid = struct.unpack("!I", _pkt[5:9])[0]
632                  if sid >> 31 != 0:
633                      # Invalid Reserved bit
634                      return cls
635                  _pkt = _pkt[length:]
636              return H2Frame
637          return cls
638  
639      # tcp_reassemble is used by TCPSession in session.py
640      @classmethod
641      def tcp_reassemble(cls, data, metadata, _):
642          detect_end = metadata.get("detect_end", None)
643          is_unknown = metadata.get("detect_unknown", True)
644          # General idea of the following is explained at
645          # https://datatracker.ietf.org/doc/html/rfc2616#section-4.4
646          if not detect_end or is_unknown:
647              metadata["detect_unknown"] = False
648              http_packet = cls(data)
649              # Detect packing method
650              if not isinstance(http_packet.payload, _HTTPContent):
651                  return http_packet
652              is_response = isinstance(http_packet.payload, cls.clsresp)
653              # Packets may have a Content-Length we must honnor
654              length = http_packet.Content_Length
655              # Heuristic to try and detect instant HEAD responses, as those include a
656              # Content-Length that must not be honored. This is a bit crappy, and assumes
657              # that a 'HEAD' will never include an Encoding...
658              if (
659                  is_response and
660                  data.endswith(b"\r\n\r\n") and
661                  not http_packet[HTTPResponse]._get_encodings()
662              ):
663                  detect_end = lambda _: True
664              elif length is not None:
665                  # The packet provides a Content-Length attribute: let's
666                  # use it. When the total size of the frags is high enough,
667                  # we have the packet
668                  length = int(length)
669                  # Subtract the length of the "HTTP*" layer
670                  if http_packet.payload.payload or length == 0:
671                      http_length = len(data) - http_packet.payload._original_len
672                      detect_end = lambda dat: len(dat) - http_length >= length
673                  else:
674                      # The HTTP layer isn't fully received.
675                      detect_end = lambda dat: False
676                      metadata["detect_unknown"] = True
677              else:
678                  # It's not Content-Length based. It could be chunked
679                  encodings = http_packet[cls].payload._get_encodings()
680                  chunked = ("chunked" in encodings)
681                  if chunked:
682                      detect_end = lambda dat: dat.endswith(b"0\r\n\r\n")
683                  # HTTP Requests that do not have any content,
684                  # end with a double CRLF. Same for HEAD responses
685                  elif isinstance(http_packet.payload, cls.clsreq):
686                      detect_end = lambda dat: dat.endswith(b"\r\n\r\n")
687                      # In case we are handling a HTTP Request,
688                      # we want to continue assessing the data,
689                      # to handle requests with a body (POST)
690                      metadata["detect_unknown"] = True
691                  elif is_response and http_packet.Status_Code == b"101":
692                      # If it's an upgrade response, it may also hold a
693                      # different protocol data.
694                      # make sure all headers are present
695                      detect_end = lambda dat: dat.find(b"\r\n\r\n")
696                  else:
697                      # If neither Content-Length nor chunked is specified,
698                      # it means it's the TCP packet that contains the data,
699                      # or that the information hasn't been given yet.
700                      detect_end = lambda dat: metadata.get("tcp_end", False)
701                      metadata["detect_unknown"] = True
702              metadata["detect_end"] = detect_end
703              if detect_end(data):
704                  return http_packet
705          else:
706              if detect_end(data):
707                  http_packet = cls(data)
708                  return http_packet

710      def guess_payload_class(self, payload):
711          """Decides if the payload is an HTTP Request or Response, or 
712          something else. 
713          """
714          try:
715              prog = re.compile(
716                  br"^(?:" + self.reqmethods + br") " +
717                  br"(?:.+?) " +
718                  self.hdr + br"/\d\.\d$"
719              )
720              crlfIndex = payload.index(b"\r\n")
721              req = payload[:crlfIndex]
722              result = prog.match(req)
723              if result:
724                  return self.clsreq
725              else:
726                  prog = re.compile(b"^" + self.hdr + br"/\d\.\d \d\d\d .*$")
727                  result = prog.match(req)
728                  if result:
729                      return self.clsresp
730          except ValueError:
731              # Anything that isn't HTTP but on port 80
732              pass
733          return Raw

746   
747      :param mech: one of HTTP_AUTH_MECHS 
748      :param ssl: whether to use HTTPS or not 
749      :param ssp: the SSP object to use for binding 
750      """

752      def __init__(
753          self,
754          mech=HTTP_AUTH_MECHS.NONE,
755          verb=True,
756          sslcontext=None,
757          ssp=None,
758          no_check_certificate=False,
759      ):
760          self.sock = None
761          self._sockinfo = None
762          self.authmethod = mech
763          self.verb = verb
764          self.sslcontext = sslcontext
765          self.ssp = ssp
766          self.sspcontext = None
767          self.no_check_certificate = no_check_certificate

769      def _connect_or_reuse(self, host, port=None, tls=False, timeout=5):
770          # Get the port
771          if port is None:
772              port = 443 if tls else 80
773          # If the current socket matches, keep it.
774          if self._sockinfo == (host, port):
775              return
776          # A new socket is needed
777          if self._sockinfo:
778              self.close()
779          sock = socket.socket()
780          sock.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1)
781          sock.settimeout(timeout)
782          if self.verb:
783              print(
784                  "\u2503 Connecting to %s on port %s%s..."
785                  % (
786                      host,
787                      port,
788                      " with SSL" if tls else "",
789                  )
790              )
791          sock.connect((host, port))
792          if self.verb:
793              print(
794                  conf.color_theme.green(
795                      "\u2514 Connected from %s" % repr(sock.getsockname())
796                  )
797              )
798          if tls:
799              if self.sslcontext is None:
800                  if self.no_check_certificate:
801                      context = ssl.SSLContext(ssl.PROTOCOL_TLS_CLIENT)
802                      context.check_hostname = False
803                      context.verify_mode = ssl.CERT_NONE
804                  else:
805                      context = ssl.create_default_context()
806              else:
807                  context = self.sslcontext
808              sock = context.wrap_socket(sock, server_hostname=host)
809              self.sock = SSLStreamSocket(sock, HTTP)
810          else:
811              self.sock = StreamSocket(sock, HTTP)
812          # Store information regarding the current socket
813          self._sockinfo = (host, port)
814  
815      def sr1(self, req, **kwargs):
816          if self.verb:
817              print(conf.color_theme.opening(">> %s" % req.summary()))
818          resp = self.sock.sr1(
819              HTTP() / req,
820              verbose=0,
821              **kwargs,
822          )
823          if self.verb:
824              print(
825                  conf.color_theme.success(
826                      "<< %s" % (resp and resp.summary())
827                  )
828              )
829          return resp

831      def request(self, url, data=b"", timeout=5, follow_redirects=True, **headers):
832          """ 
833          Perform a HTTP(s) request. 
834          """
835          # Parse request url
836          m = re.match(r"(https?)://([^/:]+)(?:\:(\d+))?(?:/(.*))?", url)
837          if not m:
838              raise ValueError("Bad URL !")
839          transport, host, port, path = m.groups()
840          if transport == "https":
841              tls = True
842          else:
843              tls = False
844  
845          path = path or "/"
846          port = port and int(port)
847  
848          # Connect (or reuse) socket
849          self._connect_or_reuse(host, port=port, tls=tls, timeout=timeout)
850  
851          # Build request
852          http_headers = {
853              "Accept_Encoding": b'gzip, deflate',
854              "Cache_Control": b'no-cache',
855              "Pragma": b'no-cache',
856              "Connection": b'keep-alive',
857              "Host": host,
858              "Path": path,
859          }
860          http_headers.update(headers)
861          req = HTTP() / HTTPRequest(**http_headers)
862          if data:
863              req /= data
864  
865          while True:
866              # Perform the request.
867              resp = self.sr1(req)
868              if not resp:
869                  break
870              # First case: auth was required. Handle that
871              if resp.Status_Code in [b"401", b"407"]:
872                  # Authentication required
873                  if self.authmethod in [
874                      HTTP_AUTH_MECHS.NTLM,
875                      HTTP_AUTH_MECHS.NEGOTIATE,
876                  ]:
877                      # Parse authenticate
878                      if b" " in resp.WWW_Authenticate:
879                          method, data = resp.WWW_Authenticate.split(b" ", 1)
880                          try:
881                              ssp_blob = GSSAPI_BLOB(base64.b64decode(data))
882                          except Exception:
883                              raise Scapy_Exception("Invalid WWW-Authenticate")
884                      else:
885                          method = resp.WWW_Authenticate
886                          ssp_blob = None
887                      if plain_str(method) != self.authmethod.value:
888                          raise Scapy_Exception("Invalid WWW-Authenticate")
889                      # SPNEGO / Kerberos / NTLM
890                      self.sspcontext, token, status = self.ssp.GSS_Init_sec_context(
891                          self.sspcontext,
892                          ssp_blob,
893                          req_flags=0,
894                      )
895                      if status not in [GSS_S_COMPLETE, GSS_S_CONTINUE_NEEDED]:
896                          raise Scapy_Exception("Authentication failure")
897                      req.Authorization = (
898                          self.authmethod.value.encode() + b" " +
899                          base64.b64encode(bytes(token))
900                      )
901                      continue
902              # Second case: follow redirection
903              if resp.Status_Code in [b"301", b"302"] and follow_redirects:
904                  return self.request(
905                      resp.Location.decode(),
906                      data=data,
907                      timeout=timeout,
908                      follow_redirects=follow_redirects,
909                      **headers,
910                  )
911              break
912          return resp

915          if self.verb:
916              print("X Connection to %s closed\n" % repr(self.sock.ins.getpeername()))
917          self.sock.close()

920  def http_request(host, path="/", port=None, timeout=3,
921                   display=False, tls=False, verbose=0, **headers):
922      """ 
923      Util to perform an HTTP request. 
924   
925      :param host: the host to connect to 
926      :param path: the path of the request (default /) 
927      :param port: the port (default 80/443) 
928      :param timeout: timeout before None is returned 
929      :param display: display the result in the default browser (default False) 
930      :param iface: interface to use. Changing this turns on "raw" 
931      :param headers: any additional headers passed to the request 
932   
933      :returns: the HTTPResponse packet 
934      """
935      client = HTTP_Client(HTTP_AUTH_MECHS.NONE, verb=verbose)
936      if port is None:
937          port = 443 if tls else 80
938      ans = client.request(
939          "http%s://%s:%s%s" % (tls and "s" or "", host, port, path),
940          timeout=timeout,
941      )
942  
943      if ans:
944          if display:
945              if Raw not in ans:
946                  warning("No HTTP content returned. Cannot display")
947                  return ans
948              # Write file
949              file = get_temp_file(autoext=".html")
950              with open(file, "wb") as fd:
951                  fd.write(ans.load)
952              # Open browser
953              if WINDOWS:
954                  os.startfile(file)
955              else:
956                  with ContextManagerSubprocess(conf.prog.universal_open):
957                      subprocess.Popen([conf.prog.universal_open, file])
958          return ans

975      """ 
976      HTTP server automaton 
977   
978      :param ssp: the SSP to serve. If None, unauthenticated (or basic). 
979      :param mech: the HTTP_AUTH_MECHS to use (default: NONE) 
980   
981      Other parameters: 
982   
983      :param BASIC_IDENTITIES: a dict that contains {"user": "password"} for Basic 
984                               authentication.

988      pkt_cls = HTTP

990      def __init__(
991          self,
992          mech=HTTP_AUTH_MECHS.NONE,
993          verb=True,
994          ssp=None,
995          *args,
996          **kwargs,
997      ):
998          self.verb = verb
999          if "sock" not in kwargs:
1000             raise ValueError(
1001                 "HTTP_Server cannot be started directly ! Use HTTP_Server.spawn"
1002             )
1003         self.ssp = ssp
1004         self.authmethod = mech.value
1005         self.sspcontext = None
1006         self.basic = False
1007         self.BASIC_IDENTITIES = kwargs.pop("BASIC_IDENTITIES", {})
1008         self.BASIC_REALM = kwargs.pop("BASIC_REALM", "default")
1009         if mech == HTTP_AUTH_MECHS.BASIC:
1010             if not self.BASIC_IDENTITIES:
1011                 raise ValueError("Please provide 'BASIC_IDENTITIES' !")
1012             if ssp is not None:
1013                 raise ValueError("Can't use 'BASIC_IDENTITIES' with 'ssp' !")
1014             self.basic = True
1015         elif mech == HTTP_AUTH_MECHS.NONE:
1016             if ssp is not None:
1017                 raise ValueError("Cannot use ssp with mech=NONE !")
1018         # Initialize
1019         Automaton.__init__(self, *args, **kwargs)

1060     def _ask_authorization(self, proxy, data):
1061         if proxy:
1062             self.send(
1063                 HTTPResponse(
1064                     Status_Code=b"407",
1065                     Reason_Phrase=b"Proxy Authentication Required",
1066                     Proxy_Authenticate=data,
1067                 )
1068             )
1069         else:
1070             self.send(
1071                 HTTPResponse(
1072                     Status_Code=b"401",
1073                     Reason_Phrase=b"Unauthorized",
1074                     WWW_Authenticate=data,
1075                 )
1076             )

1078     @ATMT.receive_condition(AUTH, prio=1)
1079     def received_unauthenticated(self, pkt):
1080         if HTTPRequest in pkt:
1081             self.vprint(pkt.summary())
1082             if pkt.Method == b"CONNECT":
1083                 # HTTP tunnel (proxy)
1084                 proxy = True
1085             else:
1086                 # HTTP non-tunnel
1087                 proxy = False
1088             # Get authorization
1089             if proxy:
1090                 authorization = pkt.Proxy_Authorization
1091             else:
1092                 authorization = pkt.Authorization
1093             if not authorization:
1094                 # Initial ask.
1095                 data = self.authmethod
1096                 if self.basic:
1097                     data += " realm='%s'" % self.BASIC_REALM
1098                 self._ask_authorization(proxy, data)
1099                 return
1100             # Parse authorization
1101             method, data = authorization.split(b" ", 1)
1102             if plain_str(method) != self.authmethod:
1103                 raise self.AUTH_ERROR(proxy)
1104             try:
1105                 data = base64.b64decode(data)
1106             except Exception:
1107                 raise self.AUTH_ERROR(proxy)
1108             # Now process the authorization
1109             if not self.basic:
1110                 try:
1111                     ssp_blob = GSSAPI_BLOB(data)
1112                 except Exception:
1113                     self.sspcontext = None
1114                     self._ask_authorization(proxy, self.authmethod)
1115                     raise self.AUTH_ERROR(proxy)
1116                 # And call the SSP
1117                 self.sspcontext, tok, status = self.ssp.GSS_Accept_sec_context(
1118                     self.sspcontext, ssp_blob
1119                 )
1120             else:
1121                 # This is actually Basic authentication
1122                 try:
1123                     next(
1124                         True
1125                         for k, v in self.BASIC_IDENTITIES.items()
1126                         if ("%s:%s" % (k, v)).encode() == data
1127                     )
1128                     tok, status = None, GSS_S_COMPLETE
1129                 except StopIteration:
1130                     tok, status = None, GSS_S_FAILURE
1131             # Send answer
1132             if status not in [GSS_S_COMPLETE, GSS_S_CONTINUE_NEEDED]:
1133                 raise self.AUTH_ERROR(proxy)
1134             elif status == GSS_S_CONTINUE_NEEDED:
1135                 data = self.authmethod.encode()
1136                 if tok:
1137                     data += b" " + base64.b64encode(bytes(tok))
1138                 self._ask_authorization(proxy, data)
1139                 raise self.AUTH()
1140             else:
1141                 # Authenticated !
1142                 self.authenticated = True
1143                 self.vprint("AUTH OK")
1144                 raise self.SERVE(pkt)
