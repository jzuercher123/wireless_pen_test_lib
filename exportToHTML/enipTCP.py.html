117  class ItemData(Packet):
118      """Common Packet Format"""
119      name = "Item Data"
120      fields_desc = [
121          LEShortEnumField("typeId", 0, _typeIdList),
122          LEShortField("length", 0),
123          XLEStrLenField("data", "", length_from=lambda pkt: pkt.length),
124      ]
125  
126      def extract_padding(self, s):
127          return '', s

278  class ENIPTCP(Packet):
279      """Ethernet/IP packet over TCP"""
280      name = "ENIPTCP"
281      fields_desc = [
282          LEShortEnumField("commandId", None, _commandIdList),
283          LEShortField("length", 0),
284          XLEIntField("session", 0),
285          LEIntEnumField("status", None, _statusList),
286          LELongField("senderContext", 0),
287          LEIntField("options", 0),
288      ]
289  
290      def post_build(self, pkt, pay):
291          if self.length is None and pay:
292              pkt = pkt[:2] + struct.pack("<H", len(pay)) + pkt[4:]
293          return pkt + pay
