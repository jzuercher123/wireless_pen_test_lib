47   class CoreConfig(TypedDict, total=False):
48       """ 
49       Base class for schema configuration options. 
50    
51       Attributes: 
52           title: The name of the configuration. 
53           strict: Whether the configuration should strictly adhere to specified rules. 
54           extra_fields_behavior: The behavior for handling extra fields. 
55           typed_dict_total: Whether the TypedDict should be considered total. Default is `True`. 
56           from_attributes: Whether to use attributes for models, dataclasses, and tagged union keys. 
57           loc_by_alias: Whether to use the used alias (or first alias for "field required" errors) instead of 
58               `field_names` to construct error `loc`s. Default is `True`. 
59           revalidate_instances: Whether instances of models and dataclasses should re-validate. Default is 'never'. 
60           validate_default: Whether to validate default values during validation. Default is `False`. 
61           populate_by_name: Whether an aliased field may be populated by its name as given by the model attribute, 
62               as well as the alias. (Replaces 'allow_population_by_field_name' in Pydantic v1.) Default is `False`. 
63           str_max_length: The maximum length for string fields. 
64           str_min_length: The minimum length for string fields. 
65           str_strip_whitespace: Whether to strip whitespace from string fields. 
66           str_to_lower: Whether to convert string fields to lowercase. 
67           str_to_upper: Whether to convert string fields to uppercase. 
68           allow_inf_nan: Whether to allow infinity and NaN values for float fields. Default is `True`. 
69           ser_json_timedelta: The serialization option for `timedelta` values. Default is 'iso8601'. 
70           ser_json_bytes: The serialization option for `bytes` values. Default is 'utf8'. 
71           ser_json_inf_nan: The serialization option for infinity and NaN values 
72               in float fields. Default is 'null'. 
73           val_json_bytes: The validation option for `bytes` values, complementing ser_json_bytes. Default is 'utf8'. 
74           hide_input_in_errors: Whether to hide input data from `ValidationError` representation. 
75           validation_error_cause: Whether to add user-python excs to the __cause__ of a ValidationError. 
76               Requires exceptiongroup backport pre Python 3.11. 
77           coerce_numbers_to_str: Whether to enable coercion of any `Number` type to `str` (not applicable in `strict` mode). 
78           regex_engine: The regex engine to use for regex pattern validation. Default is 'rust-regex'. See `StringSchema`. 
79           cache_strings: Whether to cache strings. Default is `True`, `True` or `'all'` is required to cache strings 
80               during general validation since validators don't know if they're in a key or a value. 
81       """
82   
83       title: str
84       strict: bool
85       # settings related to typed dicts, model fields, dataclass fields
86       extra_fields_behavior: ExtraBehavior
87       typed_dict_total: bool  # default: True
88       # used for models, dataclasses, and tagged union keys
89       from_attributes: bool
90       # whether to use the used alias (or first alias for "field required" errors) instead of field_names
91       # to construct error `loc`s, default True
92       loc_by_alias: bool
93       # whether instances of models and dataclasses (including subclass instances) should re-validate, default 'never'
94       revalidate_instances: Literal['always', 'never', 'subclass-instances']
95       # whether to validate default values during validation, default False
96       validate_default: bool
97       # used on typed-dicts and arguments
98       populate_by_name: bool  # replaces `allow_population_by_field_name` in pydantic v1
99       # fields related to string fields only
100      str_max_length: int
101      str_min_length: int
102      str_strip_whitespace: bool
103      str_to_lower: bool
104      str_to_upper: bool
105      # fields related to float fields only
106      allow_inf_nan: bool  # default: True
107      # the config options are used to customise serialization to JSON
108      ser_json_timedelta: Literal['iso8601', 'float']  # default: 'iso8601'
109      ser_json_bytes: Literal['utf8', 'base64', 'hex']  # default: 'utf8'
110      ser_json_inf_nan: Literal['null', 'constants', 'strings']  # default: 'null'
111      val_json_bytes: Literal['utf8', 'base64', 'hex']  # default: 'utf8'
112      # used to hide input data from ValidationError repr
113      hide_input_in_errors: bool
114      validation_error_cause: bool  # default: False
115      coerce_numbers_to_str: bool  # default: False
116      regex_engine: Literal['rust-regex', 'python-re']  # default: 'rust-regex'
117      cache_strings: Union[bool, Literal['all', 'keys', 'none']]  # default: 'True'
