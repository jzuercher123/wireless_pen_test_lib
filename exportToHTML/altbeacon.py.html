8    """ 
9    scapy.contrib.altbeacon - AltBeacon Bluetooth LE proximity beacons. 
10    
11   The AltBeacon specification can be found at: https://github.com/AltBeacon/spec 
12   """
13   
14   from scapy.fields import (
15       ByteField,
16       MayEnd,
17       ShortField,
18       SignedByteField,
19       StrFixedLenField,
20   )
21   from scapy.layers.bluetooth import EIR_Hdr, EIR_Manufacturer_Specific_Data, \
22       UUIDField, LowEnergyBeaconHelper
23   from scapy.packet import Packet
24   
25   
26   # When building beacon frames, one should use their own manufacturer ID.
27   #
28   # However, most software (including the AltBeacon SDK) requires explicitly
29   # registering particular manufacturer IDs to listen to, and the only ID used is
30   # that of Radius Networks (the developer of the specification).
31   #
32   # To maximise compatibility, Scapy's implementation of
33   # LowEnergyBeaconHelper.build_eir (for constructing frames) uses Radius
34   # Networks' manufacturer ID.
35   #
36   # Scapy's implementation of AltBeacon **does not** require a specific
37   # manufacturer ID to detect AltBeacons - it uses
38   # EIR_Manufacturer_Specific_Data.register_magic_payload.
39   RADIUS_NETWORKS_MFG = 0x0118
40   
41   
42   class AltBeacon(Packet, LowEnergyBeaconHelper):
43       """ 
44       AltBeacon broadcast frame type. 
45    
46       https://github.com/AltBeacon/spec 
47       """
48       name = "AltBeacon"
49       magic = b"\xBE\xAC"
50       fields_desc = [
51           StrFixedLenField("header", magic, len(magic)),
52   
53           # The spec says this is 20 bytes, with >=16 bytes being an
54           # organisational unit-specific identifier. However, the Android library
55           # treats this as UUID + uint16 + uint16.
56           UUIDField("id1", None),
57   
58           # Local identifier
59           ShortField("id2", None),
60           ShortField("id3", None),
61   
62           MayEnd(SignedByteField("tx_power", None)),
63           ByteField("mfg_reserved", None),
64       ]
65   
66       @classmethod
67       def magic_check(cls, payload):
68           """ 
69           Checks if the given payload is for us (starts with our magic string). 
70           """
71           return payload.startswith(cls.magic)
72   
73       def build_eir(self):
74           """Builds a list of EIR messages to wrap this frame."""
75   
76           # Note: Company ID is not required by spec, but most tools only look
77           # for manufacturer-specific data with Radius Networks' manufacturer ID.
78           return LowEnergyBeaconHelper.base_eir + [
79               EIR_Hdr() / EIR_Manufacturer_Specific_Data(
80                   company_id=RADIUS_NETWORKS_MFG) / self
81           ]
82   
83   
84   EIR_Manufacturer_Specific_Data.register_magic_payload(AltBeacon)
