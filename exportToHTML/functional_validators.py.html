28   class AfterValidator:
29       """Usage docs: https://docs.pydantic.dev/2.9/concepts/validators/#annotated-validators 
30    
31       A metadata class that indicates that a validation should be applied **after** the inner validation logic. 
32    
33       Attributes: 
34           func: The validator function. 
35    
36       Example: 
37           ```py 
38           from typing_extensions import Annotated 
39    
40           from pydantic import AfterValidator, BaseModel, ValidationError 
41    
42           MyInt = Annotated[int, AfterValidator(lambda v: v + 1)] 
43    
44           class Model(BaseModel): 
45               a: MyInt 
46    
47           print(Model(a=1).a) 
48           #> 2 
49    
50           try: 
51               Model(a='a') 
52           except ValidationError as e: 
53               print(e.json(indent=2)) 
54               ''' 
55               [ 
56                 { 
57                   "type": "int_parsing", 
58                   "loc": [ 
59                     "a" 
60                   ], 
61                   "msg": "Input should be a valid integer, unable to parse string as an integer", 
62                   "input": "a", 
63                   "url": "https://errors.pydantic.dev/2/v/int_parsing" 
64                 } 
65               ] 
66               ''' 
67           ``` 
68       """
69   
70       func: core_schema.NoInfoValidatorFunction | core_schema.WithInfoValidatorFunction
71   
72       def __get_pydantic_core_schema__(self, source_type: Any, handler: GetCoreSchemaHandler) -> core_schema.CoreSchema:
73           schema = handler(source_type)
74           info_arg = _inspect_validator(self.func, 'after')
75           if info_arg:
76               func = cast(core_schema.WithInfoValidatorFunction, self.func)
77               return core_schema.with_info_after_validator_function(func, schema=schema, field_name=handler.field_name)
78           else:
79               func = cast(core_schema.NoInfoValidatorFunction, self.func)
80               return core_schema.no_info_after_validator_function(func, schema=schema)

154  class PlainValidator:
155      """Usage docs: https://docs.pydantic.dev/2.9/concepts/validators/#annotated-validators 
156   
157      A metadata class that indicates that a validation should be applied **instead** of the inner validation logic. 
158   
159      Attributes: 
160          func: The validator function. 
161          json_schema_input_type: The input type of the function. This is only used to generate the appropriate 
162              JSON Schema (in validation mode). If not provided, will default to `Any`. 
163   
164      Example: 
165          ```py 
166          from typing_extensions import Annotated 
167   
168          from pydantic import BaseModel, PlainValidator 
169   
170          MyInt = Annotated[int, PlainValidator(lambda v: int(v) + 1)] 
171   
172          class Model(BaseModel): 
173              a: MyInt 
174   
175          print(Model(a='1').a) 
176          #> 2 
177          ``` 
178      """
179  
180      func: core_schema.NoInfoValidatorFunction | core_schema.WithInfoValidatorFunction
181      json_schema_input_type: Any = Any
182  
183      def __get_pydantic_core_schema__(self, source_type: Any, handler: GetCoreSchemaHandler) -> core_schema.CoreSchema:
184          # Note that for some valid uses of PlainValidator, it is not possible to generate a core schema for the
185          # source_type, so calling `handler(source_type)` will error, which prevents us from generating a proper
186          # serialization schema. To work around this for use cases that will not involve serialization, we simply
187          # catch any PydanticSchemaGenerationError that may be raised while attempting to build the serialization schema
188          # and abort any attempts to handle special serialization.
189          from pydantic import PydanticSchemaGenerationError
190  
191          try:
192              schema = handler(source_type)
193              # TODO if `schema['serialization']` is one of `'include-exclude-dict/sequence',
194              # schema validation will fail. That's why we use 'type ignore' comments below.
195              serialization = schema.get(
196                  'serialization',
197                  core_schema.wrap_serializer_function_ser_schema(
198                      function=lambda v, h: h(v),
199                      schema=schema,
200                      return_schema=handler.generate_schema(source_type),
201                  ),
202              )
203          except PydanticSchemaGenerationError:
204              serialization = None
205  
206          input_schema = handler.generate_schema(self.json_schema_input_type)
207          metadata = _core_metadata.build_metadata_dict(js_input_core_schema=input_schema)
208  
209          info_arg = _inspect_validator(self.func, 'plain')
210          if info_arg:
211              func = cast(core_schema.WithInfoValidatorFunction, self.func)
212              return core_schema.with_info_plain_validator_function(
213                  func,
214                  field_name=handler.field_name,
215                  serialization=serialization,  # pyright: ignore[reportArgumentType]
216                  metadata=metadata,
217              )
218          else:
219              func = cast(core_schema.NoInfoValidatorFunction, self.func)
220              return core_schema.no_info_plain_validator_function(
221                  func,
222                  serialization=serialization,  # pyright: ignore[reportArgumentType]
223                  metadata=metadata,
224              )
225  
226      @classmethod
227      def _from_decorator(cls, decorator: _decorators.Decorator[_decorators.FieldValidatorDecoratorInfo]) -> Self:
228          return cls(
229              func=decorator.func,
230              json_schema_input_type=decorator.info.json_schema_input_type,

247   
248      from typing_extensions import Annotated 
249   
250      from pydantic import BaseModel, ValidationError, WrapValidator

254              # we don't want to bother with further validation, just return the new value

257              return handler(v) 
258          except ValidationError:

263   
264      class Model(BaseModel): 
265          a: MyTimestamp 
266

271      ```

389  def field_validator(
390      field: str,
391      /,
392      *fields: str,
393      mode: FieldValidatorModes = 'after',
394      check_fields: bool | None = None,
395      json_schema_input_type: Any = PydanticUndefined,
396  ) -> Callable[[Any], Any]:
397      """Usage docs: https://docs.pydantic.dev/2.9/concepts/validators/#field-validators 
398   
399      Decorate methods on the class indicating that they should be used to validate fields. 
400   
401      Example usage: 
402      ```py 
403      from typing import Any 
404   
405      from pydantic import ( 
406          BaseModel, 
407          ValidationError, 
408          field_validator, 
409      ) 
410   
411      class Model(BaseModel): 
412          a: str 
413   
414          @field_validator('a') 
415          @classmethod 
416          def ensure_foobar(cls, v: Any): 
417              if 'foobar' not in v: 
418                  raise ValueError('"foobar" not found in a') 
419              return v 
420   
421      print(repr(Model(a='this is foobar good'))) 
422      #> Model(a='this is foobar good') 
423   
424      try: 
425          Model(a='snap') 
426      except ValidationError as exc_info: 
427          print(exc_info) 
428          ''' 
429          1 validation error for Model 
430          a 
431            Value error, "foobar" not found in a [type=value_error, input_value='snap', input_type=str] 
432          ''' 
433      ``` 
434   
435      For more in depth examples, see [Field Validators](../concepts/validators.md#field-validators). 
436   
437      Args: 
438          field: The first field the `field_validator` should be called on; this is separate 
439              from `fields` to ensure an error is raised if you don't pass at least one. 
440          *fields: Additional field(s) the `field_validator` should be called on. 
441          mode: Specifies whether to validate the fields before or after validation. 
442          check_fields: Whether to check that the fields actually exist on the model. 
443          json_schema_input_type: The input type of the function. This is only used to generate 
444              the appropriate JSON Schema (in validation mode) and can only specified 
445              when `mode` is either `'before'`, `'plain'` or `'wrap'`. 
446   
447      Returns: 
448          A decorator that can be used to decorate a function to be used as a field_validator. 
449   
450      Raises: 
451          PydanticUserError: 
452              - If `@field_validator` is used bare (with no fields). 
453              - If the args passed to `@field_validator` as fields are not strings. 
454              - If `@field_validator` applied to instance methods. 
455      """
456      if isinstance(field, FunctionType):
457          raise PydanticUserError(
458              '`@field_validator` should be used with fields and keyword arguments, not bare. '
459              "E.g. usage should be `@validator('<field_name>', ...)`",
460              code='validator-no-fields',
461          )
462  
463      if mode not in ('before', 'plain', 'wrap') and json_schema_input_type is not PydanticUndefined:
464          raise PydanticUserError(
465              f"`json_schema_input_type` can't be used when mode is set to {mode!r}",
466              code='validator-input-type',
467          )
468  
469      if json_schema_input_type is PydanticUndefined and mode == 'plain':
470          json_schema_input_type = Any
471  
472      fields = field, *fields
473      if not all(isinstance(field, str) for field in fields):
474          raise PydanticUserError(
475              '`@field_validator` fields should be passed as separate string args. '
476              "E.g. usage should be `@validator('<field_name_1>', '<field_name_2>', ...)`",
477              code='validator-invalid-fields',
478          )
479  
480      def dec(
481          f: Callable[..., Any] | staticmethod[Any, Any] | classmethod[Any, Any, Any],
482      ) -> _decorators.PydanticDescriptorProxy[Any]:
483          if _decorators.is_instance_method_from_sig(f):
484              raise PydanticUserError(
485                  '`@field_validator` cannot be applied to instance methods', code='validator-instance-method'
486              )
487  
488          # auto apply the @classmethod decorator
489          f = _decorators.ensure_classmethod_based_on_signature(f)
490  
491          dec_info = _decorators.FieldValidatorDecoratorInfo(
492              fields=fields, mode=mode, check_fields=check_fields, json_schema_input_type=json_schema_input_type
493          )
494          return _decorators.PydanticDescriptorProxy(f, dec_info)
495  
496      return dec

650  def model_validator(
651      *,
652      mode: Literal['wrap', 'before', 'after'],
653  ) -> Any:
654      """Usage docs: https://docs.pydantic.dev/2.9/concepts/validators/#model-validators 
655   
656      Decorate model methods for validation purposes. 
657   
658      Example usage: 
659      ```py 
660      from typing_extensions import Self 
661   
662      from pydantic import BaseModel, ValidationError, model_validator 
663   
664      class Square(BaseModel): 
665          width: float 
666          height: float 
667   
668          @model_validator(mode='after') 
669          def verify_square(self) -> Self: 
670              if self.width != self.height: 
671                  raise ValueError('width and height do not match') 
672              return self 
673   
674      s = Square(width=1, height=1) 
675      print(repr(s)) 
676      #> Square(width=1.0, height=1.0) 
677
