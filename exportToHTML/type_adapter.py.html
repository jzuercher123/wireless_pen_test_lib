45   def _get_schema(type_: Any, config_wrapper: _config.ConfigWrapper, parent_depth: int) -> CoreSchema:
46       """`BaseModel` uses its own `__module__` to find out where it was defined 
47       and then looks for symbols to resolve forward references in those globals. 
48       On the other hand this function can be called with arbitrary objects, 
49       including type aliases, where `__module__` (always `typing.py`) is not useful. 
50       So instead we look at the globals in our parent stack frame. 
51    
52       This works for the case where this function is called in a module that 
53       has the target of forward references in its scope, but 
54       does not always work for more complex cases. 
55    
56       For example, take the following: 
57    
58       a.py 
59       ```python 
60       from typing import Dict, List 
61    
62       IntList = List[int] 
63       OuterDict = Dict[str, 'IntList'] 
64       ``` 
65    
66       b.py 
67       ```python test="skip" 
68       from a import OuterDict 
69    
70       from pydantic import TypeAdapter 
71    
72       IntList = int  # replaces the symbol the forward reference is looking for 
73       v = TypeAdapter(OuterDict) 
74       v({'x': 1})  # should fail but doesn't 
75       ``` 
76    
77       If `OuterDict` were a `BaseModel`, this would work because it would resolve 
78       the forward reference within the `a.py` namespace. 
79       But `TypeAdapter(OuterDict)` can't determine what module `OuterDict` came from. 
80    
81       In other words, the assumption that _all_ forward references exist in the 
82       module we are being called from is not technically always true. 
83       Although most of the time it is and it works fine for recursive models and such, 
84       `BaseModel`'s behavior isn't perfect either and _can_ break in similar ways, 
85       so there is no right or wrong between the two. 
86    
87       But at the very least this behavior is _subtly_ different from `BaseModel`'s. 
88       """
89       local_ns = _typing_extra.parent_frame_namespace(parent_depth=parent_depth)
90       global_ns = sys._getframe(max(parent_depth - 1, 1)).f_globals.copy()
91       global_ns.update(local_ns or {})
92       gen = (config_wrapper.schema_generator or _generate_schema.GenerateSchema)(
93           config_wrapper, types_namespace=global_ns, typevars_map={}
94       )
95       schema = gen.generate_schema(type_)
96       schema = gen.clean_schema(schema)
97       return schema

100  def _getattr_no_parents(obj: Any, attribute: str) -> Any:
101      """Returns the attribute value without attempting to look up attributes from parent types."""
102      if hasattr(obj, '__dict__'):
103          try:
104              return obj.__dict__[attribute]
105          except KeyError:
106              pass
107  
108      slots = getattr(obj, '__slots__', None)
109      if slots is not None and attribute in slots:
110          return getattr(obj, attribute)
111      else:
112          raise AttributeError(attribute)

115  def _type_has_config(type_: Any) -> bool:
116      """Returns whether the type has config."""
117      type_ = _typing_extra.annotated_type(type_) or type_
118      try:
119          return issubclass(type_, BaseModel) or is_dataclass(type_) or is_typeddict(type_)
120      except TypeError:
121          # type is not a class
122          return False

125  # This is keeping track of the frame depth for the TypeAdapter functions. This is required for _parent_depth used for
126  # ForwardRef resolution. We may enter the TypeAdapter schema building via different TypeAdapter functions. Hence, we
127  # need to keep track of the frame depth relative to the originally provided _parent_depth.
128  def _frame_depth(
129      depth: int,
130  ) -> Callable[[Callable[Concatenate[TypeAdapterT, P], R]], Callable[Concatenate[TypeAdapterT, P], R]]:
131      def wrapper(func: Callable[Concatenate[TypeAdapterT, P], R]) -> Callable[Concatenate[TypeAdapterT, P], R]:
132          @wraps(func)
133          def wrapped(self: TypeAdapterT, *args: P.args, **kwargs: P.kwargs) -> R:
134              with self._with_frame_depth(depth + 1):  # depth + 1 for the wrapper function
135                  return func(self, *args, **kwargs)
136  
137          return wrapped
138  
139      return wrapper

142  @final
143  class TypeAdapter(Generic[T]):
144      """Usage docs: https://docs.pydantic.dev/2.9/concepts/type_adapter/ 
145   
146      Type adapters provide a flexible way to perform validation and serialization based on a Python type. 
147   
148      A `TypeAdapter` instance exposes some of the functionality from `BaseModel` instance methods 
149      for types that do not have such methods (such as dataclasses, primitive types, and more). 
150   
151      **Note:** `TypeAdapter` instances are not types, and cannot be used as type annotations for fields. 
152   
153      **Note:** By default, `TypeAdapter` does not respect the 
154      [`defer_build=True`][pydantic.config.ConfigDict.defer_build] setting in the 
155      [`model_config`][pydantic.BaseModel.model_config] or in the `TypeAdapter` constructor `config`. You need to also 
156      explicitly set [`experimental_defer_build_mode=('model', 'type_adapter')`][pydantic.config.ConfigDict.experimental_defer_build_mode] of the 
157      config to defer the model validator and serializer construction. Thus, this feature is opt-in to ensure backwards 
158      compatibility. 
159   
160      Attributes: 
161          core_schema: The core schema for the type. 
162          validator (SchemaValidator): The schema validator for the type. 
163          serializer: The schema serializer for the type. 
164      """
165

189      def __init__(
190          self,
191          type: Any,
192          *,
193          config: ConfigDict | None = None,
194          _parent_depth: int = 2,
195          module: str | None = None,
196      ) -> None:
197          """Initializes the TypeAdapter object. 
198   
199          Args: 
200              type: The type associated with the `TypeAdapter`. 
201              config: Configuration for the `TypeAdapter`, should be a dictionary conforming to [`ConfigDict`][pydantic.config.ConfigDict]. 
202              _parent_depth: depth at which to search the parent namespace to construct the local namespace. 
203              module: The module that passes to plugin if provided. 
204   
205          !!! note 
206              You cannot use the `config` argument when instantiating a `TypeAdapter` if the type you're using has its own 
207              config that cannot be overridden (ex: `BaseModel`, `TypedDict`, and `dataclass`). A 
208              [`type-adapter-config-unused`](../errors/usage_errors.md#type-adapter-config-unused) error will be raised in this case. 
209   
210          !!! note 
211              The `_parent_depth` argument is named with an underscore to suggest its private nature and discourage use. 
212              It may be deprecated in a minor version, so we only recommend using it if you're 
213              comfortable with potential change in behavior / support. 
214   
215          ??? tip "Compatibility with `mypy`" 
216              Depending on the type used, `mypy` might raise an error when instantiating a `TypeAdapter`. As a workaround, you can explicitly 
217              annotate your variable: 
218   
219              ```py 
220              from typing import Union 
221   
222              from pydantic import TypeAdapter 
223   
224              ta: TypeAdapter[Union[str, int]] = TypeAdapter(Union[str, int])  # type: ignore[arg-type] 
225              ``` 
226   
227          Returns: 
228              A type adapter configured for the specified `type`. 
229          """
230          if _type_has_config(type) and config is not None:
231              raise PydanticUserError(
232                  'Cannot use `config` when the type is a BaseModel, dataclass or TypedDict.'
233                  ' These types can have their own config and setting the config via the `config`'
234                  ' parameter to TypeAdapter will not override it, thus the `config` you passed to'
235                  ' TypeAdapter becomes meaningless, which is probably not what you want.',
236                  code='type-adapter-config-unused',
237              )
238  
239          self._type = type
240          self._config = config
241          self._parent_depth = _parent_depth
242          if module is None:
243              f = sys._getframe(1)
244              self._module_name = cast(str, f.f_globals.get('__name__', ''))
245          else:
246              self._module_name = module
247  
248          self._core_schema: CoreSchema | None = None
249          self._validator: SchemaValidator | PluggableSchemaValidator | None = None
250          self._serializer: SchemaSerializer | None = None
251  
252          if not self._defer_build():
253              # Immediately initialize the core schema, validator and serializer
254              with self._with_frame_depth(1):  # +1 frame depth for this __init__
255                  # Model itself may be using deferred building. For backward compatibility we don't rebuild model mocks
256                  # here as part of __init__ even though TypeAdapter itself is not using deferred building.
257                  self._init_core_attrs(rebuild_mocks=False)
258  
259      @contextmanager
260      def _with_frame_depth(self, depth: int) -> Iterator[None]:
261          self._parent_depth += depth
262          try:
263              yield
264          finally:
265              self._parent_depth -= depth
266  
267      @_frame_depth(1)
268      def _init_core_attrs(self, rebuild_mocks: bool) -> None:
269          try:
270              self._core_schema = _getattr_no_parents(self._type, '__pydantic_core_schema__')
271              self._validator = _getattr_no_parents(self._type, '__pydantic_validator__')
272              self._serializer = _getattr_no_parents(self._type, '__pydantic_serializer__')
273          except AttributeError:
274              config_wrapper = _config.ConfigWrapper(self._config)
275              core_config = config_wrapper.core_config(None)
276  
277              self._core_schema = _get_schema(self._type, config_wrapper, parent_depth=self._parent_depth)
278              self._validator = create_schema_validator(
279                  schema=self._core_schema,
280                  schema_type=self._type,
281                  schema_type_module=self._module_name,
282                  schema_type_name=str(self._type),
283                  schema_kind='TypeAdapter',
284                  config=core_config,
285                  plugin_settings=config_wrapper.plugin_settings,
286              )
287              self._serializer = SchemaSerializer(self._core_schema, core_config)
288  
289          if rebuild_mocks and isinstance(self._core_schema, _mock_val_ser.MockCoreSchema):
290              self._core_schema.rebuild()
291              self._init_core_attrs(rebuild_mocks=False)
292              assert not isinstance(self._core_schema, _mock_val_ser.MockCoreSchema)
293              assert not isinstance(self._validator, _mock_val_ser.MockValSer)
294              assert not isinstance(self._serializer, _mock_val_ser.MockValSer)
295  
296      @cached_property
297      @_frame_depth(2)  # +2 for @cached_property and core_schema(self)
298      def core_schema(self) -> CoreSchema:
299          """The pydantic-core schema used to build the SchemaValidator and SchemaSerializer."""
300          if self._core_schema is None or isinstance(self._core_schema, _mock_val_ser.MockCoreSchema):
301              self._init_core_attrs(rebuild_mocks=True)  # Do not expose MockCoreSchema from public function
302          assert self._core_schema is not None and not isinstance(self._core_schema, _mock_val_ser.MockCoreSchema)
303          return self._core_schema

305      @cached_property
306      @_frame_depth(2)  # +2 for @cached_property + validator(self)
307      def validator(self) -> SchemaValidator | PluggableSchemaValidator:
308          """The pydantic-core SchemaValidator used to validate instances of the model."""
309          if not isinstance(self._validator, (SchemaValidator, PluggableSchemaValidator)):
310              self._init_core_attrs(rebuild_mocks=True)  # Do not expose MockValSer from public function
311          assert isinstance(self._validator, (SchemaValidator, PluggableSchemaValidator))
312          return self._validator

314      @cached_property
315      @_frame_depth(2)  # +2 for @cached_property + serializer(self)
316      def serializer(self) -> SchemaSerializer:
317          """The pydantic-core SchemaSerializer used to dump instances of the model."""
318          if not isinstance(self._serializer, SchemaSerializer):
319              self._init_core_attrs(rebuild_mocks=True)  # Do not expose MockValSer from public function
320          assert isinstance(self._serializer, SchemaSerializer)
321          return self._serializer

323      def _defer_build(self) -> bool:
324          config = self._config if self._config is not None else self._model_config()
325          return self._is_defer_build_config(config) if config is not None else False
326  
327      def _model_config(self) -> ConfigDict | None:
328          type_: Any = _typing_extra.annotated_type(self._type) or self._type  # Eg FastAPI heavily uses Annotated
329          if _utils.lenient_issubclass(type_, BaseModel):
330              return type_.model_config
331          return getattr(type_, '__pydantic_config__', None)
332  
333      @staticmethod
334      def _is_defer_build_config(config: ConfigDict) -> bool:
335          # TODO reevaluate this logic when we have a better understanding of how defer_build should work with TypeAdapter
336          # Should we drop the special experimental_defer_build_mode check?
337          return config.get('defer_build', False) is True and 'type_adapter' in config.get(
338              'experimental_defer_build_mode', ()
339          )

544      @staticmethod
545      def json_schemas(
546          inputs: Iterable[tuple[JsonSchemaKeyT, JsonSchemaMode, TypeAdapter[Any]]],
547          /,
548          *,
549          by_alias: bool = True,
550          title: str | None = None,
551          description: str | None = None,
552          ref_template: str = DEFAULT_REF_TEMPLATE,
553          schema_generator: type[GenerateJsonSchema] = GenerateJsonSchema,
554      ) -> tuple[dict[tuple[JsonSchemaKeyT, JsonSchemaMode], JsonSchemaValue], JsonSchemaValue]:
555          """Generate a JSON schema including definitions from multiple type adapters. 
556   
557          Args: 
558              inputs: Inputs to schema generation. The first two items will form the keys of the (first) 
559                  output mapping; the type adapters will provide the core schemas that get converted into 
560                  definitions in the output JSON schema. 
561              by_alias: Whether to use alias names. 
562              title: The title for the schema. 
563              description: The description for the schema. 
564              ref_template: The format string used for generating $ref strings. 
565              schema_generator: The generator class used for creating the schema. 
566   
567          Returns: 
568              A tuple where: 
569   
570                  - The first element is a dictionary whose keys are tuples of JSON schema key type and JSON mode, and 
571                      whose values are the JSON schema corresponding to that pair of inputs. (These schemas may have 
572                      JsonRef references to definitions that are defined in the second returned element.) 
573                  - The second element is a JSON schema containing all definitions referenced in the first returned 
574                      element, along with the optional title and description keys. 
575   
576          """
577          schema_generator_instance = schema_generator(by_alias=by_alias, ref_template=ref_template)
578  
579          inputs_ = []
580          for key, mode, adapter in inputs:
581              with adapter._with_frame_depth(1):  # +1 for json_schemas staticmethod
582                  inputs_.append((key, mode, adapter.core_schema))
583  
584          json_schemas_map, definitions = schema_generator_instance.generate_definitions(inputs_)
585  
586          json_schema: dict[str, Any] = {}
587          if definitions:
588              json_schema['$defs'] = definitions
589          if title:
590              json_schema['title'] = title
591          if description:
592              json_schema['description'] = description
593  
594          return json_schemas_map, json_schema
