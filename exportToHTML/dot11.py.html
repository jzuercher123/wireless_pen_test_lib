231  def _next_radiotap_extpm(pkt, lst, cur, s):
232      """Generates the next RadioTapExtendedPresenceMask"""
233      if cur is None or (cur.present and cur.present.Ext):
234          st = len(lst) + (cur is not None)
235          return lambda *args: RadioTapExtendedPresenceMask(*args, index=st)

239  class RadioTapExtendedPresenceMask(Packet):
240      """RadioTapExtendedPresenceMask should be instantiated by passing an 
241      `index=` kwarg, stating which place the item has in the list. 
242   
243      Passing index will update the b[x] fields accordingly to the index. 
244        e.g. 
245         >>> a = RadioTapExtendedPresenceMask(present="b0+b12+b29+Ext") 
246         >>> b = RadioTapExtendedPresenceMask(index=1, present="b33+b45+b59+b62") 
247         >>> pkt = RadioTap(present="Ext", Ext=[a, b]) 
248      """
249      name = "RadioTap Extended presence mask"
250      fields_desc = [FlagsField('present', None, -32,
251                                ["b%s" % i for i in range(0, 31)] + ["Ext"])]
252  
253      def __init__(self, _pkt=None, index=0, **kwargs):
254          self._restart_indentation(index)
255          Packet.__init__(self, _pkt, **kwargs)
256  
257      def _restart_indentation(self, index):
258          st = index * 32
259          self.fields_desc[0].names = ["b%s" % (i + st) for i in range(0, 31)] + ["Ext"]  # noqa: E501
260  
261      def guess_payload_class(self, pay):
262          return conf.padding_layer

755      def answers(self, other):
756          if isinstance(other, Dot11):
757              if self.type == 0:  # management
758                  if self.addr1.lower() != other.addr2.lower():  # check resp DA w/ req SA  # noqa: E501
759                      return 0
760                  if (other.subtype, self.subtype) in [(0, 1), (2, 3), (4, 5)]:
761                      return 1
762                  if self.subtype == other.subtype == 11:  # auth
763                      return self.payload.answers(other.payload)
764              elif self.type == 1:  # control
765                  return 0
766              elif self.type == 2:  # data

772      def address_meaning(self, index):
773          """ 
774          Return the meaning of the address[index] considering the context 
775          """
776          if index not in [1, 2, 3, 4]:
777              raise ValueError("Wrong index: should be [1, 2, 3, 4]")
778          index = index - 1
779          if self.type == 0:  # Management
780              return _dot11_addr_meaning[0][index]
781          elif self.type == 1:  # Control
782              if (self.type, self.subtype) == (1, 6) and self.cfe == 6:
783                  return ["RA", "NAV-SA", "NAV-DA"][index]
784              return _dot11_addr_meaning[1][index]
785          elif self.type == 2:  # Data
786              meaning = _dot11_addr_meaning[2][index][
787                  self.FCfield.to_DS
788              ][self.FCfield.from_DS]
789              if meaning and index in [2, 3]:  # Address 3-4
790                  if isinstance(self.payload, Dot11QoS):
791                      # MSDU and Short A-MSDU
792                      if self.payload.A_MSDU_Present:
793                          meaning = "BSSID"
794              return meaning
795          elif self.type == 3:  # Extension
796              return _dot11_addr_meaning[3][index]
797          return None

799      def unwep(self, key=None, warn=1):
800          if self.FCfield & 0x40 == 0:
801              if warn:
802                  warning("No WEP to remove")
803              return
804          if isinstance(self.payload.payload, NoPayload):
805              if key or conf.wepkey:
806                  self.payload.decrypt(key)
807              if isinstance(self.payload.payload, NoPayload):
808                  if warn:
809                      warning("Dot11 can't be decrypted. Check conf.wepkey.")
810                  return
811          self.FCfield &= ~0x40
812          self.payload = self.payload.payload

866      def network_stats(self):
867          """Return a dictionary containing a summary of the Dot11 
868          elements fields 
869          """
870          summary = {}
871          crypto = set()
872          p = self.payload
873          while isinstance(p, Dot11Elt):
874              # Avoid overriding already-set SSID values because it is not part
875              # of the standard and it protects from parsing bugs,
876              # see https://github.com/secdev/scapy/issues/2683
877              if p.ID == 0 and "ssid" not in summary:
878                  summary["ssid"] = plain_str(p.info)
879              elif p.ID == 3:
880                  summary["channel"] = ord(p.info)
881              elif isinstance(p, Dot11EltCountry):
882                  summary["country"] = plain_str(p.country_string[:2])
883                  country_descriptor_types = {
884                      b"I": "Indoor",
885                      b"O": "Outdoor",
886                      b"X": "Non-country",
887                      b"\xff": "Ignored"
888                  }
889                  summary["country_desc_type"] = country_descriptor_types.get(
890                      p.country_string[-1:]
891                  )
892              elif isinstance(p, Dot11EltRates):
893                  rates = [(x & 0x7f) / 2. for x in p.rates]
894                  if "rates" in summary:
895                      summary["rates"].extend(rates)
896                  else:
897                      summary["rates"] = rates
898              elif isinstance(p, Dot11EltRSN):
899                  wpa_version = "WPA2"
900                  # WPA3-only:
901                  # - AP shall at least enable AKM suite selector 00-0F-AC:8
902                  # - AP shall not enable AKM suite selector 00-0F-AC:2 and
903                  #   00-0F-AC:6
904                  # - AP shall set MFPC and MFPR to 1
905                  # - AP shall not enable WEP and TKIP
906                  # WPA3-transition:
907                  # - AP shall at least enable AKM suite selector 00-0F-AC:2
908                  #   and 00-0F-AC:8
909                  # - AP shall set MFPC to 1 and MFPR to 0
910                  if any(x.suite == 8 for x in p.akm_suites) and \
911                          all(x.suite not in [2, 6] for x in p.akm_suites) and \
912                          p.mfp_capable and p.mfp_required and \
913                          all(x.cipher not in [1, 2, 5]
914                              for x in p.pairwise_cipher_suites):
915                      # WPA3 only mode
916                      wpa_version = "WPA3"
917                  elif any(x.suite == 8 for x in p.akm_suites) and \
918                          any(x.suite == 2 for x in p.akm_suites) and \
919                          p.mfp_capable and not p.mfp_required:
920                      # WPA3 transition mode
921                      wpa_version = "WPA3-transition"
922                  # Append suite
923                  if p.akm_suites:
924                      auth = p.akm_suites[0].sprintf("%suite%")
925                      crypto.add(wpa_version + "/%s" % auth)
926                  else:
927                      crypto.add(wpa_version)
928              elif p.ID == 221:
929                  if isinstance(p, Dot11EltMicrosoftWPA):
930                      if p.akm_suites:
931                          auth = p.akm_suites[0].sprintf("%suite%")
932                          crypto.add("WPA/%s" % auth)
933                      else:
934                          crypto.add("WPA")

937              if self.cap.privacy:
938                  crypto.add("WEP")
939              else:
940                  crypto.add("OPN")
941          if crypto:
942              summary["crypto"] = crypto
943          return summary
