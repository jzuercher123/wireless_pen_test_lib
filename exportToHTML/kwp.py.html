49   class KWP(ISOTP):
50       services = ObservableDict(
51           {0x10: 'StartDiagnosticSession',
52            0x11: 'ECUReset',
53            0x14: 'ClearDiagnosticInformation',
54            0x17: 'ReadStatusOfDiagnosticTroubleCodes',
55            0x18: 'ReadDiagnosticTroubleCodesByStatus',
56            0x1A: 'ReadECUIdentification',
57            0x21: 'ReadDataByLocalIdentifier',
58            0x22: 'ReadDataByIdentifier',
59            0x23: 'ReadMemoryByAddress',
60            0x27: 'SecurityAccess',
61            0x28: 'DisableNormalMessageTransmission',
62            0x29: 'EnableNormalMessageTransmission',
63            0x2C: 'DynamicallyDefineLocalIdentifier',
64            0x2E: 'WriteDataByIdentifier',
65            0x30: 'InputOutputControlByLocalIdentifier',
66            0x31: 'StartRoutineByLocalIdentifier',
67            0x32: 'StopRoutineByLocalIdentifier',
68            0x33: 'RequestRoutineResultsByLocalIdentifier',
69            0x34: 'RequestDownload',
70            0x35: 'RequestUpload',
71            0x36: 'TransferData',
72            0x37: 'RequestTransferExit',
73            0x3B: 'WriteDataByLocalIdentifier',
74            0x3D: 'WriteMemoryByAddress',
75            0x3E: 'TesterPresent',
76            0x85: 'ControlDTCSetting',
77            0x86: 'ResponseOnEvent',
78            0x50: 'StartDiagnosticSessionPositiveResponse',
79            0x51: 'ECUResetPositiveResponse',
80            0x54: 'ClearDiagnosticInformationPositiveResponse',
81            0x57: 'ReadStatusOfDiagnosticTroubleCodesPositiveResponse',
82            0x58: 'ReadDiagnosticTroubleCodesByStatusPositiveResponse',
83            0x5A: 'ReadECUIdentificationPositiveResponse',
84            0x61: 'ReadDataByLocalIdentifierPositiveResponse',
85            0x62: 'ReadDataByIdentifierPositiveResponse',
86            0x63: 'ReadMemoryByAddressPositiveResponse',
87            0x67: 'SecurityAccessPositiveResponse',
88            0x68: 'DisableNormalMessageTransmissionPositiveResponse',
89            0x69: 'EnableNormalMessageTransmissionPositiveResponse',
90            0x6C: 'DynamicallyDefineLocalIdentifierPositiveResponse',
91            0x6E: 'WriteDataByIdentifierPositiveResponse',
92            0x70: 'InputOutputControlByLocalIdentifierPositiveResponse',
93            0x71: 'StartRoutineByLocalIdentifierPositiveResponse',
94            0x72: 'StopRoutineByLocalIdentifierPositiveResponse',
95            0x73: 'RequestRoutineResultsByLocalIdentifierPositiveResponse',
96            0x74: 'RequestDownloadPositiveResponse',
97            0x75: 'RequestUploadPositiveResponse',
98            0x76: 'TransferDataPositiveResponse',
99            0x77: 'RequestTransferExitPositiveResponse',
100           0x7B: 'WriteDataByLocalIdentifierPositiveResponse',
101           0x7D: 'WriteMemoryByAddressPositiveResponse',
102           0x7E: 'TesterPresentPositiveResponse',
103           0xC5: 'ControlDTCSettingPositiveResponse',
104           0xC6: 'ResponseOnEventPositiveResponse',
105           0x7f: 'NegativeResponse'})  # type: Dict[int, str]
106      name = 'KWP'
107      fields_desc = [
108          XByteEnumField('service', 0, services)
109      ]
110  
111      def answers(self, other):
112          # type: (Packet) -> bool
113          if not isinstance(other, type(self)):
114              return False
115          if self.service == 0x7f:
116              return self.payload.answers(other)
117          if self.service == (other.service + 0x40):
118              if isinstance(self.payload, NoPayload) or \
119                      isinstance(other.payload, NoPayload):
120                  return len(self) <= len(other)
121              else:
122                  return self.payload.answers(other.payload)
123          return False
124  
125      def hashret(self):
126          # type: () -> bytes
127          if self.service == 0x7f:
128              return struct.pack('B', self.requestServiceId & ~0x40)
129          else:
130              return struct.pack('B', self.service & ~0x40)

992  class KWP_TesterPresentSender(PeriodicSenderThread):
993      def __init__(self, sock, pkt=KWP() / KWP_TP(responseRequired=0x02),
994                   interval=2):
995          # type: (Any, _PacketIterable, float) -> None
996          """ Thread that sends TesterPresent packets periodically 
997   
998          :param sock: socket where packet is sent periodically 
999          :param pkt: packet to send 
1000         :param interval: interval between two packets 
1001         """
1002         PeriodicSenderThread.__init__(self, sock, pkt, interval)
1003 
1004     def run(self):
1005         # type: () -> None
1006         while not self._stopped.is_set():
1007             for p in self._pkts:
1008                 self._socket.sr1(p, timeout=0.3, verbose=False)
1009                 self._stopped.wait(timeout=self._interval)
1010                 if self._stopped.is_set() or self._socket.closed:
1011                     break
