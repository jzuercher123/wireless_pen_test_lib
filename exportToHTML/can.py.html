64   class CAN(Packet):
65       """A implementation of CAN messages. 
66    
67       Dissection of CAN messages from Wireshark captures and Linux PF_CAN sockets 
68       are supported from protocol specification. 
69       See https://wiki.wireshark.org/CANopen for further information on 
70       the Wireshark dissector. Linux PF_CAN and Wireshark use different 
71       endianness for the first 32 bit of a CAN message. This dissector can be 
72       configured for both use cases. 
73    
74       Configuration ``swap-bytes``: 
75           Wireshark dissection: 
76               >>> conf.contribs['CAN']['swap-bytes'] = False 
77    
78           PF_CAN Socket dissection: 
79               >>> conf.contribs['CAN']['swap-bytes'] = True 
80    
81       Configuration ``remove-padding``: 
82       Linux PF_CAN Sockets always return 16 bytes per CAN frame receive. 
83       This implicates that CAN frames get padded from the Linux PF_CAN socket 
84       with zeros up to 8 bytes of data. The real length from the CAN frame on 
85       the wire is given by the length field. To obtain only the CAN frame from 
86       the wire, this additional padding has to be removed. Nevertheless, for 
87       corner cases, it might be useful to also get the padding. This can be 
88       configured through the **remove-padding** configuration. 
89    
90       Truncate CAN frame based on length field: 
91           >>> conf.contribs['CAN']['remove-padding'] = True 
92    
93       Show entire CAN frame received from socket: 
94           >>> conf.contribs['CAN']['remove-padding'] = False 
95    
96       """
97       fields_desc = [
98           FlagsField('flags', 0, 3, ['error',
99                                      'remote_transmission_request',
100                                     'extended']),
101          XBitField('identifier', 0, 29),
102          FieldLenField('length', None, length_of='data', fmt='B'),
103          ThreeBytesField('reserved', 0),
104          StrLenField('data', b'', length_from=lambda pkt: int(pkt.length)),
105      ]
106  
107      @classmethod
108      def dispatch_hook(cls,
109                        _pkt=None,  # type: Optional[bytes]
110                        *args,  # type: Any
111                        **kargs  # type: Any
112                        ):  # type: (...) -> Type[Packet]
113          if _pkt:
114              fdf_set = len(_pkt) > 5 and _pkt[5] & 0x04 and \
115                  not _pkt[5] & 0xf8
116              if fdf_set:
117                  return CANFD
118              elif len(_pkt) > 4 and _pkt[4] > 8:
119                  return CANFD
120          return CAN
121  
122      @staticmethod
123      def inv_endianness(pkt):
124          # type: (bytes) -> bytes
125          """Invert the order of the first four bytes of a CAN packet 
126   
127          This method is meant to be used specifically to convert a CAN packet 
128          between the pcap format and the SocketCAN format 
129   
130          :param pkt: bytes str of the CAN packet 
131          :return: bytes str with the first four bytes swapped 
132          """
133          len_partial = len(pkt) - 4  # len of the packet, CAN ID excluded
134          return struct.pack('<I{}s'.format(len_partial),
135                             *struct.unpack('>I{}s'.format(len_partial), pkt))
136  
137      def pre_dissect(self, s):
138          # type: (bytes) -> bytes
139          """Implements the swap-bytes functionality when dissecting """
140          if conf.contribs['CAN']['swap-bytes']:
141              data = CAN.inv_endianness(s)  # type: bytes
142              return data
143          return s
144  
145      def post_dissect(self, s):
146          # type: (bytes) -> bytes
147          self.raw_packet_cache = None  # Reset packet to allow post_build
148          return s
149  
150      def post_build(self, pkt, pay):
151          # type: (bytes, bytes) -> bytes
152          """Implements the swap-bytes functionality for Packet build. 
153   
154          This is based on a copy of the Packet.self_build default method. 
155          The goal is to affect only the CAN layer data and keep 
156          under layers (e.g CookedLinux) unchanged 
157          """
158          if conf.contribs['CAN']['swap-bytes']:
159              data = CAN.inv_endianness(pkt)  # type: bytes
160              return data + pay
161          return pkt + pay
162  
163      def extract_padding(self, p):
164          # type: (bytes) -> Tuple[bytes, Optional[bytes]]
165          if conf.contribs['CAN']['remove-padding']:
166              return b'', None
167          else:
168              return b'', p

175  class CANFD(CAN):
176      """ 
177      This class is used for distinction of CAN FD packets. 
178      """
179      fields_desc = [
180          FlagsField('flags', 0, 3, ['error',
181                                     'remote_transmission_request',
182                                     'extended']),
183          XBitField('identifier', 0, 29),
184          FieldLenField('length', None, length_of='data', fmt='B'),
185          FlagsField('fd_flags', 4, 8, ['bit_rate_switch',
186                                        'error_state_indicator',
187                                        'fd_frame']),
188          ShortField('reserved', 0),
189          StrLenField('data', b'', length_from=lambda pkt: int(pkt.length)),
190      ]
191  
192      def post_build(self, pkt, pay):
193          # type: (bytes, bytes) -> bytes
194  
195          data = super(CANFD, self).post_build(pkt, pay)
196  
197          length = data[4]
198  
199          if 8 < length <= 24:
200              wire_length = length + (-length) % 4
201          elif 24 < length <= 64:
202              wire_length = length + (-length) % 8
203          elif length > 64:
204              raise NotImplementedError
205          else:
206              wire_length = length
207  
208          pad = b"\x00" * (wire_length - length)
209  
210          return data[0:4] + chb(wire_length) + data[5:] + pad

216  class SignalField(ScalingField):
217      """SignalField is a base class for signal data, usually transmitted from 
218      CAN messages in automotive applications. Most vehicle manufacturers 
219      describe their vehicle internal signals by so called data base CAN (DBC) 
220      files. All necessary functions to easily create Scapy dissectors similar 
221      to signal descriptions from DBC files are provided by this base class. 
222   
223      SignalField instances should only be used together with SignalPacket 
224      classes since SignalPackets enforce length checks for CAN messages. 
225   
226      """
227      __slots__ = ["start", "size"]
228  
229      def __init__(self, name, default, start, size, scaling=1, unit="",
230                   offset=0, ndigits=3, fmt="B"):
231          # type: (str, Union[int, float], int, int, Union[int, float], str, Union[int, float], int, str) -> None  # noqa: E501
232          ScalingField.__init__(self, name, default, scaling, unit, offset,
233                                ndigits, fmt)
234          self.start = start
235          self.size = abs(size)
236  
237          if fmt[-1] == "f" and self.size != 32:
238              raise Scapy_Exception("SignalField size has to be 32 for floats")
239  
240      _lookup_table = [7, 6, 5, 4, 3, 2, 1, 0,
241                       15, 14, 13, 12, 11, 10, 9, 8,
242                       23, 22, 21, 20, 19, 18, 17, 16,
243                       31, 30, 29, 28, 27, 26, 25, 24,
244                       39, 38, 37, 36, 35, 34, 33, 32,
245                       47, 46, 45, 44, 43, 42, 41, 40,
246                       55, 54, 53, 52, 51, 50, 49, 48,
247                       63, 62, 61, 60, 59, 58, 57, 56]
248  
249      @staticmethod
250      def _msb_lookup(start):
251          # type: (int) -> int
252          try:
253              return SignalField._lookup_table.index(start)
254          except ValueError:
255              raise Scapy_Exception("Only 64 bits for all SignalFields "
256                                    "are supported")
257  
258      @staticmethod
259      def _lsb_lookup(start, size):
260          # type: (int, int) -> int
261          try:
262              return SignalField._lookup_table[SignalField._msb_lookup(start) +
263                                               size - 1]
264          except IndexError:
265              raise Scapy_Exception("Only 64 bits for all SignalFields "
266                                    "are supported")
267  
268      @staticmethod
269      def _convert_to_unsigned(number, bit_length):
270          # type: (int, int) -> int
271          if number & (1 << (bit_length - 1)):
272              mask = (2 ** bit_length)  # type: int
273              return mask + number
274          return number
275  
276      @staticmethod
277      def _convert_to_signed(number, bit_length):
278          # type: (int, int) -> int
279          mask = (2 ** bit_length) - 1  # type: int
280          if number & (1 << (bit_length - 1)):
281              return number | ~mask
282          return number & mask
283  
284      def _is_little_endian(self):
285          # type: () -> bool
286          return self.fmt[0] == "<"
287  
288      def _is_signed_number(self):
289          # type: () -> bool
290          return self.fmt[-1].islower()
291  
292      def _is_float_number(self):
293          # type: () -> bool
294          return self.fmt[-1] == "f"
295  
296      def addfield(self, pkt, s, val):
297          # type: (Packet, bytes, Optional[Union[int, float]]) -> bytes
298          if not isinstance(pkt, SignalPacket):
299              raise Scapy_Exception("Only use SignalFields in a SignalPacket")
300  
301          val = self.i2m(pkt, val)
302  
303          if self._is_little_endian():
304              msb_pos = self.start + self.size - 1
305              lsb_pos = self.start
306              shift = lsb_pos
307              fmt = "<Q"
308          else:
309              msb_pos = self.start
310              lsb_pos = self._lsb_lookup(self.start, self.size)
311              shift = (64 - self._msb_lookup(msb_pos) - self.size)
312              fmt = ">Q"
313  
314          field_len = max(msb_pos, lsb_pos) // 8 + 1
315          if len(s) < field_len:
316              s += b"\x00" * (field_len - len(s))
317  
318          if self._is_float_number():
319              int_val = struct.unpack(self.fmt[0] + "I",
320                                      struct.pack(self.fmt, val))[0]  # type: int
321          elif self._is_signed_number():
322              int_val = self._convert_to_unsigned(int(val), self.size)
323          else:
324              int_val = cast(int, val)
325  
326          pkt_val = struct.unpack(fmt, (s + b"\x00" * 8)[:8])[0]
327          pkt_val |= int_val << shift
328          tmp_s = struct.pack(fmt, pkt_val)
329          return tmp_s[:len(s)]
330  
331      def getfield(self, pkt, s):
332          # type: (Packet, bytes) -> Tuple[bytes, Union[int, float]]
333          if not isinstance(pkt, SignalPacket):
334              raise Scapy_Exception("Only use SignalFields in a SignalPacket")
335  
336          if isinstance(s, tuple):
337              s, _ = s
338  
339          if self._is_little_endian():
340              msb_pos = self.start + self.size - 1
341              lsb_pos = self.start
342              shift = self.start
343              fmt = "<Q"
344          else:
345              msb_pos = self.start
346              lsb_pos = self._lsb_lookup(self.start, self.size)
347              shift = (64 - self._msb_lookup(self.start) - self.size)
348              fmt = ">Q"
349  
350          field_len = max(msb_pos, lsb_pos) // 8 + 1
351  
352          if pkt.wirelen is None:
353              pkt.wirelen = field_len
354  
355          pkt.wirelen = max(pkt.wirelen, field_len)
356  
357          fld_val = struct.unpack(fmt, (s + b"\x00" * 8)[:8])[0] >> shift
358          fld_val &= ((1 << self.size) - 1)
359  
360          if self._is_float_number():
361              fld_val = struct.unpack(self.fmt,
362                                      struct.pack(self.fmt[0] + "I", fld_val))[0]
363          elif self._is_signed_number():
364              fld_val = self._convert_to_signed(fld_val, self.size)
365  
366          return s, self.m2i(pkt, fld_val)
367  
368      def randval(self):
369          # type: () -> Union[RandBinFloat, RandFloat]
370          if self._is_float_number():
371              return RandBinFloat(0, 0)
372  
373          if self._is_signed_number():
374              min_val = -2**(self.size - 1)
375              max_val = 2**(self.size - 1) - 1
376          else:
377              min_val = 0
378              max_val = 2 ** self.size - 1
379  
380          min_val = round(min_val * self.scaling + self.offset, self.ndigits)
381          max_val = round(max_val * self.scaling + self.offset, self.ndigits)
382  
383          return RandFloat(min(min_val, max_val), max(min_val, max_val))
384  
385      def i2len(self, pkt, x):
386          # type: (Packet, Any) -> int
387          return int(float(self.size) / 8)

438  class SignalPacket(Packet):
439      """Special implementation of Packet. 
440   
441      This class enforces the correct wirelen of a CAN message for 
442      signal transmitting in automotive applications. 
443      Furthermore, the dissection order of SignalFields in fields_desc is 
444      deduced by the start index of a field. 
445      """
446  
447      def pre_dissect(self, s):
448          # type: (bytes) -> bytes
449          if not all(isinstance(f, SignalField) or
450                     (isinstance(f, ConditionalField) and
451                      isinstance(f.fld, SignalField))
452                     for f in self.fields_desc):
453              raise Scapy_Exception("Use only SignalFields in a SignalPacket")
454          return s
455  
456      def post_dissect(self, s):
457          # type: (bytes) -> bytes
458          """SignalFields can be dissected on packets with unordered fields. 
459   
460          The order of SignalFields is defined from the start parameter. 
461          After a build, the consumed bytes of the length of all SignalFields 
462          have to be removed from the SignalPacket. 
463          """
464          if self.wirelen is not None and self.wirelen > 8:
465              raise Scapy_Exception("Only 64 bits for all SignalFields "
466                                    "are supported")
467          self.raw_packet_cache = None  # Reset packet to allow post_build
468          return s[self.wirelen:]

471  class SignalHeader(CAN):
472      """Special implementation of a CAN Packet to allow dynamic binding. 
473   
474      This class can be provided to CANSockets as basecls. 
475   
476      Example: 
477          >>> class floatSignals(SignalPacket): 
478          >>>     fields_desc = [ 
479          >>>         LEFloatSignalField("floatSignal2", default=0, start=32), 
480          >>>         BEFloatSignalField("floatSignal1", default=0, start=7)] 
481          >>> 
482          >>> bind_layers(SignalHeader, floatSignals, identifier=0x321) 
483          >>> 
484          >>> dbc_sock = CANSocket("can0", basecls=SignalHeader) 
485   
486      All CAN messages received from this dbc_sock CANSocket will be interpreted 
487      as SignalHeader. Through Scapys ``bind_layers`` mechanism, all CAN messages 
488      with CAN identifier 0x321 will interpret the payload bytes of these 
489      CAN messages as floatSignals packet. 
490      """
491      fields_desc = [
492          FlagsField('flags', 0, 3, ['error',
493                                     'remote_transmission_request',
494                                     'extended']),
495          XBitField('identifier', 0, 29),
496          LenField('length', None, fmt='B'),
497          FlagsField('fd_flags', 0, 8, ['bit_rate_switch',
498                                        'error_state_indicator',
499                                        'fd_frame']),
500          ShortField('reserved', 0)
501      ]
502  
503      @classmethod
504      def dispatch_hook(cls,
505                        _pkt=None,  # type: Optional[bytes]
506                        *args,  # type: Any
507                        **kargs  # type: Any
508                        ):  # type: (...) -> Type[Packet]
509          return SignalHeader
510  
511      def extract_padding(self, s):
512          # type: (bytes) -> Tuple[bytes, Optional[bytes]]
513          return s, None

516  def rdcandump(filename, count=-1, interface=None):
517      # type: (str, int, Optional[str]) -> PacketList
518      """ Read a candump log file and return a packet list. 
519   
520      :param filename: Filename of the file to read from. 
521                       Also gzip files are accepted. 
522      :param count: Read only <count> packets. Specify -1 to read all packets. 
523      :param interface: Return only packets from a specified interface 
524      :return: A PacketList object containing the read files 
525      """
526      with CandumpReader(filename, interface) as fdesc:
527          return fdesc.read_all(count=count)

530  class CandumpReader:
531      """A stateful candump reader. Each packet is returned as a CAN packet. 
532   
533      Creates a CandumpReader object 
534   
535      :param filename: filename of a candump logfile, compressed or 
536                       uncompressed, or a already opened file object. 
537      :param interface: Name of a interface, if candump contains messages 
538                        of multiple interfaces and only one messages from a 
539                        specific interface are wanted. 
540      """
541  
542      nonblocking_socket = True
543  
544      def __init__(self, filename, interface=None):
545          # type: (str, Optional[Union[List[str], str]]) -> None
546          self.filename, self.f = self.open(filename)
547          self.ifilter = None  # type: Optional[List[str]]
548          if interface is not None:
549              if isinstance(interface, str):
550                  self.ifilter = [interface]
551              else:
552                  self.ifilter = interface
553  
554      def __iter__(self):
555          # type: () -> CandumpReader
556          return self
557  
558      @staticmethod
559      def open(filename):
560          # type: (Union[IO[bytes], str]) -> Tuple[str, _ByteStream]
561          """Open function to handle three types of input data. 
562   
563          If filename of a regular candump log file is provided, this function 
564          opens the file and returns the file object. 
565          If filename of a gzip compressed candump log file is provided, the 
566          required gzip open function is used to obtain the necessary file 
567          object, which gets returned. 
568          If a fileobject or ByteIO is provided, the filename is gathered for 
569          internal use. No further steps are performed on this object. 
570   
571          :param filename: Can be a string, specifying a candump log file or a 
572                           gzip compressed candump log file. Also already opened 
573                           file objects are allowed. 
574          :return: A opened file object for further use. 
575          """
576          """Open (if necessary) filename."""
577          if isinstance(filename, str):
578              try:
579                  fdesc = gzip.open(filename, "rb")  # type: _ByteStream
580                  # try read to cause exception
581                  fdesc.read(1)
582                  fdesc.seek(0)
583              except IOError:
584                  fdesc = open(filename, "rb")
585              return filename, fdesc
586          else:
587              name = getattr(filename, "name", "No name")
588              return name, filename
589  
590      def next(self):
591          # type: () -> Packet
592          """Implements the iterator protocol on a set of packets 
593   
594          :return: Next readable CAN Packet from the specified file 
595          """
596          try:
597              pkt = None
598              while pkt is None:
599                  pkt = self.read_packet()
600          except EOFError:
601              raise StopIteration
602  
603          return pkt
604      __next__ = next
605  
606      def read_packet(self, size=CAN_MTU):
607          # type: (int) -> Optional[Packet]
608          """Read a packet from the specified file. 
609   
610          This function will raise EOFError when no more packets are available. 
611   
612          :param size: Not used. Just here to follow the function signature for 
613                       SuperSocket emulation. 
614          :return: A single packet read from the file or None if filters apply 
615          """
616          line = self.f.readline()
617          line = line.lstrip()
618          if len(line) < 16:
619              raise EOFError
620  
621          is_log_file_format = line[0] == ord(b"(")
622          fd_flags = None
623          if is_log_file_format:
624              t_b, intf, f = line.split()
625              if b'##' in f:
626                  idn, data = f.split(b'##')
627                  fd_flags = data[0]
628                  data = data[1:]
629              else:
630                  idn, data = f.split(b'#')
631              le = None
632              t = float(t_b[1:-1])  # type: Optional[float]
633          else:
634              h, data = line.split(b']')
635              intf, idn, le = h.split()
636              t = None
637  
638          if self.ifilter is not None and \
639                  intf.decode('ASCII') not in self.ifilter:
640              return None
641  
642          data = data.replace(b' ', b'')
643          data = data.strip()
644  
645          if len(data) <= 8 and fd_flags is None:
646              pkt = CAN(identifier=int(idn, 16), data=hex_bytes(data))
647          else:
648              pkt = CANFD(identifier=int(idn, 16), fd_flags=fd_flags,
649                          data=hex_bytes(data))
650  
651          if le is not None:
652              pkt.length = int(le[1:])
653          else:
654              pkt.length = len(pkt.data)
655  
656          if len(idn) > 3:
657              pkt.flags = 0b100
658  
659          if t is not None:
660              pkt.time = t
661  
662          return pkt

664      def dispatch(self, callback):
665          # type: (Callable[[Packet], None]) -> None
666          """Call the specified callback routine for each packet read 
667   
668          This is just a convenience function for the main loop 
669          that allows for easy launching of packet processing in a 
670          thread. 
671          """
672          for p in self:
673              callback(p)

675      def read_all(self, count=-1):
676          # type: (int) -> PacketList
677          """Read a specific number or all packets from a candump file. 
678   
679          :param count: Specify a specific number of packets to be read. 
680                        All packets can be read by count=-1. 
681          :return: A PacketList object containing read CAN messages 
682          """
683          res = []
684          while count != 0:
685              try:
686                  p = self.read_packet()
687                  if p is None:
688                      continue
689              except EOFError:
690                  break
691              count -= 1
692              res.append(p)
693          return PacketList(res, name=os.path.basename(self.filename))
