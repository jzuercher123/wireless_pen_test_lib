61   class ISOTPSoftSocket(SuperSocket):
62       """ 
63       This class is a wrapper around the ISOTPSocketImplementation, for the 
64       reasons described below. 
65    
66       The ISOTPSoftSocket aims to be fully compatible with the Linux ISOTP 
67       sockets provided by the can-isotp kernel module, while being usable on any 
68       operating system. 
69       Therefore, this socket needs to be able to respond to an incoming FF frame 
70       with a FC frame even before the recv() method is called. 
71       A thread is needed for receiving CAN frames in the background, and since 
72       the lower layer CAN implementation is not guaranteed to have a functioning 
73       POSIX select(), each ISOTP socket needs its own CAN receiver thread. 
74       SuperSocket automatically calls the close() method when the GC destroys an 
75       ISOTPSoftSocket. However, note that if any thread holds a reference to 
76       an ISOTPSoftSocket object, it will not be collected by the GC. 
77    
78       The implementation of the ISOTP protocol, along with the necessary 
79       thread, are stored in the ISOTPSocketImplementation class, and therefore: 
80    
81       * There no reference from ISOTPSocketImplementation to ISOTPSoftSocket 
82       * ISOTPSoftSocket can be normally garbage collected 
83       * Upon destruction, ISOTPSoftSocket.close() will be called 
84       * ISOTPSoftSocket.close() will call ISOTPSocketImplementation.close() 
85       * RX background thread can be stopped by the garbage collector 
86    
87       Initialize an ISOTPSoftSocket using the provided underlying can socket. 
88    
89       Example (with NativeCANSocket underneath): 
90           >>> conf.contribs['ISOTP'] = {'use-can-isotp-kernel-module': False} 
91           >>> load_contrib('isotp') 
92           >>> with ISOTPSocket("can0", tx_id=0x641, rx_id=0x241) as sock: 
93           >>>     sock.send(...) 
94    
95       Example (with PythonCANSocket underneath): 
96           >>> conf.contribs['ISOTP'] = {'use-can-isotp-kernel-module': False} 
97           >>> conf.contribs['CANSocket'] = {'use-python-can': True} 
98           >>> load_contrib('isotp') 
99           >>> with ISOTPSocket(CANSocket(bustype='socketcan', channel="can0"), tx_id=0x641, rx_id=0x241) as sock: 
100          >>>     sock.send(...) 
101   
102      :param can_socket: a CANSocket instance, preferably filtering only can 
103                         frames with identifier equal to rx_id 
104      :param tx_id: the CAN identifier of the sent CAN frames 
105      :param rx_id: the CAN identifier of the received CAN frames 
106      :param ext_address: the extended address of the sent ISOTP frames 
107      :param rx_ext_address: the extended address of the received ISOTP frames 
108      :param bs: block size sent in Flow Control ISOTP frames 
109      :param stmin: minimum desired separation time sent in 
110                    Flow Control ISOTP frames 
111      :param padding: If True, pads sending packets with 0x00 which not 
112                      count to the payload. 
113                      Does not affect receiving packets. 
114      :param listen_only: Does not send Flow Control frames if a First Frame is 
115                          received 
116      :param basecls: base class of the packets emitted by this socket 
117      :param fd: enables the CanFD support for this socket 
118      """  # noqa: E501
119

178      def recv_raw(self, x=0xffff):
179          # type: (int) -> Tuple[Optional[Type[Packet]], Optional[bytes], Optional[float]]  # noqa: E501
180          """Receive a complete ISOTP message, blocking until a message is 
181          received or the specified timeout is reached. 
182          If self.timeout is 0, then this function doesn't block and returns the 
183          first frame in the receive buffer or None if there isn't any."""
184          if not self.closed:
185              tup = self.impl.recv()
186              if tup is not None:
187                  return self.basecls, tup[0], float(tup[1])
188          return self.basecls, None, None
189  
190      def recv(self, x=0xffff, **kwargs):
191          # type: (int, **Any) -> Optional[Packet]
192          msg = super(ISOTPSoftSocket, self).recv(x, **kwargs)
193          if msg is None:
194              return None
195  
196          if hasattr(msg, "tx_id"):
197              msg.tx_id = self.tx_id
198          if hasattr(msg, "rx_id"):
199              msg.rx_id = self.rx_id
200          if hasattr(msg, "ext_address"):
201              msg.ext_address = self.ext_address
202          if hasattr(msg, "rx_ext_address"):
203              msg.rx_ext_address = self.rx_ext_address
204          return msg
205  
206      @staticmethod
207      def select(sockets, remain=None):
208          # type: (List[SuperSocket], Optional[float]) -> List[SuperSocket]
209          """This function is called during sendrecv() routine to wait for 
210          sockets to be ready to receive 
211          """
212          obj_pipes = [x.impl.rx_queue for x in sockets if
213                       isinstance(x, ISOTPSoftSocket) and not x.closed]
214  
215          ready_pipes = select_objects(obj_pipes, remain)
216  
217          return [x for x in sockets if isinstance(x, ISOTPSoftSocket) and
218                  not x.closed and x.impl.rx_queue in ready_pipes]

221  class TimeoutScheduler:
222      """A timeout scheduler which uses a single thread for all timeouts, unlike 
223      python's own Timer objects which use a thread each."""
224      GRACE = .1
225      _mutex = RLock()
226      _event = Event()
227      _thread = None  # type: Optional[Thread]
228  
229      # use heapq functions on _handles!
230      _handles = []  # type: List[TimeoutScheduler.Handle]
231  
232      logger = logging.getLogger("scapy.contrib.automotive.timeout_scheduler")
233  
234      @classmethod
235      def schedule(cls, timeout, callback):
236          # type: (float, Callable[[], None]) -> TimeoutScheduler.Handle
237          """Schedules the execution of a timeout. 
238   
239          The function `callback` will be called in `timeout` seconds. 
240   
241          Returns a handle that can be used to remove the timeout."""
242          when = cls._time() + timeout
243          handle = cls.Handle(when, callback)
244  
245          with cls._mutex:
246              # Add the handler to the heap, keeping the invariant
247              # Time complexity is O(log n)
248              heapq.heappush(cls._handles, handle)
249              must_interrupt = cls._handles[0] == handle
250  
251              # Start the scheduling thread if it is not started already
252              if cls._thread is None:
253                  t = Thread(target=cls._task, name="TimeoutScheduler._task")
254                  must_interrupt = False
255                  cls._thread = t
256                  cls._event.clear()
257                  t.start()
258  
259          if must_interrupt:
260              # if the new timeout got in front of the one we are currently
261              # waiting on, the current wait operation must be aborted and
262              # updated with the new timeout
263              cls._event.set()
264              time.sleep(0)  # call "yield"
265  
266          # Return the handle to the timeout so that the user can cancel it
267          return handle
268  
269      @classmethod
270      def cancel(cls, handle):
271          # type: (TimeoutScheduler.Handle) -> None
272          """Provided its handle, cancels the execution of a timeout."""
273  
274          with cls._mutex:
275              if handle in cls._handles:
276                  # Time complexity is O(n)
277                  handle._cb = None
278                  cls._handles.remove(handle)
279                  heapq.heapify(cls._handles)
280  
281                  if len(cls._handles) == 0:
282                      # set the event to stop the wait - this kills the thread
283                      cls._event.set()
284              else:
285                  raise Scapy_Exception("Handle not found")
286  
287      @classmethod
288      def clear(cls):
289          # type: () -> None
290          """Cancels the execution of all timeouts."""
291          with cls._mutex:
292              cls._handles = []
293  
294          # set the event to stop the wait - this kills the thread
295          cls._event.set()
296  
297      @classmethod
298      def _peek_next(cls):
299          # type: () -> Optional[TimeoutScheduler.Handle]
300          """Returns the next timeout to execute, or `None` if list is empty, 
301          without modifying the list"""
302          with cls._mutex:
303              return cls._handles[0] if cls._handles else None
304  
305      @classmethod
306      def _wait(cls, handle):
307          # type: (Optional[TimeoutScheduler.Handle]) -> None
308          """Waits until it is time to execute the provided handle, or until 
309          another thread calls _event.set()"""
310  
311          now = cls._time()
312  
313          # Check how much time until the next timeout
314          if handle is None:
315              to_wait = cls.GRACE
316          else:
317              to_wait = handle._when - now
318  
319          # Wait until the next timeout,
320          # or until event.set() gets called in another thread.
321          if to_wait > 0:
322              cls.logger.debug("Thread going to sleep @ %f " +
323                               "for %fs", now, to_wait)
324              interrupted = cls._event.wait(to_wait)
325              new = cls._time()
326              cls.logger.debug("Thread awake @ %f, slept for" +
327                               " %f, interrupted=%d", new, new - now,
328                               interrupted)
329  
330          # Clear the event so that we can wait on it again,
331          # Must be done before doing the callbacks to avoid losing a set().
332          cls._event.clear()
333  
334      @classmethod
335      def _task(cls):
336          # type: () -> None
337          """Executed in a background thread, this thread will automatically 
338          start when the first timeout is added and stop when the last timeout 
339          is removed or executed."""
340  
341          cls.logger.debug("Thread spawning @ %f", cls._time())
342  
343          time_empty = None
344  
345          try:
346              while 1:
347                  handle = cls._peek_next()
348                  if handle is None:
349                      now = cls._time()
350                      if time_empty is None:
351                          time_empty = now
352                      # 100 ms of grace time before killing the thread
353                      if cls.GRACE < now - time_empty:
354                          return
355                  else:
356                      time_empty = None
357                  cls._wait(handle)
358                  cls._poll()
359  
360          finally:
361              # Worst case scenario: if this thread dies, the next scheduled
362              # timeout will start a new one
363              cls.logger.debug("Thread died @ %f", cls._time())
364              cls._thread = None
365  
366      @classmethod
367      def _poll(cls):
368          # type: () -> None
369          """Execute all the callbacks that were due until now"""
370  
371          while 1:
372              with cls._mutex:
373                  now = cls._time()
374                  if len(cls._handles) == 0 or cls._handles[0]._when > now:
375                      # There is nothing to execute yet
376                      return
377  
378                  # Time complexity is O(log n)
379                  handle = heapq.heappop(cls._handles)
380                  callback = None
381                  if handle is not None:
382                      callback = handle._cb
383                      handle._cb = True
384  
385              # Call the callback here, outside the mutex
386              if callable(callback):
387                  try:
388                      callback()
389                  except Exception:
390                      traceback.print_exc()
391  
392      @staticmethod
393      def _time():
394          # type: () -> float
395          return time.monotonic()
396  
397      class Handle:
398          """Handle for a timeout, consisting of a callback and a time when it 
399          should be executed."""
400          __slots__ = ['_when', '_cb']
401  
402          def __init__(self,
403                       when,  # type: float
404                       cb  # type: Optional[Union[Callable[[], None], bool]]
405                       ):
406              # type: (...) -> None
407              self._when = when
408              self._cb = cb
409  
410          def cancel(self):
411              # type: () -> bool
412              """Cancels this timeout, preventing it from executing its 
413              callback"""
414              if self._cb is None:
415                  raise Scapy_Exception(
416                      "cancel() called on previous canceled Handle")
417              else:
418                  with TimeoutScheduler._mutex:
419                      if isinstance(self._cb, bool):
420                          # Handle was already executed.
421                          # We don't need to cancel anymore
422                          return False
423                      else:
424                          self._cb = None
425                          TimeoutScheduler.cancel(self)
426                          return True
427  
428          def __lt__(self, other):
429              # type: (Any) -> bool
430              if not isinstance(other, TimeoutScheduler.Handle):
431                  raise TypeError()
432              return self._when < other._when
433  
434          def __le__(self, other):
435              # type: (Any) -> bool
436              if not isinstance(other, TimeoutScheduler.Handle):
437                  raise TypeError()
438              return self._when <= other._when
439  
440          def __gt__(self, other):
441              # type: (Any) -> bool
442              if not isinstance(other, TimeoutScheduler.Handle):
443                  raise TypeError()
444              return self._when > other._when
445  
446          def __ge__(self, other):
447              # type: (Any) -> bool
448              if not isinstance(other, TimeoutScheduler.Handle):
449                  raise TypeError()
450              return self._when >= other._when

453  class ISOTPSocketImplementation:
454      """ 
455      Implementation of an ISOTP "state machine". 
456   
457      Most of the ISOTP logic was taken from 
458      https://github.com/hartkopp/can-isotp/blob/master/net/can/isotp.c 
459   
460      This class is separated from ISOTPSoftSocket to make sure the background 
461      thread can't hold a reference to ISOTPSoftSocket, allowing it to be 
462      collected by the GC. 
463   
464      :param can_socket: a CANSocket instance, preferably filtering only can 
465                         frames with identifier equal to rx_id 
466      :param tx_id: the CAN identifier of the sent CAN frames 
467      :param rx_id: the CAN identifier of the received CAN frames 
468      :param padding: If True, pads sending packets with 0x00 which not 
469                      count to the payload. 
470                      Does not affect receiving packets. 
471      :param ext_address: Extended Address byte to be added at the 
472              beginning of every CAN frame _sent_ by this object. Can be None 
473              in order to disable extended addressing on sent frames. 
474      :param rx_ext_address: Extended Address byte expected to be found at 
475              the beginning of every CAN frame _received_ by this object. Can 
476              be None in order to disable extended addressing on received 
477              frames. 
478      :param bs: Block Size byte to be included in every Control 
479              Flow Frame sent by this object. The default value of 0 means 
480              that all the data will be received in a single block. 
481      :param stmin: Time Minimum Separation byte to be 
482              included in every Control Flow Frame sent by this object. The 
483              default value of 0 indicates that the peer will not wait any 
484              time between sending frames. 
485      :param listen_only: Disables send of flow control frames 
486      """
487  
488      def __init__(self,
489                   can_socket,  # type: "CANSocket"
490                   tx_id,  # type: int
491                   rx_id,  # type: int
492                   padding=False,  # type: bool
493                   ext_address=None,  # type: Optional[int]
494                   rx_ext_address=None,  # type: Optional[int]
495                   bs=0,  # type: int
496                   stmin=0,  # type: int
497                   listen_only=False,  # type: bool
498                   fd=False  # type: bool
499                   ):
500          # type: (...) -> None
501          self.can_socket = can_socket
502          self.rx_id = rx_id
503          self.tx_id = tx_id
504          self.padding = padding
505          self.fc_timeout = 1
506          self.cf_timeout = 1
507  
508          self.fd = fd
509  
510          self.max_dlen = CAN_FD_MAX_DLEN if fd else CAN_MAX_DLEN
511  
512          self.filter_warning_emitted = False
513          self.closed = False
514  
515          self.rx_ext_address = rx_ext_address
516          self.ea_hdr = b""
517          if ext_address is not None:
518              self.ea_hdr = struct.pack("B", ext_address)
519          self.listen_only = listen_only
520  
521          self.rxfc_bs = bs
522          self.rxfc_stmin = stmin
523  
524          self.rx_queue = ObjectPipe[Tuple[bytes, Union[float, EDecimal]]]()
525          self.rx_len = -1
526          self.rx_buf = None  # type: Optional[bytes]
527          self.rx_sn = 0
528          self.rx_bs = 0
529          self.rx_idx = 0
530          self.rx_ts = 0.0  # type: Union[float, EDecimal]
531          self.rx_state = ISOTP_IDLE
532  
533          self.tx_queue = ObjectPipe[bytes]()
534          self.txfc_bs = 0
535          self.txfc_stmin = 0
536          self.tx_gap = 0.
537  
538          self.tx_buf = None  # type: Optional[bytes]
539          self.tx_sn = 0
540          self.tx_bs = 0
541          self.tx_idx = 0
542          self.rx_ll_dl = 0
543          self.tx_state = ISOTP_IDLE
544  
545          self.rx_tx_poll_rate = 0.005
546          self.tx_timeout_handle = None  # type: Optional[TimeoutScheduler.Handle]  # noqa: E501
547          self.rx_timeout_handle = None  # type: Optional[TimeoutScheduler.Handle]  # noqa: E501
548          self.rx_handle = TimeoutScheduler.schedule(
549              self.rx_tx_poll_rate, self.can_recv)
550          self.tx_handle = TimeoutScheduler.schedule(
551              self.rx_tx_poll_rate, self._send)
552          self.last_rx_call = 0.0
553  
554      def failure_analysis(self):
555          # type: () -> None
556          log_isotp.debug("Failure analysis")
557          log_isotp.debug("Last_rx_call: %s", str(self.last_rx_call))
558          log_isotp.debug("self.rx_handle: %s", str(self.rx_handle))
559          log_isotp.debug("self.rx_handle._cb: %s", str(self.rx_handle._cb))
560          log_isotp.debug("self.rx_handle._when: %s", str(self.rx_handle._when))
561          log_isotp.debug("Now: %s", TimeoutScheduler._time())
562  
563      def __del__(self):
564          # type: () -> None
565          self.close()
566  
567      def can_send(self, load):
568          # type: (bytes) -> None
569          def _get_padding_size(pl_size):
570              # type: (int) -> int
571              if not self.fd:
572                  return CAN_MAX_DLEN
573              else:
574                  fd_accepted_sizes = [0, 8, 12, 16, 20, 24, 32, 48, 64]
575                  pos = bisect_left(fd_accepted_sizes, pl_size)
576                  if pos == 0:
577                      return fd_accepted_sizes[0]
578                  if pos == len(fd_accepted_sizes):
579                      return fd_accepted_sizes[-1]
580                  return fd_accepted_sizes[pos]
581  
582          if self.padding:
583              load += b"\xCC" * (_get_padding_size(len(load)) - len(load))
584          if self.tx_id is None or self.tx_id <= 0x7ff:
585              self.can_socket.send(CAN(identifier=self.tx_id, data=load))
586          else:
587              self.can_socket.send(CAN(identifier=self.tx_id, flags="extended",
588                                       data=load))
589  
590      def can_recv(self):
591          # type: () -> None
592          self.last_rx_call = TimeoutScheduler._time()
593          if self.can_socket.select([self.can_socket], 0):
594              pkt = self.can_socket.recv()
595              if pkt:
596                  self.on_can_recv(pkt)
597          if not self.closed and not self.can_socket.closed:
598              if self.can_socket.select([self.can_socket], 0):
599                  poll_time = 0.0
600              else:
601                  poll_time = self.rx_tx_poll_rate
602              self.rx_handle = TimeoutScheduler.schedule(
603                  poll_time, self.can_recv)
604          else:
605              try:
606                  self.rx_handle.cancel()
607              except Scapy_Exception:
608                  pass
609  
610      def on_can_recv(self, p):
611          # type: (Packet) -> None
612          if p.identifier != self.rx_id:
613              if not self.filter_warning_emitted and conf.verb >= 2:
614                  log_isotp.warning("You should put a filter for identifier=%x on your "
615                                    "CAN socket", self.rx_id)
616                  self.filter_warning_emitted = True
617          else:
618              self.on_recv(p)
619  
620      def close(self):
621          # type: () -> None
622          try:
623              if select_objects([self.tx_queue], 0):
624                  log_isotp.warning("TX queue not empty")
625                  time.sleep(0.1)
626          except OSError:
627              pass
628  
629          try:
630              if select_objects([self.rx_queue], 0):
631                  log_isotp.warning("RX queue not empty")
632          except OSError:
633              pass
634  
635          self.closed = True
636          try:
637              self.rx_handle.cancel()
638          except Scapy_Exception:
639              pass
640          try:
641              self.tx_handle.cancel()
642          except Scapy_Exception:
643              pass
644  
645      def _rx_timer_handler(self):
646          # type: () -> None
647          """Method called every time the rx_timer times out, due to the peer not 
648          sending a consecutive frame within the expected time window"""
649  
650          if self.rx_state == ISOTP_WAIT_DATA:
651              # we did not get new data frames in time.
652              # reset rx state
653              self.rx_state = ISOTP_IDLE
654              if conf.verb > 2:
655                  log_isotp.warning("RX state was reset due to timeout")
656  
657      def _tx_timer_handler(self):
658          # type: () -> None
659          """Method called every time the tx_timer times out, which can happen in 
660          two situations: either a Flow Control frame was not received in time, 
661          or the Separation Time Min is expired and a new frame must be sent."""
662  
663          if (self.tx_state == ISOTP_WAIT_FC or
664                  self.tx_state == ISOTP_WAIT_FIRST_FC):
665              # we did not get any flow control frame in time
666              # reset tx state
667              self.tx_state = ISOTP_IDLE
668              log_isotp.warning("TX state was reset due to timeout")
669              return
670          elif self.tx_state == ISOTP_SENDING:
671              # push out the next segmented pdu
672              src_off = len(self.ea_hdr)
673              max_bytes = (self.max_dlen - 1) - src_off
674              if self.tx_buf is None:
675                  self.tx_state = ISOTP_IDLE
676                  log_isotp.warning("TX buffer is not filled")
677                  return
678              while 1:
679                  load = self.ea_hdr
680                  load += struct.pack("B", N_PCI_CF + self.tx_sn)
681                  load += self.tx_buf[self.tx_idx:self.tx_idx + max_bytes]
682                  self.can_send(load)
683  
684                  self.tx_sn = (self.tx_sn + 1) % 16
685                  self.tx_bs += 1
686                  self.tx_idx += max_bytes
687  
688                  if len(self.tx_buf) <= self.tx_idx:
689                      # we are done
690                      self.tx_state = ISOTP_IDLE
691                      return
692  
693                  if self.txfc_bs != 0 and self.tx_bs >= self.txfc_bs:
694                      # stop and wait for FC
695                      self.tx_state = ISOTP_WAIT_FC
696                      self.tx_timeout_handle = TimeoutScheduler.schedule(
697                          self.fc_timeout, self._tx_timer_handler)
698                      return
699  
700                  if self.tx_gap == 0:
701                      continue
702                  else:
703                      # stop and wait for tx gap
704                      self.tx_timeout_handle = TimeoutScheduler.schedule(
705                          self.tx_gap, self._tx_timer_handler)
706                      return

708      def on_recv(self, cf):
709          # type: (Packet) -> None
710          """Function that must be called every time a CAN frame is received, to 
711          advance the state machine."""
712  
713          data = bytes(cf.data)
714  
715          if len(data) < 2:
716              return
717  
718          ae = 0
719          if self.rx_ext_address is not None:
720              ae = 1
721              if len(data) < 3:
722                  return
723              if data[0] != self.rx_ext_address:
724                  return
725  
726          n_pci = data[ae] & 0xf0
727  
728          if n_pci == N_PCI_FC:
729              self._recv_fc(data[ae:])
730          elif n_pci == N_PCI_SF:
731              self._recv_sf(data[ae:], cf.time)
732          elif n_pci == N_PCI_FF:
733              self._recv_ff(data[ae:], cf.time)
734          elif n_pci == N_PCI_CF:
735              self._recv_cf(data[ae:])
736  
737      def _recv_fc(self, data):
738          # type: (bytes) -> None
739          """Process a received 'Flow Control' frame"""
740          log_isotp.debug("Processing FC")
741  
742          if (self.tx_state != ISOTP_WAIT_FC and
743                  self.tx_state != ISOTP_WAIT_FIRST_FC):
744              return
745  
746          if self.tx_timeout_handle is not None:
747              self.tx_timeout_handle.cancel()
748              self.tx_timeout_handle = None
749  
750          if len(data) < 3:
751              self.tx_state = ISOTP_IDLE
752              log_isotp.warning("CF frame discarded because it was too short")
753              return
754  
755          # get communication parameters only from the first FC frame
756          if self.tx_state == ISOTP_WAIT_FIRST_FC:
757              self.txfc_bs = data[1]
758              self.txfc_stmin = data[2]
759  
760          if ((self.txfc_stmin > 0x7F) and
761                  ((self.txfc_stmin < 0xF1) or (self.txfc_stmin > 0xF9))):
762              self.txfc_stmin = 0x7F
763  
764          if data[2] <= 127:
765              self.tx_gap = data[2] / 1000
766          elif 0xf1 <= data[2] <= 0xf9:
767              self.tx_gap = (data[2] & 0x0f) / 10000
768          else:
769              self.tx_gap = 0.
770  
771          self.tx_state = ISOTP_WAIT_FC
772  
773          isotp_fc = data[0] & 0x0f
774  
775          if isotp_fc == ISOTP_FC_CTS:
776              self.tx_bs = 0
777              self.tx_state = ISOTP_SENDING
778              # start cyclic timer for sending CF frame
779              self.tx_timeout_handle = TimeoutScheduler.schedule(
780                  self.tx_gap, self._tx_timer_handler)
781          elif isotp_fc == ISOTP_FC_WT:
782              # start timer to wait for next FC frame
783              self.tx_state = ISOTP_WAIT_FC
784              self.tx_timeout_handle = TimeoutScheduler.schedule(
785                  self.fc_timeout, self._tx_timer_handler)
786          elif isotp_fc == ISOTP_FC_OVFLW:
787              # overflow in receiver side
788              self.tx_state = ISOTP_IDLE
789              log_isotp.warning("Overflow happened at the receiver side")
790              return
791          else:
792              self.tx_state = ISOTP_IDLE
793              log_isotp.warning("Unknown FC frame type")
794              return
795  
796      def _recv_sf(self, data, ts):
797          # type: (bytes, Union[float, EDecimal]) -> None
798          """Process a received 'Single Frame' frame"""
799          log_isotp.debug("Processing SF")
800  
801          if self.rx_timeout_handle is not None:
802              self.rx_timeout_handle.cancel()
803              self.rx_timeout_handle = None
804  
805          if self.rx_state != ISOTP_IDLE:
806              if conf.verb > 2:
807                  log_isotp.warning("RX state was reset because "
808                                    "single frame was received")
809              self.rx_state = ISOTP_IDLE
810  
811          length = data[0] & 0xf
812          is_fd_frame = self.fd and length == 0 and len(data) >= 2
813  
814          if is_fd_frame:
815              length = data[1]
816  
817          if len(data) - 1 < length:
818              return
819  
820          msg = None
821          if is_fd_frame:
822              msg = data[2:2 + length]
823          else:
824              msg = data[1:1 + length]
825          self.rx_queue.send((msg, ts))
826  
827      def _recv_ff(self, data, ts):
828          # type: (bytes, Union[float, EDecimal]) -> None
829          """Process a received 'First Frame' frame"""
830          log_isotp.debug("Processing FF")
831  
832          if self.rx_timeout_handle is not None:
833              self.rx_timeout_handle.cancel()
834              self.rx_timeout_handle = None
835  
836          if self.rx_state != ISOTP_IDLE:
837              if conf.verb > 2:
838                  log_isotp.warning("RX state was reset because first frame was received")
839              self.rx_state = ISOTP_IDLE
840  
841          if len(data) < 7:
842              return
843          self.rx_ll_dl = len(data)
844  
845          # get the FF_DL
846          self.rx_len = (data[0] & 0x0f) * 256 + data[1]
847          ff_pci_sz = 2
848  
849          # Check for FF_DL escape sequence supporting 32 bit PDU length
850          if self.rx_len == 0:
851              # FF_DL = 0 => get real length from next 4 bytes
852              self.rx_len = data[2] << 24
853              self.rx_len += data[3] << 16
854              self.rx_len += data[4] << 8
855              self.rx_len += data[5]
856              ff_pci_sz = 6
857  
858          # copy the first received data bytes
859          data_bytes = data[ff_pci_sz:]
860          self.rx_idx = len(data_bytes)
861          self.rx_buf = data_bytes
862          self.rx_ts = ts
863  
864          # initial setup for this pdu reception
865          self.rx_sn = 1
866          self.rx_state = ISOTP_WAIT_DATA
867  
868          # no creation of flow control frames
869          if not self.listen_only:
870              # send our first FC frame
871              load = self.ea_hdr
872              load += struct.pack("BBB", N_PCI_FC, self.rxfc_bs, self.rxfc_stmin)
873              self.can_send(load)
874  
875          # wait for a CF
876          self.rx_bs = 0
877          self.rx_timeout_handle = TimeoutScheduler.schedule(
878              self.cf_timeout, self._rx_timer_handler)
879  
880      def _recv_cf(self, data):
881          # type: (bytes) -> None
882          """Process a received 'Consecutive Frame' frame"""
883          log_isotp.debug("Processing CF")
884  
885          if self.rx_state != ISOTP_WAIT_DATA:
886              return
887  
888          if self.rx_timeout_handle is not None:
889              self.rx_timeout_handle.cancel()
890              self.rx_timeout_handle = None
891  
892          # CFs are never longer than the FF
893          if len(data) > self.rx_ll_dl:
894              return
895  
896          # CFs have usually the LL_DL length
897          if len(data) < self.rx_ll_dl:
898              # this is only allowed for the last CF
899              if self.rx_len - self.rx_idx > self.rx_ll_dl:
900                  if conf.verb > 2:
901                      log_isotp.warning("Received a CF with insufficient length")
902                  return
903  
904          if data[0] & 0x0f != self.rx_sn:
905              # Wrong sequence number
906              if conf.verb > 2:
907                  log_isotp.warning("RX state was reset because wrong sequence "
908                                    "number was received")
909              self.rx_state = ISOTP_IDLE
910              return
911  
912          if self.rx_buf is None:
913              if conf.verb > 2:
914                  log_isotp.warning("rx_buf not filled with data!")
915              self.rx_state = ISOTP_IDLE
916              return
917  
918          self.rx_sn = (self.rx_sn + 1) % 16
919          self.rx_buf += data[1:]
920          self.rx_idx = len(self.rx_buf)
921  
922          if self.rx_idx >= self.rx_len:
923              # we are done
924              self.rx_buf = self.rx_buf[0:self.rx_len]
925              self.rx_state = ISOTP_IDLE
926              self.rx_queue.send((self.rx_buf, self.rx_ts))
927              self.rx_buf = None
928              return
929  
930          # perform blocksize handling, if enabled
931          if self.rxfc_bs != 0:
932              self.rx_bs += 1
933  
934              # check if we reached the end of the block
935              if self.rx_bs >= self.rxfc_bs and not self.listen_only:
936                  # send our FC frame
937                  load = self.ea_hdr
938                  load += struct.pack("BBB", N_PCI_FC, self.rxfc_bs,
939                                      self.rxfc_stmin)
940                  self.rx_bs = 0
941                  self.can_send(load)
942  
943          # wait for another CF
944          log_isotp.debug("Wait for another CF")
945          self.rx_timeout_handle = TimeoutScheduler.schedule(
946              self.cf_timeout, self._rx_timer_handler)
947  
948      def begin_send(self, x):
949          # type: (bytes) -> None
950          """Begins sending an ISOTP message. This method does not block."""
951          if self.tx_state != ISOTP_IDLE:
952              log_isotp.warning("Socket is already sending, retry later")
953              return
954  
955          self.tx_state = ISOTP_SENDING
956          length = len(x)
957          if length > ISOTP_MAX_DLEN_2015:
958              log_isotp.warning("Too much data for ISOTP message")
959  
960          sf_size_check = self.max_dlen - 1
961  
962          if len(self.ea_hdr) + length + int(self.fd) <= sf_size_check:
963              # send a single frame
964              data = self.ea_hdr
965              if not self.fd or length <= 7:
966                  data += struct.pack("B", length)
967              else:
968                  data += struct.pack("BB", 0, length)
969              data += x
970              self.tx_state = ISOTP_IDLE
971              self.can_send(data)
972              return
973  
974          # send the first frame
975          data = self.ea_hdr
976          if length > ISOTP_MAX_DLEN:
977              data += struct.pack(">HI", 0x1000, length)
978          else:
979              data += struct.pack(">H", 0x1000 | length)
980          load = x[0:self.max_dlen - len(data)]
981          data += load
982          self.can_send(data)
983  
984          self.tx_buf = x
985          self.tx_sn = 1
986          self.tx_bs = 0
987          self.tx_idx = len(load)
988  
989          self.tx_state = ISOTP_WAIT_FIRST_FC
990          self.tx_timeout_handle = TimeoutScheduler.schedule(
991              self.fc_timeout, self._tx_timer_handler)
992  
993      def _send(self):
994          # type: () -> None
995          if self.tx_state == ISOTP_IDLE:
996              if select_objects([self.tx_queue], 0):
997                  pkt = self.tx_queue.recv()
998                  if pkt:
999                      self.begin_send(pkt)
1000 
1001         if not self.closed:
1002             self.tx_handle = TimeoutScheduler.schedule(
1003                 self.rx_tx_poll_rate, self._send)
1004         else:
1005             try:
1006                 self.tx_handle.cancel()
1007             except Scapy_Exception:
1008                 pass

1013         happens."""
