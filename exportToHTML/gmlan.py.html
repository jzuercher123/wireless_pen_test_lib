54   class GMLAN(ISOTP):
55       @staticmethod
56       def determine_len(x):
57           if conf.contribs['GMLAN']['GMLAN_ECU_AddressingScheme'] is None:
58               log_automotive.warning(
59                   "Define conf.contribs['GMLAN']['GMLAN_ECU_AddressingScheme']! "
60                   "Assign either 2,3 or 4")
61           if conf.contribs['GMLAN']['GMLAN_ECU_AddressingScheme'] \
62                   not in [2, 3, 4]:
63               log_automotive.warning(
64                   "Define conf.contribs['GMLAN']['GMLAN_ECU_AddressingScheme']! "
65                   "Assign either 2,3 or 4")
66           return conf.contribs['GMLAN']['GMLAN_ECU_AddressingScheme'] == x

68       services = ObservableDict(
69           {0x04: 'ClearDiagnosticInformation',
70            0x10: 'InitiateDiagnosticOperation',
71            0x12: 'ReadFailureRecordData',
72            0x1a: 'ReadDataByIdentifier',
73            0x20: 'ReturnToNormalOperation',
74            0x22: 'ReadDataByParameterIdentifier',
75            0x23: 'ReadMemoryByAddress',
76            0x27: 'SecurityAccess',
77            0x28: 'DisableNormalCommunication',
78            0x2c: 'DynamicallyDefineMessage',
79            0x2d: 'DefinePIDByAddress',
80            0x34: 'RequestDownload',
81            0x36: 'TransferData',
82            0x3b: 'WriteDataByIdentifier',
83            0x3e: 'TesterPresent',
84            0x44: 'ClearDiagnosticInformationPositiveResponse',
85            0x50: 'InitiateDiagnosticOperationPositiveResponse',
86            0x52: 'ReadFailureRecordDataPositiveResponse',
87            0x5a: 'ReadDataByIdentifierPositiveResponse',

89            0x62: 'ReadDataByParameterIdentifierPositiveResponse',

96            0x76: 'TransferDataPositiveResponse',
97            0x7b: 'WriteDataByIdentifierPositiveResponse',
98            0x7e: 'TesterPresentPositiveResponse',
99            0x7f: 'NegativeResponse',
100           0xa2: 'ReportProgrammingState',
101           0xa5: 'ProgrammingMode',
102           0xa9: 'ReadDiagnosticInformation',
103           0xaa: 'ReadDataByPacketIdentifier',
104           0xae: 'DeviceControl',

106           0xe5: 'ProgrammingModePositiveResponse',
107           0xe9: 'ReadDiagnosticInformationPositiveResponse',
108           0xea: 'ReadDataByPacketIdentifierPositiveResponse',
109           0xee: 'DeviceControlPositiveResponse'})
110      name = 'General Motors Local Area Network'
111      fields_desc = [
112          XByteEnumField('service', 0, services)
113      ]
114  
115      def answers(self, other):
116          if not isinstance(other, type(self)):
117              return False
118          if self.service == 0x7f:
119              return self.payload.answers(other)
120          if self.service == (other.service + 0x40):
121              if isinstance(self.payload, NoPayload) or \
122                      isinstance(other.payload, NoPayload):
123                  return True
124              else:
125                  return self.payload.answers(other.payload)
126          return False
127  
128      def hashret(self):
129          if self.service == 0x7f:
130              return struct.pack('B', self.requestServiceId & ~0x40)
131          return struct.pack('B', self.service & ~0x40)
