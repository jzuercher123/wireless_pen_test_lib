40   def i2s_frameid(x):
41       """ Get representation name of a pnio frame ID 
42    
43       :param x: a key of the PNIO_FRAME_IDS dictionary 
44       :returns: str 
45       """
46       try:
47           return PNIO_FRAME_IDS[x]
48       except KeyError:
49           pass
50       if 0x0100 <= x < 0x1000:
51           return "RT_CLASS_3 (%4x)" % x
52       if 0x8000 <= x < 0xC000:
53           return "RT_CLASS_1 (%4x)" % x
54       if 0xC000 <= x < 0xFC00:
55           return "RT_CLASS_UDP (%4x)" % x
56       if 0xFF80 <= x < 0xFF90:
57           return "FragmentationFrameID (%4x)" % x
58       return x

61   def s2i_frameid(x):
62       """ Get pnio frame ID from a representation name 
63    
64       Performs a reverse look-up in PNIO_FRAME_IDS dictionary 
65    
66       :param x: a value of PNIO_FRAME_IDS dict 
67       :returns: integer 
68       """
69       try:
70           return {
71               "RT_CLASS_3": 0x0100,
72               "RT_CLASS_1": 0x8000,
73               "RT_CLASS_UDP": 0xC000,
74               "FragmentationFrameID": 0xFF80,
75           }[x]
76       except KeyError:
77           pass
78       try:
79           return next(key for key, value in PNIO_FRAME_IDS.items()
80                       if value == x)
81       except StopIteration:
82           pass
83       return x

90   class ProfinetIO(Packet):
91       """ Basic PROFINET IO dispatcher """
92       fields_desc = [
93           XShortEnumField("frameID", 0, (i2s_frameid, s2i_frameid))
94       ]
95   
96       def guess_payload_class(self, payload):
97           # For frameID in the RT_CLASS_* range, use the RTC packet as payload
98           if self.frameID in [0xfefe, 0xfeff, 0xfefd]:
99               from scapy.contrib.pnio_dcp import ProfinetDCP
100              return ProfinetDCP
101          elif self.frameID == 0xFE01:
102              from scapy.contrib.pnio_rpc import Alarm_Low
103              return Alarm_Low
104          elif self.frameID == 0xFC01:
105              from scapy.contrib.pnio_rpc import Alarm_High
106              return Alarm_High
107          elif (
108                  (0x0100 <= self.frameID < 0x1000) or
109                  (0x8000 <= self.frameID < 0xFC00)
110  
111  
112          ):
113              return PNIORealTimeCyclicPDU
114          return super(ProfinetIO, self).guess_payload_class(payload)

129      """ IOCS and IOPS packets for PROFINET Real-Time payload """
130      name = "PNIO RTC IOxS"
131      fields_desc = [
132          # IOxS.DataState -- IEC-61158 - 6 - 10 / FDIS ED 3, Table 181
133          BitEnumField("dataState", 1, 1, ["bad", "good"]),
134          # IOxS.Instance -- IEC-61158 - 6 - 10 / FDIS ED 3, Table 180
135          BitEnumField("instance", 0, 2,
136                       ["subslot", "slot", "device", "controller"]),
137          # IOxS.reserved -- IEC-61158 - 6 - 10 / FDIS ED 3, line 2649

139          # IOxS.Extension -- IEC-61158-6-10/FDIS ED 3, Table 179
140          BitField("extension", 0, 1),

144      def is_extension_set(cls, _pkt, _lst, p, _remain):
145          ret = cls if isinstance(p, type(None)) or p.extension != 0 else None
146          return ret

149      def get_len(cls):
150          return sum(type(fld).i2len(None, 0) for fld in cls.fields_desc)

153          return conf.padding_layer

168  class PNIORealTimeCyclicPDU(Packet):
169      """ PROFINET cyclic real-time """
170      __slots__ = ["_len", "_layout"]
171      name = "PROFINET Real-Time"
172  
173      fields_desc = [
174          # C_SDU ^ CSF_SDU -- IEC-61158-6-10/FDIS ED 3, Table 163
175          PacketListField(
176              "data", [],
177              next_cls_cb=lambda pkt, lst, p, remain: pkt.next_cls_cb(
178                  lst, p, remain)
179          ),
180          # RTCPadding -- IEC - 61158 - 6 - 10 / FDIS ED 3, Table 163
181          StrFixedLenField("padding", '',
182                           length_from=lambda p: p.get_padding_length()),
183          # APDU_Status -- IEC-61158-6-10/FDIS ED 3, Table 164
184          ShortField("cycleCounter", 0),
185          FlagsField("dataStatus", 0x35, 8, [
186              "primary",
187              "redundancy",
188              "validData",
189              "reserved_1",
190              "run",
191              "no_problem",
192              "reserved_2",
193              "ignore",
194          ]),
195          ByteField("transferStatus", 0)
196      ]
197  
198      def pre_dissect(self, s):
199          # Constraint from IEC-61158-6-10/FDIS ED 3, line 690
200          self._len = min(1440, len(s))
201          return s
202  
203      def get_padding_length(self):
204          if hasattr(self, "_len"):
205              pad_len = (
206                  self._len -
207                  sum(len(raw(pkt)) for pkt in self.getfieldval("data")) -
208                  2 -  # Cycle Counter size (ShortField)
209                  1 -  # DataStatus size (FlagsField over 8 bits)
210                  1  # TransferStatus (ByteField)
211              )
212          else:
213              pad_len = len(self.getfieldval("padding"))
214  
215          # Constraints from IEC-61158-6-10/FDIS ED 3, Table 163
216          assert 0 <= pad_len <= 40
217          q = self
218          while not isinstance(q, UDP) and hasattr(q, "underlayer"):
219              q = q.underlayer
220          if isinstance(q, UDP):
221              assert 0 <= pad_len <= 12
222          return pad_len
223  
224      def next_cls_cb(self, _lst, _p, _remain):
225          if hasattr(self, "_layout") and isinstance(self._layout, list):
226              try:
227                  return self._layout.pop(0)
228              except IndexError:
229                  self._layout = None
230                  return None
231  
232          ether_layer = None
233          q = self
234          while not isinstance(q, Ether) and hasattr(q, "underlayer"):
235              q = q.underlayer
236          if isinstance(q, Ether):
237              ether_layer = q
238  
239          pnio_layer = None
240          q = self
241          while not isinstance(q, ProfinetIO) and hasattr(q, "underlayer"):
242              q = q.underlayer
243          if isinstance(q, ProfinetIO):
244              pnio_layer = q
245  
246          self._layout = [PNIORealTimeCyclicDefaultRawData]
247          if not (ether_layer is None and pnio_layer is None):
248              # Get from config the layout for these hosts and frameid
249              layout = type(self).get_layout_from_config(
250                  ether_layer.src,
251                  ether_layer.dst,
252                  pnio_layer.frameID)
253              if not isinstance(layout, type(None)):
254                  self._layout = layout
255  
256          return self._layout.pop(0)
257  
258      @staticmethod
259      def get_layout_from_config(ether_src, ether_dst, frame_id):
260          try:
261              return copy.deepcopy(
262                  conf.contribs["PNIO_RTC"][(ether_src, ether_dst, frame_id)]
263              )
264          except KeyError:
265              return None
266  
267      @staticmethod
268      def build_fixed_len_raw_type(length):
269          return type(
270              "FixedLenRawPacketLen{}".format(length),
271              (conf.raw_layer,),
272              {
273                  "name": "FixedLenRawPacketLen{}".format(length),
274                  "fields_desc": [StrFixedLenField("data", '', length=length)],
275                  "get_data_length": lambda _: length,
276                  "guess_payload_class": lambda self, p: conf.padding_layer,
277              }
278          )

335  class PROFIsafe(Packet):
336      __slots__ = ["_len"] + Packet.__slots__
337  
338      def guess_payload_class(self, p):
339          return conf.padding_layer
340  
341      def get_data_length(self):
342          """ Must be overridden in a subclass to return the correct value """
343          raise Scapy_Exception(
344              "This method must be overridden in a specific subclass"
345          )
346  
347      def get_mandatory_fields_len(self):
348          # 4 is the len of the control/status byte + the CRC length
349          return 4
350  
351      @staticmethod
352      def get_max_data_length():
353          # Constraints from IEC-61784-3-3 ED 3, Figure 18
354          return 12
355  
356      @staticmethod
357      def build_PROFIsafe_class(cls, data_length):
358          assert cls.get_max_data_length() >= data_length
359          return type(
360              "{}Len{}".format(cls.__name__, data_length),
361              (cls,),
362              {
363                  "get_data_length": lambda _: data_length,
364              }
365          )
