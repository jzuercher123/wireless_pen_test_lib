color.py 
1    """ 
2    Color definitions are  used as per CSS3 specification: 
3    http://www.w3.org/TR/css3-color/#svg-color 
4     
5    A few colors have multiple names referring to the sames colors, eg. `grey` and `gray` or `aqua` and `cyan`. 
6     
7    In these cases the LAST color when sorted alphabetically takes preferences, 
8    eg. Color((0, 255, 255)).as_named() == 'cyan' because "cyan" comes after "aqua". 
9    """
10   import math
11   import re
12   from colorsys import hls_to_rgb, rgb_to_hls
13   from typing import TYPE_CHECKING, Any, Dict, Optional, Tuple, Union, cast
14   
15   from pydantic.v1.errors import ColorError
16   from pydantic.v1.utils import Representation, almost_equal_floats
17   
18   if TYPE_CHECKING:
19       from pydantic.v1.typing import CallableGenerator, ReprArgs
20   
21   ColorTuple = Union[Tuple[int, int, int], Tuple[int, int, int, float]]
22   ColorType = Union[ColorTuple, str]
23   HslColorTuple = Union[Tuple[float, float, float], Tuple[float, float, float, float]]

26   class RGBA:
27       """ 
28       Internal use only as a representation of a color. 
29       """
30   
31       __slots__ = 'r', 'g', 'b', 'alpha', '_tuple'
32   
33       def __init__(self, r: float, g: float, b: float, alpha: Optional[float]):
34           self.r = r
35           self.g = g
36           self.b = b
37           self.alpha = alpha
38   
39           self._tuple: Tuple[float, float, float, Optional[float]] = (r, g, b, alpha)
40   
41       def __getitem__(self, item: Any) -> Any:
42           return self._tuple[item]
43

45   # these are not compiled here to avoid import slowdown, they'll be compiled the first time they're used, then cached
46   r_hex_short = r'\s*(?:#|0x)?([0-9a-f])([0-9a-f])([0-9a-f])([0-9a-f])?\s*'
47   r_hex_long = r'\s*(?:#|0x)?([0-9a-f]{2})([0-9a-f]{2})([0-9a-f]{2})([0-9a-f]{2})?\s*'
48   _r_255 = r'(\d{1,3}(?:\.\d+)?)'
49   _r_comma = r'\s*,\s*'
50   r_rgb = fr'\s*rgb\(\s*{_r_255}{_r_comma}{_r_255}{_r_comma}{_r_255}\)\s*'
51   _r_alpha = r'(\d(?:\.\d+)?|\.\d+|\d{1,2}%)'
52   r_rgba = fr'\s*rgba\(\s*{_r_255}{_r_comma}{_r_255}{_r_comma}{_r_255}{_r_comma}{_r_alpha}\s*\)\s*'
53   _r_h = r'(-?\d+(?:\.\d+)?|-?\.\d+)(deg|rad|turn)?'
54   _r_sl = r'(\d{1,3}(?:\.\d+)?)%'
55   r_hsl = fr'\s*hsl\(\s*{_r_h}{_r_comma}{_r_sl}{_r_comma}{_r_sl}\s*\)\s*'
56   r_hsla = fr'\s*hsl\(\s*{_r_h}{_r_comma}{_r_sl}{_r_comma}{_r_sl}{_r_comma}{_r_alpha}\s*\)\s*'
57   
58   # colors where the two hex characters are the same, if all colors match this the short version of hex colors can be used
59   repeat_colors = {int(c * 2, 16) for c in '0123456789abcdef'}
60   rads = 2 * math.pi
61   
62   
63   class Color(Representation):
64       __slots__ = '_original', '_rgba'
65   
66       def __init__(self, value: ColorType) -> None:
67           self._rgba: RGBA
68           self._original: ColorType
69           if isinstance(value, (tuple, list)):
70               self._rgba = parse_tuple(value)
71           elif isinstance(value, str):
72               self._rgba = parse_str(value)
73           elif isinstance(value, Color):
74               self._rgba = value._rgba
75               value = value._original
76           else:
77               raise ColorError(reason='value must be a tuple, list or string')
78   
79           # if we've got here value must be a valid color
80           self._original = value
81   
82       @classmethod
83       def __modify_schema__(cls, field_schema: Dict[str, Any]) -> None:
84           field_schema.update(type='string', format='color')
85   
86       def original(self) -> ColorType:
87           """ 
88           Original value passed to Color 
89           """
90           return self._original
91   
92       def as_named(self, *, fallback: bool = False) -> str:
93           if self._rgba.alpha is None:
94               rgb = cast(Tuple[int, int, int], self.as_rgb_tuple())
95               try:
96                   return COLORS_BY_VALUE[rgb]
97               except KeyError as e:
98                   if fallback:
99                       return self.as_hex()
100                  else:
101                      raise ValueError('no named color found, use fallback=True, as_hex() or as_rgb()') from e
102          else:
103              return self.as_hex()
104  
105      def as_hex(self) -> str:
106          """ 
107          Hex string representing the color can be 3, 4, 6 or 8 characters depending on whether the string 
108          a "short" representation of the color is possible and whether there's an alpha channel. 
109          """
110          values = [float_to_255(c) for c in self._rgba[:3]]
111          if self._rgba.alpha is not None:
112              values.append(float_to_255(self._rgba.alpha))
113  
114          as_hex = ''.join(f'{v:02x}' for v in values)
115          if all(c in repeat_colors for c in values):
116              as_hex = ''.join(as_hex[c] for c in range(0, len(as_hex), 2))
117          return '#' + as_hex
118  
119      def as_rgb(self) -> str:
120          """ 
121          Color as an rgb(<r>, <g>, <b>) or rgba(<r>, <g>, <b>, <a>) string. 
122          """
123          if self._rgba.alpha is None:
124              return f'rgb({float_to_255(self._rgba.r)}, {float_to_255(self._rgba.g)}, {float_to_255(self._rgba.b)})'
125          else:
126              return (
127                  f'rgba({float_to_255(self._rgba.r)}, {float_to_255(self._rgba.g)}, {float_to_255(self._rgba.b)}, '
128                  f'{round(self._alpha_float(), 2)})'
129              )
130  
131      def as_rgb_tuple(self, *, alpha: Optional[bool] = None) -> ColorTuple:
132          """ 
133          Color as an RGB or RGBA tuple; red, green and blue are in the range 0 to 255, alpha if included is 
134          in the range 0 to 1. 
135   
136          :param alpha: whether to include the alpha channel, options are 
137            None - (default) include alpha only if it's set (e.g. not None) 
138            True - always include alpha, 
139            False - always omit alpha, 
140          """
141          r, g, b = (float_to_255(c) for c in self._rgba[:3])
142          if alpha is None:
143              if self._rgba.alpha is None:
144                  return r, g, b
145              else:
146                  return r, g, b, self._alpha_float()
147          elif alpha:
148              return r, g, b, self._alpha_float()
149          else:
150              # alpha is False
151              return r, g, b
152  
153      def as_hsl(self) -> str:
154          """ 
155          Color as an hsl(<h>, <s>, <l>) or hsl(<h>, <s>, <l>, <a>) string. 
156          """
157          if self._rgba.alpha is None:
158              h, s, li = self.as_hsl_tuple(alpha=False)  # type: ignore
159              return f'hsl({h * 360:0.0f}, {s:0.0%}, {li:0.0%})'
160          else:
161              h, s, li, a = self.as_hsl_tuple(alpha=True)  # type: ignore
162              return f'hsl({h * 360:0.0f}, {s:0.0%}, {li:0.0%}, {round(a, 2)})'
163  
164      def as_hsl_tuple(self, *, alpha: Optional[bool] = None) -> HslColorTuple:
165          """ 
166          Color as an HSL or HSLA tuple, e.g. hue, saturation, lightness and optionally alpha; all elements are in 
167          the range 0 to 1. 
168   
169          NOTE: this is HSL as used in HTML and most other places, not HLS as used in python's colorsys. 
170   
171          :param alpha: whether to include the alpha channel, options are 
172            None - (default) include alpha only if it's set (e.g. not None) 
173            True - always include alpha, 
174            False - always omit alpha, 
175          """
176          h, l, s = rgb_to_hls(self._rgba.r, self._rgba.g, self._rgba.b)
177          if alpha is None:
178              if self._rgba.alpha is None:
179                  return h, s, l
180              else:
181                  return h, s, l, self._alpha_float()
182          if alpha:
183              return h, s, l, self._alpha_float()
184          else:
185              # alpha is False
186              return h, s, l
187  
188      def _alpha_float(self) -> float:
189          return 1 if self._rgba.alpha is None else self._rgba.alpha
190  
191      @classmethod
192      def __get_validators__(cls) -> 'CallableGenerator':
193          yield cls
194  
195      def __str__(self) -> str:
196          return self.as_named(fallback=True)
197  
198      def __repr_args__(self) -> 'ReprArgs':
199          return [(None, self.as_named(fallback=True))] + [('rgb', self.as_rgb_tuple())]  # type: ignore
200  
201      def __eq__(self, other: Any) -> bool:
202          return isinstance(other, Color) and self.as_rgb_tuple() == other.as_rgb_tuple()
203  
204      def __hash__(self) -> int:
205          return hash(self.as_rgb_tuple())

208  def parse_tuple(value: Tuple[Any, ...]) -> RGBA:
209      """ 
210      Parse a tuple or list as a color. 
211      """
212      if len(value) == 3:
213          r, g, b = (parse_color_value(v) for v in value)
214          return RGBA(r, g, b, None)
215      elif len(value) == 4:
216          r, g, b = (parse_color_value(v) for v in value[:3])
217          return RGBA(r, g, b, parse_float_alpha(value[3]))
218      else:
219          raise ColorError(reason='tuples must have length 3 or 4')
220  
221  
222  def parse_str(value: str) -> RGBA:
223      """ 
224      Parse a string to an RGBA tuple, trying the following formats (in this order): 
225      * named color, see COLORS_BY_NAME below 
226      * hex short eg. `<prefix>fff` (prefix can be `#`, `0x` or nothing) 
227      * hex long eg. `<prefix>ffffff` (prefix can be `#`, `0x` or nothing) 
228      * `rgb(<r>, <g>, <b>) ` 
229      * `rgba(<r>, <g>, <b>, <a>)` 
230      """
231      value_lower = value.lower()
232      try:
233          r, g, b = COLORS_BY_NAME[value_lower]
234      except KeyError:
235          pass
236      else:
237          return ints_to_rgba(r, g, b, None)
238  
239      m = re.fullmatch(r_hex_short, value_lower)
240      if m:
241          *rgb, a = m.groups()
242          r, g, b = (int(v * 2, 16) for v in rgb)
243          if a:
244              alpha: Optional[float] = int(a * 2, 16) / 255
245          else:
246              alpha = None
247          return ints_to_rgba(r, g, b, alpha)
248  
249      m = re.fullmatch(r_hex_long, value_lower)
250      if m:
251          *rgb, a = m.groups()
252          r, g, b = (int(v, 16) for v in rgb)
253          if a:
254              alpha = int(a, 16) / 255
255          else:
256              alpha = None
257          return ints_to_rgba(r, g, b, alpha)
258  
259      m = re.fullmatch(r_rgb, value_lower)
260      if m:
261          return ints_to_rgba(*m.groups(), None)  # type: ignore
262  
263      m = re.fullmatch(r_rgba, value_lower)
264      if m:
265          return ints_to_rgba(*m.groups())  # type: ignore
266  
267      m = re.fullmatch(r_hsl, value_lower)
268      if m:
269          h, h_units, s, l_ = m.groups()
270          return parse_hsl(h, h_units, s, l_)
271  
272      m = re.fullmatch(r_hsla, value_lower)
273      if m:
274          h, h_units, s, l_, a = m.groups()
275          return parse_hsl(h, h_units, s, l_, parse_float_alpha(a))
276  
277      raise ColorError(reason='string not recognised as a valid color')
278

280  def ints_to_rgba(r: Union[int, str], g: Union[int, str], b: Union[int, str], alpha: Optional[float]) -> RGBA:
281      return RGBA(parse_color_value(r), parse_color_value(g), parse_color_value(b), parse_float_alpha(alpha))
282

284  def parse_color_value(value: Union[int, str], max_val: int = 255) -> float:
285      """ 
286      Parse a value checking it's a valid int in the range 0 to max_val and divide by max_val to give a number 
287      in the range 0 to 1 
288      """
289      try:
290          color = float(value)
291      except ValueError:
292          raise ColorError(reason='color values must be a valid number')
293      if 0 <= color <= max_val:
294          return color / max_val
295      else:
296          raise ColorError(reason=f'color values must be in the range 0 to {max_val}')
297  
298  
299  def parse_float_alpha(value: Union[None, str, float, int]) -> Optional[float]:
300      """ 
301      Parse a value checking it's a valid float in the range 0 to 1 
302      """
303      if value is None:
304          return None
305      try:
306          if isinstance(value, str) and value.endswith('%'):
307              alpha = float(value[:-1]) / 100
308          else:
309              alpha = float(value)
310      except ValueError:
311          raise ColorError(reason='alpha values must be a valid float')
312  
313      if almost_equal_floats(alpha, 1):
314          return None
315      elif 0 <= alpha <= 1:
316          return alpha
317      else:
318          raise ColorError(reason='alpha values must be in the range 0 to 1')
319

321  def parse_hsl(h: str, h_units: str, sat: str, light: str, alpha: Optional[float] = None) -> RGBA:
322      """ 
323      Parse raw hue, saturation, lightness and alpha values and convert to RGBA. 
324      """
325      s_value, l_value = parse_color_value(sat, 100), parse_color_value(light, 100)
326  
327      h_value = float(h)
328      if h_units in {None, 'deg'}:
329          h_value = h_value % 360 / 360
330      elif h_units == 'rad':
331          h_value = h_value % rads / rads
332      else:
333          # turns
334          h_value = h_value % 1
335  
336      r, g, b = hls_to_rgb(h_value, l_value, s_value)
337      return RGBA(r, g, b, alpha)

340  def float_to_255(c: float) -> int:
341      return int(round(c * 255))
