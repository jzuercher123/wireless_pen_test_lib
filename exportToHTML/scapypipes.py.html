28   class SniffSource(Source):
29       """Read packets from an interface and send them to low exit. 
30    
31       .. code:: 
32    
33                +-----------+ 
34             >>-|           |->> 
35                |           | 
36              >-|  [iface]--|-> 
37                +-----------+ 
38    
39       If neither of the ``iface`` or ``socket`` parameters are specified, then 
40       Scapy will capture from the first network interface. 
41    
42       :param iface: A layer 2 interface to sniff packets from. Mutually 
43                     exclusive with the ``socket`` parameter. 
44       :param filter: Packet filter to use while capturing. See ``L2listen``. 
45                      Not used with ``socket`` parameter. 
46       :param socket: A ``SuperSocket`` to sniff packets from. 
47       """
48   
49       def __init__(self,
50                    iface=None,  # type: Optional[str]
51                    filter=None,  # type: Optional[Any]
52                    socket=None,  # type: Optional[SuperSocket]
53                    name=None,  # type: Optional[Any]
54                    ):
55           # type: (...) -> None
56           Source.__init__(self, name=name)
57   
58           if (iface or filter) and socket:
59               raise ValueError("iface and filter options are mutually exclusive "
60                                "with socket")
61   
62           self.s = cast(SuperSocket, socket)
63           self.iface = iface
64           self.filter = filter
65   
66       def start(self):
67           # type: () -> None
68           if not self.s:
69               self.s = conf.L2listen(iface=self.iface, filter=self.filter)
70   
71       def stop(self):
72           # type: () -> None
73           if self.s:
74               self.s.close()
75   
76       def fileno(self):
77           # type: () -> int
78           return self.s.fileno()
79   
80       def deliver(self):
81           # type: () -> None
82           try:
83               pkt = self.s.recv()
84               if pkt is not None:
85                   self._send(pkt)
86           except EOFError:
87               self.is_exhausted = True

168  class WrpcapSink(Sink):
169      """ 
170      Writes :py:class:`Packet` on the low entry to a ``pcap`` file. 
171      Ignores all messages on the high entry. 
172   
173      .. note:: 
174   
175          Due to limitations of the ``pcap`` format, all packets **must** be of 
176          the same link type. This class will not mutate packets to conform with 
177          the expected link type. 
178   
179      .. code:: 
180   
181           +----------+ 
182        >>-|          |->> 
183           |          | 
184         >-|--[pcap]  |-> 
185           +----------+ 
186   
187      :param fname: Filename to write packets to. 
188      :type fname: str 
189      :param linktype: See :py:attr:`linktype`. 
190      :type linktype: None or int 
191   
192      .. py:attribute:: linktype 
193   
194          Set an explicit link-type (``DLT_``) for packets.  This must be an 
195          ``int`` or ``None``. 
196   
197          This is the same as the :py:func:`wrpcap` ``linktype`` parameter. 
198   
199          If ``None`` (the default), the linktype will be auto-detected on the 
200          first packet. This field will *not* be updated with the result of this 
201          auto-detection. 
202   
203          This attribute has no effect after calling :py:meth:`PipeEngine.start`. 
204      """
205  
206      def __init__(self, fname, name=None, linktype=None, **kwargs):
207          # type: (str, Optional[str], Optional[int], **Any) -> None
208          Sink.__init__(self, name=name)
209          self.fname = fname
210          self.f = None  # type: Optional[PcapWriter]
211          self.linktype = linktype
212          self.kwargs = kwargs
213  
214      def start(self):
215          # type: () -> None
216          self.f = PcapWriter(self.fname, linktype=self.linktype, **self.kwargs)
217  
218      def stop(self):
219          # type: () -> None
220          if self.f:
221              self.f.flush()
222              self.f.close()
223  
224      def push(self, msg):
225          # type: (Packet) -> None
226          if msg and self.f:
227              self.f.write(msg)

230  class WiresharkSink(WrpcapSink):
231      """ 
232      Streams :py:class:`Packet` from the low entry to Wireshark. 
233   
234      Packets are written into a ``pcap`` stream (like :py:class:`WrpcapSink`), 
235      and streamed to a new Wireshark process on its ``stdin``. 
236   
237      Wireshark is run with the ``-ki -`` arguments, which cause it to treat 
238      ``stdin`` as a capture device.  Arguments in :py:attr:`args` will be 
239      appended after this. 
240   
241      Extends :py:mod:`WrpcapSink`. 
242   
243      .. code:: 
244   
245           +----------+ 
246        >>-|          |->> 
247           |          | 
248         >-|--[pcap]  |-> 
249           +----------+ 
250   
251      :param linktype: See :py:attr:`WrpcapSink.linktype`. 
252      :type linktype: None or int 
253      :param args: See :py:attr:`args`. 
254      :type args: None or list[str] 
255   
256      .. py:attribute:: args 
257   
258          Additional arguments for the Wireshark process. 
259   
260          This must be either ``None`` (the default), or a ``list`` of ``str``. 
261   
262          This attribute has no effect after calling :py:meth:`PipeEngine.start`. 
263   
264          See :manpage:`wireshark(1)` for more details. 
265      """
266  
267      def __init__(self, name=None, linktype=None, args=None):
268          # type: (Optional[Any], Optional[int], Optional[List[str]]) -> None
269          WrpcapSink.__init__(self, fname="", name=name, linktype=linktype)
270          self.args = args
271  
272      def start(self):
273          # type: () -> None
274          # Wireshark must be running first, because PcapWriter will block until
275          # data has been read!
276          with ContextManagerSubprocess(conf.prog.wireshark):
277              args = [conf.prog.wireshark, "-Slki", "-"]
278              if self.args:
279                  args.extend(self.args)
280  
281              proc = subprocess.Popen(
282                  args,
283                  stdin=subprocess.PIPE,
284                  stdout=None,
285                  stderr=None,
286              )
287  
288          self.fname = proc.stdin  # type: ignore
289          WrpcapSink.start(self)

292  class UDPDrain(Drain):
293      """UDP payloads received on high entry are sent over UDP 
294   
295      .. code:: 
296   
297           +-------------+ 
298        >>-|--[payload]--|->> 
299           |      X      | 
300         >-|----[UDP]----|-> 
301           +-------------+ 
302      """
303  
304      def __init__(self, ip="127.0.0.1", port=1234):
305          # type: (str, int) -> None
306          Drain.__init__(self)
307          self.ip = ip
308          self.port = port
309  
310      def push(self, msg):
311          # type: (Packet) -> None
312          from scapy.layers.inet import IP, UDP
313          if IP in msg and msg[IP].proto == 17 and UDP in msg:
314              payload = msg[UDP].payload
315              self._high_send(raw(payload))
316  
317      def high_push(self, msg):
318          # type: (Packet) -> None
319          from scapy.layers.inet import IP, UDP
320          p = IP(dst=self.ip) / UDP(sport=1234, dport=self.port) / msg
321          self._send(p)

355      """TCP connect to addr:port and use it as source and sink

359           +-------------+

380          # type: () -> None

382              self.fd.close()
383

385          # type: (bytes) -> None

387  
388      def fileno(self):
389          # type: () -> int

393          # type: () -> None

395              msg = self.fd.recv(65536)
396          except socket.error:

398              raise
399          if msg:
400              self._send(msg)

403  class TCPListenPipe(TCPConnectPipe):
404      """TCP listen on [addr:]port and use first connection as source and sink; 
405      send peer address to high output 
406   
407      .. code:: 
408   
409           +------^------+ 
410        >>-|    +-[peer]-|->> 
411           |   /         | 
412         >-|-[addr:port]-|-> 
413           +-------------+ 
414      """
415      __selectable_force_select__ = True
416  
417      def __init__(self, addr="", port=0, name=None):
418          # type: (str, int, Optional[str]) -> None
419          TCPConnectPipe.__init__(self, addr, port, name)
420          self.connected = False
421          self.q: Queue[Any] = Queue()
422  
423      def start(self):
424          # type: () -> None
425          self.connected = False
426          self.fd = socket.socket()
427          self.fd.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
428          self.fd.bind((self.addr, self.port))
429          self.fd.listen(1)
430  
431      def push(self, msg):
432          # type: (bytes) -> None
433          if self.connected:
434              self.fd.send(msg)
435          else:
436              self.q.put(msg)
437  
438      def deliver(self):
439          # type: () -> None
440          if self.connected:
441              try:
442                  msg = self.fd.recv(65536)
443              except socket.error:
444                  self.stop()
445                  raise
446              if msg:
447                  self._send(msg)
448          else:
449              fd, frm = self.fd.accept()
450              self._high_send(frm)
451              self.fd.close()
452              self.fd = fd
453              self.connected = True
454              self._trigger(frm)
455              while True:
456                  try:
457                      self.fd.send(self.q.get(block=False))
458                  except Empty:
459                      break

503  class UDPServerPipe(TCPListenPipe):
504      """UDP bind to [addr:]port and use as source and sink 
505      Use (ip, port) from first received IP packet as destination for all data 
506   
507      .. code:: 
508   
509           +------^------+ 
510        >>-|    +-[peer]-|->> 
511           |   /         | 
512         >-|-[addr:port]-|-> 
513           +-------------+ 
514      """
515  
516      def __init__(self, addr="", port=0, name=None):
517          # type: (str, int, Optional[str]) -> None
518          TCPListenPipe.__init__(self, addr, port, name)
519          self._destination = None  # type: Any
520  
521      def start(self):
522          # type: () -> None
523          self.fd = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
524          self.fd.bind((self.addr, self.port))
525  
526      def push(self, msg):
527          # type: (bytes) -> None
528          if self._destination:
529              self.fd.sendto(msg, self._destination)
530          else:
531              self.q.put(msg)
532  
533      def deliver(self):
534          # type: () -> None
535          if self._destination:
536              try:
537                  msg = self.fd.recv(65536)
538              except socket.error:
539                  self.stop()
540                  raise
541              if msg:
542                  self._send(msg)
543          else:
544              msg, dest = self.fd.recvfrom(65536)
545              if msg:
546                  self._send(msg)
547              self._destination = dest
548              self._trigger(dest)
549              self._high_send(dest)
550              while True:
551                  try:
552                      msg = self.q.get(block=False)
553                      self.fd.sendto(msg, self._destination)
554                  except Empty:
555                      break

647  class TriggeredQueueingValve(Drain):
648      """Let messages alternatively pass or queued, changing on trigger 
649   
650      .. code:: 
651   
652           +------^-------+ 
653        >>-|-[pass/queue]-|->> 
654           |      |       | 
655         >-|-[pass/queue]-|-> 
656           +------^-------+ 
657      """
658  
659      def __init__(self, start_state=True, name=None):
660          # type: (bool, Optional[Any]) -> None
661          Drain.__init__(self, name=name)
662          self.opened = start_state
663          self.q: Queue[Any] = Queue()
664  
665      def start(self):
666          # type: () -> None
667          self.q = Queue()
668  
669      def push(self, msg):
670          # type: (str) -> None
671          if self.opened:
672              self._send(msg)
673          else:
674              self.q.put((True, msg))
675  
676      def high_push(self, msg):
677          # type: (str) -> None
678          if self.opened:
679              self._send(msg)
680          else:
681              self.q.put((False, msg))
682  
683      def on_trigger(self, msg):
684          # type: (bool) -> None
685          self.opened ^= True
686          self._trigger(msg)
687          while True:
688              try:
689                  low, msg = self.q.get(block=False)
690              except Empty:
691                  break
692              else:
693                  if low:
694                      self._send(msg)
695                  else:
696                      self._high_send(msg)

700      r"""Let messages alternatively high or low, changing on trigger 
701   
702      .. code:: 
703   
704           +------^------+ 
705        >>-|-\    |    /-|->> 
706           |  [up/down]  |

725          # type: (bool) -> None
726          self.low ^= True
