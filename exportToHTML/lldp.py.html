154      def guess_payload_class(self, payload):
155          # type is a 7-bit bitfield spanning bits 1..7 -> div 2
156          try:
157              lldpdu_tlv_type = orb(payload[0]) // 2
158              class_type = LLDPDU_CLASS_TYPES.get(lldpdu_tlv_type, conf.raw_layer)
159              if isinstance(class_type, list):
160                  for cls in class_type:
161                      if cls._match_organization_specific(payload):
162                          return cls
163              else:
164                  return class_type
165          except IndexError:
166              return conf.raw_layer

168      @staticmethod
169      def _dot1q_headers_size(layer):
170          """ 
171          calculate size of lower dot1q layers (if present) 
172          :param layer: the layer to start at 
173          :return: size of vlan headers, layer below lowest vlan header 
174          """
175  
176          vlan_headers_size = 0
177          under_layer = layer
178  
179          while under_layer and isinstance(under_layer, Dot1Q):
180              vlan_headers_size += LLDPDU.DOT1Q_HEADER_LEN
181              under_layer = under_layer.underlayer
182  
183          return vlan_headers_size, under_layer
184  
185      def post_build(self, pkt, pay):
186  
187          under_layer = self.underlayer
188  
189          if under_layer is None:
190              if conf.contribs['LLDP'].strict_mode():
191                  raise LLDPMissingLowerLayer('No lower layer (Ethernet '
192                                              'or Dot1Q) provided.')
193              else:
194                  return pkt + pay
195  
196          if isinstance(under_layer, LLDPDU):
197              return pkt + pay
198  
199          frame_size, under_layer = LLDPDU._dot1q_headers_size(under_layer)
200  
201          if not under_layer or not isinstance(under_layer, Ether):
202              if conf.contribs['LLDP'].strict_mode():
203                  raise LLDPMissingLowerLayer('No Ethernet layer provided.')
204              else:
205                  return pkt + pay
206  
207          frame_size += LLDPDU.ETHER_HEADER_LEN
208          frame_size += len(pkt) + len(pay) + LLDPDU.ETHER_FSC_LEN
209          if frame_size < LLDPDU.ETHER_FRAME_MIN_LEN:
210              return pkt + pay + b'\x00' * (LLDPDU.ETHER_FRAME_MIN_LEN - frame_size)  # noqa: E501
211          return pkt + pay

213      @staticmethod
214      def _frame_structure_check(structure_description):
215          """ 
216          check if the structure of the frame is conform to the basic 
217          frame structure defined by the standard 
218          :param structure_description: string-list reflecting LLDP-msg structure 
219          """
220  
221          standard_frame_structure = [LLDPDUChassisID.__name__,
222                                      LLDPDUPortID.__name__,
223                                      LLDPDUTimeToLive.__name__,
224                                      '<...>']
225  
226          if len(structure_description) < 3:
227              raise LLDPInvalidFrameStructure(
228                  'Invalid frame structure.\ngot: {}\nexpected: '
229                  '{}'.format(' '.join(structure_description),
230                              ' '.join(standard_frame_structure)))
231  
232          for idx, layer_name in enumerate(standard_frame_structure):
233  
234              if layer_name == '<...>':
235                  break
236              if layer_name != structure_description[idx]:
237                  raise LLDPInvalidFrameStructure(
238                      'Invalid frame structure.\ngot: {}\nexpected: '
239                      '{}'.format(' '.join(structure_description),
240                                  ' '.join(standard_frame_structure)))
241  
242      @staticmethod
243      def _tlv_multiplicities_check(tlv_type_count):
244          """ 
245          check if multiplicity of present TLVs conforms to the standard 
246          :param tlv_type_count: dict containing counte-per-TLV 
247          """
248  
249          # * : 0..n, 1 : one and only one.
250          standard_multiplicities = {
251              LLDPDUEndOfLLDPDU.__name__: '*',
252              LLDPDUChassisID.__name__: 1,
253              LLDPDUPortID.__name__: 1,
254              LLDPDUTimeToLive.__name__: 1,
255              LLDPDUPortDescription: '*',
256              LLDPDUSystemName: '*',
257              LLDPDUSystemDescription: '*',
258              LLDPDUSystemCapabilities: '*',
259              LLDPDUManagementAddress: '*'
260          }
261  
262          for tlv_type_name in standard_multiplicities:
263  
264              standard_tlv_multiplicity = \
265                  standard_multiplicities[tlv_type_name]
266              if standard_tlv_multiplicity == '*':
267                  continue
268  
269              try:
270                  if tlv_type_count[tlv_type_name] != standard_tlv_multiplicity:
271                      raise LLDPInvalidTLVCount(
272                          'Invalid number of entries for TLV type '
273                          '{} - expected {} entries, got '
274                          '{}'.format(tlv_type_name,
275                                      standard_tlv_multiplicity,
276                                      tlv_type_count[tlv_type_name]))
277  
278              except KeyError:
279                  raise LLDPInvalidTLVCount('Missing TLV layer of type '
280                                            '{}.'.format(tlv_type_name))

282      def pre_dissect(self, s):
283  
284          if conf.contribs['LLDP'].strict_mode():
285              if self.__class__.__name__ == 'LLDPDU':
286                  LLDPDU.LAYER_STACK = []
287                  LLDPDU.LAYER_MULTIPLICITIES = {}
288              else:
289                  LLDPDU.LAYER_STACK.append(self.__class__.__name__)
290                  try:
291                      LLDPDU.LAYER_MULTIPLICITIES[self.__class__.__name__] += 1
292                  except KeyError:
293                      LLDPDU.LAYER_MULTIPLICITIES[self.__class__.__name__] = 1
294  
295          return s
296  
297      def dissection_done(self, pkt):
298  
299          if self.__class__.__name__ == 'LLDPDU' and \
300                  conf.contribs['LLDP'].strict_mode():
301              LLDPDU._frame_structure_check(LLDPDU.LAYER_STACK)
302              LLDPDU._tlv_multiplicities_check(LLDPDU.LAYER_MULTIPLICITIES)
303  
304          super(LLDPDU, self).dissection_done(pkt)
305  
306      def _check(self):
307          """Overwritten by LLDPU objects"""
308          pass
309  
310      def post_dissect(self, s):
311          self._check()
312          return super(LLDPDU, self).post_dissect(s)
313  
314      def do_build(self):
315          self._check()
316          return super(LLDPDU, self).do_build()

319  def _ldp_id_adjustlen(pkt, x):
320      """Return the length of the `id` field, 
321      according to its real encoded type"""
322      f, v = pkt.getfield_and_val('id')
323      length = f.i2len(pkt, v) + 1
324      if (isinstance(pkt, LLDPDUPortID) and pkt.subtype == 0x4) or \
325              (isinstance(pkt, LLDPDUChassisID) and pkt.subtype == 0x5):
326          # Take the ConditionalField into account
327          length += 1
328      return length
329  
330  
331  def _ldp_id_lengthfrom(pkt):
332      length = pkt._length
333      if length is None:
334          return 0
335      # Subtract the subtype field
336      length -= 1
337      if (isinstance(pkt, LLDPDUPortID) and pkt.subtype == 0x4) or \
338              (isinstance(pkt, LLDPDUChassisID) and pkt.subtype == 0x5):
339          # Take the ConditionalField into account
340          length -= 1
341      return length

345      """ 
346          ieee 802.1ab-2016 - sec. 8.5.2 / p. 26 
347      """
348      LLDP_CHASSIS_ID_TLV_SUBTYPES = {
349          0x00: 'reserved',
350          0x01: 'chassis component',
351          0x02: 'interface alias',
352          0x03: 'port component',
353          0x04: 'MAC address',
354          0x05: 'network address',
355          0x06: 'interface name',
356          0x07: 'locally assigned',
357      }
358  
359      SUBTYPE_RESERVED = 0x00
360      SUBTYPE_CHASSIS_COMPONENT = 0x01
361      SUBTYPE_INTERFACE_ALIAS = 0x02
362      SUBTYPE_PORT_COMPONENT = 0x03
363      SUBTYPE_MAC_ADDRESS = 0x04
364      SUBTYPE_NETWORK_ADDRESS = 0x05
365      SUBTYPE_INTERFACE_NAME = 0x06
366      SUBTYPE_LOCALLY_ASSIGNED = 0x07
367  
368      fields_desc = [
369          BitEnumField('_type', 0x01, 7, LLDPDU.TYPES),
370          BitFieldLenField('_length', None, 9, length_of='id',
371                           adjust=lambda pkt, x: _ldp_id_adjustlen(pkt, x)),
372          ByteEnumField('subtype', 0x00, LLDP_CHASSIS_ID_TLV_SUBTYPES),
373          ConditionalField(
374              ByteEnumField('family', 0, LLDPDU.IANA_ADDRESS_FAMILY_NUMBERS),
375              lambda pkt: pkt.subtype == 0x05
376          ),
377          MultipleTypeField([
378              (
379                  MACField('id', None),
380                  lambda pkt: pkt.subtype == 0x04
381              ),
382              (
383                  IPField('id', None),
384                  lambda pkt: pkt.subtype == 0x05 and pkt.family == 0x01
385              ),
386              (
387                  IP6Field('id', None),
388                  lambda pkt: pkt.subtype == 0x05 and pkt.family == 0x02
389              ),
390          ], StrLenField('id', '', length_from=_ldp_id_lengthfrom)
391          )
392      ]
393  
394      def _check(self):
395          """ 
396          run layer specific checks 
397          """
398          if conf.contribs['LLDP'].strict_mode() and not self.id:
399              raise LLDPInvalidLengthField('id must be >= 1 characters long')

1168     def post_dissect(self, s):
1169         s = super(LLDPDUPowerViaMDIMeasure, self).post_dissect(s)
1170         self.power_price_index = 0xffff if self.power_price_index == 0xffff \

1174     @staticmethod
1175     def _match_organization_specific(payload):
1176         ''' 
1177         match organization specific TLV 
1178         '''
1179         return (orb(payload[5]) == 8 and orb(payload[1]) == 26
1180                 and bytes_int(payload[2:5]) ==
1181                 LLDPDUGenericOrganisationSpecific.ORG_UNIQUE_CODE_IEEE_802_3)

1183     def _check(self):
1184         """ 
1185         run layer specific checks 
1186         """
1187         if conf.contribs['LLDP'].strict_mode() and self._length != 26:
1188             raise LLDPInvalidLengthField('length must be 26 - got '
1189                                          '{}'.format(self._length))
1190         # IEEE802.3bt-2018 - sec. 79.3.8
1191         for field, description, max_value in [('voltage_uncertainty',
1192                                                'voltage uncertainty',
1193                                                65),
1194                                               ('voltage_measurement',
1195                                                'voltage measurement',
1196                                                65),
1197                                               ('current_uncertainty',
1198                                                'current uncertainty',
1199                                                6.5),
1200                                               ('current_measurement',
1201                                                'current measurement',
1202                                                6.5),
1203                                               ('energy_uncertainty',
1204                                                'energy uncertainty',
1205                                                6500000),
1206                                               ('power_uncertainty',
1207                                                'power uncertainty',
1208                                                650),
1209                                               ('power_measurement',
1210                                                'power measurement',
1211                                                650)]:
1212             val = getattr(self, field) or 0
1213             if (conf.contribs['LLDP'].strict_mode() and val > max_value):
1214                 raise LLDPInvalidFieldValue(
1215                     'exceeded maximum {} of {} - got '
1216                     '{}'.format(description, max_value, val))
1217             val = self.power_price_index or 0xffff
1218             if val > 65000 and val != 0xffff:
1219                 raise LLDPInvalidFieldValue(
1220                     'exceeded maximum power price index of {} - got '
1221                     '{}'.format(LLDPDUPowerViaMDIMeasure._decode_ppi(65000),
1222                                 LLDPDUPowerViaMDIMeasure._decode_ppi(val)))
