99   def all_literal_values(type_: type[Any]) -> list[Any]:
100      """This method is used to retrieve all Literal values as 
101      Literal can be used recursively (see https://www.python.org/dev/peps/pep-0586) 
102      e.g. `Literal[Literal[Literal[1, 2, 3], "foo"], 5, None]`. 
103      """
104      if not is_literal_type(type_):
105          return [type_]
106  
107      values = literal_values(type_)
108      return [x for value in values for x in all_literal_values(value)]

147      if _check_classvar(ann_type) or _check_classvar(get_origin(ann_type)):
148          return True

151      # forward references, see #3679

153          return True

170  def parent_frame_namespace(*, parent_depth: int = 2, force: bool = False) -> dict[str, Any] | None:
171      """We allow use of items in parent namespace to get around the issue with `get_type_hints` only looking in the 
172      global module namespace. See https://github.com/pydantic/pydantic/issues/2678#issuecomment-1008139014 -> Scope 
173      and suggestion at the end of the next comment by @gvanrossum. 
174   
175      WARNING 1: it matters exactly where this is called. By default, this function will build a namespace from the 
176      parent of where it is called. 
177   
178      WARNING 2: this only looks in the parent namespace, not other parents since (AFAIK) there's no way to collect a 
179      dict of exactly what's in scope. Using `f_back` would work sometimes but would be very wrong and confusing in many 
180      other cases. See https://discuss.python.org/t/is-there-a-way-to-access-parent-nested-namespaces/20659. 
181   
182      There are some cases where we want to force fetching the parent namespace, ex: during a `model_rebuild` call. 
183      In this case, we want both the namespace of the class' module, if applicable, and the parent namespace of the 
184      module where the rebuild is called. 
185   
186      In other cases, like during initial schema build, if a class is defined at the top module level, we don't need to 
187      fetch that module's namespace, because the class' __module__ attribute can be used to access the parent namespace. 
188      This is done in `_typing_extra.get_module_ns_of`. Thus, there's no need to cache the parent frame namespace in this case. 
189      """
190      frame = sys._getframe(parent_depth)
191  
192      # note, we don't copy frame.f_locals here (or during the last return call), because we don't expect the namespace to be modified down the line
193      # if this becomes a problem, we could implement some sort of frozen mapping structure to enforce this
194      if force:
195          return frame.f_locals
196  
197      # if either of the following conditions are true, the class is defined at the top module level
198      # to better understand why we need both of these checks, see
199      # https://github.com/pydantic/pydantic/pull/10113#discussion_r1714981531
200      if frame.f_back is None or frame.f_code.co_name == '<module>':
201          return None
202  
203      return frame.f_locals

206  def get_module_ns_of(obj: Any) -> dict[str, Any]:
207      """Get the namespace of the module where the object is defined. 
208   
209      Caution: this function does not return a copy of the module namespace, so it should not be mutated. 
210      The burden of enforcing this is on the caller. 
211      """
212      module_name = getattr(obj, '__module__', None)
213      if module_name:
214          try:
215              return sys.modules[module_name].__dict__
216          except KeyError:
217              # happens occasionally, see https://github.com/pydantic/pydantic/issues/2363
218              return {}
219      return {}

222  def merge_cls_and_parent_ns(cls: type[Any], parent_namespace: dict[str, Any] | None = None) -> dict[str, Any]:
223      ns = get_module_ns_of(cls).copy()
224      if parent_namespace is not None:
225          ns.update(parent_namespace)
226      ns[cls.__name__] = cls
227      return ns

230  def get_cls_type_hints_lenient(
231      obj: Any, globalns: dict[str, Any] | None = None, mro: Iterable[type[Any]] | None = None
232  ) -> dict[str, Any]:
233      """Collect annotations from a class, including those from parent classes. 
234   
235      Unlike `typing.get_type_hints`, this function will not error if a forward reference is not resolvable. 
236      """
237      hints = {}
238      if mro is None:
239          mro = reversed(obj.__mro__)
240      for base in mro:
241          ann = base.__dict__.get('__annotations__')
242          localns = dict(vars(base))
243          if ann is not None and ann is not GetSetDescriptorType:
244              for name, value in ann.items():
245                  hints[name] = eval_type_lenient(value, globalns, localns)
246      return hints

249  def eval_type_lenient(value: Any, globalns: dict[str, Any] | None = None, localns: dict[str, Any] | None = None) -> Any:
250      """Behaves like typing._eval_type, except it won't raise an error if a forward reference can't be resolved."""
251      if value is None:
252          value = NoneType
253      elif isinstance(value, str):
254          value = _make_forward_ref(value, is_argument=False, is_class=True)
255  
256      try:
257          return eval_type_backport(value, globalns, localns)
258      except NameError:
259          # the point of this function is to be tolerant to this case
260          return value

263  def eval_type_backport(
264      value: Any,
265      globalns: dict[str, Any] | None = None,
266      localns: dict[str, Any] | None = None,
267      type_params: tuple[Any] | None = None,
268  ) -> Any:
269      """An enhanced version of `typing._eval_type` which will fall back to using the `eval_type_backport` 
270      package if it's installed to let older Python versions use newer typing constructs. 
271   
272      Specifically, this transforms `X | Y` into `typing.Union[X, Y]` and `list[X]` into `typing.List[X]` 
273      (as well as all the types made generic in PEP 585) if the original syntax is not supported in the 
274      current Python version. 
275   
276      This function will also display a helpful error if the value passed fails to evaluate. 
277      """
278      try:
279          return _eval_type_backport(value, globalns, localns, type_params)
280      except TypeError as e:
281          if 'Unable to evaluate type annotation' in str(e):
282              raise
283  
284          # If it is a `TypeError` and value isn't a `ForwardRef`, it would have failed during annotation definition.
285          # Thus we assert here for type checking purposes:
286          assert isinstance(value, typing.ForwardRef)
287  
288          message = f'Unable to evaluate type annotation {value.__forward_arg__!r}.'
289          if sys.version_info >= (3, 11):
290              e.add_note(message)
291              raise
292          else:
293              raise TypeError(message) from e
294

296  def _eval_type_backport(
297      value: Any,
298      globalns: dict[str, Any] | None = None,
299      localns: dict[str, Any] | None = None,
300      type_params: tuple[Any] | None = None,
301  ) -> Any:
302      try:
303          return _eval_type(value, globalns, localns, type_params)
304      except TypeError as e:
305          if not (isinstance(value, typing.ForwardRef) and is_backport_fixable_error(e)):
306              raise
307  
308          try:
309              from eval_type_backport import eval_type_backport
310          except ImportError:
311              raise TypeError(
312                  f'Unable to evaluate type annotation {value.__forward_arg__!r}. If you are making use '
313                  'of the new typing syntax (unions using `|` since Python 3.10 or builtins subscripting '
314                  'since Python 3.9), you should either replace the use of new syntax with the existing '
315                  '`typing` constructs or install the `eval_type_backport` package.'
316              ) from e
317  
318          return eval_type_backport(value, globalns, localns, try_default=False)

348  def get_function_type_hints(
349      function: Callable[..., Any], *, include_keys: set[str] | None = None, types_namespace: dict[str, Any] | None = None
350  ) -> dict[str, Any]:
351      """Like `typing.get_type_hints`, but doesn't convert `X` to `Optional[X]` if the default value is `None`, also 
352      copes with `partial`. 
353      """
354      try:
355          if isinstance(function, partial):
356              annotations = function.func.__annotations__
357          else:
358              annotations = function.__annotations__
359      except AttributeError:
360          type_hints = get_type_hints(function)
361          if isinstance(function, type):
362              # `type[...]` is a callable, which returns an instance of itself.
363              # At some point, we might even look into the return type of `__new__`
364              # if it returns something else.
365              type_hints.setdefault('return', function)
366          return type_hints
367  
368      globalns = get_module_ns_of(function)
369      type_hints = {}
370      type_params: tuple[Any] = getattr(function, '__type_params__', ())  # type: ignore
371      for name, value in annotations.items():
372          if include_keys is not None and name not in include_keys:
373              continue
374          if value is None:
375              value = NoneType
376          elif isinstance(value, str):
377              value = _make_forward_ref(value)
378  
379          type_hints[name] = eval_type_backport(value, globalns, types_namespace, type_params)
380  
381      return type_hints

386      def _make_forward_ref(
387          arg: Any,
388          is_argument: bool = True,
389          *,
390          is_class: bool = False,
391      ) -> typing.ForwardRef:
392          """Wrapper for ForwardRef that accounts for the `is_class` argument missing in older versions. 
393          The `module` argument is omitted as it breaks <3.9.8, =3.10.0 and isn't used in the calls below. 
394   
395          See https://github.com/python/cpython/pull/28560 for some background. 
396          The backport happened on 3.9.8, see: 
397          https://github.com/pydantic/pydantic/discussions/6244#discussioncomment-6275458, 
398          and on 3.10.1 for the 3.10 branch, see: 
399          https://github.com/pydantic/pydantic/issues/6912 
400   
401          Implemented as EAFP with memory. 
402          """
403          return typing.ForwardRef(arg, is_argument)
404

418      @typing.no_type_check
419      def get_type_hints(  # noqa: C901
420          obj: Any,
421          globalns: dict[str, Any] | None = None,
422          localns: dict[str, Any] | None = None,
423          include_extras: bool = False,
424      ) -> dict[str, Any]:  # pragma: no cover
425          """Taken verbatim from python 3.10.8 unchanged, except: 
426          * type annotations of the function definition above. 
427          * prefixing `typing.` where appropriate 
428          * Use `_make_forward_ref` instead of `typing.ForwardRef` to handle the `is_class` argument. 
429   
430          https://github.com/python/cpython/blob/aaaf5174241496afca7ce4d4584570190ff972fe/Lib/typing.py#L1773-L1875 
431   
432          DO NOT CHANGE THIS METHOD UNLESS ABSOLUTELY NECESSARY. 
433          ====================================================== 
434   
435          Return type hints for an object. 
436   
437          This is often the same as obj.__annotations__, but it handles 
438          forward references encoded as string literals, adds Optional[t] if a 
439          default value equal to None is set and recursively replaces all 
440          'Annotated[T, ...]' with 'T' (unless 'include_extras=True'). 
441   
442          The argument may be a module, class, method, or function. The annotations 
443          are returned as a dictionary. For classes, annotations include also 
444          inherited members. 
445   
446          TypeError is raised if the argument is not of a type that can contain 
447          annotations, and an empty dictionary is returned if no annotations are 
448          present. 
449   
450          BEWARE -- the behavior of globalns and localns is counterintuitive 
451          (unless you are familiar with how eval() and exec() work).  The 
452          search order is locals first, then globals. 
453   
454          - If no dict arguments are passed, an attempt is made to use the 
455            globals from obj (or the respective module's globals for classes), 
456            and these are also used as the locals.  If the object does not appear 
457            to have globals, an empty dictionary is used.  For classes, the search 
458            order is globals first then locals. 
459   
460          - If one dict argument is passed, it is used for both globals and 
461            locals. 
462   
463          - If two dict arguments are passed, they specify globals and 
464            locals, respectively. 
465          """
466          if getattr(obj, '__no_type_check__', None):
467              return {}
468          # Classes require a special treatment.
469          if isinstance(obj, type):
470              hints = {}
471              for base in reversed(obj.__mro__):
472                  if globalns is None:
473                      base_globals = getattr(sys.modules.get(base.__module__, None), '__dict__', {})
474                  else:
475                      base_globals = globalns
476                  ann = base.__dict__.get('__annotations__', {})
477                  if isinstance(ann, types.GetSetDescriptorType):
478                      ann = {}
479                  base_locals = dict(vars(base)) if localns is None else localns
480                  if localns is None and globalns is None:
481                      # This is surprising, but required.  Before Python 3.10,
482                      # get_type_hints only evaluated the globalns of
483                      # a class.  To maintain backwards compatibility, we reverse
484                      # the globalns and localns order so that eval() looks into
485                      # *base_globals* first rather than *base_locals*.
486                      # This only affects ForwardRefs.
487                      base_globals, base_locals = base_locals, base_globals
488                  for name, value in ann.items():
489                      if value is None:
490                          value = type(None)
491                      if isinstance(value, str):
492                          value = _make_forward_ref(value, is_argument=False, is_class=True)
493  
494                      value = eval_type_backport(value, base_globals, base_locals)
495                      hints[name] = value
496              if not include_extras and hasattr(typing, '_strip_annotations'):
497                  return {
498                      k: typing._strip_annotations(t)  # type: ignore
499                      for k, t in hints.items()
500                  }
501              else:
502                  return hints
503  
504          if globalns is None:
505              if isinstance(obj, types.ModuleType):
506                  globalns = obj.__dict__
507              else:
508                  nsobj = obj
509                  # Find globalns for the unwrapped object.
510                  while hasattr(nsobj, '__wrapped__'):
511                      nsobj = nsobj.__wrapped__
512                  globalns = getattr(nsobj, '__globals__', {})
513              if localns is None:
514                  localns = globalns
515          elif localns is None:
516              localns = globalns
517          hints = getattr(obj, '__annotations__', None)
518          if hints is None:
519              # Return empty annotations for something that _could_ have them.
520              if isinstance(obj, typing._allowed_types):  # type: ignore
521                  return {}
522              else:
523                  raise TypeError(f'{obj!r} is not a module, class, method, ' 'or function.')
524          defaults = typing._get_defaults(obj)  # type: ignore
525          hints = dict(hints)
526          for name, value in hints.items():
527              if value is None:
528                  value = type(None)
529              if isinstance(value, str):
530                  # class-level forward refs were handled above, this must be either
531                  # a module-level annotation or a function argument annotation
532  
533                  value = _make_forward_ref(
534                      value,
535                      is_argument=not isinstance(obj, types.ModuleType),
536                      is_class=False,
537                  )
538              value = eval_type_backport(value, globalns, localns)
539              if name in defaults and defaults[name] is None:
540                  value = typing.Optional[value]
541              hints[name] = value
542          return hints if include_extras else {k: typing._strip_annotations(t) for k, t in hints.items()}  # type: ignore
