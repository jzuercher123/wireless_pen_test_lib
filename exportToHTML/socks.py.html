39   class SOCKS(Packet):
40       fields_desc = [
41           ByteEnumField("vn", 0x5,
42                         {0x4: "v4 - Request", 0x0: "v4 - Reply", 0x5: "v5"}),
43       ]
44   
45       def guess_payload_class(self, pkt):
46           d_port = s_port = True
47           if self.underlayer and isinstance(self.underlayer, TCP):
48               ports = conf.contribs['socks']['serverports']
49               d_port = self.underlayer.dport in ports
50               s_port = self.underlayer.sport in ports
51           if self.vn == 0x5:
52               if d_port:
53                   return SOCKS5Request
54               elif s_port:
55                   return SOCKS5Reply
56           elif self.vn == 0x4:
57               if d_port:
58                   return SOCKS4Request
59           elif self.vn == 0x0:
60               if s_port:
61                   return SOCKS4Reply
62           warning("No TCP underlayer, or dport/sport not in "
63                   "conf.contribs['socks']['serverports']. "
64                   "Assuming a SOCKS v5 request layer")
65           return SOCKS5Request
66   
67       def add_payload(self, payload):
68           if self.underlayer and isinstance(self.underlayer, TCP):
69               if isinstance(payload, (SOCKS5Request, SOCKS4Request)):
70                   self.underlayer.dport = 1080
71                   self.underlayer.sport = 1081
72               elif isinstance(payload, (SOCKS5Reply, SOCKS4Reply)):
73                   self.underlayer.sport = 1080
74                   self.underlayer.dport = 1081
75           Packet.add_payload(self, payload)

89   class SOCKS4Request(Packet):
90       name = "SOCKS 4 - Request"
91       overload_fields = {SOCKS: {"vn": 0x4}}
92       fields_desc = [
93           ByteEnumField("cd", 1, _socks4_cd_request),
94           ShortField("dstport", 80),
95           IPField("dst", "0.0.0.0"),
96           StrNullField("userid", ""),
97       ]
98   
99   
100  _socks4_cd_reply = {
101      90: "Request granted",
102      91: "Request rejected",
103      92: "Request rejected - SOCKS server cannot connect to identd",
104      93: "Request rejected - user-ids mismatch"
105  }
106  
107  
108  class SOCKS4Reply(Packet):
109      name = "SOCKS 4 - Reply"
110      overload_fields = {SOCKS: {"vn": 0x0}}
111      fields_desc = [
112          ByteEnumField("cd", 90, _socks4_cd_reply),
113      ] + SOCKS4Request.fields_desc[1:-2]  # Reuse dstport, dst and userid

118  _socks5_cdtypes = {
119      1: "Connect",
120      2: "Bind",
121      3: "UDP associate",
122  }
123  
124  
125  class SOCKS5Request(Packet):
126      name = "SOCKS 5 - Request"
127      overload_fields = {SOCKS: {"vn": 0x5}}
128      fields_desc = [
129          ByteEnumField("cd", 0x0, _socks5_cdtypes),
130          ByteField("res", 0),
131          ByteEnumField("atyp", 0x1,
132                        {0x1: "IPv4", 0x3: "DomainName", 0x4: "IPv6"}),
133          MultipleTypeField(
134              [
135                  # IPv4
136                  (IPField("addr", "0.0.0.0"), lambda pkt: pkt.atyp == 0x1),
137                  # DNS
138                  (DNSStrField("addr", ""), lambda pkt: pkt.atyp == 0x3),
139                  # IPv6
140                  (IP6Field("addr", "::"), lambda pkt: pkt.atyp == 0x4),
141              ],
142              StrField("addr", "")
143          ),
144          ShortField("port", 80),
145      ]
146  
147  
148  _socks5_rep = {
149      0: "succeeded",
150      1: "general server failure",
151      2: "connection not allowed by ruleset",
152      3: "network unreachable",
153      4: "host unreachable",
154      5: "connection refused",
155      6: "TTL expired",
156      7: "command not supported",
157      8: "address type not supported",
158  }

161  class SOCKS5Reply(Packet):
162      name = "SOCKS 5 - Reply"
163      overload_fields = {SOCKS: {"vn": 0x5}}
164      # All fields are the same except the first one
165      fields_desc = [
166          ByteEnumField("rep", 0x0, _socks5_rep),
167      ] + SOCKS5Request.fields_desc[1:]

171  
172  class SOCKS5UDP(Packet):
173      name = "SOCKS 5 - UDP Header"
174      fields_desc = [
175          ShortField("res", 0),
176          ByteField("frag", 0),
177      ] + SOCKS5Request.fields_desc[2:]  # Reuse the atyp, addr and port fields
178  
179      def guess_payload_class(self, s):
180          if self.port == 0:
181              return conf.raw_layer
182          return UDP(sport=self.port, dport=self.port).guess_payload_class(None)
