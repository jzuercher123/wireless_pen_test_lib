7    """ 
8    This is a register for DH groups from RFC 3526 and RFC 4306. 
9    At this time the groups from RFC 7919 have not been registered by openssl, 
10   thus they cannot be imported from the cryptography library. 
11    
12   We also provide TLS identifiers for these DH groups and also the ECDH groups. 
13   (Note that the equivalent of _ffdh_groups for ECDH is ec._CURVE_TYPES.) 
14   """
15   
16   
17   from scapy.config import conf
18   from scapy.compat import bytes_int, int_bytes
19   from scapy.error import warning
20   from scapy.utils import long_converter
21   if conf.crypto_valid:
22       from cryptography.hazmat.backends import default_backend
23       from cryptography.hazmat.primitives.asymmetric import dh, ec
24       from cryptography.hazmat.primitives import serialization
25       from cryptography.hazmat.primitives.asymmetric.dh import DHParameterNumbers
26   if conf.crypto_valid_advanced:
27       from cryptography.hazmat.primitives.asymmetric import x25519
28       from cryptography.hazmat.primitives.asymmetric import x448

31   _ffdh_groups = {}
32   
33   
34   class _FFDHParamsMetaclass(type):
35       def __new__(cls, ffdh_name, bases, dct):
36           the_class = super(_FFDHParamsMetaclass, cls).__new__(cls, ffdh_name,
37                                                                bases, dct)
38           if conf.crypto_valid and ffdh_name != "_FFDHParams":
39               pn = DHParameterNumbers(the_class.m, the_class.g)
40               params = pn.parameters(default_backend())
41               _ffdh_groups[ffdh_name] = [params, the_class.mLen]
42           return the_class
43   
44   
45   class _FFDHParams(metaclass=_FFDHParamsMetaclass):
46       pass

416  def _tls_named_groups_import(group, pubbytes):
417      if group in _tls_named_ffdh_groups:
418          # https://datatracker.ietf.org/doc/html/rfc8446#section-4.2.8.1
419          params = _ffdh_groups[_tls_named_ffdh_groups[group]][0]
420          pn = params.parameter_numbers()
421          y = bytes_int(pubbytes)
422          public_numbers = dh.DHPublicNumbers(y, pn)
423          return public_numbers.public_key(default_backend())
424      elif group in _tls_named_curves:
425          # https://datatracker.ietf.org/doc/html/rfc8446#section-4.2.8.2
426          if _tls_named_curves[group] in ["x25519", "x448"]:
427              if conf.crypto_valid_advanced:
428                  if _tls_named_curves[group] == "x25519":
429                      import_point = x25519.X25519PublicKey.from_public_bytes
430                  else:
431                      import_point = x448.X448PublicKey.from_public_bytes
432                  return import_point(pubbytes)
433          else:
434              curve = ec._CURVE_TYPES[_tls_named_curves[group]]
435              try:
436                  # cryptography < 42
437                  curve = curve()
438              except TypeError:
439                  pass
440              try:  # cryptography >= 2.5
441                  return ec.EllipticCurvePublicKey.from_encoded_point(
442                      curve,
443                      pubbytes
444                  )
445              except AttributeError:
446                  pub_num = ec.EllipticCurvePublicNumbers.from_encoded_point(
447                      curve,
448                      pubbytes
449                  ).public_numbers()
450                  return pub_num.public_key(default_backend())

453  def _tls_named_groups_pubbytes(privkey):
454      if isinstance(privkey, dh.DHPrivateKey):
455          # https://datatracker.ietf.org/doc/html/rfc8446#section-4.2.8.1
456          pubkey = privkey.public_key()
457          return int_bytes(pubkey.public_numbers().y, privkey.key_size // 8)
458      elif isinstance(privkey, (x25519.X25519PrivateKey,
459                                x448.X448PrivateKey)):
460          # https://datatracker.ietf.org/doc/html/rfc8446#section-4.2.8.2
461          pubkey = privkey.public_key()
462          return pubkey.public_bytes(
463              serialization.Encoding.Raw,
464              serialization.PublicFormat.Raw
465          )
466      else:
467          pubkey = privkey.public_key()
468          try:
469              # cryptography >= 2.5
470              return pubkey.public_bytes(
471                  serialization.Encoding.X962,
472                  serialization.PublicFormat.UncompressedPoint
473              )
474          except TypeError:
475              # older versions
476              return pubkey.public_numbers().encode_point()

479  def _tls_named_groups_generate(group):
480      if group in _tls_named_ffdh_groups:
481          params = _ffdh_groups[_tls_named_ffdh_groups[group]][0]
482          return params.generate_private_key()
483      elif group in _tls_named_curves:
484          group_name = _tls_named_curves[group]
485          if group_name in ["x25519", "x448"]:
486              if conf.crypto_valid_advanced:
487                  if group_name == "x25519":
488                      return x25519.X25519PrivateKey.generate()
489                  else:
490                      return x448.X448PrivateKey.generate()
491              else:
492                  warning(
493                      "Your cryptography version doesn't support " + group_name
494                  )
495          else:
496              curve = ec._CURVE_TYPES[_tls_named_curves[group]]
497              try:
498                  # cryptography < 42
499                  curve = curve()
500              except TypeError:
501                  pass
502              return ec.generate_private_key(curve, default_backend())
