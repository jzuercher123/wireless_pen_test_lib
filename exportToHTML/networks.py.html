98   _host_regex = (
99       r'(?:'
100      r'(?P<ipv4>(?:\d{1,3}\.){3}\d{1,3})(?=$|[/:#?])|'  # ipv4
101      r'(?P<ipv6>\[[A-F0-9]*:[A-F0-9:]+\])(?=$|[/:#?])|'  # ipv6
102      r'(?P<domain>[^\s/:?#]+)'  # domain, validation occurs later
103      r')?'
104      r'(?::(?P<port>\d+))?'  # port
105  )
106  _scheme_regex = r'(?:(?P<scheme>[a-z][a-z0-9+\-.]+)://)?'  # scheme https://tools.ietf.org/html/rfc3986#appendix-A
107  _user_info_regex = r'(?:(?P<user>[^\s:/]*)(?::(?P<password>[^\s/]*))?@)?'
108  _path_regex = r'(?P<path>/[^\s?#]*)?'
109  _query_regex = r'(?:\?(?P<query>[^\s#]*))?'
110  _fragment_regex = r'(?:#(?P<fragment>[^\s#]*))?'

113  def url_regex() -> Pattern[str]:
114      global _url_regex_cache
115      if _url_regex_cache is None:
116          _url_regex_cache = re.compile(
117              rf'{_scheme_regex}{_user_info_regex}{_host_regex}{_path_regex}{_query_regex}{_fragment_regex}',
118              re.IGNORECASE,
119          )
120      return _url_regex_cache

123  def multi_host_url_regex() -> Pattern[str]:
124      """ 
125      Compiled multi host url regex. 
126   
127      Additionally to `url_regex` it allows to match multiple hosts. 
128      E.g. host1.db.net,host2.db.net 
129      """
130      global _multi_host_url_regex_cache
131      if _multi_host_url_regex_cache is None:
132          _multi_host_url_regex_cache = re.compile(
133              rf'{_scheme_regex}{_user_info_regex}'
134              r'(?P<hosts>([^/]*))'  # validation occurs later
135              rf'{_path_regex}{_query_regex}{_fragment_regex}',
136              re.IGNORECASE,
137          )
138      return _multi_host_url_regex_cache

141  def ascii_domain_regex() -> Pattern[str]:
142      global _ascii_domain_regex_cache
143      if _ascii_domain_regex_cache is None:
144          ascii_chunk = r'[_0-9a-z](?:[-_0-9a-z]{0,61}[_0-9a-z])?'
145          ascii_domain_ending = r'(?P<tld>\.[a-z]{2,63})?\.?'
146          _ascii_domain_regex_cache = re.compile(
147              fr'(?:{ascii_chunk}\.)*?{ascii_chunk}{ascii_domain_ending}', re.IGNORECASE
148          )
149      return _ascii_domain_regex_cache

152  def int_domain_regex() -> Pattern[str]:
153      global _int_domain_regex_cache
154      if _int_domain_regex_cache is None:
155          int_chunk = r'[_0-9a-\U00040000](?:[-_0-9a-\U00040000]{0,61}[_0-9a-\U00040000])?'
156          int_domain_ending = r'(?P<tld>(\.[^\W\d_]{2,63})|(\.(?:xn--)[_0-9a-z-]{2,63}))?\.?'
157          _int_domain_regex_cache = re.compile(fr'(?:{int_chunk}\.)*?{int_chunk}{int_domain_ending}', re.IGNORECASE)
158      return _int_domain_regex_cache

171  class AnyUrl(str):
172      strip_whitespace = True
173      min_length = 1
174      max_length = 2**16
175      allowed_schemes: Optional[Collection[str]] = None
176      tld_required: bool = False
177      user_required: bool = False
178      host_required: bool = True
179      hidden_parts: Set[str] = set()
180  
181      __slots__ = ('scheme', 'user', 'password', 'host', 'tld', 'host_type', 'port', 'path', 'query', 'fragment')
182  
183      @no_type_check
184      def __new__(cls, url: Optional[str], **kwargs) -> object:
185          return str.__new__(cls, cls.build(**kwargs) if url is None else url)
186  
187      def __init__(
188          self,
189          url: str,
190          *,
191          scheme: str,
192          user: Optional[str] = None,
193          password: Optional[str] = None,
194          host: Optional[str] = None,
195          tld: Optional[str] = None,
196          host_type: str = 'domain',
197          port: Optional[str] = None,
198          path: Optional[str] = None,
199          query: Optional[str] = None,
200          fragment: Optional[str] = None,
201      ) -> None:
202          str.__init__(url)
203          self.scheme = scheme
204          self.user = user
205          self.password = password
206          self.host = host
207          self.tld = tld
208          self.host_type = host_type
209          self.port = port
210          self.path = path
211          self.query = query
212          self.fragment = fragment

214      @classmethod
215      def build(
216          cls,
217          *,
218          scheme: str,
219          user: Optional[str] = None,
220          password: Optional[str] = None,
221          host: str,
222          port: Optional[str] = None,
223          path: Optional[str] = None,
224          query: Optional[str] = None,
225          fragment: Optional[str] = None,
226          **_kwargs: str,
227      ) -> str:
228          parts = Parts(
229              scheme=scheme,
230              user=user,
231              password=password,
232              host=host,
233              port=port,
234              path=path,
235              query=query,
236              fragment=fragment,
237              **_kwargs,  # type: ignore[misc]
238          )
239  
240          url = scheme + '://'
241          if user:
242              url += user
243          if password:
244              url += ':' + password
245          if user or password:
246              url += '@'
247          url += host
248          if port and ('port' not in cls.hidden_parts or cls.get_default_parts(parts).get('port') != port):
249              url += ':' + port
250          if path:
251              url += path
252          if query:
253              url += '?' + query
254          if fragment:
255              url += '#' + fragment
256          return url

258      @classmethod
259      def __modify_schema__(cls, field_schema: Dict[str, Any]) -> None:
260          update_not_none(field_schema, minLength=cls.min_length, maxLength=cls.max_length, format='uri')

262      @classmethod
263      def __get_validators__(cls) -> 'CallableGenerator':
264          yield cls.validate

266      @classmethod
267      def validate(cls, value: Any, field: 'ModelField', config: 'BaseConfig') -> 'AnyUrl':
268          if value.__class__ == cls:
269              return value
270          value = str_validator(value)
271          if cls.strip_whitespace:
272              value = value.strip()
273          url: str = cast(str, constr_length_validator(value, field, config))
274  
275          m = cls._match_url(url)
276          # the regex should always match, if it doesn't please report with details of the URL tried
277          assert m, 'URL regex failed unexpectedly'
278  
279          original_parts = cast('Parts', m.groupdict())
280          parts = cls.apply_default_parts(original_parts)
281          parts = cls.validate_parts(parts)
282  
283          if m.end() != len(url):
284              raise errors.UrlExtraError(extra=url[m.end() :])
285  
286          return cls._build_url(m, url, parts)

290          """ 
291          Validate hosts and build the AnyUrl object. Split from `validate` so this method

293          """
294          host, tld, host_type, rebuild = cls.validate_host(parts)

296          return cls(

298              scheme=parts['scheme'],
299              user=parts['user'],
300              password=parts['password'],
301              host=host,
302              tld=tld,
303              host_type=host_type,
304              port=parts['port'],
305              path=parts['path'],
306              query=parts['query'],
307              fragment=parts['fragment'],
308          )

315      def _validate_port(port: Optional[str]) -> None:

319      @classmethod
320      def validate_parts(cls, parts: 'Parts', validate_port: bool = True) -> 'Parts':
321          """ 
322          A method used to validate parts of a URL. 
323          Could be overridden to set default values for parts if missing 
324          """
325          scheme = parts['scheme']
326          if scheme is None:
327              raise errors.UrlSchemeError()
328  
329          if cls.allowed_schemes and scheme.lower() not in cls.allowed_schemes:
330              raise errors.UrlSchemePermittedError(set(cls.allowed_schemes))
331  
332          if validate_port:
333              cls._validate_port(parts['port'])
334  
335          user = parts['user']
336          if cls.user_required and user is None:
337              raise errors.UrlUserInfoError()
338  
339          return parts

341      @classmethod
342      def validate_host(cls, parts: 'Parts') -> Tuple[str, Optional[str], str, bool]:
343          tld, host_type, rebuild = None, None, False
344          for f in ('domain', 'ipv4', 'ipv6'):
345              host = parts[f]  # type: ignore[literal-required]
346              if host:
347                  host_type = f
348                  break
349  
350          if host is None:
351              if cls.host_required:
352                  raise errors.UrlHostError()
353          elif host_type == 'domain':
354              is_international = False
355              d = ascii_domain_regex().fullmatch(host)
356              if d is None:
357                  d = int_domain_regex().fullmatch(host)
358                  if d is None:
359                      raise errors.UrlHostError()
360                  is_international = True
361  
362              tld = d.group('tld')
363              if tld is None and not is_international:
364                  d = int_domain_regex().fullmatch(host)
365                  assert d is not None
366                  tld = d.group('tld')
367                  is_international = True
368  
369              if tld is not None:
370                  tld = tld[1:]
371              elif cls.tld_required:
372                  raise errors.UrlHostTldError()
373  
374              if is_international:
375                  host_type = 'int_domain'
376                  rebuild = True
377                  host = host.encode('idna').decode('ascii')
378                  if tld is not None:
379                      tld = tld.encode('idna').decode('ascii')
380  
381          return host, tld, host_type, rebuild  # type: ignore

384      def get_default_parts(parts: 'Parts') -> 'Parts':

387      @classmethod
388      def apply_default_parts(cls, parts: 'Parts') -> 'Parts':
389          for key, value in cls.get_default_parts(parts).items():
390              if not parts[key]:  # type: ignore[literal-required]
391                  parts[key] = value  # type: ignore[literal-required]
392          return parts

423  class MultiHostDsn(AnyUrl):
424      __slots__ = AnyUrl.__slots__ + ('hosts',)
425  
426      def __init__(self, *args: Any, hosts: Optional[List['HostParts']] = None, **kwargs: Any):
427          super().__init__(*args, **kwargs)
428          self.hosts = hosts

430      @staticmethod
431      def _match_url(url: str) -> Optional[Match[str]]:
432          return multi_host_url_regex().match(url)
433  
434      @classmethod
435      def validate_parts(cls, parts: 'Parts', validate_port: bool = True) -> 'Parts':
436          return super().validate_parts(parts, validate_port=False)
437  
438      @classmethod
439      def _build_url(cls, m: Match[str], url: str, parts: 'Parts') -> 'MultiHostDsn':
440          hosts_parts: List['HostParts'] = []
441          host_re = host_regex()
442          for host in m.groupdict()['hosts'].split(','):
443              d: Parts = host_re.match(host).groupdict()  # type: ignore
444              host, tld, host_type, rebuild = cls.validate_host(d)
445              port = d.get('port')
446              cls._validate_port(port)
447              hosts_parts.append(
448                  {
449                      'host': host,
450                      'host_type': host_type,
451                      'tld': tld,
452                      'rebuild': rebuild,
453                      'port': port,
454                  }
455              )
456  
457          if len(hosts_parts) > 1:
458              return cls(
459                  None if any([hp['rebuild'] for hp in hosts_parts]) else url,
460                  scheme=parts['scheme'],
461                  user=parts['user'],
462                  password=parts['password'],
463                  path=parts['path'],
464                  query=parts['query'],
465                  fragment=parts['fragment'],
466                  host_type=None,
467                  hosts=hosts_parts,
468              )
469          else:
470              # backwards compatibility with single host
471              host_part = hosts_parts[0]
472              return cls(
473                  None if host_part['rebuild'] else url,
474                  scheme=parts['scheme'],
475                  user=parts['user'],
476                  password=parts['password'],
477                  host=host_part['host'],
478                  tld=host_part['tld'],
479                  host_type=host_part['host_type'],
480                  port=host_part.get('port'),
481                  path=parts['path'],
482                  query=parts['query'],
483                  fragment=parts['fragment'],
484              )

711  def validate_email(value: Union[str]) -> Tuple[str, str]:
712      """ 
713      Email address validation using https://pypi.org/project/email-validator/ 
714      Notes: 
715      * raw ip address (literal) domain parts are not allowed. 
716      * "John Doe <local_part@domain.com>" style "pretty" email addresses are processed 
717      * spaces are striped from the beginning and end of addresses but no error is raised 
718      """
719      if email_validator is None:
720          import_email_validator()
721  
722      if len(value) > MAX_EMAIL_LENGTH:
723          raise errors.EmailError()
724  
725      m = pretty_email_regex.fullmatch(value)
726      name: Union[str, None] = None
727      if m:
728          name, value = m.groups()
729      email = value.strip()
730      try:
731          parts = email_validator.validate_email(email, check_deliverability=False)
732      except email_validator.EmailNotValidError as e:
733          raise errors.EmailError from e
734  
735      if hasattr(parts, 'normalized'):
736          # email-validator >= 2
737          email = parts.normalized
738          assert email is not None
739          name = name or parts.local_part
740          return name, email
741      else:
742          # email-validator >1, <2
743          at_index = email.index('@')
744          local_part = email[:at_index]  # RFC 5321, local part must be case-sensitive.
745          global_part = email[at_index:].lower()
746  
747          return name or local_part, local_part + global_part
