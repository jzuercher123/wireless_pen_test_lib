338  class Block(Packet):
339      """A generic block packet for PNIO RPC"""
340      fields_desc = [
341          BlockHeader,
342          StrLenField("load", "", length_from=lambda pkt: pkt.block_length - 2),
343      ]
344      # default block_type
345      block_type = 0
346  
347      def post_build(self, p, pay):
348          # update the block_length if needed
349          if self.block_length is None:
350              # block_length and block_type are not part of the length count
351              length = len(p) - 4
352              p = p[:2] + struct.pack("!H", length) + p[4:]
353  
354          return Packet.post_build(self, p, pay)
355  
356      def extract_padding(self, s):
357          # all fields after block_length are included in the length and must be
358          # subtracted from the pdu length l
359          length = self.payload_length()
360          return s[:length], s[length:]
361  
362      def payload_length(self):
363          """ A function for each block, to determine the length of 
364          the payload """
365          return 0  # default, no payload

370  class IODControlReq(Block):
371      """IODControl request block"""
372      fields_desc = [
373          BlockHeader,
374          StrFixedLenField("padding", "", length=2),
375          UUIDField("ARUUID", None),
376          ShortField("SessionKey", 0),
377          XShortField("AlarmSequenceNumber", 0),
378          # ControlCommand
379          BitField("ControlCommand_reserved", 0, 9),
380          BitField("ControlCommand_PrmBegin", 0, 1),
381          BitField("ControlCommand_ReadyForRT_CLASS_3", 0, 1),
382          BitField("ControlCommand_ReadyForCompanion", 0, 1),
383          BitField("ControlCommand_Done", 0, 1),
384          BitField("ControlCommand_Release", 0, 1),
385          BitField("ControlCommand_ApplicationReady", 0, 1),
386          BitField("ControlCommand_PrmEnd", 0, 1),
387          XShortField("ControlBlockProperties", 0)
388      ]
389  
390      def post_build(self, p, pay):
391          # Try to find the right block type
392          if self.block_type is None:
393              if self.ControlCommand_PrmBegin:
394                  p = struct.pack("!H", 0x0118) + p[2:]
395              elif self.ControlCommand_ReadyForRT_CLASS_3:
396                  p = struct.pack("!H", 0x0117) + p[2:]
397              elif self.ControlCommand_ReadyForCompanion:
398                  p = struct.pack("!H", 0x0116) + p[2:]
399              elif self.ControlCommand_Release:
400                  p = struct.pack("!H", 0x0114) + p[2:]
401              elif self.ControlCommand_ApplicationReady:
402                  if self.AlarmSequenceNumber > 0:
403                      p = struct.pack("!H", 0x0113) + p[2:]
404                  else:
405                      p = struct.pack("!H", 0x0112) + p[2:]
406              elif self.ControlCommand_PrmEnd:
407                  if self.AlarmSequenceNumber > 0:
408                      p = struct.pack("!H", 0x0111) + p[2:]
409                  else:
410                      p = struct.pack("!H", 0x0110) + p[2:]
411          return Block.post_build(self, p, pay)
412  
413      def get_response(self):
414          """Generate the response block of this request. 
415          Careful: it only sets the fields which can be set from the request 
416          """
417          res = IODControlRes()
418          for field in ["ARUUID", "SessionKey", "AlarmSequenceNumber"]:
419              res.setfieldval(field, self.getfieldval(field))
420  
421          res.block_type = self.block_type + 0x8000
422          return res

1421 def _guess_block_class(_pkt, *args, **kargs):
1422     cls = Block  # Default block type
1423 
1424     # Special cases
1425     if _pkt[:2] == b'\x00\x08':  # IODWriteReq
1426         if _pkt[34:36] == b'\xe0@':  # IODWriteMultipleReq
1427             cls = IODWriteMultipleReq
1428         else:
1429             cls = IODWriteReq
1430 
1431     elif _pkt[:2] == b'\x00\x09':  # IODReadReq
1432         cls = IODReadReq
1433 
1434     elif _pkt[:2] == b'\x80\x08':    # IODWriteRes
1435         if _pkt[34:36] == b'\xe0@':  # IODWriteMultipleRes
1436             cls = IODWriteMultipleRes
1437         else:
1438             cls = IODWriteRes
1439 
1440     elif _pkt[:2] == b'\x80\x09':  # IODReadRes
1441         cls = IODReadRes
1442 
1443     # Common cases
1444     else:
1445         btype = bytes_hex(_pkt[:2]).decode("utf8")
1446         if btype in PNIO_RPC_BLOCK_ASSOCIATION:
1447             cls = PNIO_RPC_BLOCK_ASSOCIATION[btype]
1448 
1449     return cls(_pkt, *args, **kargs)

1453     """determine the symbol for the endianness of a the DCE/RPC"""
1454     try:
1455         endianness = pkt.underlayer.endian
1456     except AttributeError:
1457         # handle the case where a PNIO class is
1458         # built without its DCE-RPC under-layer
1459         # i.e there is no endianness indication
1460         return "!"
1461     if endianness == 0:  # big endian

1464         return "<"
1465     else:
