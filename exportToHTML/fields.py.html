180      def __repr_args__(self) -> 'ReprArgs':
181          field_defaults_to_hide: Dict[str, Any] = {
182              'repr': True,
183              **self.__field_constraints__,
184          }
185  
186          attrs = ((s, getattr(self, s)) for s in self.__slots__)
187          return [(a, v) for a, v in attrs if v != field_defaults_to_hide.get(a, None)]

189      def get_constraints(self) -> Set[str]:
190          """ 
191          Gets the constraints set on the field by comparing the constraint value with its default value 
192   
193          :return: the constraints set on field_info 
194          """
195          return {attr for attr, default in self.__field_constraints__.items() if getattr(self, attr) != default}

197      def update_from_config(self, from_config: Dict[str, Any]) -> None:
198          """ 
199          Update this FieldInfo based on a dict from get_field_info, only fields which have not been set are dated. 
200          """
201          for attr_name, value in from_config.items():
202              try:
203                  current_value = getattr(self, attr_name)
204              except AttributeError:
205                  # attr_name is not an attribute of FieldInfo, it should therefore be added to extra
206                  # (except if extra already has this value!)
207                  self.extra.setdefault(attr_name, value)
208              else:
209                  if current_value is self.__field_constraints__.get(attr_name, None):
210                      setattr(self, attr_name, value)
211                  elif attr_name == 'exclude':
212                      self.exclude = ValueItems.merge(value, current_value)
213                  elif attr_name == 'include':
214                      self.include = ValueItems.merge(value, current_value, intersect=True)

216      def _validate(self) -> None:
217          if self.default is not Undefined and self.default_factory is not None:
218              raise ValueError('cannot specify both default and default_factory')

222      default: Any = Undefined,
223      *,
224      default_factory: Optional[NoArgAnyCallable] = None,
225      alias: Optional[str] = None,
226      title: Optional[str] = None,
227      description: Optional[str] = None,
228      exclude: Optional[Union['AbstractSetIntStr', 'MappingIntStrAny', Any]] = None,
229      include: Optional[Union['AbstractSetIntStr', 'MappingIntStrAny', Any]] = None,
230      const: Optional[bool] = None,
231      gt: Optional[float] = None,
232      ge: Optional[float] = None,
233      lt: Optional[float] = None,
234      le: Optional[float] = None,
235      multiple_of: Optional[float] = None,
236      allow_inf_nan: Optional[bool] = None,
237      max_digits: Optional[int] = None,
238      decimal_places: Optional[int] = None,
239      min_items: Optional[int] = None,
240      max_items: Optional[int] = None,
241      unique_items: Optional[bool] = None,
242      min_length: Optional[int] = None,
243      max_length: Optional[int] = None,
244      allow_mutation: bool = True,
245      regex: Optional[str] = None,
246      discriminator: Optional[str] = None,
247      repr: bool = True,
248      **extra: Any,
249  ) -> Any:
250      """ 
251      Used to provide extra information about a field, either for the model schema or complex validation. Some arguments 
252      apply only to number fields (``int``, ``float``, ``Decimal``) and some apply only to ``str``. 
253   
254      :param default: since this is replacing the field’s default, its first argument is used 
255        to set the default, use ellipsis (``...``) to indicate the field is required 
256      :param default_factory: callable that will be called when a default value is needed for this field 
257        If both `default` and `default_factory` are set, an error is raised. 
258      :param alias: the public name of the field 
259      :param title: can be any string, used in the schema 
260      :param description: can be any string, used in the schema 
261      :param exclude: exclude this field while dumping. 
262        Takes same values as the ``include`` and ``exclude`` arguments on the ``.dict`` method. 
263      :param include: include this field while dumping. 
264        Takes same values as the ``include`` and ``exclude`` arguments on the ``.dict`` method. 
265      :param const: this field is required and *must* take it's default value 
266      :param gt: only applies to numbers, requires the field to be "greater than". The schema 
267        will have an ``exclusiveMinimum`` validation keyword 
268      :param ge: only applies to numbers, requires the field to be "greater than or equal to". The 
269        schema will have a ``minimum`` validation keyword 
270      :param lt: only applies to numbers, requires the field to be "less than". The schema 
271        will have an ``exclusiveMaximum`` validation keyword 
272      :param le: only applies to numbers, requires the field to be "less than or equal to". The 
273        schema will have a ``maximum`` validation keyword 
274      :param multiple_of: only applies to numbers, requires the field to be "a multiple of". The 
275        schema will have a ``multipleOf`` validation keyword 
276      :param allow_inf_nan: only applies to numbers, allows the field to be NaN or infinity (+inf or -inf), 
277          which is a valid Python float. Default True, set to False for compatibility with JSON. 
278      :param max_digits: only applies to Decimals, requires the field to have a maximum number 
279        of digits within the decimal. It does not include a zero before the decimal point or trailing decimal zeroes. 
280      :param decimal_places: only applies to Decimals, requires the field to have at most a number of decimal places 
281        allowed. It does not include trailing decimal zeroes. 
282      :param min_items: only applies to lists, requires the field to have a minimum number of 
283        elements. The schema will have a ``minItems`` validation keyword 
284      :param max_items: only applies to lists, requires the field to have a maximum number of 
285        elements. The schema will have a ``maxItems`` validation keyword 
286      :param unique_items: only applies to lists, requires the field not to have duplicated 
287        elements. The schema will have a ``uniqueItems`` validation keyword 
288      :param min_length: only applies to strings, requires the field to have a minimum length. The 
289        schema will have a ``minLength`` validation keyword 
290      :param max_length: only applies to strings, requires the field to have a maximum length. The 
291        schema will have a ``maxLength`` validation keyword 
292      :param allow_mutation: a boolean which defaults to True. When False, the field raises a TypeError if the field is 
293        assigned on an instance.  The BaseModel Config must set validate_assignment to True 
294      :param regex: only applies to strings, requires the field match against a regular expression 
295        pattern string. The schema will have a ``pattern`` validation keyword 
296      :param discriminator: only useful with a (discriminated a.k.a. tagged) `Union` of sub models with a common field. 
297        The `discriminator` is the name of this common field to shorten validation and improve generated schema 
298      :param repr: show this field in the representation 
299      :param **extra: any additional keyword arguments will be added as is to the schema 
300      """
301      field_info = FieldInfo(
302          default,
303          default_factory=default_factory,
304          alias=alias,
305          title=title,
306          description=description,
307          exclude=exclude,
308          include=include,
309          const=const,
310          gt=gt,
311          ge=ge,
312          lt=lt,
313          le=le,
314          multiple_of=multiple_of,
315          allow_inf_nan=allow_inf_nan,

317          decimal_places=decimal_places,
318          min_items=min_items,
319          max_items=max_items,
320          unique_items=unique_items,
321          min_length=min_length,
322          max_length=max_length,
323          allow_mutation=allow_mutation,
324          regex=regex,
325          discriminator=discriminator,
326          repr=repr,
327          **extra,
328      )
329      field_info._validate()
330      return field_info

439      @staticmethod
440      def _get_field_info(
441          field_name: str, annotation: Any, value: Any, config: Type['BaseConfig']
442      ) -> Tuple[FieldInfo, Any]:
443          """ 
444          Get a FieldInfo from a root typing.Annotated annotation, value, or config default. 
445   
446          The FieldInfo may be set in typing.Annotated or the value, but not both. If neither contain 
447          a FieldInfo, a new one will be created using the config. 
448   
449          :param field_name: name of the field for use in error messages 
450          :param annotation: a type hint such as `str` or `Annotated[str, Field(..., min_length=5)]` 
451          :param value: the field's assigned value 
452          :param config: the model's config object 
453          :return: the FieldInfo contained in the `annotation`, the value, or a new one from the config. 
454          """
455          field_info_from_config = config.get_field_info(field_name)
456  
457          field_info = None
458          if get_origin(annotation) is Annotated:
459              field_infos = [arg for arg in get_args(annotation)[1:] if isinstance(arg, FieldInfo)]
460              if len(field_infos) > 1:
461                  raise ValueError(f'cannot specify multiple `Annotated` `Field`s for {field_name!r}')
462              field_info = next(iter(field_infos), None)
463              if field_info is not None:
464                  field_info = copy.copy(field_info)
465                  field_info.update_from_config(field_info_from_config)
466                  if field_info.default not in (Undefined, Required):
467                      raise ValueError(f'`Field` default cannot be set in `Annotated` for {field_name!r}')
468                  if value is not Undefined and value is not Required:
469                      # check also `Required` because of `validate_arguments` that sets `...` as default value
470                      field_info.default = value
471  
472          if isinstance(value, FieldInfo):
473              if field_info is not None:
474                  raise ValueError(f'cannot specify `Annotated` and value `Field`s together for {field_name!r}')
475              field_info = value
476              field_info.update_from_config(field_info_from_config)
477          elif field_info is None:
478              field_info = FieldInfo(value, **field_info_from_config)
479          value = None if field_info.default_factory is not None else field_info.default
480          field_info._validate()
481          return field_info, value

483      @classmethod
484      def infer(
485          cls,
486          *,
487          name: str,
488          value: Any,
489          annotation: Any,
490          class_validators: Optional[Dict[str, Validator]],
491          config: Type['BaseConfig'],
492      ) -> 'ModelField':
493          from pydantic.v1.schema import get_annotation_from_field_info
494  
495          field_info, value = cls._get_field_info(name, annotation, value, config)
496          required: 'BoolUndefined' = Undefined
497          if value is Required:
498              required = True
499              value = None
500          elif value is not Undefined:
501              required = False
502          annotation = get_annotation_from_field_info(annotation, field_info, name, config.validate_assignment)
503  
504          return cls(
505              name=name,
506              type_=annotation,
507              alias=field_info.alias,
508              class_validators=class_validators,
509              default=value,
510              default_factory=field_info.default_factory,
511              required=required,
512              model_config=config,
513              field_info=field_info,
514          )

516      def set_config(self, config: Type['BaseConfig']) -> None:
517          self.model_config = config
518          info_from_config = config.get_field_info(self.name)
519          config.prepare_field(self)
520          new_alias = info_from_config.get('alias')
521          new_alias_priority = info_from_config.get('alias_priority') or 0
522          if new_alias and new_alias_priority >= (self.field_info.alias_priority or 0):
523              self.field_info.alias = new_alias
524              self.field_info.alias_priority = new_alias_priority
525              self.alias = new_alias
526          new_exclude = info_from_config.get('exclude')
527          if new_exclude is not None:
528              self.field_info.exclude = ValueItems.merge(self.field_info.exclude, new_exclude)
529          new_include = info_from_config.get('include')
530          if new_include is not None:
531              self.field_info.include = ValueItems.merge(self.field_info.include, new_include, intersect=True)

537      def prepare(self) -> None:
538          """ 
539          Prepare the field but inspecting self.default, self.type_ etc. 
540   
541          Note: this method is **not** idempotent (because _type_analysis is not idempotent), 
542          e.g. calling it it multiple times may modify the field and configure it incorrectly. 
543          """
544          self._set_default_and_type()
545          if self.type_.__class__ is ForwardRef or self.type_.__class__ is DeferredType:
546              # self.type_ is currently a ForwardRef and there's nothing we can do now,
547              # user will need to call model.update_forward_refs()
548              return
549  
550          self._type_analysis()
551          if self.required is Undefined:
552              self.required = True
553          if self.default is Undefined and self.default_factory is None:
554              self.default = None
555          self.populate_validators()
556  
557      def _set_default_and_type(self) -> None:
558          """ 
559          Set the default value, infer the type if needed and check if `None` value is valid. 
560          """
561          if self.default_factory is not None:
562              if self.type_ is Undefined:
563                  raise errors_.ConfigError(
564                      f'you need to set the type of field {self.name!r} when using `default_factory`'
565                  )
566              return
567  
568          default_value = self.get_default()
569  
570          if default_value is not None and self.type_ is Undefined:
571              self.type_ = default_value.__class__
572              self.outer_type_ = self.type_
573              self.annotation = self.type_
574  
575          if self.type_ is Undefined:
576              raise errors_.ConfigError(f'unable to infer type for attribute "{self.name}"')
577  
578          if self.required is False and default_value is None:
579              self.allow_none = True
580  
581      def _type_analysis(self) -> None:  # noqa: C901 (ignore complexity)
582          # typing interface is horrible, we have to do some ugly checks
583          if lenient_issubclass(self.type_, JsonWrapper):
584              self.type_ = self.type_.inner_type
585              self.parse_json = True
586          elif lenient_issubclass(self.type_, Json):
587              self.type_ = Any
588              self.parse_json = True
589          elif isinstance(self.type_, TypeVar):
590              if self.type_.__bound__:
591                  self.type_ = self.type_.__bound__
592              elif self.type_.__constraints__:
593                  self.type_ = Union[self.type_.__constraints__]
594              else:
595                  self.type_ = Any
596          elif is_new_type(self.type_):
597              self.type_ = new_type_supertype(self.type_)
598  
599          if self.type_ is Any or self.type_ is object:
600              if self.required is Undefined:
601                  self.required = False
602              self.allow_none = True
603              return
604          elif self.type_ is Pattern or self.type_ is re.Pattern:
605              # python 3.7 only, Pattern is a typing object but without sub fields
606              return
607          elif is_literal_type(self.type_):
608              return
609          elif is_typeddict(self.type_):
610              return
611  
612          if is_finalvar(self.type_):
613              self.final = True
614  
615              if self.type_ is Final:
616                  self.type_ = Any
617              else:
618                  self.type_ = get_args(self.type_)[0]
619  
620              self._type_analysis()
621              return
622  
623          origin = get_origin(self.type_)
624  
625          if origin is Annotated or is_typeddict_special(origin):
626              self.type_ = get_args(self.type_)[0]
627              self._type_analysis()
628              return
629  
630          if self.discriminator_key is not None and not is_union(origin):
631              raise TypeError('`discriminator` can only be used with `Union` type with more than one variant')
632  
633          # add extra check for `collections.abc.Hashable` for python 3.10+ where origin is not `None`
634          if origin is None or origin is CollectionsHashable:
635              # field is not "typing" object eg. Union, Dict, List etc.
636              # allow None for virtual superclasses of NoneType, e.g. Hashable
637              if isinstance(self.type_, type) and isinstance(None, self.type_):
638                  self.allow_none = True
639              return
640          elif origin is Callable:
641              return
642          elif is_union(origin):
643              types_ = []
644              for type_ in get_args(self.type_):
645                  if is_none_type(type_) or type_ is Any or type_ is object:
646                      if self.required is Undefined:
647                          self.required = False
648                      self.allow_none = True
649                  if is_none_type(type_):
650                      continue
651                  types_.append(type_)
652  
653              if len(types_) == 1:
654                  # Optional[]
655                  self.type_ = types_[0]
656                  # this is the one case where the "outer type" isn't just the original type
657                  self.outer_type_ = self.type_
658                  # re-run to correctly interpret the new self.type_
659                  self._type_analysis()
660              else:
661                  self.sub_fields = [self._create_sub_type(t, f'{self.name}_{display_as_type(t)}') for t in types_]
662  
663                  if self.discriminator_key is not None:
664                      self.prepare_discriminated_union_sub_fields()
665              return
666          elif issubclass(origin, Tuple):  # type: ignore
667              # origin == Tuple without item type
668              args = get_args(self.type_)
669              if not args:  # plain tuple
670                  self.type_ = Any
671                  self.shape = SHAPE_TUPLE_ELLIPSIS
672              elif len(args) == 2 and args[1] is Ellipsis:  # e.g. Tuple[int, ...]
673                  self.type_ = args[0]
674                  self.shape = SHAPE_TUPLE_ELLIPSIS
675                  self.sub_fields = [self._create_sub_type(args[0], f'{self.name}_0')]
676              elif args == ((),):  # Tuple[()] means empty tuple
677                  self.shape = SHAPE_TUPLE
678                  self.type_ = Any
679                  self.sub_fields = []
680              else:
681                  self.shape = SHAPE_TUPLE
682                  self.sub_fields = [self._create_sub_type(t, f'{self.name}_{i}') for i, t in enumerate(args)]
683              return
684          elif issubclass(origin, List):
685              # Create self validators
686              get_validators = getattr(self.type_, '__get_validators__', None)
687              if get_validators:
688                  self.class_validators.update(
689                      {f'list_{i}': Validator(validator, pre=True) for i, validator in enumerate(get_validators())}
690                  )
691  
692              self.type_ = get_args(self.type_)[0]
693              self.shape = SHAPE_LIST
694          elif issubclass(origin, Set):
695              # Create self validators
696              get_validators = getattr(self.type_, '__get_validators__', None)
697              if get_validators:
698                  self.class_validators.update(
699                      {f'set_{i}': Validator(validator, pre=True) for i, validator in enumerate(get_validators())}
700                  )
701  
702              self.type_ = get_args(self.type_)[0]
703              self.shape = SHAPE_SET
704          elif issubclass(origin, FrozenSet):
705              # Create self validators
706              get_validators = getattr(self.type_, '__get_validators__', None)
707              if get_validators:
708                  self.class_validators.update(
709                      {f'frozenset_{i}': Validator(validator, pre=True) for i, validator in enumerate(get_validators())}
710                  )
711  
712              self.type_ = get_args(self.type_)[0]
713              self.shape = SHAPE_FROZENSET
714          elif issubclass(origin, Deque):
715              self.type_ = get_args(self.type_)[0]
716              self.shape = SHAPE_DEQUE
717          elif issubclass(origin, Sequence):
718              self.type_ = get_args(self.type_)[0]
719              self.shape = SHAPE_SEQUENCE
720          # priority to most common mapping: dict
721          elif origin is dict or origin is Dict:
722              self.key_field = self._create_sub_type(get_args(self.type_)[0], 'key_' + self.name, for_keys=True)
723              self.type_ = get_args(self.type_)[1]
724              self.shape = SHAPE_DICT
725          elif issubclass(origin, DefaultDict):
726              self.key_field = self._create_sub_type(get_args(self.type_)[0], 'key_' + self.name, for_keys=True)
727              self.type_ = get_args(self.type_)[1]
728              self.shape = SHAPE_DEFAULTDICT
729          elif issubclass(origin, Counter):
730              self.key_field = self._create_sub_type(get_args(self.type_)[0], 'key_' + self.name, for_keys=True)
731              self.type_ = int
732              self.shape = SHAPE_COUNTER
733          elif issubclass(origin, Mapping):
734              self.key_field = self._create_sub_type(get_args(self.type_)[0], 'key_' + self.name, for_keys=True)
735              self.type_ = get_args(self.type_)[1]
736              self.shape = SHAPE_MAPPING
737          # Equality check as almost everything inherits form Iterable, including str
738          # check for Iterable and CollectionsIterable, as it could receive one even when declared with the other
739          elif origin in {Iterable, CollectionsIterable}:
740              self.type_ = get_args(self.type_)[0]
741              self.shape = SHAPE_ITERABLE
742              self.sub_fields = [self._create_sub_type(self.type_, f'{self.name}_type')]
743          elif issubclass(origin, Type):  # type: ignore
744              return
745          elif hasattr(origin, '__get_validators__') or self.model_config.arbitrary_types_allowed:
746              # Is a Pydantic-compatible generic that handles itself
747              # or we have arbitrary_types_allowed = True
748              self.shape = SHAPE_GENERIC
749              self.sub_fields = [self._create_sub_type(t, f'{self.name}_{i}') for i, t in enumerate(get_args(self.type_))]
750              self.type_ = origin
751              return
752          else:
753              raise TypeError(f'Fields of type "{origin}" are not supported.')
754  
755          # type_ has been refined eg. as the type of a List and sub_fields needs to be populated
756          self.sub_fields = [self._create_sub_type(self.type_, '_' + self.name)]
757  
758      def prepare_discriminated_union_sub_fields(self) -> None:
759          """ 
760          Prepare the mapping <discriminator key> -> <ModelField> and update `sub_fields` 
761          Note that this process can be aborted if a `ForwardRef` is encountered 
762          """
763          assert self.discriminator_key is not None
764  
765          if self.type_.__class__ is DeferredType:
766              return
767  
768          assert self.sub_fields is not None
769          sub_fields_mapping: Dict[str, 'ModelField'] = {}
770          all_aliases: Set[str] = set()
771  
772          for sub_field in self.sub_fields:
773              t = sub_field.type_
774              if t.__class__ is ForwardRef:
775                  # Stopping everything...will need to call `update_forward_refs`
776                  return
777  
778              alias, discriminator_values = get_discriminator_alias_and_values(t, self.discriminator_key)
779              all_aliases.add(alias)
780              for discriminator_value in discriminator_values:
781                  sub_fields_mapping[discriminator_value] = sub_field
782  
783          self.sub_fields_mapping = sub_fields_mapping
784          self.discriminator_alias = get_unique_discriminator_alias(all_aliases, self.discriminator_key)
785  
786      def _create_sub_type(self, type_: Type[Any], name: str, *, for_keys: bool = False) -> 'ModelField':
787          if for_keys:
788              class_validators = None
789          else:
790              # validators for sub items should not have `each_item` as we want to check only the first sublevel
791              class_validators = {
792                  k: Validator(
793                      func=v.func,
794                      pre=v.pre,
795                      each_item=False,
796                      always=v.always,
797                      check_fields=v.check_fields,
798                      skip_on_failure=v.skip_on_failure,
799                  )
800                  for k, v in self.class_validators.items()
801                  if v.each_item
802              }
803  
804          field_info, _ = self._get_field_info(name, type_, None, self.model_config)
805  
806          return self.__class__(
807              type_=type_,
808              name=name,
809              class_validators=class_validators,
810              model_config=self.model_config,
811              field_info=field_info,
812          )
813  
814      def populate_validators(self) -> None:
815          """ 
816          Prepare self.pre_validators, self.validators, and self.post_validators based on self.type_'s  __get_validators__ 
817          and class validators. This method should be idempotent, e.g. it should be safe to call multiple times 
818          without mis-configuring the field. 
819          """
820          self.validate_always = getattr(self.type_, 'validate_always', False) or any(
821              v.always for v in self.class_validators.values()
822          )
823  
824          class_validators_ = self.class_validators.values()
825          if not self.sub_fields or self.shape == SHAPE_GENERIC:
826              get_validators = getattr(self.type_, '__get_validators__', None)
827              v_funcs = (
828                  *[v.func for v in class_validators_ if v.each_item and v.pre],
829                  *(get_validators() if get_validators else list(find_validators(self.type_, self.model_config))),
830                  *[v.func for v in class_validators_ if v.each_item and not v.pre],
831              )
832              self.validators = prep_validators(v_funcs)
833  
834          self.pre_validators = []
835          self.post_validators = []
836  
837          if self.field_info and self.field_info.const:
838              self.post_validators.append(make_generic_validator(constant_validator))
839  
840          if class_validators_:
841              self.pre_validators += prep_validators(v.func for v in class_validators_ if not v.each_item and v.pre)
842              self.post_validators += prep_validators(v.func for v in class_validators_ if not v.each_item and not v.pre)
843  
844          if self.parse_json:
845              self.pre_validators.append(make_generic_validator(validate_json))
846  
847          self.pre_validators = self.pre_validators or None
848          self.post_validators = self.post_validators or None
849  
850      def validate(
851          self, v: Any, values: Dict[str, Any], *, loc: 'LocStr', cls: Optional['ModelOrDc'] = None
852      ) -> 'ValidateReturn':
853          assert self.type_.__class__ is not DeferredType
854  
855          if self.type_.__class__ is ForwardRef:
856              assert cls is not None
857              raise ConfigError(
858                  f'field "{self.name}" not yet prepared so type is still a ForwardRef, '
859                  f'you might need to call {cls.__name__}.update_forward_refs().'
860              )
861  
862          errors: Optional['ErrorList']
863          if self.pre_validators:
864              v, errors = self._apply_validators(v, values, loc, cls, self.pre_validators)
865              if errors:
866                  return v, errors
867  
868          if v is None:
869              if is_none_type(self.type_):
870                  # keep validating
871                  pass
872              elif self.allow_none:
873                  if self.post_validators:
874                      return self._apply_validators(v, values, loc, cls, self.post_validators)
875                  else:
876                      return None, None
877              else:
878                  return v, ErrorWrapper(NoneIsNotAllowedError(), loc)
879  
880          if self.shape == SHAPE_SINGLETON:
881              v, errors = self._validate_singleton(v, values, loc, cls)
882          elif self.shape in MAPPING_LIKE_SHAPES:
883              v, errors = self._validate_mapping_like(v, values, loc, cls)
884          elif self.shape == SHAPE_TUPLE:
885              v, errors = self._validate_tuple(v, values, loc, cls)
886          elif self.shape == SHAPE_ITERABLE:
887              v, errors = self._validate_iterable(v, values, loc, cls)
888          elif self.shape == SHAPE_GENERIC:
889              v, errors = self._apply_validators(v, values, loc, cls, self.validators)
890          else:
891              #  sequence, list, set, generator, tuple with ellipsis, frozen set
892              v, errors = self._validate_sequence_like(v, values, loc, cls)
893  
894          if not errors and self.post_validators:
895              v, errors = self._apply_validators(v, values, loc, cls, self.post_validators)
896          return v, errors
897  
898      def _validate_sequence_like(  # noqa: C901 (ignore complexity)
899          self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
900      ) -> 'ValidateReturn':
901          """ 
902          Validate sequence-like containers: lists, tuples, sets and generators 
903          Note that large if-else blocks are necessary to enable Cython 
904          optimization, which is why we disable the complexity check above. 
905          """
906          if not sequence_like(v):
907              e: errors_.PydanticTypeError
908              if self.shape == SHAPE_LIST:
909                  e = errors_.ListError()
910              elif self.shape in (SHAPE_TUPLE, SHAPE_TUPLE_ELLIPSIS):
911                  e = errors_.TupleError()
912              elif self.shape == SHAPE_SET:
913                  e = errors_.SetError()
914              elif self.shape == SHAPE_FROZENSET:
915                  e = errors_.FrozenSetError()
916              else:
917                  e = errors_.SequenceError()
918              return v, ErrorWrapper(e, loc)
919  
920          loc = loc if isinstance(loc, tuple) else (loc,)
921          result = []
922          errors: List[ErrorList] = []
923          for i, v_ in enumerate(v):
924              v_loc = *loc, i
925              r, ee = self._validate_singleton(v_, values, v_loc, cls)
926              if ee:
927                  errors.append(ee)
928              else:
929                  result.append(r)
930  
931          if errors:
932              return v, errors
933  
934          converted: Union[List[Any], Set[Any], FrozenSet[Any], Tuple[Any, ...], Iterator[Any], Deque[Any]] = result
935  
936          if self.shape == SHAPE_SET:
937              converted = set(result)
938          elif self.shape == SHAPE_FROZENSET:
939              converted = frozenset(result)
940          elif self.shape == SHAPE_TUPLE_ELLIPSIS:
941              converted = tuple(result)
942          elif self.shape == SHAPE_DEQUE:
943              converted = deque(result, maxlen=getattr(v, 'maxlen', None))
944          elif self.shape == SHAPE_SEQUENCE:
945              if isinstance(v, tuple):
946                  converted = tuple(result)
947              elif isinstance(v, set):
948                  converted = set(result)
949              elif isinstance(v, Generator):
950                  converted = iter(result)
951              elif isinstance(v, deque):
952                  converted = deque(result, maxlen=getattr(v, 'maxlen', None))
953          return converted, None
954  
955      def _validate_iterable(
956          self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
957      ) -> 'ValidateReturn':
958          """ 
959          Validate Iterables. 
960   
961          This intentionally doesn't validate values to allow infinite generators. 
962          """
963  
964          try:
965              iterable = iter(v)
966          except TypeError:
967              return v, ErrorWrapper(errors_.IterableError(), loc)
968          return iterable, None
969  
970      def _validate_tuple(
971          self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
972      ) -> 'ValidateReturn':
973          e: Optional[Exception] = None
974          if not sequence_like(v):
975              e = errors_.TupleError()
976          else:
977              actual_length, expected_length = len(v), len(self.sub_fields)  # type: ignore
978              if actual_length != expected_length:
979                  e = errors_.TupleLengthError(actual_length=actual_length, expected_length=expected_length)
980  
981          if e:
982              return v, ErrorWrapper(e, loc)
983  
984          loc = loc if isinstance(loc, tuple) else (loc,)
985          result = []
986          errors: List[ErrorList] = []
987          for i, (v_, field) in enumerate(zip(v, self.sub_fields)):  # type: ignore
988              v_loc = *loc, i
989              r, ee = field.validate(v_, values, loc=v_loc, cls=cls)
990              if ee:
991                  errors.append(ee)
992              else:
993                  result.append(r)
994  
995          if errors:
996              return v, errors
997          else:
998              return tuple(result), None

1000     def _validate_mapping_like(
1001         self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
1002     ) -> 'ValidateReturn':
1003         try:
1004             v_iter = dict_validator(v)
1005         except TypeError as exc:
1006             return v, ErrorWrapper(exc, loc)
1007 
1008         loc = loc if isinstance(loc, tuple) else (loc,)
1009         result, errors = {}, []
1010         for k, v_ in v_iter.items():
1011             v_loc = *loc, '__key__'
1012             key_result, key_errors = self.key_field.validate(k, values, loc=v_loc, cls=cls)  # type: ignore
1013             if key_errors:
1014                 errors.append(key_errors)
1015                 continue
1016 
1017             v_loc = *loc, k
1018             value_result, value_errors = self._validate_singleton(v_, values, v_loc, cls)
1019             if value_errors:
1020                 errors.append(value_errors)
1021                 continue
1022 
1023             result[key_result] = value_result
1024         if errors:
1025             return v, errors
1026         elif self.shape == SHAPE_DICT:
1027             return result, None
1028         elif self.shape == SHAPE_DEFAULTDICT:
1029             return defaultdict(self.type_, result), None
1030         elif self.shape == SHAPE_COUNTER:
1031             return CollectionCounter(result), None
1032         else:
1033             return self._get_mapping_value(v, result), None

1035     def _get_mapping_value(self, original: T, converted: Dict[Any, Any]) -> Union[T, Dict[Any, Any]]:
1036         """ 
1037         When type is `Mapping[KT, KV]` (or another unsupported mapping), we try to avoid 
1038         coercing to `dict` unwillingly. 
1039         """
1040         original_cls = original.__class__
1041 
1042         if original_cls == dict or original_cls == Dict:
1043             return converted
1044         elif original_cls in {defaultdict, DefaultDict}:
1045             return defaultdict(self.type_, converted)
1046         else:
1047             try:
1048                 # Counter, OrderedDict, UserDict, ...
1049                 return original_cls(converted)  # type: ignore
1050             except TypeError:
1051                 raise RuntimeError(f'Could not convert dictionary to {original_cls.__name__!r}') from None
1052 
1053     def _validate_singleton(
1054         self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
1055     ) -> 'ValidateReturn':
1056         if self.sub_fields:
1057             if self.discriminator_key is not None:
1058                 return self._validate_discriminated_union(v, values, loc, cls)
1059 
1060             errors = []
1061 
1062             if self.model_config.smart_union and is_union(get_origin(self.type_)):
1063                 # 1st pass: check if the value is an exact instance of one of the Union types
1064                 # (e.g. to avoid coercing a bool into an int)
1065                 for field in self.sub_fields:
1066                     if v.__class__ is field.outer_type_:
1067                         return v, None
1068 
1069                 # 2nd pass: check if the value is an instance of any subclass of the Union types
1070                 for field in self.sub_fields:
1071                     # This whole logic will be improved later on to support more complex `isinstance` checks
1072                     # It will probably be done once a strict mode is added and be something like:
1073                     # ```
1074                     #     value, error = field.validate(v, values, strict=True)
1075                     #     if error is None:
1076                     #         return value, None
1077                     # ```
1078                     try:
1079                         if isinstance(v, field.outer_type_):
1080                             return v, None
1081                     except TypeError:
1082                         # compound type
1083                         if lenient_isinstance(v, get_origin(field.outer_type_)):
1084                             value, error = field.validate(v, values, loc=loc, cls=cls)
1085                             if not error:
1086                                 return value, None
1087 
1088             # 1st pass by default or 3rd pass with `smart_union` enabled:
1089             # check if the value can be coerced into one of the Union types
1090             for field in self.sub_fields:
1091                 value, error = field.validate(v, values, loc=loc, cls=cls)
1092                 if error:
1093                     errors.append(error)
1094                 else:
1095                     return value, None
1096             return v, errors
1097         else:
1098             return self._apply_validators(v, values, loc, cls, self.validators)

1100     def _validate_discriminated_union(
1101         self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc']
1102     ) -> 'ValidateReturn':
1103         assert self.discriminator_key is not None
1104         assert self.discriminator_alias is not None
1105 
1106         try:
1107             try:
1108                 discriminator_value = v[self.discriminator_alias]
1109             except KeyError:
1110                 if self.model_config.allow_population_by_field_name:
1111                     discriminator_value = v[self.discriminator_key]
1112                 else:
1113                     raise
1114         except KeyError:
1115             return v, ErrorWrapper(MissingDiscriminator(discriminator_key=self.discriminator_key), loc)
1116         except TypeError:
1117             try:
1118                 # BaseModel or dataclass
1119                 discriminator_value = getattr(v, self.discriminator_key)
1120             except (AttributeError, TypeError):
1121                 return v, ErrorWrapper(MissingDiscriminator(discriminator_key=self.discriminator_key), loc)
1122 
1123         if self.sub_fields_mapping is None:
1124             assert cls is not None
1125             raise ConfigError(
1126                 f'field "{self.name}" not yet prepared so type is still a ForwardRef, '
1127                 f'you might need to call {cls.__name__}.update_forward_refs().'
1128             )
1129 
1130         try:
1131             sub_field = self.sub_fields_mapping[discriminator_value]
1132         except (KeyError, TypeError):
1133             # KeyError: `discriminator_value` is not in the dictionary.
1134             # TypeError: `discriminator_value` is unhashable.
1135             assert self.sub_fields_mapping is not None
1136             return v, ErrorWrapper(
1137                 InvalidDiscriminator(
1138                     discriminator_key=self.discriminator_key,
1139                     discriminator_value=discriminator_value,
1140                     allowed_values=list(self.sub_fields_mapping),
1141                 ),
1142                 loc,
1143             )
1144         else:
1145             if not isinstance(loc, tuple):
1146                 loc = (loc,)
1147             return sub_field.validate(v, values, loc=(*loc, display_as_type(sub_field.type_)), cls=cls)
1148 
1149     def _apply_validators(
1150         self, v: Any, values: Dict[str, Any], loc: 'LocStr', cls: Optional['ModelOrDc'], validators: 'ValidatorsList'
1151     ) -> 'ValidateReturn':
1152         for validator in validators:
1153             try:
1154                 v = validator(cls, v, values, self, self.model_config)
1155             except (ValueError, TypeError, AssertionError) as exc:
1156                 return v, ErrorWrapper(exc, loc)
1157         return v, None

1159     def is_complex(self) -> bool:
1160         """ 
1161         Whether the field is "complex" eg. env variables should be parsed as JSON. 
1162         """
1163         from pydantic.v1.main import BaseModel
1164 
1165         return (
1166             self.shape != SHAPE_SINGLETON
1167             or hasattr(self.type_, '__pydantic_model__')
1168             or lenient_issubclass(self.type_, (BaseModel, list, set, frozenset, dict))
1169         )

1171     def _type_display(self) -> PyObjectStr:
1172         t = display_as_type(self.type_)
1173 
1174         if self.shape in MAPPING_LIKE_SHAPES:
1175             t = f'Mapping[{display_as_type(self.key_field.type_)}, {t}]'  # type: ignore
1176         elif self.shape == SHAPE_TUPLE:
