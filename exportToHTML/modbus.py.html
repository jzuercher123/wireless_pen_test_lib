366  class ModbusPDU14ReadFileRecordRequest(Packet):
367      name = "Read File Record"
368      fields_desc = [XByteField("funcCode", 0x14),
369                     ByteField("byteCount", None)]
370  
371      def guess_payload_class(self, payload):
372          if self.byteCount > 0:
373              return ModbusReadFileSubRequest
374          else:
375              return Packet.guess_payload_class(self, payload)
376  
377      def post_build(self, p, pay):
378          if self.byteCount is None:
379              tmp_len = len(pay)
380              p = p[:1] + struct.pack("!B", tmp_len) + p[3:]
381          return p + pay

855  class ModbusADURequest(Packet):
856      name = "ModbusADU"
857      fields_desc = [
858          # needs to be unique
859          XShortField("transId", 0x0000),
860          # needs to be zero (Modbus)
861          XShortField("protoId", 0x0000),
862          # is calculated with payload
863          ShortField("len", None),
864          # 0xFF (recommended as non-significant value) or 0x00
865          XByteField("unitId", 0xff),
866      ]
867  
868      def guess_payload_class(self, payload):
869          function_code = orb(payload[0])
870  
871          if function_code == 0x2B:
872              sub_code = orb(payload[1])
873              try:
874                  return _mei_types_request[sub_code]
875              except KeyError:
876                  pass
877          try:
878              return _modbus_request_classes[function_code]
879          except KeyError:
880              pass
881          if function_code in _reserved_funccode_request:
882              return ModbusPDUReservedFunctionCodeRequest
883          return ModbusPDUUserDefinedFunctionCodeRequest
884  
885      def post_build(self, p, pay):
886          if self.len is None:
887              tmp_len = len(pay) + 1  # +len(p)
888              p = p[:4] + struct.pack("!H", tmp_len) + p[6:]
889          return p + pay

892  class ModbusADUResponse(Packet):
893      name = "ModbusADU"
894      fields_desc = [
895          # needs to be unique
896          XShortField("transId", 0x0000),
897          # needs to be zero (Modbus)
898          XShortField("protoId", 0x0000),
899          # is calculated with payload
900          ShortField("len", None),
901          # 0xFF or 0x00 should be used for Modbus over TCP/IP
902          XByteField("unitId", 0xff),
903      ]
904  
905      def guess_payload_class(self, payload):
906          function_code = orb(payload[0])
907  
908          if function_code == 0x2B:
909              sub_code = orb(payload[1])
910              try:
911                  return _mei_types_response[sub_code]
912              except KeyError:
913                  pass
914          try:
915              return _modbus_response_classes[function_code]
916          except KeyError:
917              pass
918          try:
919              return _modbus_error_classes[function_code]
920          except KeyError:
921              pass
922          if function_code in _reserved_funccode_response:
923              return ModbusPDUReservedFunctionCodeResponse
924          elif function_code in _reserved_funccode_error:
925              return ModbusPDUReservedFunctionCodeError
926          if function_code < 0x80:
927              return ModbusPDUUserDefinedFunctionCodeResponse
928          return ModbusPDUUserDefinedFunctionCodeError
929  
930      def post_build(self, p, pay):
931          if self.len is None:
932              tmp_len = len(pay) + 1  # +len(p)
933              p = p[:4] + struct.pack("!H", tmp_len) + p[6:]
934          return p + pay
