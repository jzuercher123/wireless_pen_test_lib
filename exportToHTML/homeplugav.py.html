493  class ModuleOperationRequest(Packet):
494      name = "ModuleOperationRequest"
495      fields_desc = [XIntField("reserved", 0),
496                     XByteField("NumOpData", 0x01),
497                     LEShortEnumField("operation", 0x0000, OperationList),
498                     LEShortField("OPDataLength", None),
499                     XIntField("reserved_1", 0),
500                     ConditionalField(LEIntField("SessionID", 0),
501                                      lambda pkt:(0x0011 == pkt.operation)),
502                     ConditionalField(XByteField("ModuleIDX", 0),
503                                      lambda pkt:(0x0011 == pkt.operation)),
504                     LEShortField("ModuleID", 0x7002),
505                     LEShortField("ModuleSubID", 0x0000),
506                     ConditionalField(LEShortField("ReadDataLen", 0x0578),
507                                      lambda pkt:(0x0000 == pkt.operation)),
508                     ConditionalField(LEIntField("ReadOffset", 0x00000000),
509                                      lambda pkt:(0x0000 == pkt.operation)),
510                     ConditionalField(FieldLenField("WriteDataLen", None,
511                                                    count_of="ModuleData",
512                                                    fmt="<H"),
513                                      lambda pkt:(0x0011 == pkt.operation)),
514                     ConditionalField(LEIntField("WriteOffset", 0x00000000),
515                                      lambda pkt:(0x0011 == pkt.operation)),
516                     ConditionalField(StrLenField("ModuleData", b"\x00",
517                                                  length_from=lambda pkt: pkt.WriteDataLen),  # noqa: E501
518                                      lambda pkt:(0x0011 == pkt.operation))]
519  
520      def post_build(self, p, pay):
521          if self.operation == 0x0000:
522              if self.OPDataLength is None:
523                  _len = 18
524                  p = p[:7] + struct.pack('!H', _len) + p[9:]
525          if self.operation == 0x0011:
526              if self.OPDataLength is None:
527                  _len = 23 + len(self.ModuleData)
528                  p = p[:7] + struct.pack('!H', _len) + p[9:]
529              if self.WriteDataLen is None:
530                  _len = len(self.ModuleData)
531                  p = p[:22] + struct.pack('!H', _len) + p[24:]
532          return p + pay

591      cksum = 0
592      for i in range(0, len(data), 4):
593          cksum = (cksum ^ struct.unpack('<I', data[i:i + 4])[0]) & 0xffffffff
594      return (~cksum) & 0xffffffff
