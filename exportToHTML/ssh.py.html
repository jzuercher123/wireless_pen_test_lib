39   class _SSHHeaderField(FieldListField):
40       def getfield(self, pkt, s):
41           val = []
42           while s:
43               s, v = self.field.getfield(pkt, s)
44               val.append(v)
45               if v[:4] == b"SSH-":
46                   return s, val
47           return s, val

56   class _ComaStrField(StrLenField):
57       islist = 1
58   
59       def m2i(self, pkt, x):
60           return super(_ComaStrField, self).m2i(pkt, x).split(b",")
61   
62       def i2m(self, pkt, x):
63           return super(_ComaStrField, self).i2m(pkt, b",".join(x))

76   class SSHPacketStringField(PacketField):
77       __slots__ = ["sub_cls"]
78   
79       def __init__(self, name, sub_cls):
80           self.sub_cls = sub_cls
81           super(SSHPacketStringField, self).__init__(name, SSHString(), SSHString)
82   
83       def m2i(self, pkt, x):
84           x = super(SSHPacketStringField, self).m2i(pkt, x)
85           x.value = self.sub_cls(x.value)
86           return x

157  class SSH(Packet):
158      name = "SSH - Binary Packet"
159      fields_desc = [
160          IntField("packet_length", None),
161          ByteField("padding_length", None),
162          PacketLenField(
163              "pay",
164              None,
165              _SSHPayload,
166              length_from=lambda pkt: pkt.packet_length - pkt.padding_length - 1,
167          ),
168          StrLenField("random_padding", b"", length_from=lambda pkt: pkt.padding_length),
169          # StrField("mac", b""),
170      ]
171  
172      @classmethod
173      def dispatch_hook(cls, _pkt=None, *args, **kargs):
174          if _pkt and len(_pkt) >= 4 and _pkt[:4] == b"SSH-":
175              return SSHVersionExchange
176          return cls
177  
178      def mysummary(self):
179          if self.pay:
180              if isinstance(self.pay, conf.raw_layer):
181                  return "SSH type " + str(self.pay.load[0]), [TCP, SSH]
182              return "SSH " + self.pay.sprintf("%type%"), [TCP, SSH]
183          return "SSH", [TCP, SSH]

189  class SSHVersionExchange(Packet):
190      name = "SSH - Protocol Version Exchange"
191      fields_desc = [
192          _SSHHeaderField(
193              "lines",
194              [],
195              StrCRLFField("", b""),
196          )
197      ]
198  
199      def mysummary(self):
200          return "SSH - Version Exchange %s" % plain_str(self.lines[-1]), [TCP]
