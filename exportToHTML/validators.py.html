59   def str_validator(v: Any) -> Union[str]:
60       if isinstance(v, str):
61           if isinstance(v, Enum):
62               return v.value
63           else:
64               return v
65       elif isinstance(v, (float, int, Decimal)):
66           # is there anything else we want to add here? If you think so, create an issue.
67           return str(v)
68       elif isinstance(v, (bytes, bytearray)):
69           return v.decode()
70       else:
71           raise errors.StrError()

106  def bool_validator(v: Any) -> bool:
107      if v is True or v is False:
108          return v
109      if isinstance(v, bytes):
110          v = v.decode()
111      if isinstance(v, str):
112          v = v.lower()
113      try:
114          if v in BOOL_TRUE:
115              return True
116          if v in BOOL_FALSE:
117              return False
118      except TypeError:
119          raise errors.BoolError()
120      raise errors.BoolError()

127  def int_validator(v: Any) -> int:
128      if isinstance(v, int) and not (v is True or v is False):
129          return v
130  
131      # see https://github.com/pydantic/pydantic/issues/1477 and in turn, https://github.com/python/cpython/issues/95778
132      # this check should be unnecessary once patch releases are out for 3.7, 3.8, 3.9 and 3.10
133      # but better to check here until then.
134      # NOTICE: this does not fully protect user from the DOS risk since the standard library JSON implementation
135      # (and other std lib modules like xml) use `int()` and are likely called before this, the best workaround is to
136      # 1. update to the latest patch release of python once released, 2. use a different JSON library like ujson
137      if isinstance(v, (str, bytes, bytearray)) and len(v) > max_str_int:
138          raise errors.IntegerError()
139  
140      try:
141          return int(v)
142      except (TypeError, ValueError, OverflowError):
143          raise errors.IntegerError()

170      if allow_inf_nan is None:
171          allow_inf_nan = config.allow_inf_nan

174          raise errors.NumberNotFiniteError()

178  def number_multiple_validator(v: 'Number', field: 'ModelField') -> 'Number':
179      field_type: ConstrainedNumber = field.type_
180      if field_type.multiple_of is not None:
181          mod = float(v) / float(field_type.multiple_of) % 1
182          if not almost_equal_floats(mod, 0.0) and not almost_equal_floats(mod, 1.0):
183              raise errors.NumberNotMultipleError(multiple_of=field_type.multiple_of)
184      return v

187  def number_size_validator(v: 'Number', field: 'ModelField') -> 'Number':
188      field_type: ConstrainedNumber = field.type_
189      if field_type.gt is not None and not v > field_type.gt:
190          raise errors.NumberNotGtError(limit_value=field_type.gt)
191      elif field_type.ge is not None and not v >= field_type.ge:
192          raise errors.NumberNotGeError(limit_value=field_type.ge)
193  
194      if field_type.lt is not None and not v < field_type.lt:
195          raise errors.NumberNotLtError(limit_value=field_type.lt)
196      if field_type.le is not None and not v <= field_type.le:
197          raise errors.NumberNotLeError(limit_value=field_type.le)
198  
199      return v

315  def uuid_validator(v: Any, field: 'ModelField') -> UUID:
316      try:
317          if isinstance(v, str):
318              v = UUID(v)
319          elif isinstance(v, (bytes, bytearray)):
320              try:
321                  v = UUID(v.decode())
322              except ValueError:
323                  # 16 bytes in big-endian order as the bytes argument fail
324                  # the above check
325                  v = UUID(bytes=v)
326      except ValueError:
327          raise errors.UUIDError()
328  
329      if not isinstance(v, UUID):
330          raise errors.UUIDError()
331  
332      required_version = getattr(field.type_, '_required_version', None)
333      if required_version and v.version != required_version:
334          raise errors.UUIDVersionError(required_version=required_version)
335  
336      return v

480  def make_literal_validator(type_: Any) -> Callable[[Any], Any]:
481      permitted_choices = all_literal_values(type_)
482  
483      # To have a O(1) complexity and still return one of the values set inside the `Literal`,
484      # we create a dict with the set values (a set causes some problems with the way intersection works).
485      # In some cases the set value and checked value can indeed be different (see `test_literal_validator_str_enum`)
486      allowed_choices = {v: v for v in permitted_choices}
487  
488      def literal_validator(v: Any) -> Any:
489          try:
490              return allowed_choices[v]
491          except (KeyError, TypeError):
492              raise errors.WrongConstantError(given=v, permitted=permitted_choices)
493  
494      return literal_validator

594  def make_namedtuple_validator(
595      namedtuple_cls: Type[NamedTupleT], config: Type['BaseConfig']
596  ) -> Callable[[Tuple[Any, ...]], NamedTupleT]:
597      from pydantic.v1.annotated_types import create_model_from_namedtuple
598  
599      NamedTupleModel = create_model_from_namedtuple(
600          namedtuple_cls,
601          __config__=config,
602          __module__=namedtuple_cls.__module__,
603      )
604      namedtuple_cls.__pydantic_model__ = NamedTupleModel  # type: ignore[attr-defined]
605  
606      def namedtuple_validator(values: Tuple[Any, ...]) -> NamedTupleT:
607          annotations = NamedTupleModel.__annotations__
608  
609          if len(values) > len(annotations):
610              raise errors.ListMaxLengthError(limit_value=len(annotations))
611  
612          dict_values: Dict[str, Any] = dict(zip(annotations, values))
613          validated_dict_values: Dict[str, Any] = dict(NamedTupleModel(**dict_values))
614          return namedtuple_cls(**validated_dict_values)
615  
616      return namedtuple_validator

637  class IfConfig:
638      def __init__(self, validator: AnyCallable, *config_attr_names: str, ignored_value: Any = False) -> None:
639          self.validator = validator
640          self.config_attr_names = config_attr_names
641          self.ignored_value = ignored_value
642  
643      def check(self, config: Type['BaseConfig']) -> bool:
644          return any(getattr(config, name) not in {None, self.ignored_value} for name in self.config_attr_names)

698  def find_validators(  # noqa: C901 (ignore complexity)
699      type_: Type[Any], config: Type['BaseConfig']
700  ) -> Generator[AnyCallable, None, None]:
701      from pydantic.v1.dataclasses import is_builtin_dataclass, make_dataclass_validator
702  
703      if type_ is Any or type_ is object:
704          return
705      type_type = type_.__class__
706      if type_type == ForwardRef or type_type == TypeVar:
707          return
708  
709      if is_none_type(type_):
710          yield none_validator
711          return
712      if type_ is Pattern or type_ is re.Pattern:
713          yield pattern_validator
714          return
715      if type_ is Hashable or type_ is CollectionsHashable:
716          yield hashable_validator
717          return
718      if is_callable_type(type_):
719          yield callable_validator
720          return
721      if is_literal_type(type_):
722          yield make_literal_validator(type_)
723          return
724      if is_builtin_dataclass(type_):
725          yield from make_dataclass_validator(type_, config)
726          return
727      if type_ is Enum:
728          yield enum_validator
729          return
730      if type_ is IntEnum:
731          yield int_enum_validator
732          return
733      if is_namedtuple(type_):
734          yield tuple_validator
735          yield make_namedtuple_validator(type_, config)
736          return
737      if is_typeddict(type_):
738          yield make_typeddict_validator(type_, config)
739          return
740  
741      class_ = get_class(type_)
742      if class_ is not None:
743          if class_ is not Any and isinstance(class_, type):
744              yield make_class_validator(class_)
745          else:
746              yield any_class_validator
747          return
748  
749      for val_type, validators in _VALIDATORS:
750          try:
751              if issubclass(type_, val_type):
752                  for v in validators:
753                      if isinstance(v, IfConfig):
754                          if v.check(config):
755                              yield v.validator
756                      else:
757                          yield v
758                  return
759          except TypeError:
760              raise RuntimeError(f'error checking inheritance of {type_!r} (type: {display_as_type(type_)})')
761  
762      if config.arbitrary_types_allowed:
763          yield make_arbitrary_type_validator(type_)
764      else:
765          raise RuntimeError(f'no validator found for {type_}, see `arbitrary_types_allowed` in Config')
