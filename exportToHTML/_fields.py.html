38   def pydantic_general_metadata(**metadata: Any) -> BaseMetadata:
39       """Create a new `_PydanticGeneralMetadata` class with the given metadata. 
40    
41       Args: 
42           **metadata: The metadata to add. 
43    
44       Returns: 
45           The new `_PydanticGeneralMetadata` class.

47       return _general_metadata_cls()(metadata)  # type: ignore

51   def _general_metadata_cls() -> type[BaseMetadata]:
52       """Do it this way to avoid importing `annotated_types` at import time."""
53       from annotated_types import BaseMetadata
54   
55       class _PydanticGeneralMetadata(PydanticMetadata, BaseMetadata):
56           """Pydantic general metadata like `max_digits`."""
57   
58           def __init__(self, metadata: Any):
59               self.__dict__ = metadata

64   def _update_fields_from_docstrings(cls: type[Any], fields: dict[str, FieldInfo], config_wrapper: ConfigWrapper) -> None:
65       if config_wrapper.use_attribute_docstrings:
66           fields_docs = extract_docstrings_from_cls(cls)
67           for ann_name, field_info in fields.items():
68               if field_info.description is None and ann_name in fields_docs:
69                   field_info.description = fields_docs[ann_name]

72   def collect_model_fields(  # noqa: C901
73       cls: type[BaseModel],
74       bases: tuple[type[Any], ...],
75       config_wrapper: ConfigWrapper,
76       types_namespace: dict[str, Any] | None,
77       *,
78       typevars_map: dict[Any, Any] | None = None,
79   ) -> tuple[dict[str, FieldInfo], set[str]]:
80       """Collect the fields of a nascent pydantic model. 
81    
82       Also collect the names of any ClassVars present in the type hints. 
83    
84       The returned value is a tuple of two items: the fields dict, and the set of ClassVar names. 
85    
86       Args: 
87           cls: BaseModel or dataclass. 
88           bases: Parents of the class, generally `cls.__bases__`. 
89           config_wrapper: The config wrapper instance. 
90           types_namespace: Optional extra namespace to look for types in. 
91           typevars_map: A dictionary mapping type variables to their concrete types. 
92    
93       Returns: 
94           A tuple contains fields and class variables. 
95    
96       Raises: 
97           NameError: 
98               - If there is a conflict between a field name and protected namespaces. 
99               - If there is a field other than `root` in `RootModel`. 
100              - If a field shadows an attribute in the parent model. 
101      """
102      BaseModel = import_cached_base_model()
103      FieldInfo_ = import_cached_field_info()
104  
105      type_hints = get_cls_type_hints_lenient(cls, types_namespace)
106  
107      # https://docs.python.org/3/howto/annotations.html#accessing-the-annotations-dict-of-an-object-in-python-3-9-and-older
108      # annotations is only used for finding fields in parent classes
109      annotations = cls.__dict__.get('__annotations__', {})
110      fields: dict[str, FieldInfo] = {}
111  
112      class_vars: set[str] = set()
113      for ann_name, ann_type in type_hints.items():
114          if ann_name == 'model_config':
115              # We never want to treat `model_config` as a field
116              # Note: we may need to change this logic if/when we introduce a `BareModel` class with no
117              # protected namespaces (where `model_config` might be allowed as a field name)
118              continue
119          for protected_namespace in config_wrapper.protected_namespaces:
120              if ann_name.startswith(protected_namespace):
121                  for b in bases:
122                      if hasattr(b, ann_name):
123                          if not (issubclass(b, BaseModel) and ann_name in b.model_fields):
124                              raise NameError(
125                                  f'Field "{ann_name}" conflicts with member {getattr(b, ann_name)}'
126                                  f' of protected namespace "{protected_namespace}".'
127                              )
128                  else:
129                      valid_namespaces = tuple(
130                          x for x in config_wrapper.protected_namespaces if not ann_name.startswith(x)
131                      )
132                      warnings.warn(
133                          f'Field "{ann_name}" in {cls.__name__} has conflict with protected namespace "{protected_namespace}".'
134                          '\n\nYou may be able to resolve this warning by setting'
135                          f" `model_config['protected_namespaces'] = {valid_namespaces}`.",
136                          UserWarning,
137                      )
138          if is_classvar(ann_type):
139              class_vars.add(ann_name)
140              continue
141          if _is_finalvar_with_default_val(ann_type, getattr(cls, ann_name, PydanticUndefined)):
142              class_vars.add(ann_name)
143              continue
144          if not is_valid_field_name(ann_name):
145              continue
146          if cls.__pydantic_root_model__ and ann_name != 'root':
147              raise NameError(
148                  f"Unexpected field with name {ann_name!r}; only 'root' is allowed as a field of a `RootModel`"
149              )
150  
151          # when building a generic model with `MyModel[int]`, the generic_origin check makes sure we don't get
152          # "... shadows an attribute" warnings
153          generic_origin = getattr(cls, '__pydantic_generic_metadata__', {}).get('origin')
154          for base in bases:
155              dataclass_fields = {
156                  field.name for field in (dataclasses.fields(base) if dataclasses.is_dataclass(base) else ())
157              }
158              if hasattr(base, ann_name):
159                  if base is generic_origin:
160                      # Don't warn when "shadowing" of attributes in parametrized generics
161                      continue
162  
163                  if ann_name in dataclass_fields:
164                      # Don't warn when inheriting stdlib dataclasses whose fields are "shadowed" by defaults being set
165                      # on the class instance.
166                      continue
167  
168                  if ann_name not in annotations:
169                      # Don't warn when a field exists in a parent class but has not been defined in the current class
170                      continue
171  
172                  warnings.warn(
173                      f'Field name "{ann_name}" in "{cls.__qualname__}" shadows an attribute in parent '
174                      f'"{base.__qualname__}"',
175                      UserWarning,
176                  )
177  
178          try:
179              default = getattr(cls, ann_name, PydanticUndefined)
180              if default is PydanticUndefined:
181                  raise AttributeError
182          except AttributeError:
183              if ann_name in annotations:
184                  field_info = FieldInfo_.from_annotation(ann_type)
185              else:
186                  # if field has no default value and is not in __annotations__ this means that it is
187                  # defined in a base class and we can take it from there
188                  model_fields_lookup: dict[str, FieldInfo] = {}
189                  for x in cls.__bases__[::-1]:
190                      model_fields_lookup.update(getattr(x, 'model_fields', {}))
191                  if ann_name in model_fields_lookup:
192                      # The field was present on one of the (possibly multiple) base classes
193                      # copy the field to make sure typevar substitutions don't cause issues with the base classes
194                      field_info = copy(model_fields_lookup[ann_name])
195                  else:
196                      # The field was not found on any base classes; this seems to be caused by fields not getting
197                      # generated thanks to models not being fully defined while initializing recursive models.
198                      # Nothing stops us from just creating a new FieldInfo for this type hint, so we do this.
199                      field_info = FieldInfo_.from_annotation(ann_type)
200          else:
201              _warn_on_nested_alias_in_annotation(ann_type, ann_name)
202              field_info = FieldInfo_.from_annotated_attribute(ann_type, default)
203              # attributes which are fields are removed from the class namespace:
204              # 1. To match the behaviour of annotation-only fields
205              # 2. To avoid false positives in the NameError check above
206              try:
207                  delattr(cls, ann_name)
208              except AttributeError:
209                  pass  # indicates the attribute was on a parent class
210  
211          # Use cls.__dict__['__pydantic_decorators__'] instead of cls.__pydantic_decorators__
212          # to make sure the decorators have already been built for this exact class
213          decorators: DecoratorInfos = cls.__dict__['__pydantic_decorators__']
214          if ann_name in decorators.computed_fields:
215              raise ValueError("you can't override a field with a computed field")
216          fields[ann_name] = field_info
217  
218      if typevars_map:
219          for field in fields.values():
220              field.apply_typevars_map(typevars_map, types_namespace)
221  
222      _update_fields_from_docstrings(cls, fields, config_wrapper)
223  
224      return fields, class_vars

227  def _warn_on_nested_alias_in_annotation(ann_type: type[Any], ann_name: str) -> None:
228      FieldInfo = import_cached_field_info()
229  
230      args = getattr(ann_type, '__args__', None)
231      if args:
232          for anno_arg in args:
233              if _typing_extra.is_annotated(anno_arg):
234                  for anno_type_arg in _typing_extra.get_args(anno_arg):
235                      if isinstance(anno_type_arg, FieldInfo) and anno_type_arg.alias is not None:
236                          warnings.warn(
237                              f'`alias` specification on field "{ann_name}" must be set on outermost annotation to take effect.',
238                              UserWarning,
239                          )
240                          return

243  def _is_finalvar_with_default_val(type_: type[Any], val: Any) -> bool:
244      FieldInfo = import_cached_field_info()
245  
246      if not is_finalvar(type_):
247          return False
248      elif val is PydanticUndefined:
249          return False
250      elif isinstance(val, FieldInfo) and (val.default is PydanticUndefined and val.default_factory is None):
251          return False
252      else:
253          return True

256  def collect_dataclass_fields(
257      cls: type[StandardDataclass],
258      types_namespace: dict[str, Any] | None,
259      *,
260      typevars_map: dict[Any, Any] | None = None,
261      config_wrapper: ConfigWrapper | None = None,
262  ) -> dict[str, FieldInfo]:
263      """Collect the fields of a dataclass. 
264   
265      Args: 
266          cls: dataclass. 
267          types_namespace: Optional extra namespace to look for types in. 
268          typevars_map: A dictionary mapping type variables to their concrete types. 
269          config_wrapper: The config wrapper instance. 
270   
271      Returns: 
272          The dataclass fields. 
273      """
274      FieldInfo_ = import_cached_field_info()
275  
276      fields: dict[str, FieldInfo] = {}
277      dataclass_fields: dict[str, dataclasses.Field] = cls.__dataclass_fields__
278      cls_localns = dict(vars(cls))  # this matches get_cls_type_hints_lenient, but all tests pass with `= None` instead
279  
280      source_module = sys.modules.get(cls.__module__)
281      if source_module is not None:
282          types_namespace = {**source_module.__dict__, **(types_namespace or {})}
283  
284      for ann_name, dataclass_field in dataclass_fields.items():
285          ann_type = _typing_extra.eval_type_lenient(dataclass_field.type, types_namespace, cls_localns)
286          if is_classvar(ann_type):
287              continue
288  
289          if (
290              not dataclass_field.init
291              and dataclass_field.default == dataclasses.MISSING
292              and dataclass_field.default_factory == dataclasses.MISSING
293          ):
294              # TODO: We should probably do something with this so that validate_assignment behaves properly
295              #   Issue: https://github.com/pydantic/pydantic/issues/5470
296              continue
297  
298          if isinstance(dataclass_field.default, FieldInfo_):
299              if dataclass_field.default.init_var:
300                  if dataclass_field.default.init is False:
301                      raise PydanticUserError(
302                          f'Dataclass field {ann_name} has init=False and init_var=True, but these are mutually exclusive.',
303                          code='clashing-init-and-init-var',
304                      )
305  
306                  # TODO: same note as above re validate_assignment
307                  continue
308              field_info = FieldInfo_.from_annotated_attribute(ann_type, dataclass_field.default)
309          else:
310              field_info = FieldInfo_.from_annotated_attribute(ann_type, dataclass_field)
311  
312          fields[ann_name] = field_info
313  
314          if field_info.default is not PydanticUndefined and isinstance(getattr(cls, ann_name, field_info), FieldInfo_):
315              # We need this to fix the default when the "default" from __dataclass_fields__ is a pydantic.FieldInfo
316              setattr(cls, ann_name, field_info.default)
317  
318      if typevars_map:
319          for field in fields.values():
320              field.apply_typevars_map(typevars_map, types_namespace)
321  
322      if config_wrapper is not None:
323          _update_fields_from_docstrings(cls, fields, config_wrapper)
324  
325      return fields
