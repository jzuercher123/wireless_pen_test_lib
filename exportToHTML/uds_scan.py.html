91   class UDS_DSCEnumerator(UDS_Enumerator, StateGeneratingServiceEnumerator):
92       _description = "Available sessions"
93       _supported_kwargs = copy.copy(ServiceEnumerator._supported_kwargs)
94       _supported_kwargs.update({
95           'delay_state_change': (int, lambda x: x >= 0),
96           'overwrite_timeout': (bool, None)
97       })
98       _supported_kwargs["scan_range"] = (
99           (list, tuple, range), lambda x: max(x) < 0x100 and min(x) >= 0)
100  
101      _supported_kwargs_doc = ServiceEnumerator._supported_kwargs_doc + """ 
102          :param int delay_state_change: Specifies an additional delay after 
103                                         after a session is modified from 
104                                         the transition function. In unit-test 
105                                         scenarios, this delay should be set to 
106                                         zero. 
107          :param bool overwrite_timeout: True by default. This enumerator 
108                                         overwrites the timeout argument, since 
109                                         most ECUs take some time until a session 
110                                         is changed. This ensures that more 
111                                         results are gathered by default. In 
112                                         unit-test scenarios, this value should 
113                                         be set to False, in order to use the 
114                                         timeout specified by the 'timeout' 
115                                         argument."""

117      def _get_initial_requests(self, **kwargs):
118          # type: (Any) -> Iterable[Packet]
119          session_range = kwargs.pop("scan_range", range(2, 0x100))
120          return UDS() / UDS_DSC(diagnosticSessionType=session_range)
121  
122      def execute(self, socket, state, **kwargs):
123          # type: (_SocketUnion, EcuState, Any) -> None
124  
125          # fix configuration in kwargs to avoid overwrite from user
126          kwargs["exit_if_service_not_supported"] = False
127          kwargs["retry_if_busy_returncode"] = False
128  
129          # Apply a fixed timeout for this execute.
130          # Unit-tests may want to overwrite the timeout to speed up testing
131          if kwargs.pop("overwrite_timeout", True):
132              kwargs["timeout"] = 3
133  
134          super(UDS_DSCEnumerator, self).execute(socket, state, **kwargs)
135  
136      execute.__doc__ = _supported_kwargs_doc
137  
138      def _get_table_entry_y(self, tup):
139          # type: (_AutomotiveTestCaseScanResult) -> str
140          return "0x%02x: %s" % (
141              tup[1].diagnosticSessionType,
142              tup[1].sprintf("%UDS_DSC.diagnosticSessionType%"))

144      @staticmethod
145      def enter_state(socket,  # type: _SocketUnion
146                      configuration,  # type: AutomotiveTestCaseExecutorConfiguration  # noqa: E501
147                      request  # type: Packet
148                      ):  # type: (...) -> bool
149          try:
150              timeout = configuration[UDS_DSCEnumerator.__name__]["timeout"]
151          except KeyError:
152              timeout = 3
153          ans = socket.sr1(request, timeout=timeout, verbose=False)
154          if ans is not None:
155              if configuration.verbose:
156                  log_automotive.debug(
157                      "Try to enter session req: %s, resp: %s" %
158                      (repr(request), repr(ans)))
159              return cast(int, ans.service) != 0x7f
160          else:
161              return False
162  
163      def get_new_edge(self,
164                       socket,  # type: _SocketUnion
165                       config  # type: AutomotiveTestCaseExecutorConfiguration
166                       ):  # type: (...) -> Optional[_Edge]
167          edge = super(UDS_DSCEnumerator, self).get_new_edge(socket, config)
168          if edge:
169              state, new_state = edge
170              # Force TesterPresent if session is changed
171              new_state.tp = 1  # type: ignore
172              return state, new_state
173          return None
174  
175      @staticmethod
176      def enter_state_with_tp(sock,  # type: _SocketUnion
177                              conf,  # type: AutomotiveTestCaseExecutorConfiguration  # noqa: E501
178                              kwargs  # type: Dict[str, Any]
179                              ):  # type: (...) -> bool
180          UDS_TPEnumerator.enter(sock, conf, kwargs)
181          # Wait 5 seconds, since some ECUs require time
182          # to switch to the bootloader
183          try:
184              delay = conf[UDS_DSCEnumerator.__name__]["delay_state_change"]
185          except KeyError:
186              delay = 5
187          conf.stop_event.wait(delay)
188          state_changed = UDS_DSCEnumerator.enter_state(
189              sock, conf, kwargs["req"])
190          if not state_changed:
191              UDS_TPEnumerator.cleanup(sock, conf)
192          return state_changed
193  
194      def get_transition_function(self, socket, edge):
195          # type: (_SocketUnion, _Edge) -> Optional[_TransitionTuple]
196          return UDS_DSCEnumerator.enter_state_with_tp, {
197              "req": self._results[-1].req,
198              "desc": "DSC=%d" % self._results[-1].req.diagnosticSessionType
199          }, UDS_TPEnumerator.cleanup

202  class UDS_TPEnumerator(UDS_Enumerator, StateGeneratingServiceEnumerator):
203      _description = "TesterPresent supported"
204  
205      def _get_initial_requests(self, **kwargs):
206          # type: (Any) -> Iterable[Packet]
207          return [UDS() / UDS_TP()]
208  
209      def _get_table_entry_y(self, tup):
210          # type: (_AutomotiveTestCaseScanResult) -> str
211          return "TesterPresent:"
212  
213      @staticmethod
214      def enter(socket,  # type: _SocketUnion
215                configuration,  # type: AutomotiveTestCaseExecutorConfiguration
216                _  # type: Dict[str, Any]
217                ):  # type: (...) -> bool
218          if configuration.unittest:
219              configuration["tps"] = None
220              socket.sr1(UDS() / UDS_TP(), timeout=0.1, verbose=False)
221              return True
222  
223          UDS_TPEnumerator.cleanup(socket, configuration)
224          configuration["tps"] = UDS_TesterPresentSender(socket, interval=3)
225          configuration["tps"].start()
226          return True
227  
228      @staticmethod
229      def cleanup(_, configuration):
230          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration) -> bool
231          try:
232              configuration["tps"].stop()
233              configuration["tps"] = None
234          except (AttributeError, KeyError):
235              pass
236              # log_automotive.debug("Cleanup TP-Sender Error: %s", e)
237          return True
238  
239      def get_transition_function(self, socket, edge):
240          # type: (_SocketUnion, _Edge) -> Optional[_TransitionTuple]
241          return self.enter, {"desc": "TP"}, self.cleanup

307      _description = "Available services and negative response per state"
308      _supported_kwargs = copy.copy(ServiceEnumerator._supported_kwargs)

316          :param int request_length: Specifies the maximum length of arequest 
317                                     packet. The enumerator will generate all 
318                                     packets from a length of 1 (UDS Service

320                                     `request_length`."""

328      def _get_initial_requests(self, **kwargs):
329          # type: (Any) -> Iterable[Packet]
330          # Only generate services with unset positive response bit (0x40) as
331          # default scan_range
332          scan_range = kwargs.pop("scan_range",
333                                  (x for x in range(0x100) if not x & 0x40))
334          request_length = kwargs.pop("request_length", 1)
335          return itertools.chain.from_iterable(
336              ([UDS(service=x) / Raw(b"\x00" * req_len)
337                for req_len in range(request_length)] for x in scan_range))
338  
339      def _evaluate_response(self,
340                             state,  # type: EcuState
341                             request,  # type: Packet
342                             response,  # type: Optional[Packet]
343                             **kwargs  # type: Optional[Dict[str, Any]]
344                             ):  # type: (...) -> bool
345          if response and response.service == 0x51:
346              log_automotive.warning(
347                  "ECUResetPositiveResponse detected! This might have changed "
348                  "the state of the ECU under test.")
349  
350          # remove args from kwargs since they will be overwritten
351          kwargs["exit_if_service_not_supported"] = False  # type: ignore
352  
353          return super(UDS_ServiceEnumerator, self)._evaluate_response(
354              state, request, response, **kwargs)

389  class UDS_RDBISelectiveEnumerator(StagedAutomotiveTestCase):
390      @staticmethod
391      def __connector_rnd_to_seq(rdbi_random,  # type: AutomotiveTestCaseABC
392                                 _  # type: AutomotiveTestCaseABC
393                                 ):  # type: (...) -> Dict[str, Any]
394          rdbi_random = cast(UDS_Enumerator, rdbi_random)
395          identifiers_with_positive_response = \
396              [p.resp.dataIdentifier
397               for p in rdbi_random.results_with_positive_response]
398  
399          scan_range = UDS_RDBISelectiveEnumerator. \
400              points_to_blocks(identifiers_with_positive_response)
401          return {"scan_range": scan_range}
402  
403      @staticmethod
404      def points_to_blocks(pois):
405          # type: (Sequence[int]) -> Iterable[int]
406  
407          if len(pois) == 0:
408              # quick path for better performance
409              return []
410  
411          block_size = UDS_RDBIRandomEnumerator.block_size
412          generators = []
413          for start in range(0, 2 ** 16, block_size):
414              end = start + block_size
415              pr_in_block = any((start <= identifier < end
416                                 for identifier in pois))
417              if pr_in_block:
418                  generators.append(range(start, end))
419          scan_range = list(itertools.chain.from_iterable(generators))
420          return scan_range
421  
422      def __init__(self):
423          # type: () -> None
424          super(UDS_RDBISelectiveEnumerator, self).__init__(
425              [UDS_RDBIRandomEnumerator(), UDS_RDBIEnumerator()],
426              [None, self.__connector_rnd_to_seq])

447      def _get_initial_requests(self, **kwargs):
448          # type: (Any) -> Iterable[Packet]
449  
450          samples_per_block = {
451              4: 29, 5: 22, 6: 19, 8: 11, 9: 11, 10: 13, 11: 14, 12: 31, 13: 4,
452              14: 26, 16: 30, 17: 4, 18: 20, 19: 5, 20: 49, 21: 54, 22: 9, 23: 4,
453              24: 10, 25: 8, 28: 6, 29: 3, 32: 11, 36: 4, 37: 3, 40: 9, 41: 9,
454              42: 3, 44: 2, 47: 3, 48: 4, 49: 3, 52: 8, 64: 35, 66: 2, 68: 24,
455              69: 19, 70: 30, 71: 28, 72: 16, 73: 4, 74: 6, 75: 27, 76: 41,
456              77: 11, 78: 6, 81: 2, 88: 3, 90: 2, 92: 16, 97: 15, 98: 20, 100: 6,
457              101: 5, 102: 5, 103: 10, 106: 10, 108: 4, 124: 3, 128: 7, 136: 15,
458              137: 14, 138: 27, 139: 10, 148: 9, 150: 2, 152: 2, 168: 23,
459              169: 15, 170: 16, 171: 16, 172: 2, 176: 3, 177: 4, 178: 2, 187: 2,
460              232: 3, 235: 2, 240: 8, 252: 25, 256: 7, 257: 2, 287: 6, 290: 2,
461              316: 2, 319: 3, 323: 3, 324: 19, 326: 2, 327: 2, 330: 4, 331: 10,
462              332: 3, 334: 8, 338: 3, 832: 6, 833: 2, 900: 4, 956: 4, 958: 3,
463              964: 12, 965: 13, 966: 34, 967: 3, 972: 10, 1000: 3, 1012: 23,
464              1013: 14, 1014: 15
465          }
466          to_scan = []
467          block_size = UDS_RDBIRandomEnumerator.block_size
468  
469          probe_start = kwargs.pop("probe_start", 0)
470          probe_end = kwargs.pop("probe_end", 0x10000)
471          probe_range = range(probe_start, probe_end, block_size)
472  
473          for block_index, start in enumerate(probe_range):
474              end = start + block_size
475              count_samples = samples_per_block.get(block_index, 1)
476              to_scan += random.sample(range(start, end), count_samples)
477  
478          # Use locality effect
479          # If an identifier brought a positive response in any state,
480          # it is likely that in another state it is available as well
481          positive_identifiers = [t.resp.dataIdentifier for t in
482                                  self.results_with_positive_response]
483          to_scan += positive_identifiers
484  
485          # make all identifiers unique with set()
486          # Sort for better logs
487          to_scan = sorted(list(set(to_scan)))
488          return (UDS() / UDS_RDBI(identifiers=[x]) for x in to_scan)

657  class UDS_SA_XOR_Enumerator(UDS_SAEnumerator, StateGenerator):
658      _description = "XOR SecurityAccess supported"
659      _transition_function_args = dict()  # type: Dict[_Edge, Dict[str, Any]]
660  
661      @staticmethod
662      def get_key_pkt(seed, level=1):
663          # type: (Packet, int) -> Optional[Packet]
664  
665          def key_function_int(s):
666              # type: (int) -> int
667              return 0xffffffff & ~s
668  
669          def key_function_short(s):
670              # type: (int) -> int
671              return 0xffff & ~s
672  
673          try:
674              s = seed.securitySeed
675          except AttributeError:
676              return None
677  
678          fmt = None
679          key_function = None  # Optional[Callable[[int], int]]
680  
681          if len(s) == 2:
682              fmt = "H"
683              key_function = key_function_short
684  
685          if len(s) == 4:
686              fmt = "I"
687              key_function = key_function_int
688  
689          if key_function is not None and fmt is not None:
690              key = struct.pack(fmt, key_function(struct.unpack(fmt, s)[0]))
691              return cast(Packet, UDS() / UDS_SA(securityAccessType=level + 1,
692                                                 securityKey=key))
693          else:
694              return None
695  
696      def get_security_access(self, sock, level=1, seed_pkt=None):
697          # type: (_SocketUnion, int, Optional[Packet]) -> bool
698          log_automotive.info(
699              "Try bootloader security access for level %d" % level)
700          if seed_pkt is None:
701              seed_pkt = self.get_seed_pkt(sock, level)
702              if not seed_pkt:
703                  return False
704  
705          if not any(seed_pkt.securitySeed):
706              log_automotive.info(
707                  "Security access for level %d already granted!" % level)
708              return True
709  
710          key_pkt = self.get_key_pkt(seed_pkt, level)
711          if key_pkt is None:
712              return False
713  
714          try:
715              res = sock.sr1(key_pkt, timeout=5, verbose=False)
716              if sock.closed:
717                  log_automotive.critical("Socket closed during scan.")
718                  raise Scapy_Exception("Socket closed during scan")
719          except (OSError, ValueError, Scapy_Exception) as e:
720              try:
721                  last_seed_req = self._results[-1].req
722                  last_state = self._results[-1].state
723                  if not self._populate_retry(last_state, last_seed_req):
724                      log_automotive.exception(
725                          "Exception during retry. This is bad")
726              except IndexError:
727                  log_automotive.warning("Couldn't populate retry.")
728              raise e
729  
730          return self.evaluate_security_access_response(
731              res, seed_pkt, key_pkt)
732  
733      def transition_function(self, sock, _, kwargs):
734          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration, Dict[str, Any]) -> bool  # noqa: E501
735          spec = inspect.getfullargspec(self.get_security_access)
736  
737          func_kwargs = {k: kwargs[k] for k in spec.args if k in kwargs.keys()}
738          return self.get_security_access(sock, **func_kwargs)
739  
740      def get_new_edge(self, socket, config):
741          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration) -> Optional[_Edge]  # noqa: E501
742          last_resp = self._results[-1].resp
743          last_state = self._results[-1].state
744  
745          if last_resp is None or last_resp.service == 0x7f:
746              return None
747  
748          try:
749              if last_resp.service != 0x67 or \
750                      last_resp.securityAccessType % 2 != 1:
751                  return None
752  
753              seed = last_resp
754              sec_lvl = seed.securityAccessType
755  
756              if self.get_security_access(socket, sec_lvl, seed):
757                  log_automotive.debug("Security Access found.")
758                  # create edge
759                  new_state = copy.copy(last_state)
760                  new_state.security_level = seed.securityAccessType + 1  # type: ignore  # noqa: E501
761                  if last_state == new_state:
762                      return None
763                  edge = (last_state, new_state)
764                  self._transition_function_args[edge] = \
765                      {"level": sec_lvl, "desc": "SA=%d" % sec_lvl}
766                  return edge
767          except AttributeError:
768              pass
769  
770          return None
771  
772      def get_transition_function(self, socket, edge):
773          # type: (_SocketUnion, _Edge) -> Optional[_TransitionTuple]
774          return self.transition_function, \
775              self._transition_function_args[edge], None
