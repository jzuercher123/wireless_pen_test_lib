75   # Custom fields
76   class VariableFieldLenField(FieldLenField):
77       """ 
78       MQTT-SN length field either has 1 byte for values [0x02, 0xff] or 3 bytes 
79       for values [0x0100, 0xffff]. If the first byte is 0x01 the length value 
80       comes in network byte-order in the next 2 bytes. MQTT-SN packets are at 
81       least 2 bytes long (length field + type field). 
82       """
83       def addfield(self, pkt, s, val):
84           val = self.i2m(pkt, val)
85           if (val < 2) or (val > 0xffff):
86               raise Scapy_Exception("%s: invalid length field value" %
87                                     self.__class__.__name__)
88           elif val > 0xff:
89               return s + b"\x01" + struct.pack("!H", val)
90           else:
91               return s + chb(val)
92   
93       def getfield(self, pkt, s):
94           if orb(s[0]) == 0x01:
95               if len(s) < 3:
96                   raise Scapy_Exception("%s: malformed length field" %
97                                         self.__class__.__name__)
98               return s[3:], (orb(s[1]) << 8) | orb(s[2])
99           else:
100              return s[1:], orb(s[0])
101  
102      def randval(self):
103          return RandVariableFieldLen()
104  
105      def __init__(self, *args, **kwargs):
106          super(VariableFieldLenField, self).__init__(*args, **kwargs)

109  class RandVariableFieldLen(RandNum):
110      def __init__(self):

181  def _mqttsn_length_from(size_until):
182      def fun(pkt):
183          if (hasattr(pkt.underlayer, "len")):
184              if pkt.underlayer.len > 0xff:
185                  return pkt.underlayer.len - size_until - 4
186              elif (pkt.underlayer.len > 1) and (pkt.underlayer.len < 0xffff):
187                  return pkt.underlayer.len - size_until - 2
188          # assume string to be of length 0
189          return len(pkt.payload) - size_until + 1
190      return fun
191  
192  
193  def _mqttsn_len_adjust(pkt, x):
194      res = x + len(pkt.payload)
195      if (pkt.type == DISCONNECT) and \
196         (getattr(pkt.payload, "duration", None) is None):
197          res -= 2    # duration is optional with DISCONNECT
198      elif (pkt.type == ENCAPS_MSG) and \
199           (getattr(pkt.payload, "w_node_id", None) is not None):
200          res = x + len(pkt.payload.w_node_id) + 1
201      if res > 0xff:
202          res += 2
203      return res

206  class MQTTSN(Packet):
207      name = "MQTT-SN header"
208      fields_desc = [
209          # Since the size of the len field depends on the next layer, we
210          # need to "cheat" with the length_of parameter and use adjust
211          # parameter to calculate the value.
212          VariableFieldLenField("len", None, length_of="len",
213                                adjust=_mqttsn_len_adjust),
214          XByteEnumField("type", 0, PACKET_TYPE),
215      ]
