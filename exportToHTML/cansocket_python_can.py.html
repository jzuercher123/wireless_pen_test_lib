44   class SocketMapper(object):
45       """Internal Helper class to map a python-can bus object to 
46       a list of SocketWrapper instances 
47       """
48       def __init__(self, bus, sockets):
49           # type: (can_BusABC, List[SocketWrapper]) -> None
50           """Initializes the SocketMapper helper class 
51    
52           :param bus: A python-can Bus object 
53           :param sockets: A list of SocketWrapper objects which want to receive 
54                           messages from the provided python-can Bus object. 
55           """
56           self.bus = bus
57           self.sockets = sockets
58   
59       def mux(self):
60           # type: () -> None
61           """Multiplexer function. Tries to receive from its python-can bus 
62           object. If a message is received, this message gets forwarded to 
63           all receive queues of the SocketWrapper objects. 
64           """
65           msgs = []
66           while True:
67               try:
68                   msg = self.bus.recv(timeout=0)
69                   if msg is None:
70                       break
71                   else:
72                       msgs.append(msg)
73               except Exception as e:
74                   warning("[MUX] python-can exception caught: %s" % e)
75   
76           for sock in self.sockets:
77               with sock.lock:
78                   for msg in msgs:
79                       if sock._matches_filters(msg):
80                           sock.rx_queue.append(msg)

83   class _SocketsPool(object):
84       """Helper class to organize all SocketWrapper and SocketMapper objects"""
85       def __init__(self):
86           # type: () -> None
87           self.pool = dict()  # type: Dict[str, SocketMapper]
88           self.pool_mutex = threading.Lock()
89           self.last_call = 0.0
90   
91       def internal_send(self, sender, msg):
92           # type: (SocketWrapper, can_Message) -> None
93           """Internal send function. 
94    
95           A given SocketWrapper wants to send a CAN message. The python-can 
96           Bus object is obtained from an internal pool of SocketMapper objects. 
97           The given message is sent on the python-can Bus object and also 
98           inserted into the message queues of all other SocketWrapper objects 
99           which are connected to the same python-can bus object 
100          by the SocketMapper. 
101   
102          :param sender: SocketWrapper which initiated a send of a CAN message 
103          :param msg: CAN message to be sent 
104          """
105          if sender.name is None:
106              raise TypeError("SocketWrapper.name should never be None")
107  
108          with self.pool_mutex:
109              try:
110                  mapper = self.pool[sender.name]
111                  mapper.bus.send(msg)
112                  for sock in mapper.sockets:
113                      if sock == sender:
114                          continue
115                      if not sock._matches_filters(msg):
116                          continue
117  
118                      with sock.lock:
119                          sock.rx_queue.append(msg)
120              except KeyError:
121                  warning("[SND] Socket %s not found in pool" % sender.name)
122              except can_CanError as e:
123                  warning("[SND] python-can exception caught: %s" % e)
124  
125      def multiplex_rx_packets(self):
126          # type: () -> None
127          """This calls the mux() function of all SocketMapper 
128          objects in this SocketPool 
129          """
130          if time.monotonic() - self.last_call < 0.001:
131              # Avoid starvation if multiple threads are doing selects, since
132              # this object is singleton and all python-CAN sockets are using
133              # the same instance and locking the same locks.
134              return
135          with self.pool_mutex:
136              for t in self.pool.values():
137                  t.mux()
138          self.last_call = time.monotonic()
139  
140      def register(self, socket, *args, **kwargs):
141          # type: (SocketWrapper, Tuple[Any, ...], Dict[str, Any]) -> None
142          """Registers a SocketWrapper object. Every SocketWrapper describes to 
143          a python-can bus object. This python-can bus object can only exist 
144          once. In case this object already exists in this SocketsPool, organized 
145          by a SocketMapper object, the new SocketWrapper is inserted in the 
146          list of subscribers of the SocketMapper. Otherwise a new python-can 
147          Bus object is created from the provided args and kwargs and inserted, 
148          encapsulated in a SocketMapper, into this SocketsPool. 
149   
150          :param socket: SocketWrapper object which needs to be registered. 
151          :param args: Arguments for the python-can Bus object 
152          :param kwargs: Keyword arguments for the python-can Bus object 
153          """
154          if "interface" in kwargs.keys():
155              k = str(kwargs.get("interface", "unknown_interface")) + "_" + \
156                  str(kwargs.get("channel", "unknown_channel"))
157          else:
158              k = str(kwargs.get("bustype", "unknown_bustype")) + "_" + \
159                  str(kwargs.get("channel", "unknown_channel"))
160          with self.pool_mutex:
161              if k in self.pool:
162                  t = self.pool[k]
163                  t.sockets.append(socket)
164                  filters = [s.filters for s in t.sockets
165                             if s.filters is not None]
166                  if filters:
167                      t.bus.set_filters(reduce(add, filters))
168                  socket.name = k
169              else:
170                  bus = can_Bus(*args, **kwargs)
171                  socket.name = k
172                  self.pool[k] = SocketMapper(bus, [socket])
173  
174      def unregister(self, socket):
175          # type: (SocketWrapper) -> None
176          """Unregisters a SocketWrapper from its subscription to a SocketMapper. 
177   
178          If a SocketMapper doesn't have any subscribers, the python-can Bus 
179          get shutdown. 
180   
181          :param socket: SocketWrapper to be unregistered 
182          """
183          if socket.name is None:
184              raise TypeError("SocketWrapper.name should never be None")
185  
186          with self.pool_mutex:
187              try:
188                  t = self.pool[socket.name]
189                  t.sockets.remove(socket)
190                  if not t.sockets:
191                      t.bus.shutdown()
192                      del self.pool[socket.name]
193              except KeyError:
194                  warning("Socket %s already removed from pool" % socket.name)
195

200  class SocketWrapper(can_BusABC):
201      """Helper class to wrap a python-can Bus object as socket"""
202  
203      def __init__(self, *args, **kwargs):
204          # type: (Tuple[Any, ...], Dict[str, Any]) -> None
205          """Initializes a new python-can based socket, described by the provided 
206          arguments and keyword arguments. This SocketWrapper gets automatically 
207          registered in the SocketsPool. 
208   
209          :param args: Arguments for the python-can Bus object 
210          :param kwargs: Keyword arguments for the python-can Bus object 
211          """
212          super(SocketWrapper, self).__init__(*args, **kwargs)
213          self.lock = threading.Lock()
214          self.rx_queue = deque()  # type: deque[can_Message]
215          self.name = None  # type: Optional[str]
216          SocketsPool.register(self, *args, **kwargs)
217  
218      def _recv_internal(self, timeout):
219          # type: (int) -> Tuple[Optional[can_Message], bool]
220          """Internal blocking receive method, 
221          following the ``can_BusABC`` interface of python-can. 
222   
223          This triggers the multiplex function of the general SocketsPool. 
224   
225          :param timeout: Time to wait for a packet 
226          :return: Returns a tuple of either a can_Message or None and a bool to 
227                   indicate if filtering was already applied. 
228          """
229          if not self.rx_queue:
230              # Early return without locking if it looks like rx_queue is empty
231              return None, True
232  
233          with self.lock:
234              # It could be that 2 threads are using this same socket, so it's
235              # necessary to check again if the queue was emptied between the
236              # previous check and now
237              if len(self.rx_queue) == 0:
238                  return None, True
239              msg = self.rx_queue.popleft()
240              return msg, True
241  
242      def send(self, msg, timeout=None):
243          # type: (can_Message, Optional[int]) -> None
244          """Send function, following the ``can_BusABC`` interface of python-can. 
245   
246          :param msg: Message to be sent. 
247          :param timeout: Not used. 
248          """
249          SocketsPool.internal_send(self, msg)
250  
251      def shutdown(self):
252          # type: () -> None
253          """Shutdown function, following the ``can_BusABC`` interface of 
254          python-can. 
255          """
256          SocketsPool.unregister(self)
257          super().shutdown()

260  class PythonCANSocket(SuperSocket):
261      """Initializes a python-can bus object as Scapy PythonCANSocket. 
262   
263      All provided keyword arguments, except *basecls* are forwarded to 
264      the python-can can_Bus init function. For further details on python-can 
265      check: https://python-can.readthedocs.io/ 
266   
267      Example: 
268          >>> socket = PythonCANSocket(bustype='socketcan', channel='vcan0', bitrate=250000) 
269      """  # noqa: E501
270      desc = "read/write packets at a given CAN interface " \
271             "using a python-can bus object"
272      nonblocking_socket = True
273  
274      def __init__(self, **kwargs):
275          # type: (Dict[str, Any]) -> None
276          self.basecls = cast(Optional[Type[Packet]], kwargs.pop("basecls", CAN))
277          self.can_iface = SocketWrapper(**kwargs)
278  
279      def recv_raw(self, x=0xffff):
280          # type: (int) -> Tuple[Optional[Type[Packet]], Optional[bytes], Optional[float]]  # noqa: E501
281          """Returns a tuple containing (cls, pkt_data, time)"""
282          msg = self.can_iface.recv()
283  
284          hdr = msg.is_extended_id << 31 | msg.is_remote_frame << 30 | \
285              msg.is_error_frame << 29 | msg.arbitration_id
286  
287          if conf.contribs['CAN']['swap-bytes']:
288              hdr = struct.unpack("<I", struct.pack(">I", hdr))[0]
289  
290          dlc = msg.dlc << 24 | msg.is_fd << 18 | \
291              msg.error_state_indicator << 17 | msg.bitrate_switch << 16
292          pkt_data = struct.pack("!II", hdr, dlc) + bytes(msg.data)
293          return self.basecls, pkt_data, msg.timestamp
294  
295      def send(self, x):
296          # type: (Packet) -> int
297          bx = bytes(x)
298          msg = can_Message(is_remote_frame=x.flags == 0x2,
299                            is_extended_id=x.flags == 0x4,
300                            is_error_frame=x.flags == 0x1,
301                            arbitration_id=x.identifier,
302                            is_fd=bx[5] & 4 > 0,
303                            error_state_indicator=bx[5] & 2 > 0,
304                            bitrate_switch=bx[5] & 1 > 0,
305                            dlc=x.length,
306                            data=bx[8:])
307          msg.timestamp = time.time()
308          try:
309              x.sent_time = msg.timestamp
310          except AttributeError:
311              pass
312          self.can_iface.send(msg)
313          return len(x)
314  
315      @staticmethod
316      def select(sockets, remain=conf.recv_poll_rate):
317          # type: (List[SuperSocket], Optional[float]) -> List[SuperSocket]
318          """This function is called during sendrecv() routine to select 
319          the available sockets. 
320   
321          :param sockets: an array of sockets that need to be selected 
322          :returns: an array of sockets that were selected and 
323              the function to be called next to get the packets (i.g. recv) 
324          """
325          ready_sockets = \
326              [s for s in sockets if isinstance(s, PythonCANSocket) and
327               len(s.can_iface.rx_queue)]
328          # checking the queue length without locking might sound
329          # dangerous, but for the purpose of this select, if another
330          # thread is reading the same socket, then even proper locking
331          # wouldn't help
332          if not ready_sockets:
333              # yield this thread to avoid starvation
334              time.sleep(0)
335  
336          SocketsPool.multiplex_rx_packets()
337          return cast(List[SuperSocket], ready_sockets)
338  
339      def close(self):
340          # type: () -> None
341          """Closes this socket"""
342          if self.closed:
343              return
344          super(PythonCANSocket, self).close()
345          self.can_iface.shutdown()
