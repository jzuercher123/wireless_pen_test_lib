180          base class for all DCP Blocks 
181      """
182      fields_desc = [

188          StrLenField("data", "", length_from=lambda x: x.dcp_block_length),
189      ]

191      def extract_padding(self, s):
192          return '', s

453  def guess_dcp_block_class(packet, **kargs):
454      """ 
455      returns the correct dcp block class needed to dissect the current tag 
456      if nothing can be found -> dcp base block will be used 
457   
458      :param packet: the current packet 
459      :return: dcp block class 
460      """
461      # packet = unicode(packet, "utf-8")
462      option = orb(packet[0])
463      suboption = orb(packet[1])
464  
465      # NOTE implement the other functions if needed
466  
467      class_switch_case = {
468          # IP
469          0x01:
470              {
471                  0x01: "DCPMACBlock",
472                  0x02: "DCPIPBlock"
473              },
474          # Device Properties
475          0x02:
476              {
477                  0x01: "DCPManufacturerSpecificBlock",
478                  0x02: "DCPNameOfStationBlock",
479                  0x03: "DCPDeviceIDBlock",
480                  0x04: "DCPDeviceRoleBlock",
481                  0x05: "DCPDeviceOptionsBlock",
482                  0x06: "DCPAliasNameBlock",
483                  0x07: "DCPDeviceInstanceBlock",
484                  0x08: "DCPOEMIDBlock"
485              },
486          # DHCP
487          0x03:
488              {
489                  0x0c: "Host name",
490                  0x2b: "Vendor specific",
491                  0x36: "Server identifier",
492                  0x37: "Parameter request list",
493                  0x3c: "Class identifier",
494                  0x3d: "DHCP client identifier",
495                  0x51: "FQDN, Fully Qualified Domain Name",
496                  0x61: "UUID/GUID-based Client",
497                  0xff: "Control DHCP for address resolution"
498              },
499          # Control
500          0x05:
501              {
502                  0x00: "Reserved (0x00)",
503                  0x01: "Start Transaction (0x01)",
504                  0x02: "End Transaction (0x02)",
505                  0x03: "Signal (0x03)",
506                  0x04: "DCPControlBlock",
507                  0x05: "Reset Factory Settings (0x05)",
508                  0x06: "Reset to Factory (0x06)"
509              },
510          # Device Inactive
511          0x06:
512              {
513                  0x00: "Reserved (0x00)",
514                  0x01: "DCPDeviceInitiativeBlock"
515              },
516          # ALL Selector
517          0xff:
518              {
519                  0xff: "ALL Selector (0xff)"
520              }
521      }
522  
523      try:
524          c = class_switch_case[option][suboption]
525      except KeyError:
526          c = "DCPBaseBlock"
527  
528      cls = globals()[c]
529      return cls(packet, **kargs)

534  class ProfinetDCP(Packet):
535      """ 
536      Profinet DCP Packet 
537   
538      Requests are handled via ConditionalField because here only 1 Block is used 
539      every time. 
540   
541      Response can contain 1..n Blocks, for that you have to use one ProfinetDCP 
542      Layer with one or multiple DCP*Block Layers:: 
543   
544          ProfinetDCP / DCPNameOfStationBlock / DCPDeviceIDBlock ... 
545   
546      Example for a DCP Identify All Request:: 
547   
548          Ether(dst="01:0e:cf:00:00:00") / 
549          ProfinetIO(frameID=DCP_IDENTIFY_REQUEST_FRAME_ID) / 
550          ProfinetDCP(service_id=DCP_SERVICE_ID_IDENTIFY, 
551              service_type=DCP_REQUEST, option=255, sub_option=255, 
552              dcp_data_length=4) 
553   
554      Example for a DCP Identify Response:: 
555   
556          Ether(dst=dst_mac) / 
557          ProfinetIO(frameID=DCP_IDENTIFY_RESPONSE_FRAME_ID) / 
558          ProfinetDCP( 
559              service_id=DCP_SERVICE_ID_IDENTIFY, 
560              service_type=DCP_RESPONSE) / 
561          DCPNameOfStationBlock(name_of_station="device1") 
562   
563      Example for a DCP Set Request:: 
564   
565          Ether(dst=mac) / 
566          ProfinetIO(frameID=DCP_GET_SET_FRAME_ID) / 
567          ProfinetDCP(service_id=DCP_SERVICE_ID_SET, service_type=DCP_REQUEST, 
568              option=2, sub_option=2, dcp_data_length=14, dcp_block_length=10, 
569              name_of_station=name, reserved=0) 
570   
571      """
572  
573      name = "Profinet DCP"
574      # a DCP PDU consists of some fields and 1..n DCP Blocks
575      fields_desc = [
576          ByteEnumField("service_id", 5, DCP_SERVICE_ID),
577          ByteEnumField("service_type", 0, DCP_SERVICE_TYPE),
578          XIntField("xid", 0x01000001),
579          # XShortField('reserved', 0),
580  
581          ShortField('reserved', 0),
582          LenField("dcp_data_length", None),
583  
584          # DCP REQUEST specific
585          ConditionalField(ByteEnumField("option", 2, DCP_OPTIONS),
586                           lambda pkt: pkt.service_type == 0),
587          ConditionalField(
588              MultiEnumField("sub_option", 3, DCP_SUBOPTIONS, fmt='B',
589                             depends_on=lambda p: p.option),
590              lambda pkt: pkt.service_type == 0),
591  
592          # calculate the len fields - workaround
593          ConditionalField(LenField("dcp_block_length", 0),
594                           lambda pkt: pkt.service_type == 0),
595  
596          # DCP SET REQUEST #
597          ConditionalField(ShortEnumField("block_qualifier", 1,
598                                          BLOCK_QUALIFIERS),
599                           lambda pkt: pkt.service_id == 4 and
600                           pkt.service_type == 0),
601          # (Common) Name Of Station
602          ConditionalField(
603              MultipleTypeField(
604                  [
605                      (StrLenField("name_of_station", "et200sp",
606                                   length_from=lambda x: x.dcp_block_length - 2),
607                       lambda pkt: pkt.service_id == 4),
608                  ],
609                  StrLenField("name_of_station", "et200sp",
610                              length_from=lambda x: x.dcp_block_length),
611              ),
612              lambda pkt: pkt.service_type == 0 and pkt.option == 2 and
613              pkt.sub_option == 2
614          ),
615          # DCP SET REQUEST #
616          # MAC
617          ConditionalField(MACField("mac", "00:00:00:00:00:00"),
618                           lambda pkt: pkt.service_id == 4 and
619                           pkt.service_type == 0 and pkt.option == 1 and
620                           pkt.sub_option == 1),
621          # IP
622          ConditionalField(IPField("ip", "192.168.0.2"),
623                           lambda pkt: pkt.service_id == 4 and
624                           pkt.service_type == 0 and pkt.option == 1 and
625                           pkt.sub_option in [2, 3]),
626          ConditionalField(IPField("netmask", "255.255.255.0"),
627                           lambda pkt: pkt.service_id == 4 and
628                           pkt.service_type == 0 and pkt.option == 1 and
629                           pkt.sub_option in [2, 3]),
630          ConditionalField(IPField("gateway", "192.168.0.1"),
631                           lambda pkt: pkt.service_id == 4 and
632                           pkt.service_type == 0 and pkt.option == 1 and
633                           pkt.sub_option in [2, 3]),
634  
635          # Full IP
636          ConditionalField(FieldListField("dnsaddr", [], IPField("", "0.0.0.0"),
637                                          count_from=lambda x: 4),
638                           lambda pkt: pkt.service_id == 4 and
639                           pkt.service_type == 0 and pkt.option == 1 and
640                           pkt.sub_option == 3),
641  
642          # DCP IDENTIFY REQUEST #
643          # Name of station (handled above)
644  
645          # Alias name
646          ConditionalField(StrLenField("alias_name", "et200sp",
647                                       length_from=lambda x: x.dcp_block_length),
648                           lambda pkt: pkt.service_id == 5 and
649                           pkt.service_type == 0 and pkt.option == 2 and
650                           pkt.sub_option == 6),
651
