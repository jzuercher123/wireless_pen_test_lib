126  @crypto_validator
127  def ComputeSessionKeyAES(HashNt, ClientChallenge, ServerChallenge):
128      M4SS = HashNt
129      h = hmac.HMAC(M4SS, hashes.SHA256())
130      h.update(ClientChallenge)
131      h.update(ServerChallenge)
132      return h.finalize()[:16]

135  # [MS-NRPC] sect 3.1.4.3.2
136  @crypto_validator
137  def ComputeSessionKeyStrongKey(HashNt, ClientChallenge, ServerChallenge):
138      M4SS = HashNt
139      digest = hashes.Hash(hashes.MD5())
140      digest.update(b"\x00\x00\x00\x00")
141      digest.update(ClientChallenge)
142      digest.update(ServerChallenge)
143      h = hmac.HMAC(M4SS, hashes.MD5())
144      h.update(digest.finalize())
145      return h.finalize()

156  # [MS-NRPC] sect 3.1.4.4.2
157  def InitLMKey(KeyIn):
158      KeyOut = bytearray(b"\x00" * 8)
159      KeyOut[0] = KeyIn[0] >> 0x01
160      KeyOut[1] = ((KeyIn[0] & 0x01) << 6) | (KeyIn[1] >> 2)
161      KeyOut[2] = ((KeyIn[1] & 0x03) << 5) | (KeyIn[2] >> 3)
162      KeyOut[3] = ((KeyIn[2] & 0x07) << 4) | (KeyIn[3] >> 4)
163      KeyOut[4] = ((KeyIn[3] & 0x0F) << 3) | (KeyIn[4] >> 5)
164      KeyOut[5] = ((KeyIn[4] & 0x1F) << 2) | (KeyIn[5] >> 6)
165      KeyOut[6] = ((KeyIn[5] & 0x3F) << 1) | (KeyIn[6] >> 7)
166      KeyOut[7] = KeyIn[6] & 0x7F
167      for i in range(8):
168          KeyOut[i] = (KeyOut[i] << 1) & 0xFE
169      return KeyOut

172  @crypto_validator
173  def ComputeNetlogonCredentialDES(Input, Sk):
174      k3 = InitLMKey(Sk[0:7])
175      k4 = InitLMKey(Sk[7:14])
176      output1 = Cipher(DES(k3), modes.ECB()).encryptor().update(Input)
177      return Cipher(DES(k4), modes.ECB()).encryptor().update(output1)

181  def _credentialAddition(cred, i):
182      return (
183          struct.pack(
184              "<I",
185              (i + struct.unpack("<I", cred[:4])[0]) & 0xFFFFFFFF,
186          )
187          + cred[4:]
188      )

194  def ComputeCopySeqNumber(ClientSequenceNumber, client):
195      low = struct.pack(">L", ClientSequenceNumber & 0xFFFFFFFF)
196      high = struct.pack(
197          ">L",
198          ((ClientSequenceNumber >> 32) & 0xFFFFFFFF) | (0x80000000 if client else 0),
199      )
200      return low + high

203  @crypto_validator
204  def ComputeNetlogonChecksumAES(nl_auth_sig, message, SessionKey, Confounder=None):
205      h = hmac.HMAC(SessionKey, hashes.SHA256())
206      h.update(nl_auth_sig[:8])
207      if Confounder:
208          h.update(Confounder)
209      h.update(message)
210      return h.finalize()
211  
212  
213  @crypto_validator
214  def ComputeNetlogonChecksumMD5(nl_auth_sig, message, SessionKey, Confounder=None):
215      digest = hashes.Hash(hashes.MD5())
216      digest.update(b"\x00\x00\x00\x00")
217      digest.update(nl_auth_sig[:8])
218      if Confounder:
219          digest.update(Confounder)
220      digest.update(message)
221      h = hmac.HMAC(SessionKey, hashes.MD5())
222      h.update(digest.finalize())
223      return h.finalize()

226  @crypto_validator
227  def ComputeNetlogonSealingKeyAES(SessionKey):
228      return bytes(bytearray((x ^ 0xF0) for x in bytearray(SessionKey)))
229

231  @crypto_validator
232  def ComputeNetlogonSealingKeyRC4(SessionKey, CopySeqNumber):
233      XorKey = bytes(bytearray((x ^ 0xF0) for x in bytearray(SessionKey)))
234      h = hmac.HMAC(XorKey, hashes.MD5())
235      h.update(b"\x00\x00\x00\x00")
236      h = hmac.HMAC(h.finalize(), hashes.MD5())
237      h.update(CopySeqNumber)
238      return h.finalize()

241  @crypto_validator
242  def ComputeNetlogonSequenceNumberKeyMD5(SessionKey, Checksum):
243      h = hmac.HMAC(SessionKey, hashes.MD5())
244      h.update(b"\x00\x00\x00\x00")
245      h = hmac.HMAC(h.finalize(), hashes.MD5())
246      h.update(Checksum)
247      return h.finalize()

282      def _secure(self, Context, msgs, Seal):
283          """ 
284          Internal function used by GSS_WrapEx and GSS_GetMICEx 
285   
286          [MS-NRPC] 3.3.4.2.1 
287          """
288          # Concatenate the ToSign
289          ToSign = b"".join(x.data for x in msgs if x.sign)
290  
291          Confounder = None
292          if Seal:
293              Confounder = os.urandom(8)
294  
295          if Context.AES:
296              # 1. If AES is negotiated
297              signature = NL_AUTH_SIGNATURE(
298                  SignatureAlgorithm=0x0013,
299                  SealAlgorithm=0x001A if Seal else 0xFFFF,
300              )
301          else:
302              # 2. If AES is not negotiated
303              signature = NL_AUTH_SIGNATURE(
304                  SignatureAlgorithm=0x0077,
305                  SealAlgorithm=0x007A if Seal else 0xFFFF,
306              )
307          # 3. Pad filled with 0xff (OK)
308          # 4. Flags with 0x00 (OK)
309          # 5. SequenceNumber
310          SequenceNumber = ComputeCopySeqNumber(
311              Context.ClientSequenceNumber, Context.IsClient
312          )
313          # 6. The ClientSequenceNumber MUST be incremented by 1
314          Context.ClientSequenceNumber += 1
315          # 7. Signature
316          if Context.AES:
317              signature.Checksum = ComputeNetlogonChecksumAES(
318                  bytes(signature), ToSign, self.SessionKey, Confounder
319              )[:8]
320          else:
321              signature.Checksum = ComputeNetlogonChecksumMD5(
322                  bytes(signature), ToSign, self.SessionKey, Confounder
323              )[:8]
324          # 8. If the Confidentiality option is requested, the Confounder field and
325          # the data MUST be encrypted
326          if Seal:
327              if Context.AES:
328                  EncryptionKey = ComputeNetlogonSealingKeyAES(self.SessionKey)
329              else:
330                  EncryptionKey = ComputeNetlogonSealingKeyRC4(
331                      self.SessionKey, SequenceNumber
332                  )
333              # Encrypt Confounder and data
334              if Context.AES:
335                  IV = SequenceNumber * 2
336                  encryptor = Cipher(
337                      algorithms.AES(EncryptionKey), mode=modes.CFB8(IV)
338                  ).encryptor()
339                  # Confounder
340                  signature.Confounder = encryptor.update(Confounder)
341                  # data
342                  for msg in msgs:
343                      if msg.conf_req_flag:
344                          msg.data = encryptor.update(msg.data)
345              else:
346                  handle = RC4Init(EncryptionKey)
347                  # Confounder
348                  signature.Confounder = RC4(handle, Confounder)
349                  # DOC IS WRONG !
350                  # > The server MUST initialize RC4 only once, before encrypting
351                  # > the Confounder field.
352                  # But, this fails ! as Samba put it:
353                  # > For RC4, Windows resets the cipherstate after encrypting
354                  # > the confounder, thus defeating the purpose of the confounder
355                  handle = RC4Init(EncryptionKey)
356                  # data
357                  for msg in msgs:
358                      if msg.conf_req_flag:
359                          msg.data = RC4(handle, msg.data)
360          # 9. The SequenceNumber MUST be encrypted.
361          if Context.AES:
362              EncryptionKey = self.SessionKey
363              IV = signature.Checksum * 2
364              cipher = Cipher(algorithms.AES(EncryptionKey), mode=modes.CFB8(IV))
365              encryptor = cipher.encryptor()
366              signature.SequenceNumber = encryptor.update(SequenceNumber)
367          else:
368              EncryptionKey = ComputeNetlogonSequenceNumberKeyMD5(
369                  self.SessionKey, signature.Checksum
370              )
371              signature.SequenceNumber = RC4K(EncryptionKey, SequenceNumber)
372  
373          return (
374              msgs,
375              signature,
376          )

378      def _unsecure(self, Context, msgs, signature, Seal):
379          """ 
380          Internal function used by GSS_UnwrapEx and GSS_VerifyMICEx 
381   
382          [MS-NRPC] 3.3.4.2.2 
383          """
384          assert isinstance(signature, NL_AUTH_SIGNATURE)
385  
386          # 1. The SignatureAlgorithm bytes MUST be verified
387          if (Context.AES and signature.SignatureAlgorithm != 0x0013) or (
388              not Context.AES and signature.SignatureAlgorithm != 0x0077
389          ):
390              raise ValueError("Invalid SignatureAlgorithm !")
391  
392          # 5. The SequenceNumber MUST be decrypted.
393          if Context.AES:
394              EncryptionKey = self.SessionKey
395              IV = signature.Checksum * 2
396              cipher = Cipher(algorithms.AES(EncryptionKey), mode=modes.CFB8(IV))
397              decryptor = cipher.decryptor()
398              SequenceNumber = decryptor.update(signature.SequenceNumber)
399          else:
400              EncryptionKey = ComputeNetlogonSequenceNumberKeyMD5(
401                  self.SessionKey, signature.Checksum
402              )
403              SequenceNumber = RC4K(EncryptionKey, signature.SequenceNumber)
404          # 6. A local copy of SequenceNumber MUST be computed
405          CopySeqNumber = ComputeCopySeqNumber(
406              Context.ClientSequenceNumber, not Context.IsClient
407          )
408          # 7. The SequenceNumber MUST be compared to CopySeqNumber
409          if SequenceNumber != CopySeqNumber:
410              raise ValueError("ERROR: SequenceNumber don't match")
411          # 8. ClientSequenceNumber MUST be incremented.
412          Context.ClientSequenceNumber += 1
413          # 9. If the Confidentiality option is requested, the Confounder and the
414          # data MUST be decrypted.
415          Confounder = None
416          if Seal:
417              if Context.AES:
418                  EncryptionKey = ComputeNetlogonSealingKeyAES(self.SessionKey)
419              else:
420                  EncryptionKey = ComputeNetlogonSealingKeyRC4(
421                      self.SessionKey, SequenceNumber
422                  )
423              # Decrypt Confounder and data
424              if Context.AES:
425                  IV = SequenceNumber * 2
426                  decryptor = Cipher(
427                      algorithms.AES(EncryptionKey), mode=modes.CFB8(IV)
428                  ).decryptor()
429                  # Confounder
430                  Confounder = decryptor.update(signature.Confounder)
431                  # data
432                  for msg in msgs:
433                      if msg.conf_req_flag:
434                          msg.data = decryptor.update(msg.data)
435              else:
436                  # Confounder
437                  EncryptionKey = ComputeNetlogonSealingKeyRC4(
438                      self.SessionKey, SequenceNumber
439                  )
440                  Confounder = RC4K(EncryptionKey, signature.Confounder)
441                  # data
442                  handle = RC4Init(EncryptionKey)
443                  for msg in msgs:
444                      if msg.conf_req_flag:
445                          msg.data = RC4(handle, msg.data)
446  
447          # Concatenate the ToSign
448          ToSign = b"".join(x.data for x in msgs if x.sign)
449  
450          # 10/11. Signature
451          if Context.AES:
452              Checksum = ComputeNetlogonChecksumAES(
453                  bytes(signature), ToSign, self.SessionKey, Confounder
454              )[:8]
455          else:
456              Checksum = ComputeNetlogonChecksumMD5(
457                  bytes(signature), ToSign, self.SessionKey, Confounder
458              )[:8]
459          if signature.Checksum != Checksum:
460              raise ValueError("ERROR: Checksum don't match")
461          return msgs

475      def GSS_Init_sec_context(
476          self, Context, val=None, req_flags: Optional[GSS_C_FLAGS] = None
477      ):
478          if Context is None:
479              Context = self.CONTEXT(True, req_flags=req_flags, AES=self.AES)
480  
481          if Context.state == self.STATE.INIT:
482              Context.state = self.STATE.CLI_SENT_NL
483              return (
484                  Context,
485                  NL_AUTH_MESSAGE(
486                      MessageType=0,
487                      Flags=3,
488                      NetbiosDomainName=self.domainname,
489                      NetbiosComputerName=self.computername,
490                  ),
491                  GSS_S_CONTINUE_NEEDED,
492              )
493          else:
494              return Context, None, GSS_S_COMPLETE
495  
496      def GSS_Accept_sec_context(self, Context, val=None):
497          if Context is None:
498              Context = self.CONTEXT(False, req_flags=0, AES=self.AES)
499  
500          if Context.state == self.STATE.INIT:
501              Context.state = self.STATE.SRV_SENT_NL
502              return (
503                  Context,
504                  NL_AUTH_MESSAGE(
505                      MessageType=1,
506                      Flags=0,
507                  ),
508                  GSS_S_COMPLETE,
509              )
510          else:
511              # Invalid state
512              return Context, None, GSS_S_FAILURE

514      def MaximumSignatureLength(self, Context: CONTEXT):
515          """ 
516          Returns the Maximum Signature length. 
517   
518          This will be used in auth_len in DceRpc5, and is necessary for 
519          PFC_SUPPORT_HEADER_SIGN to work properly. 
520          """
521          # len(NL_AUTH_SIGNATURE())
522          if Context.flags & GSS_C_FLAGS.GSS_C_CONF_FLAG:
523              if Context.AES:
524                  return 56
525              else:
526                  return 32
527          else:
528              if Context.AES:
529                  return 48
530              else:
531                  return 24

542  class NetlogonClient(DCERPC_Client):
543      """ 
544      A subclass of DCERPC_Client that supports establishing a Netlogon secure channel 
545      using the Netlogon SSP, and handling Netlogon authenticators. 
546   
547      This class therefore only supports the 'logon' rpc. 
548   
549      :param auth_level: one of DCE_C_AUTHN_LEVEL 
550   
551      :param verb: verbosity control. 
552      :param supportAES: advertise AES support in the Netlogon session. 
553   
554      Example:: 
555   
556          >>> cli = NetlogonClient() 
557          >>> cli.connect_and_bind("192.168.0.100") 
558          >>> cli.establishSecureChannel( 
559          ...     domainname="DOMAIN", computername="WIN10", 
560          ...     HashNT=bytes.fromhex("aaaaaaaaaaaaaaaaaaaaaaaaaaaaaaaa"), 
561          ... ) 
562      """
563  
564      def __init__(
565          self,
566          auth_level=DCE_C_AUTHN_LEVEL.NONE,
567          verb=True,
568          supportAES=True,
569          **kwargs,
570      ):
571          self.interface = find_dcerpc_interface("logon")
572          self.ndr64 = False  # Netlogon doesn't work with NDR64
573          self.SessionKey = None
574          self.ClientStoredCredential = None
575          self.supportAES = supportAES
576          super(NetlogonClient, self).__init__(
577              DCERPC_Transport.NCACN_IP_TCP,
578              auth_level=auth_level,
579              ndr64=self.ndr64,
580              verb=verb,
581              **kwargs,
582          )

585          """ 
586          This calls DCERPC_Client's connect_and_bind to bind the 'logon' interface. 
587          """
588          super(NetlogonClient, self).connect_and_bind(remoteIP, self.interface)

590      def alter_context(self):

593      def create_authenticator(self):
594          """ 
595          Create a NETLOGON_AUTHENTICATOR 
596          """
597          # [MS-NRPC] sect 3.1.4.5
598          ts = int(time.time())
599          self.ClientStoredCredential = _credentialAddition(
600              self.ClientStoredCredential, ts
601          )
602          return PNETLOGON_AUTHENTICATOR(
603              Credential=PNETLOGON_CREDENTIAL(
604                  data=(
605                      ComputeNetlogonCredentialAES(
606                          self.ClientStoredCredential,
607                          self.SessionKey,
608                      )
609                      if self.supportAES
610                      else ComputeNetlogonCredentialDES(
611                          self.ClientStoredCredential,
612                          self.SessionKey,
613                      )
614                  ),
615              ),
616              Timestamp=ts,
617          )

619      def validate_authenticator(self, auth):
620          """ 
621          Validate a NETLOGON_AUTHENTICATOR 
622   
623          :param auth: the NETLOGON_AUTHENTICATOR object 
624          """
625          # [MS-NRPC] sect 3.1.4.5
626          self.ClientStoredCredential = _credentialAddition(
627              self.ClientStoredCredential, 1
628          )
629          if self.supportAES:
630              tempcred = ComputeNetlogonCredentialAES(
631                  self.ClientStoredCredential, self.SessionKey
632              )
633          else:
634              tempcred = ComputeNetlogonCredentialDES(
635                  self.ClientStoredCredential, self.SessionKey
636              )
637          if tempcred != auth.Credential.data:
638              raise ValueError("Server netlogon authenticator is wrong !")

640      def establishSecureChannel(
641          self,
642          computername: str,
643          domainname: str,
644          HashNt: bytes,
645          mode=NETLOGON_SECURE_CHANNEL_METHOD.NetrServerAuthenticate3,
646          secureChannelType=NETLOGON_SECURE_CHANNEL_TYPE.WorkstationSecureChannel,
647      ):
648          """ 
649          Function to establish the Netlogon Secure Channel. 
650   
651          This uses NetrServerAuthenticate3 to negotiate the session key, then creates a 
652          NetlogonSSP that uses that session key and alters the DCE/RPC session to use it. 
653   
654          :param mode: one of NETLOGON_SECURE_CHANNEL_METHOD. This defines which method 
655                       to use to establish the secure channel. 
656          :param computername: the netbios computer account name that is used to establish 
657                               the secure channel. (e.g. WIN10) 
658          :param domainname: the netbios domain name to connect to (e.g. DOMAIN) 
659          :param HashNt: the HashNT of the computer account. 
660          """
661          # Flow documented in 3.1.4 Session-Key Negotiation
662          # and sect 3.4.5.2 for specific calls
663          clientChall = os.urandom(8)
664          # Step 1: NetrServerReqChallenge
665          netr_server_req_chall_response = self.sr1_req(
666              NetrServerReqChallenge_Request(
667                  PrimaryName=None,
668                  ComputerName=computername,
669                  ClientChallenge=PNETLOGON_CREDENTIAL(
670                      data=clientChall,
671                  ),
672                  ndr64=self.ndr64,
673                  ndrendian=self.ndrendian,
674              )
675          )
676          if (
677              NetrServerReqChallenge_Response not in netr_server_req_chall_response
678              or netr_server_req_chall_response.status != 0
679          ):
680              print(
681                  conf.color_theme.fail(
682                      "! %s"
683                      % STATUS_ERREF.get(netr_server_req_chall_response.status, "Failure")
684                  )
685              )
686              netr_server_req_chall_response.show()
687              raise ValueError
688          # Calc NegotiateFlags
689          NegotiateFlags = FlagValue(
690              0x602FFFFF,  # sensible default (Windows)
691              names=_negotiateFlags,
692          )
693          if self.supportAES:
694              NegotiateFlags += "AES"
695          # We are either using NetrServerAuthenticate3 or NetrServerAuthenticateKerberos
696          if mode == NETLOGON_SECURE_CHANNEL_METHOD.NetrServerAuthenticate3:
697              # We use the legacy NetrServerAuthenticate3 function (NetlogonSSP)
698              # Step 2: Build the session key
699              serverChall = netr_server_req_chall_response.ServerChallenge.data
700              if self.supportAES:
701                  SessionKey = ComputeSessionKeyAES(HashNt, clientChall, serverChall)
702                  self.ClientStoredCredential = ComputeNetlogonCredentialAES(
703                      clientChall, SessionKey
704                  )
705              else:
706                  SessionKey = ComputeSessionKeyStrongKey(
707                      HashNt, clientChall, serverChall
708                  )
709                  self.ClientStoredCredential = ComputeNetlogonCredentialDES(
710                      clientChall, SessionKey
711                  )
712              netr_server_auth3_response = self.sr1_req(
713                  NetrServerAuthenticate3_Request(
714                      PrimaryName=None,
715                      AccountName=computername + "$",
716                      SecureChannelType=secureChannelType,
717                      ComputerName=computername,
718                      ClientCredential=PNETLOGON_CREDENTIAL(
719                          data=self.ClientStoredCredential,
720                      ),
721                      NegotiateFlags=int(NegotiateFlags),
722                      ndr64=self.ndr64,
723                      ndrendian=self.ndrendian,
724                  )
725              )
726              if (
727                  NetrServerAuthenticate3_Response not in netr_server_auth3_response
728                  or netr_server_auth3_response.status != 0
729              ):
730                  NegotiatedFlags = None
731                  if NetrServerAuthenticate3_Response in netr_server_auth3_response:
732                      NegotiatedFlags = FlagValue(
733                          netr_server_auth3_response.NegotiateFlags,
734                          names=_negotiateFlags,
735                      )
736                      if NegotiateFlags != NegotiatedFlags:
737                          print(
738                              conf.color_theme.fail(
739                                  "! Unsupported server flags: %s"
740                                  % (NegotiatedFlags ^ NegotiateFlags)
741                              )
742                          )
743                  print(
744                      conf.color_theme.fail(
745                          "! %s"
746                          % STATUS_ERREF.get(netr_server_auth3_response.status, "Failure")
747                      )
748                  )
749                  if netr_server_auth3_response.status not in STATUS_ERREF:
750                      netr_server_auth3_response.show()
751                  raise ValueError
752              # Check Server Credential
753              if self.supportAES:
754                  if (
755                      netr_server_auth3_response.ServerCredential.data
756                      != ComputeNetlogonCredentialAES(serverChall, SessionKey)
757                  ):
758                      print(conf.color_theme.fail("! Invalid ServerCredential."))
759                      raise ValueError
760              else:
761                  if (
762                      netr_server_auth3_response.ServerCredential.data
763                      != ComputeNetlogonCredentialDES(serverChall, SessionKey)
764                  ):
765                      print(conf.color_theme.fail("! Invalid ServerCredential."))
766                      raise ValueError
767              # SessionKey negotiated !
768              self.SessionKey = SessionKey
769              # Create the NetlogonSSP and assign it to the local client
770              self.ssp = self.sock.session.ssp = NetlogonSSP(
771                  SessionKey=self.SessionKey,
772                  AES=self.supportAES,
773                  domainname=domainname,
774                  computername=computername,
775              )
776          elif mode == NETLOGON_SECURE_CHANNEL_METHOD.NetrServerAuthenticateKerberos:
777              NegotiateFlags += "Kerberos"
778              # TODO
779              raise NotImplementedError
780          # Finally alter context (to use the SSP)
781          self.alter_context()
