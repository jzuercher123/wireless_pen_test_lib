205  class _RPLGuessMsgType(Packet):
206      name = "Dummy RPL Message class"
207  
208      def guess_payload_class(self, payload):
209          if isinstance(payload, str):
210              otype = ord(payload[0])
211          else:
212              otype = payload[0]
213          return RPLOPTS.get(otype)
