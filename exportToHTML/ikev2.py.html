521  class IKEv2(_IKEv2_Packet):  # rfc4306
522      name = "IKEv2"
523      fields_desc = [
524          XStrFixedLenField("init_SPI", "", 8),
525          XStrFixedLenField("resp_SPI", "", 8),
526          ByteEnumField("next_payload", 0, IKEv2PayloadTypes),
527          XByteField("version", 0x20),
528          ByteEnumField("exch_type", 0, IKEv2ExchangeTypes),
529          FlagsField("flags", 0, 8, ["res0", "res1", "res2", "Initiator", "Version", "Response", "res6", "res7"]),  # noqa: E501
530          IntField("id", 0),
531          IntField("length", None)  # Length of total message: packets + all payloads  # noqa: E501
532      ]
533  
534      @classmethod
535      def dispatch_hook(cls, _pkt=None, *args, **kargs):
536          if _pkt and len(_pkt) >= 18:
537              version = struct.unpack("!B", _pkt[17:18])[0]
538              if version < 0x20:
539                  return ISAKMP
540          return cls
541  
542      def answers(self, other):
543          if isinstance(other, IKEv2):
544              if other.init_SPI == self.init_SPI:
545                  return 1
546          return 0
547  
548      def post_build(self, p, pay):
549          p += pay
550          if self.length is None:
551              p = p[:24] + struct.pack("!I", len(p)) + p[28:]
552          return p

555  class IKEv2_Key_Length_Attribute(IntField):
556      # We only support the fixed-length Key Length attribute (the only one currently defined)  # noqa: E501
557      def __init__(self, name):
558          IntField.__init__(self, name, 0x800E0000)
559  
560      def i2h(self, pkt, x):
561          return IntField.i2h(self, pkt, x & 0xFFFF)
562  
563      def h2i(self, pkt, x):
564          return IntField.h2i(self, pkt, (x if x is not None else 0) | 0x800E0000)  # noqa: E501

567  class IKEv2_Payload(_IKEv2_Packet):
568      name = "IKEv2 Payload"
569      fields_desc = [
570          ByteEnumField("next_payload", None, IKEv2PayloadTypes),
571          FlagsField("flags", 0, 8, ["critical"]),
572          ShortField("length", None),
573          XStrLenField("load", "", length_from=lambda pkt: pkt.length - 4),
574      ]
575  
576      def post_build(self, pkt, pay):
577          if self.length is None:
578              pkt = pkt[:2] + struct.pack("!H", len(pkt)) + pkt[4:]
579          return pkt + pay

621  class TrafficSelector(Packet):
622      @classmethod
623      def dispatch_hook(cls, _pkt=None, *args, **kargs):
624          if _pkt and len(_pkt) >= 16:
625              ts_type = struct.unpack("!B", _pkt[0:1])[0]
626              if ts_type == 7:
627                  return IPv4TrafficSelector
628              elif ts_type == 8:
629                  return IPv6TrafficSelector
630              elif ts_type == 9:
631                  return EncryptedTrafficSelector
632              else:
633                  return RawTrafficSelector
634          return IPv4TrafficSelector
635  
636      def extract_padding(self, s):
637          return '', s
