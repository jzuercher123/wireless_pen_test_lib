37   def send_multiple_ext(sock, ext_id, packet, number_of_packets):
38       # type: (SuperSocket, int, Packet, int) -> None
39       """Send multiple packets with extended addresses at once. 
40    
41       This function is used for scanning with extended addresses. 
42       It sends multiple packets at once. The number of packets 
43       is defined in the number_of_packets variable. 
44       It only iterates the extended ID, NOT the actual CAN ID of the packet. 
45       This method is used in extended scan function. 
46    
47       :param sock: CAN interface to send packets 
48       :param ext_id: Extended ISOTP-Address 
49       :param packet: Template Packet 
50       :param number_of_packets: number of packets to send in one batch 
51       """
52       end_id = min(ext_id + number_of_packets, 255)
53       for i in range(ext_id, end_id + 1):
54           packet.extended_address = i
55           sock.send(packet)

58   def get_isotp_packet(identifier=0x0, extended=False, extended_can_id=False):
59       # type: (int, bool, bool) -> Packet
60       """Craft ISO-TP packet 
61    
62       :param identifier: identifier of crafted packet 
63       :param extended: boolean if packet uses extended address 
64       :param extended_can_id: boolean if CAN should use extended Ids 
65       :return: Crafted Packet 
66       """
67   
68       if extended:
69           pkt = ISOTPHeaderEA() / ISOTP_FF()  # type: Packet
70           pkt.extended_address = 0
71           pkt.data = b'\x00\x00\x00\x00\x00'
72       else:
73           pkt = ISOTPHeader() / ISOTP_FF()
74           pkt.data = b'\x00\x00\x00\x00\x00\x00'
75       if extended_can_id:
76           pkt.flags = "extended"
77   
78       pkt.identifier = identifier
79       pkt.message_size = 100
80       return pkt

83   def filter_periodic_packets(packet_dict):
84       # type: (Dict[int, Tuple[Packet, int]]) -> None
85       """Filter to remove periodic packets from packet_dict 
86    
87       ISOTP-Filter for periodic packets (same ID, always same time-gaps) 
88       Deletes periodic packets in packet_dict 
89    
90       :param packet_dict: Dictionary, where the filter is applied 
91       """
92       filter_dict = {}  # type: Dict[int, Tuple[List[int], List[Packet]]]
93   
94       for key, value in packet_dict.items():
95           pkt = value[0]
96           idn = value[1]
97           if idn not in filter_dict:
98               filter_dict[idn] = ([key], [pkt])
99           else:
100              key_lst, pkt_lst = filter_dict[idn]
101              filter_dict[idn] = (key_lst + [key], pkt_lst + [pkt])
102  
103      for idn in filter_dict:
104          key_lst = filter_dict[idn][0]
105          pkt_lst = filter_dict[idn][1]
106          if len(pkt_lst) < 3:
107              continue
108  
109          tg = [float(p1.time) - float(p2.time)
110                for p1, p2 in zip(pkt_lst[1:], pkt_lst[:-1])]
111          if all(abs(t1 - t2) < 0.001 for t1, t2 in zip(tg[1:], tg[:-1])):
112              log_isotp.info(
113                  "[i] Identifier 0x%03x seems to be periodic. Filtered.")
114              for k in key_lst:
115                  del packet_dict[k]

118  def get_isotp_fc(
119          id_value,  # type: int
120          id_list,  # type: Union[List[int], Dict[int, Tuple[Packet, int]]]
121          noise_ids,  # type: Optional[List[int]]
122          extended,  # type: bool
123          packet,  # type: Packet
124  ):
125      # type: (...) -> None
126      """Callback for sniff function when packet received 
127   
128      If received packet is a FlowControl and not in noise_ids append it 
129      to id_list. 
130   
131      :param id_value: packet id of send packet 
132      :param id_list: list of received IDs 
133      :param noise_ids: list of packet IDs which will not be considered when 
134                        received during scan 
135      :param extended: boolean if extended scan 
136      :param packet: received packet 
137      """
138      if packet.flags and packet.flags != "extended":
139          return
140  
141      if noise_ids is not None and packet.identifier in noise_ids:
142          return
143  
144      try:
145          index = 1 if extended else 0
146          isotp_pci = orb(packet.data[index]) >> 4
147          isotp_fc = orb(packet.data[index]) & 0x0f
148          if isotp_pci == 3 and 0 <= isotp_fc <= 2:
149              log_isotp.info("Found flow-control frame from identifier "
150                             "0x%03x when testing identifier 0x%03x",
151                             packet.identifier, id_value)
152              if isinstance(id_list, dict):
153                  id_list[id_value] = (packet, packet.identifier)
154              elif isinstance(id_list, list):
155                  id_list.append(id_value)
156              else:
157                  raise TypeError("Unknown type of id_list")
158          else:
159              if noise_ids is not None:
160                  noise_ids.append(packet.identifier)
161      except Exception as e:
162          log_isotp.exception(
163              "Unknown message Exception: %s on packet: %s",
164              e, repr(packet))

167  def scan(sock,  # type: SuperSocket
168           scan_range=range(0x800),  # type: Iterable[int]
169           noise_ids=None,  # type: Optional[List[int]]
170           sniff_time=0.1,  # type: float
171           extended_can_id=False,  # type: bool
172           verify_results=True,  # type: bool
173           stop_event=None  # type: Optional[Event]
174           ):  # type: (...) -> Dict[int, Tuple[Packet, int]]
175      """Scan and return dictionary of detections 
176   
177      ISOTP-Scan - NO extended IDs 
178      found_packets = Dictionary with Send-to-ID as 
179      key and a tuple (received packet, Recv_ID) 
180   
181      :param sock: socket for can interface 
182      :param scan_range: hexadecimal range of IDs to scan. Default is 0x0 - 0x7ff 
183      :param noise_ids: list of packet IDs which will not be tested during scan 
184      :param sniff_time: time the scan waits for isotp flow control responses 
185                         after sending a first frame 
186      :param extended_can_id: Send extended can frames 
187      :param verify_results: Verify scan results. This will cause a second scan 
188                             of all possible candidates for ISOTP Sockets 
189      :param stop_event: Event object to asynchronously stop the scan 
190      :return: Dictionary with all found packets 
191      """
192      return_values = dict()  # type: Dict[int, Tuple[Packet, int]]
193      for value in scan_range:
194          if stop_event is not None and stop_event.is_set():
195              break
196          if noise_ids and value in noise_ids:
197              continue
198          sock.send(get_isotp_packet(value, False, extended_can_id))
199          sock.sniff(prn=lambda pkt: get_isotp_fc(value, return_values,
200                                                  noise_ids, False, pkt),
201                     timeout=sniff_time, store=False)
202  
203      if not verify_results:
204          return return_values
205  
206      cleaned_ret_val = dict()  # type: Dict[int, Tuple[Packet, int]]
207      retest_ids = list(set(
208          itertools.chain.from_iterable(
209              range(max(0, i - 2), i + 2) for i in return_values.keys())))
210      for value in retest_ids:
211          if stop_event is not None and stop_event.is_set():
212              break
213          sock.send(get_isotp_packet(value, False, extended_can_id))
214          sock.sniff(prn=lambda pkt: get_isotp_fc(value, cleaned_ret_val,
215                                                  noise_ids, False, pkt),
216                     timeout=sniff_time * 10, store=False)
217  
218      return cleaned_ret_val

221  def scan_extended(sock,  # type: SuperSocket
222                    scan_range=range(0x800),  # type: Iterable[int]
223                    scan_block_size=32,  # type: int
224                    extended_scan_range=range(0x100),  # type: Iterable[int]
225                    noise_ids=None,  # type: Optional[List[int]]
226                    sniff_time=0.1,  # type: float
227                    extended_can_id=False,  # type: bool
228                    stop_event=None  # type: Optional[Event]
229                    ):  # type: (...) -> Dict[int, Tuple[Packet, int]]
230      """Scan with ISOTP extended addresses and return dictionary of detections 
231   
232      If an answer-packet found -> slow scan with 
233      single packages with extended ID 0 - 255 
234      found_packets = Dictionary with Send-to-ID 
235      as key and a tuple (received packet, Recv_ID) 
236   
237      :param sock: socket for can interface 
238      :param scan_range: hexadecimal range of IDs to scan. Default is 0x0 - 0x7ff 
239      :param scan_block_size: count of packets send at once 
240      :param extended_scan_range: range to search for extended ISOTP addresses 
241      :param noise_ids: list of packet IDs which will not be tested during scan 
242      :param sniff_time: time the scan waits for isotp flow control responses 
243                         after sending a first frame 
244      :param extended_can_id: Send extended can frames 
245      :param stop_event: Event object to asynchronously stop the scan 
246      :return: Dictionary with all found packets 
247      """
248      return_values = dict()  # type: Dict[int, Tuple[Packet, int]]
249      scan_block_size = scan_block_size or 1
250      r = list(extended_scan_range)
251  
252      for value in scan_range:
253          if noise_ids and value in noise_ids:
254              continue
255  
256          pkt = get_isotp_packet(
257              value, extended=True, extended_can_id=extended_can_id)
258          id_list = []  # type: List[int]
259          for ext_isotp_id in range(r[0], r[-1], scan_block_size):
260              if stop_event is not None and stop_event.is_set():
261                  break
262              send_multiple_ext(sock, ext_isotp_id, pkt, scan_block_size)
263              sock.sniff(prn=lambda p: get_isotp_fc(ext_isotp_id, id_list,
264                                                    noise_ids, True, p),
265                         timeout=sniff_time * 3, store=False)
266              # sleep to prevent flooding
267              time.sleep(sniff_time)
268  
269          # remove duplicate IDs
270          id_list = list(set(id_list))
271          for ext_isotp_id in id_list:
272              if stop_event is not None and stop_event.is_set():
273                  break
274              for ext_id in range(max(ext_isotp_id - 2, 0),
275                                  min(ext_isotp_id + scan_block_size + 2, 256)):
276                  if stop_event is not None and stop_event.is_set():
277                      break
278                  pkt.extended_address = ext_id
279                  full_id = (value << 8) + ext_id
280                  sock.send(pkt)
281                  sock.sniff(prn=lambda pkt: get_isotp_fc(full_id,
282                                                          return_values,
283                                                          noise_ids, True,
284                                                          pkt),
285                             timeout=sniff_time * 2, store=False)
286  
287      return return_values

290  def isotp_scan(sock,  # type: SuperSocket
291                 scan_range=range(0x7ff + 1),  # type: Iterable[int]
292                 extended_addressing=False,  # type: bool
293                 extended_scan_range=range(0x100),  # type: Iterable[int]
294                 noise_listen_time=2,  # type: int
295                 sniff_time=0.1,  # type: float
296                 output_format=None,  # type: Optional[str]
297                 can_interface=None,  # type: Optional[str]
298                 extended_can_id=False,  # type: bool
299                 verify_results=True,  # type: bool
300                 verbose=False,  # type: bool
301                 stop_event=None  # type: Optional[Event]
302                 ):
303      # type: (...) -> Union[str, List[SuperSocket]]
304      """Scan for ISOTP Sockets on a bus and return findings 
305   
306      Scan for ISOTP Sockets in the defined range and returns found sockets 
307      in a specified format. The format can be: 
308   
309      - text: human readable output 
310      - code: python code for copy&paste 
311      - json: json string 
312      - sockets: if output format is not specified, ISOTPSockets will be 
313        created and returned in a list 
314   
315      :param sock: CANSocket object to communicate with the bus under scan 
316      :param scan_range: range of CAN-Identifiers to scan. Default is 0x0 - 0x7ff 
317      :param extended_addressing: scan with ISOTP extended addressing 
318      :param extended_scan_range: range for ISOTP extended addressing values 
319      :param noise_listen_time: seconds to listen for default communication on 
320                                the bus 
321      :param sniff_time: time the scan waits for isotp flow control responses 
322                         after sending a first frame 
323      :param output_format: defines the format of the returned results 
324                            (text, code or sockets). Provide a string e.g. 
325                            "text". Default is "socket". 
326      :param can_interface: interface used to create the returned code/sockets 
327      :param extended_can_id: Use Extended CAN-Frames 
328      :param verify_results: Verify scan results. This will cause a second scan 
329                             of all possible candidates for ISOTP Sockets 
330      :param verbose: displays information during scan 
331      :param stop_event: Event object to asynchronously stop the scan 
332      :return: 
333      """
334      if verbose:
335          log_isotp.setLevel(logging.DEBUG)
336  
337      log_isotp.info("Filtering background noise...")
338  
339      # Send dummy packet. In most cases, this triggers activity on the bus.
340  
341      dummy_pkt = CAN(identifier=0x123,
342                      data=b'\xaa\xbb\xcc\xdd\xee\xff\xaa\xbb')
343  
344      background_pkts = sock.sniff(
345          timeout=noise_listen_time,
346          started_callback=lambda: sock.send(dummy_pkt))
347  
348      noise_ids = list(set(pkt.identifier for pkt in background_pkts))
349  
350      if extended_addressing:
351          found_packets = scan_extended(sock, scan_range,
352                                        extended_scan_range=extended_scan_range,
353                                        noise_ids=noise_ids,
354                                        sniff_time=sniff_time,
355                                        extended_can_id=extended_can_id,
356                                        stop_event=stop_event)
357      else:
358          found_packets = scan(sock, scan_range,
359                               noise_ids=noise_ids,
360                               sniff_time=sniff_time,
361                               extended_can_id=extended_can_id,
362                               verify_results=verify_results,
363                               stop_event=stop_event)
364  
365      filter_periodic_packets(found_packets)
366  
367      if output_format == "text":
368          return generate_text_output(found_packets, extended_addressing)
369  
370      if output_format == "code":
371          return generate_code_output(found_packets, can_interface,
372                                      extended_addressing)
373  
374      if output_format == "json":
375          return generate_json_output(found_packets, can_interface,
376                                      extended_addressing)
377  
378      return generate_isotp_list(found_packets, can_interface or sock,
379                                 extended_addressing)

382  def generate_text_output(found_packets, extended_addressing=False):
383      # type: (Dict[int, Tuple[Packet, int]], bool) -> str
384      """Generate a human readable output from the result of the `scan` or the 
385      `scan_extended` function. 
386   
387      :param found_packets: result of the `scan` or `scan_extended` function 
388      :param extended_addressing: print results from a scan with 
389                                  ISOTP extended addressing 
390      :return: human readable scan results 
391      """
392      if not found_packets:
393          return "No packets found."
394  
395      text = "\nFound %s ISOTP-FlowControl Packet(s):" % len(found_packets)
396      for pack in found_packets:
397          if extended_addressing:
398              send_id = pack // 256
399              send_ext = pack - (send_id * 256)
400              ext_id = hex(orb(found_packets[pack][0].data[0]))
401              text += "\nSend to ID:             %s" \
402                      "\nSend to extended ID:    %s" \
403                      "\nReceived ID:            %s" \
404                      "\nReceived extended ID:   %s" \
405                      "\nMessage:                %s" % \
406                      (hex(send_id), hex(send_ext),
407                       hex(found_packets[pack][0].identifier), ext_id,
408                       repr(found_packets[pack][0]))
409          else:
410              text += "\nSend to ID:             %s" \
411                      "\nReceived ID:            %s" \
412                      "\nMessage:                %s" % \
413                      (hex(pack),
414                       hex(found_packets[pack][0].identifier),
415                       repr(found_packets[pack][0]))
416  
417          padding = found_packets[pack][0].length == 8
418          if padding:
419              text += "\nPadding enabled"
420          else:
421              text += "\nNo Padding"
422  
423          text += "\n"
424      return text

427  def generate_code_output(found_packets, can_interface="iface",
428                           extended_addressing=False):
429      # type: (Dict[int, Tuple[Packet, int]], Optional[str], bool) -> str
430      """Generate a copy&past-able output from the result of the `scan` or 
431      the `scan_extended` function. 
432   
433      :param found_packets: result of the `scan` or `scan_extended` function 
434      :param can_interface: description string for a CAN interface to be 
435                            used for the creation of the output. 
436      :param extended_addressing: print results from a scan with ISOTP 
437                                  extended addressing 
438      :return: Python-code as string to generate all found sockets 
439      """
440      result = ""
441      if not found_packets:
442          return result
443  
444      header = "\n\nimport can\n" \
445               "conf.contribs['CANSocket'] = {'use-python-can': %s}\n" \
446               "load_contrib('cansocket')\n" \
447               "load_contrib('isotp')\n\n" % PYTHON_CAN
448  
449      for pack in found_packets:
450          if extended_addressing:
451              send_id = pack // 256
452              send_ext = pack - (send_id * 256)
453              ext_id = orb(found_packets[pack][0].data[0])
454              result += "ISOTPSocket(%s, tx_id=0x%x, rx_id=0x%x, padding=%s, " \
455                        "ext_address=0x%x, rx_ext_address=0x%x, " \
456                        "basecls=ISOTP)\n" % \
457                        (can_interface, send_id,
458                         int(found_packets[pack][0].identifier),
459                         found_packets[pack][0].length == 8,
460                         send_ext,
461                         ext_id)
462  
463          else:
464              result += "ISOTPSocket(%s, tx_id=0x%x, rx_id=0x%x, padding=%s, " \
465                        "basecls=ISOTP)\n" % \
466                        (can_interface, pack,
467                         int(found_packets[pack][0].identifier),
468                         found_packets[pack][0].length == 8)
469      return header + result

472  def generate_json_output(found_packets,  # type: Dict[int, Tuple[Packet, int]]
473                           can_interface="iface",  # type: Optional[str]
474                           extended_addressing=False  # type: bool
475                           ):
476      # type: (...) -> str
477      """Generate a list of ISOTPSocket objects from the result of the `scan` or 
478      the `scan_extended` function. 
479   
480      :param found_packets: result of the `scan` or `scan_extended` function 
481      :param can_interface: description string for a CAN interface to be 
482                            used for the creation of the output. 
483      :param extended_addressing: print results from a scan with ISOTP 
484                                  extended addressing 
485      :return: A list of all found ISOTPSockets 
486      """
487      socket_list = []  # type: List[Dict[str, Any]]
488      for pack in found_packets:
489          pkt = found_packets[pack][0]
490  
491          dest_id = pkt.identifier
492          pad = True if pkt.length == 8 else False
493  
494          if extended_addressing:
495              source_id = pack >> 8
496              source_ext = int(pack - (source_id * 256))
497              dest_ext = orb(pkt.data[0])
498              socket_list.append({"iface": can_interface,
499                                  "tx_id": source_id,
500                                  "ext_address": source_ext,
501                                  "rx_id": dest_id,
502                                  "rx_ext_address": dest_ext,
503                                  "padding": pad,
504                                  "basecls": ISOTP.__name__})
505          else:
506              source_id = pack
507              socket_list.append({"iface": can_interface,
508                                  "tx_id": source_id,
509                                  "rx_id": dest_id,
510                                  "padding": pad,
511                                  "basecls": ISOTP.__name__})
512      return json.dumps(socket_list)

515  def generate_isotp_list(found_packets,  # type: Dict[int, Tuple[Packet, int]]
516                          can_interface,  # type: Union[SuperSocket, str]
517                          extended_addressing=False  # type: bool
518                          ):
519      # type: (...) -> List[SuperSocket]
520      """Generate a list of ISOTPSocket objects from the result of the `scan` or 
521      the `scan_extended` function. 
522   
523      :param found_packets: result of the `scan` or `scan_extended` function 
524      :param can_interface: description string for a CAN interface to be 
525                            used for the creation of the output. 
526      :param extended_addressing: print results from a scan with ISOTP 
527                                  extended addressing 
528      :return: A list of all found ISOTPSockets 
529      """
530      from scapy.contrib.isotp import ISOTPSocket
531  
532      socket_list = []  # type: List[SuperSocket]
533      for pack in found_packets:
534          pkt = found_packets[pack][0]
535  
536          dest_id = pkt.identifier
537          pad = True if pkt.length == 8 else False
538  
539          if extended_addressing:
540              source_id = pack >> 8
541              source_ext = int(pack - (source_id * 256))
542              dest_ext = orb(pkt.data[0])
543              socket_list.append(ISOTPSocket(can_interface, tx_id=source_id,
544                                             ext_address=source_ext,
545                                             rx_id=dest_id,
546                                             rx_ext_address=dest_ext,
547                                             padding=pad,
548                                             basecls=ISOTP))
549          else:
550              source_id = pack
551              socket_list.append(ISOTPSocket(can_interface, tx_id=source_id,
552                                             rx_id=dest_id, padding=pad,
553                                             basecls=ISOTP))
554      return socket_list
555
