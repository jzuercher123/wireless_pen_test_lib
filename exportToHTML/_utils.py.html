105  def deep_update(mapping: dict[KeyType, Any], *updating_mappings: dict[KeyType, Any]) -> dict[KeyType, Any]:
106      updated_mapping = mapping.copy()
107      for updating_mapping in updating_mappings:
108          for k, v in updating_mapping.items():
109              if k in updated_mapping and isinstance(updated_mapping[k], dict) and isinstance(v, dict):
110                  updated_mapping[k] = deep_update(updated_mapping[k], v)
111              else:
112                  updated_mapping[k] = v
113      return updated_mapping

123  def unique_list(
124      input_list: list[T] | tuple[T, ...],
125      *,
126      name_factory: typing.Callable[[T], str] = str,
127  ) -> list[T]:
128      """Make a list unique while maintaining order. 
129      We update the list if another one with the same name is set 
130      (e.g. model validator overridden in subclass). 
131      """
132      result: list[T] = []
133      result_names: list[str] = []
134      for v in input_list:
135          v_name = name_factory(v)
136          if v_name not in result_names:
137              result_names.append(v_name)
138              result.append(v)
139          else:
140              result[result_names.index(v_name)] = v
141  
142      return result

145  class ValueItems(_repr.Representation):
146      """Class for more convenient calculation of excluded or included fields on values."""
147  
148      __slots__ = ('_items', '_type')
149  
150      def __init__(self, value: Any, items: AbstractSetIntStr | MappingIntStrAny) -> None:
151          items = self._coerce_items(items)
152  
153          if isinstance(value, (list, tuple)):
154              items = self._normalize_indexes(items, len(value))  # type: ignore
155  
156          self._items: MappingIntStrAny = items  # type: ignore
157  
158      def is_excluded(self, item: Any) -> bool:
159          """Check if item is fully excluded. 
160   
161          :param item: key or index of a value 
162          """
163          return self.is_true(self._items.get(item))
164  
165      def is_included(self, item: Any) -> bool:
166          """Check if value is contained in self._items. 
167   
168          :param item: key or index of value 
169          """
170          return item in self._items
171  
172      def for_element(self, e: int | str) -> AbstractSetIntStr | MappingIntStrAny | None:
173          """:param e: key or index of element on value 
174          :return: raw values for element if self._items is dict and contain needed element 
175          """
176          item = self._items.get(e)  # type: ignore
177          return item if not self.is_true(item) else None
178  
179      def _normalize_indexes(self, items: MappingIntStrAny, v_length: int) -> dict[int | str, Any]:
180          """:param items: dict or set of indexes which will be normalized 
181          :param v_length: length of sequence indexes of which will be 
182   
183          >>> self._normalize_indexes({0: True, -2: True, -1: True}, 4) 
184          {0: True, 2: True, 3: True} 
185          >>> self._normalize_indexes({'__all__': True}, 4) 
186          {0: True, 1: True, 2: True, 3: True} 
187          """
188          normalized_items: dict[int | str, Any] = {}
189          all_items = None
190          for i, v in items.items():
191              if not (isinstance(v, typing.Mapping) or isinstance(v, typing.AbstractSet) or self.is_true(v)):
192                  raise TypeError(f'Unexpected type of exclude value for index "{i}" {v.__class__}')
193              if i == '__all__':
194                  all_items = self._coerce_value(v)
195                  continue
196              if not isinstance(i, int):
197                  raise TypeError(
198                      'Excluding fields from a sequence of sub-models or dicts must be performed index-wise: '
199                      'expected integer keys or keyword "__all__"'
200                  )
201              normalized_i = v_length + i if i < 0 else i
202              normalized_items[normalized_i] = self.merge(v, normalized_items.get(normalized_i))
203  
204          if not all_items:
205              return normalized_items
206          if self.is_true(all_items):
207              for i in range(v_length):
208                  normalized_items.setdefault(i, ...)
209              return normalized_items
210          for i in range(v_length):
211              normalized_item = normalized_items.setdefault(i, {})
212              if not self.is_true(normalized_item):
213                  normalized_items[i] = self.merge(all_items, normalized_item)
214          return normalized_items
215  
216      @classmethod
217      def merge(cls, base: Any, override: Any, intersect: bool = False) -> Any:
218          """Merge a `base` item with an `override` item. 
219   
220          Both `base` and `override` are converted to dictionaries if possible. 
221          Sets are converted to dictionaries with the sets entries as keys and 
222          Ellipsis as values. 
223   
224          Each key-value pair existing in `base` is merged with `override`, 
225          while the rest of the key-value pairs are updated recursively with this function. 
226   
227          Merging takes place based on the "union" of keys if `intersect` is 
228          set to `False` (default) and on the intersection of keys if 
229          `intersect` is set to `True`. 
230          """
231          override = cls._coerce_value(override)
232          base = cls._coerce_value(base)
233          if override is None:
234              return base
235          if cls.is_true(base) or base is None:
236              return override
237          if cls.is_true(override):
238              return base if intersect else override
239  
240          # intersection or union of keys while preserving ordering:
241          if intersect:
242              merge_keys = [k for k in base if k in override] + [k for k in override if k in base]
243          else:
244              merge_keys = list(base) + [k for k in override if k not in base]
245  
246          merged: dict[int | str, Any] = {}
247          for k in merge_keys:
248              merged_item = cls.merge(base.get(k), override.get(k), intersect=intersect)
249              if merged_item is not None:
250                  merged[k] = merged_item
251  
252          return merged
253  
254      @staticmethod
255      def _coerce_items(items: AbstractSetIntStr | MappingIntStrAny) -> MappingIntStrAny:
256          if isinstance(items, typing.Mapping):
257              pass
258          elif isinstance(items, typing.AbstractSet):
259              items = dict.fromkeys(items, ...)  # type: ignore
260          else:
261              class_name = getattr(items, '__class__', '???')
262              raise TypeError(f'Unexpected type of exclude value {class_name}')
263          return items  # type: ignore
264  
265      @classmethod
266      def _coerce_value(cls, value: Any) -> Any:
267          if value is None or cls.is_true(value):
268              return value
269          return cls._coerce_items(value)

272      def is_true(v: Any) -> bool:
273          return v is True or v is ...

303  def smart_deepcopy(obj: Obj) -> Obj:
304      """Return type as is for immutable built-in types 
305      Use obj.copy() for built-in empty collections 
306      Use copy.deepcopy() for non-empty collections and unknown objects. 
307      """
308      obj_type = obj.__class__
309      if obj_type in IMMUTABLE_NON_COLLECTIONS_TYPES:
310          return obj  # fastest case: obj is immutable and not collection therefore will not be copied anyway
311      try:
312          if not obj and obj_type in BUILTIN_COLLECTIONS:
313              # faster way for empty collections, no need to copy its members
314              return obj if obj_type is tuple else obj.copy()  # tuple doesn't have copy method  # type: ignore
315      except (TypeError, ValueError, RuntimeError):
316          # do we really dare to catch ALL errors? Seems a bit risky
317          pass
318  
319      return deepcopy(obj)  # slowest way when we actually might need a deepcopy

322  _SENTINEL = object()
323  
324  
325  def all_identical(left: typing.Iterable[Any], right: typing.Iterable[Any]) -> bool:
326      """Check that the items of `left` are the same objects as those in `right`. 
327   
328      >>> a, b = object(), object() 
329      >>> all_identical([a, b, a], [a, b, a]) 
330      True 
331      >>> all_identical([a, b, [a]], [a, b, [a]])  # new list object, while "equal" is not "identical" 
332      False 
333      """
334      for left_item, right_item in zip_longest(left, right, fillvalue=_SENTINEL):
335          if left_item is not right_item:
336              return False
337      return True

340  @dataclasses.dataclass(frozen=True)
341  class SafeGetItemProxy:
342      """Wrapper redirecting `__getitem__` to `get` with a sentinel value as default 
343   
344      This makes is safe to use in `operator.itemgetter` when some keys may be missing 
345      """
346  
347      # Define __slots__manually for performances
348      # @dataclasses.dataclass() only support slots=True in python>=3.10
349      __slots__ = ('wrapped',)
350  
351      wrapped: Mapping[str, Any]
352  
353      def __getitem__(self, key: str, /) -> Any:
354          return self.wrapped.get(key, _SENTINEL)
355  
356      # required to pass the object to operator.itemgetter() instances due to a quirk of typeshed
357      # https://github.com/python/mypy/issues/13713
358      # https://github.com/python/typeshed/pull/8785
359      # Since this is typing-only, hide it in a typing.TYPE_CHECKING block
360      if typing.TYPE_CHECKING:
361  
362          def __contains__(self, key: str, /) -> bool:
363              return self.wrapped.__contains__(key)
