128      def get_args(t: Type[Any]) -> Tuple[Any, ...]:
129          """Compatibility version of get_args for python 3.7. 
130   
131          Mostly compatible with the python 3.8 `typing` module version 
132          and able to handle almost all use cases. 
133          """
134          if type(t).__name__ in AnnotatedTypeNames:
135              return t.__args__ + t.__metadata__
136          if isinstance(t, _GenericAlias):
137              res = t.__args__
138              if t.__origin__ is Callable and res and res[0] is not Ellipsis:
139                  res = (list(res[:-1]), res[-1])
140              return res
141          return getattr(t, '__args__', ())

146      def _generic_get_args(tp: Type[Any]) -> Tuple[Any, ...]:
147          """ 
148          In python 3.9, `typing.Dict`, `typing.List`, ... 
149          do have an empty `__args__` by default (instead of the generic ~T for example). 
150          In order to still support `Dict` for example and consider it as `Dict[Any, Any]`, 
151          we retrieve the `_nparams` value that tells us how many parameters it needs. 
152          """
153          if hasattr(tp, '_nparams'):
154              return (Any,) * tp._nparams
155          # Special case for `tuple[()]`, which used to return ((),) with `typing.Tuple`
156          # in python 3.10- but now returns () for `tuple` and `Tuple`.
157          # This will probably be clarified in pydantic v2
158          try:
159              if tp == Tuple[()] or sys.version_info >= (3, 9) and tp == tuple[()]:  # type: ignore[misc]
160                  return ((),)
161          # there is a TypeError when compiled with cython
162          except TypeError:  # pragma: no cover
163              pass
164          return ()

166      def get_args(tp: Type[Any]) -> Tuple[Any, ...]:
167          """Get type arguments with all substitutions performed. 
168   
169          For unions, basic simplifications used by Union constructor are performed. 
170          Examples:: 
171              get_args(Dict[str, int]) == (str, int) 
172              get_args(int) == () 
173              get_args(Union[int, Union[T, int], str][int]) == (int, str) 
174              get_args(Union[int, Tuple[T, int]][str]) == (int, Tuple[str, int]) 
175              get_args(Callable[[], T][int]) == ([], int) 
176          """
177          if type(tp).__name__ in AnnotatedTypeNames:
178              return tp.__args__ + tp.__metadata__
179          # the fallback is needed for the same reasons as `get_origin` (see above)
180          return _typing_get_args(tp) or getattr(tp, '__args__', ()) or _generic_get_args(tp)

199      def convert_generics(tp: Type[Any]) -> Type[Any]:
200          """ 
201          Recursively searches for `str` type hints and replaces them with ForwardRef. 
202   
203          Examples:: 
204              convert_generics(list['Hero']) == list[ForwardRef('Hero')] 
205              convert_generics(dict['Hero', 'Team']) == dict[ForwardRef('Hero'), ForwardRef('Team')] 
206              convert_generics(typing.Dict['Hero', 'Team']) == typing.Dict[ForwardRef('Hero'), ForwardRef('Team')] 
207              convert_generics(list[str | 'Hero'] | int) == list[str | ForwardRef('Hero')] | int 
208          """
209          origin = get_origin(tp)
210          if not origin or not hasattr(tp, '__args__'):
211              return tp
212  
213          args = get_args(tp)
214  
215          # typing.Annotated needs special treatment
216          if origin is Annotated:
217              return _AnnotatedAlias(convert_generics(args[0]), args[1:])
218  
219          # recursively replace `str` instances inside of `GenericAlias` with `ForwardRef(arg)`
220          converted = tuple(
221              ForwardRef(arg) if isinstance(arg, str) and isinstance(tp, TypingGenericAlias) else convert_generics(arg)
222              for arg in args
223          )
224  
225          if converted == args:
226              return tp
227          elif isinstance(tp, TypingGenericAlias):
228              return TypingGenericAlias(origin, converted)
229          elif isinstance(tp, TypesUnionType):
230              # recreate types.UnionType (PEP604, Python >= 3.10)
231              return _UnionGenericAlias(origin, converted)
232          else:
233              try:
234                  setattr(tp, '__args__', converted)
235              except AttributeError:
236                  pass
237              return tp

360  def display_as_type(v: Type[Any]) -> str:
361      if not isinstance(v, typing_base) and not isinstance(v, WithArgsTypes) and not isinstance(v, type):
362          v = v.__class__
363  
364      if is_union(get_origin(v)):
365          return f'Union[{", ".join(map(display_as_type, get_args(v)))}]'
366  
367      if isinstance(v, WithArgsTypes):
368          # Generic alias are constructs like `list[int]`
369          return str(v).replace('typing.', '')

371      try:
372          return v.__name__

375          return str(v).replace('typing.', '')

378  def resolve_annotations(raw_annotations: Dict[str, Type[Any]], module_name: Optional[str]) -> Dict[str, Type[Any]]:
379      """ 
380      Partially taken from typing.get_type_hints. 
381   
382      Resolve string or ForwardRef annotations into type objects if possible. 
383      """
384      base_globals: Optional[Dict[str, Any]] = None
385      if module_name:
386          try:
387              module = sys.modules[module_name]
388          except KeyError:
389              # happens occasionally, see https://github.com/pydantic/pydantic/issues/2363
390              pass
391          else:
392              base_globals = module.__dict__
393  
394      annotations = {}
395      for name, value in raw_annotations.items():
396          if isinstance(value, str):
397              if (3, 10) > sys.version_info >= (3, 9, 8) or sys.version_info >= (3, 10, 1):
398                  value = ForwardRef(value, is_argument=False, is_class=True)
399              else:
400                  value = ForwardRef(value, is_argument=False)
401          try:
402              if sys.version_info >= (3, 13):
403                  value = _eval_type(value, base_globals, None, type_params=())
404              else:
405                  value = _eval_type(value, base_globals, None)
406          except NameError:
407              # this is ok, it can be fixed with update_forward_refs
408              pass
409          annotations[name] = value
410      return annotations

425  def all_literal_values(type_: Type[Any]) -> Tuple[Any, ...]:
426      """ 
427      This method is used to retrieve all Literal values as 
428      Literal can be used recursively (see https://www.python.org/dev/peps/pep-0586) 
429      e.g. `Literal[Literal[Literal[1, 2, 3], "foo"], 5, None]` 
430      """
431      if not is_literal_type(type_):
432          return (type_,)
433  
434      values = literal_values(type_)
435      return tuple(x for value in values for x in all_literal_values(value))

518  def update_field_forward_refs(field: 'ModelField', globalns: Any, localns: Any) -> None:
519      """ 
520      Try to update ForwardRefs on fields based on this ModelField, globalns and localns. 
521      """
522      prepare = False
523      if field.type_.__class__ == ForwardRef:
524          prepare = True
525          field.type_ = evaluate_forwardref(field.type_, globalns, localns or None)
526      if field.outer_type_.__class__ == ForwardRef:
527          prepare = True
528          field.outer_type_ = evaluate_forwardref(field.outer_type_, globalns, localns or None)
529      if prepare:
530          field.prepare()
531  
532      if field.sub_fields:
533          for sub_f in field.sub_fields:
534              update_field_forward_refs(sub_f, globalns=globalns, localns=localns)
535  
536      if field.discriminator_key is not None:
537          field.prepare_discriminated_union_sub_fields()

540  def update_model_forward_refs(
541      model: Type[Any],
542      fields: Iterable['ModelField'],
543      json_encoders: Dict[Union[Type[Any], str, ForwardRef], AnyCallable],
544      localns: 'DictStrAny',
545      exc_to_suppress: Tuple[Type[BaseException], ...] = (),
546  ) -> None:
547      """ 
548      Try to update model fields ForwardRefs based on model and localns. 
549      """
550      if model.__module__ in sys.modules:
551          globalns = sys.modules[model.__module__].__dict__.copy()
552      else:
553          globalns = {}
554  
555      globalns.setdefault(model.__name__, model)
556  
557      for f in fields:
558          try:
559              update_field_forward_refs(f, globalns=globalns, localns=localns)
560          except exc_to_suppress:
561              pass
562  
563      for key in set(json_encoders.keys()):
564          if isinstance(key, str):
565              fr: ForwardRef = ForwardRef(key)
566          elif isinstance(key, ForwardRef):
567              fr = key
568          else:
569              continue
570  
571          try:
572              new_key = evaluate_forwardref(fr, globalns, localns or None)
573          except exc_to_suppress:  # pragma: no cover
574              continue
575  
576          json_encoders[new_key] = json_encoders.pop(key)

579  def get_class(type_: Type[Any]) -> Union[None, bool, Type[Any]]:
580      """ 
581      Tries to get the class of a Type[T] annotation. Returns True if Type is used 
582      without brackets. Otherwise returns None. 
583      """
584      if type_ is type:
585          return True
586  
587      if get_origin(type_) is None:
588          return None
589  
590      args = get_args(type_)
591      if not args or not isinstance(args[0], type):
592          return True
593      else:
594          return args[0]

597  def get_sub_types(tp: Any) -> List[Any]:
598      """ 
599      Return all the types that are allowed by type `tp` 
600      `tp` can be a `Union` of allowed types or an `Annotated` type 
601      """
602      origin = get_origin(tp)
603      if origin is Annotated:
604          return get_sub_types(get_args(tp)[0])
605      elif is_union(origin):
606          return [x for t in get_args(tp) for x in get_sub_types(t)]
607      else:
608          return [tp]
