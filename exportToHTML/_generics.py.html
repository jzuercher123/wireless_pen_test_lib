44       class LimitedDict(dict):
45           """Limit the size/length of a dict used for caching to avoid unlimited increase in memory usage. 
46    
47           Since the dict is ordered, and we always remove elements from the beginning, this is effectively a FIFO cache. 
48           """
49   
50           def __init__(self, size_limit: int = _LIMITED_DICT_SIZE):
51               self.size_limit = size_limit
52               super().__init__()
53   
54           def __setitem__(self, key: Any, value: Any, /) -> None:
55               super().__setitem__(key, value)
56               if len(self) > self.size_limit:
57                   excess = len(self) - self.size_limit + self.size_limit // 10
58                   to_remove = list(self.keys())[:excess]
59                   for k in to_remove:
60                       del self[k]

93               hit = False
94               for mapping in self.maps:
95                   if key in mapping:
96                       del mapping[key]
97                       hit = True
98               if not hit:
99                   raise KeyError(key)

115  def create_generic_submodel(
116      model_name: str, origin: type[BaseModel], args: tuple[Any, ...], params: tuple[Any, ...]
117  ) -> type[BaseModel]:
118      """Dynamically create a submodel of a provided (generic) BaseModel. 
119   
120      This is used when producing concrete parametrizations of generic models. This function 
121      only *creates* the new subclass; the schema/validators/serialization must be updated to 
122      reflect a concrete parametrization elsewhere. 
123   
124      Args: 
125          model_name: The name of the newly created model. 
126          origin: The base class for the new model to inherit from. 
127          args: A tuple of generic metadata arguments. 
128          params: A tuple of generic metadata parameters. 
129   
130      Returns: 
131          The created submodel. 
132      """
133      namespace: dict[str, Any] = {'__module__': origin.__module__}
134      bases = (origin,)
135      meta, ns, kwds = prepare_class(model_name, bases)
136      namespace.update(ns)
137      created_model = meta(
138          model_name,
139          bases,
140          namespace,
141          __pydantic_generic_metadata__={
142              'origin': origin,
143              'args': args,
144              'parameters': params,
145          },
146          __pydantic_reset_parent_namespace__=False,
147          **kwds,
148      )
149  
150      model_module, called_globally = _get_caller_frame_info(depth=3)
151      if called_globally:  # create global reference and therefore allow pickling
152          object_by_reference = None
153          reference_name = model_name
154          reference_module_globals = sys.modules[created_model.__module__].__dict__
155          while object_by_reference is not created_model:
156              object_by_reference = reference_module_globals.setdefault(reference_name, created_model)
157              reference_name += '_'
158  
159      return created_model

162  def _get_caller_frame_info(depth: int = 2) -> tuple[str | None, bool]:
163      """Used inside a function to check whether it was called globally. 
164   
165      Args: 
166          depth: The depth to get the frame. 
167   
168      Returns: 
169          A tuple contains `module_name` and `called_globally`. 
170   
171      Raises: 
172          RuntimeError: If the function is not called inside a function. 
173      """
174      try:
175          previous_caller_frame = sys._getframe(depth)
176      except ValueError as e:
177          raise RuntimeError('This function must be used inside another function') from e
178      except AttributeError:  # sys module does not have _getframe function, so there's nothing we can do about it
179          return None, False
180      frame_globals = previous_caller_frame.f_globals
181      return frame_globals.get('__name__'), previous_caller_frame.f_locals is frame_globals

187  def iter_contained_typevars(v: Any) -> Iterator[TypeVarType]:
188      """Recursively iterate through all subtypes and type args of `v` and yield any typevars that are found. 
189   
190      This is inspired as an alternative to directly accessing the `__parameters__` attribute of a GenericAlias, 
191      since __parameters__ of (nested) generic BaseModel subclasses won't show up in that list. 
192      """
193      if isinstance(v, TypeVar):
194          yield v
195      elif is_model_class(v):
196          yield from v.__pydantic_generic_metadata__['parameters']
197      elif isinstance(v, (DictValues, list)):
198          for var in v:
199              yield from iter_contained_typevars(var)
200      else:
201          args = get_args(v)
202          for arg in args:
203              yield from iter_contained_typevars(arg)
204

206  def get_args(v: Any) -> Any:
207      pydantic_generic_metadata: PydanticGenericMetadata | None = getattr(v, '__pydantic_generic_metadata__', None)
208      if pydantic_generic_metadata:
209          return pydantic_generic_metadata.get('args')
210      return typing_extensions.get_args(v)

213  def get_origin(v: Any) -> Any:
214      pydantic_generic_metadata: PydanticGenericMetadata | None = getattr(v, '__pydantic_generic_metadata__', None)
215      if pydantic_generic_metadata:
216          return pydantic_generic_metadata.get('origin')
217      return typing_extensions.get_origin(v)

220  def get_standard_typevars_map(cls: type[Any]) -> dict[TypeVarType, Any] | None:
221      """Package a generic type's typevars and parametrization (if present) into a dictionary compatible with the 
222      `replace_types` function. Specifically, this works with standard typing generics and typing._GenericAlias. 
223      """
224      origin = get_origin(cls)
225      if origin is None:
226          return None
227      if not hasattr(origin, '__parameters__'):
228          return None
229  
230      # In this case, we know that cls is a _GenericAlias, and origin is the generic type
231      # So it is safe to access cls.__args__ and origin.__parameters__
232      args: tuple[Any, ...] = cls.__args__  # type: ignore
233      parameters: tuple[TypeVarType, ...] = origin.__parameters__
234      return dict(zip(parameters, args))

237  def get_model_typevars_map(cls: type[BaseModel]) -> dict[TypeVarType, Any] | None:
238      """Package a generic BaseModel's typevars and concrete parametrization (if present) into a dictionary compatible 
239      with the `replace_types` function. 
240   
241      Since BaseModel.__class_getitem__ does not produce a typing._GenericAlias, and the BaseModel generic info is 
242      stored in the __pydantic_generic_metadata__ attribute, we need special handling here. 
243      """
244      # TODO: This could be unified with `get_standard_typevars_map` if we stored the generic metadata
245      #   in the __origin__, __args__, and __parameters__ attributes of the model.
246      generic_metadata = cls.__pydantic_generic_metadata__
247      origin = generic_metadata['origin']
248      args = generic_metadata['args']
249      return dict(zip(iter_contained_typevars(origin), args))

252  def replace_types(type_: Any, type_map: Mapping[Any, Any] | None) -> Any:
253      """Return type with all occurrences of `type_map` keys recursively replaced with their values. 
254   
255      Args: 
256          type_: The class or generic alias. 
257          type_map: Mapping from `TypeVar` instance to concrete types. 
258   
259      Returns: 
260          A new type representing the basic structure of `type_` with all 
261          `typevar_map` keys recursively replaced. 
262   
263      Example: 
264          ```py 
265          from typing import List, Tuple, Union 
266   
267          from pydantic._internal._generics import replace_types 
268   
269          replace_types(Tuple[str, Union[List[str], float]], {str: int}) 
270          #> Tuple[int, Union[List[int], float]] 
271          ``` 
272      """
273      if not type_map:
274          return type_
275  
276      type_args = get_args(type_)
277      origin_type = get_origin(type_)
278  
279      if origin_type is typing_extensions.Annotated:
280          annotated_type, *annotations = type_args
281          annotated = replace_types(annotated_type, type_map)
282          for annotation in annotations:
283              annotated = typing_extensions.Annotated[annotated, annotation]
284          return annotated
285  
286      # Having type args is a good indicator that this is a typing module
287      # class instantiation or a generic alias of some sort.
288      if type_args:
289          resolved_type_args = tuple(replace_types(arg, type_map) for arg in type_args)
290          if all_identical(type_args, resolved_type_args):
291              # If all arguments are the same, there is no need to modify the
292              # type or create a new object at all
293              return type_
294          if (
295              origin_type is not None
296              and isinstance(type_, typing_base)
297              and not isinstance(origin_type, typing_base)
298              and getattr(type_, '_name', None) is not None
299          ):
300              # In python < 3.9 generic aliases don't exist so any of these like `list`,
301              # `type` or `collections.abc.Callable` need to be translated.
302              # See: https://www.python.org/dev/peps/pep-0585
303              origin_type = getattr(typing, type_._name)
304          assert origin_type is not None
305          # PEP-604 syntax (Ex.: list | str) is represented with a types.UnionType object that does not have __getitem__.
306          # We also cannot use isinstance() since we have to compare types.
307          if sys.version_info >= (3, 10) and origin_type is types.UnionType:
308              return _UnionGenericAlias(origin_type, resolved_type_args)
309          # NotRequired[T] and Required[T] don't support tuple type resolved_type_args, hence the condition below
310          return origin_type[resolved_type_args[0] if len(resolved_type_args) == 1 else resolved_type_args]
311  
312      # We handle pydantic generic models separately as they don't have the same
313      # semantics as "typing" classes or generic aliases
314  
315      if not origin_type and is_model_class(type_):
316          parameters = type_.__pydantic_generic_metadata__['parameters']
317          if not parameters:
318              return type_
319          resolved_type_args = tuple(replace_types(t, type_map) for t in parameters)
320          if all_identical(parameters, resolved_type_args):
321              return type_
322          return type_[resolved_type_args]
323  
324      # Handle special case for typehints that can have lists as arguments.
325      # `typing.Callable[[int, str], int]` is an example for this.
326      if isinstance(type_, (List, list)):
327          resolved_list = [replace_types(element, type_map) for element in type_]
328          if all_identical(type_, resolved_list):
329              return type_
330          return resolved_list
331  
332      # If all else fails, we try to resolve the type directly and otherwise just
333      # return the input with no modifications.
334      return type_map.get(type_, type_)

337  def has_instance_in_type(type_: Any, isinstance_target: Any) -> bool:
338      """Checks if the type, or any of its arbitrary nested args, satisfy 
339      `isinstance(<type>, isinstance_target)`. 
340      """
341      if isinstance(type_, isinstance_target):
342          return True
343  
344      type_args = get_args(type_)
345      origin_type = get_origin(type_)
346  
347      if origin_type is typing_extensions.Annotated:
348          annotated_type, *annotations = type_args
349          return has_instance_in_type(annotated_type, isinstance_target)
350  
351      # Having type args is a good indicator that this is a typing module
352      # class instantiation or a generic alias of some sort.
353      for arg in type_args:
354          if has_instance_in_type(arg, isinstance_target):
355              return True
356  
357      # Handle special case for typehints that can have lists as arguments.
358      # `typing.Callable[[int, str], int]` is an example for this.
359      if isinstance(type_, (List, list)) and not isinstance(type_, typing_extensions.ParamSpec):
360          for element in type_:
361              if has_instance_in_type(element, isinstance_target):
362                  return True
363  
364      return False

367  def check_parameters_count(cls: type[BaseModel], parameters: tuple[Any, ...]) -> None:
368      """Check the generic model parameters count is equal. 
369   
370      Args: 
371          cls: The generic model. 
372          parameters: A tuple of passed parameters to the generic model. 
373   
374      Raises: 
375          TypeError: If the passed parameters count is not equal to generic model parameters count. 
376      """
377      actual = len(parameters)
378      expected = len(cls.__pydantic_generic_metadata__['parameters'])
379      if actual != expected:
380          description = 'many' if actual > expected else 'few'
381          raise TypeError(f'Too {description} parameters for {cls}; actual {actual}, expected {expected}')

387  @contextmanager
388  def generic_recursion_self_type(
389      origin: type[BaseModel], args: tuple[Any, ...]
390  ) -> Iterator[PydanticRecursiveRef | None]:
391      """This contextmanager should be placed around the recursive calls used to build a generic type, 
392      and accept as arguments the generic origin type and the type arguments being passed to it. 
393   
394      If the same origin and arguments are observed twice, it implies that a self-reference placeholder 
395      can be used while building the core schema, and will produce a schema_ref that will be valid in the 
396      final parent schema. 
397      """
398      previously_seen_type_refs = _generic_recursion_cache.get()
399      if previously_seen_type_refs is None:
400          previously_seen_type_refs = set()
401          token = _generic_recursion_cache.set(previously_seen_type_refs)
402      else:
403          token = None
404  
405      try:
406          type_ref = get_type_ref(origin, args_override=args)
407          if type_ref in previously_seen_type_refs:
408              self_type = PydanticRecursiveRef(type_ref=type_ref)
409              yield self_type
410          else:
411              previously_seen_type_refs.add(type_ref)
412              yield None
413      finally:
414          if token:
415              _generic_recursion_cache.reset(token)
416  
417  
418  def recursively_defined_type_refs() -> set[str]:
419      visited = _generic_recursion_cache.get()
420      if not visited:
421          return set()  # not in a generic recursion, so there are no types
422  
423      return visited.copy()  # don't allow modifications

426  def get_cached_generic_type_early(parent: type[BaseModel], typevar_values: Any) -> type[BaseModel] | None:
427      """The use of a two-stage cache lookup approach was necessary to have the highest performance possible for 
428      repeated calls to `__class_getitem__` on generic types (which may happen in tighter loops during runtime), 
429      while still ensuring that certain alternative parametrizations ultimately resolve to the same type. 
430   
431      As a concrete example, this approach was necessary to make Model[List[T]][int] equal to Model[List[int]]. 
432      The approach could be modified to not use two different cache keys at different points, but the 
433      _early_cache_key is optimized to be as quick to compute as possible (for repeated-access speed), and the 
434      _late_cache_key is optimized to be as "correct" as possible, so that two types that will ultimately be the 
435      same after resolving the type arguments will always produce cache hits. 
436   
437      If we wanted to move to only using a single cache key per type, we would either need to always use the 
438      slower/more computationally intensive logic associated with _late_cache_key, or would need to accept 
439      that Model[List[T]][int] is a different type than Model[List[T]][int]. Because we rely on subclass relationships 
440      during validation, I think it is worthwhile to ensure that types that are functionally equivalent are actually 
441      equal. 
442      """
443      return _GENERIC_TYPES_CACHE.get(_early_cache_key(parent, typevar_values))

446  def get_cached_generic_type_late(
447      parent: type[BaseModel], typevar_values: Any, origin: type[BaseModel], args: tuple[Any, ...]
448  ) -> type[BaseModel] | None:
449      """See the docstring of `get_cached_generic_type_early` for more information about the two-stage cache lookup."""
450      cached = _GENERIC_TYPES_CACHE.get(_late_cache_key(origin, args, typevar_values))
451      if cached is not None:
452          set_cached_generic_type(parent, typevar_values, cached, origin, args)
453      return cached

456  def set_cached_generic_type(
457      parent: type[BaseModel],
458      typevar_values: tuple[Any, ...],
459      type_: type[BaseModel],
460      origin: type[BaseModel] | None = None,
461      args: tuple[Any, ...] | None = None,
462  ) -> None:
463      """See the docstring of `get_cached_generic_type_early` for more information about why items are cached with 
464      two different keys. 
465      """
466      _GENERIC_TYPES_CACHE[_early_cache_key(parent, typevar_values)] = type_
467      if len(typevar_values) == 1:
468          _GENERIC_TYPES_CACHE[_early_cache_key(parent, typevar_values[0])] = type_
469      if origin and args:
470          _GENERIC_TYPES_CACHE[_late_cache_key(origin, args, typevar_values)] = type_

473  def _union_orderings_key(typevar_values: Any) -> Any:
474      """This is intended to help differentiate between Union types with the same arguments in different order. 
475   
476      Thanks to caching internal to the `typing` module, it is not possible to distinguish between 
477      List[Union[int, float]] and List[Union[float, int]] (and similarly for other "parent" origins besides List) 
478      because `typing` considers Union[int, float] to be equal to Union[float, int]. 
479   
480      However, you _can_ distinguish between (top-level) Union[int, float] vs. Union[float, int]. 
481      Because we parse items as the first Union type that is successful, we get slightly more consistent behavior 
482      if we make an effort to distinguish the ordering of items in a union. It would be best if we could _always_ 
483      get the exact-correct order of items in the union, but that would require a change to the `typing` module itself. 
484      (See https://github.com/python/cpython/issues/86483 for reference.) 
485      """
486      if isinstance(typevar_values, tuple):
487          args_data = []
488          for value in typevar_values:
489              args_data.append(_union_orderings_key(value))
490          return tuple(args_data)
491      elif typing_extensions.get_origin(typevar_values) is typing.Union:
492          return get_args(typevar_values)
493      else:
494          return ()

497  def _early_cache_key(cls: type[BaseModel], typevar_values: Any) -> GenericTypesCacheKey:
498      """This is intended for minimal computational overhead during lookups of cached types. 
499   
500      Note that this is overly simplistic, and it's possible that two different cls/typevar_values 
501      inputs would ultimately result in the same type being created in BaseModel.__class_getitem__. 
502      To handle this, we have a fallback _late_cache_key that is checked later if the _early_cache_key 
503      lookup fails, and should result in a cache hit _precisely_ when the inputs to __class_getitem__ 
504      would result in the same type. 
505      """
506      return cls, typevar_values, _union_orderings_key(typevar_values)

509  def _late_cache_key(origin: type[BaseModel], args: tuple[Any, ...], typevar_values: Any) -> GenericTypesCacheKey:
510      """This is intended for use later in the process of creating a new type, when we have more information 
511      about the exact args that will be passed. If it turns out that a different set of inputs to 
512      __class_getitem__ resulted in the same inputs to the generic type creation process, we can still 
513      return the cached type, and update the cache with the _early_cache_key as well. 
514      """
515      # The _union_orderings_key is placed at the start here to ensure there cannot be a collision with an
516      # _early_cache_key, as that function will always produce a BaseModel subclass as the first item in the key,
517      # whereas this function will always produce a tuple as the first item in the key.
518      return _union_orderings_key(typevar_values), origin, args
