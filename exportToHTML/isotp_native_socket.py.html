76       # This struct is only used within the sockaddr_can struct
77       _fields_ = [("rx_id", ctypes.c_uint32),
78                   ("tx_id", ctypes.c_uint32)]

82       # This struct is only used within the sockaddr_can struct
83       # This union is to future proof for future can address information

101  class ISOTPNativeSocket(SuperSocket):
102      """ 
103      ISOTPSocket using the can-isotp kernel module 
104   
105      :param iface: a CANSocket instance or an interface name 
106      :param tx_id: the CAN identifier of the sent CAN frames 
107      :param rx_id: the CAN identifier of the received CAN frames 
108      :param ext_address: the extended address of the sent ISOTP frames 
109      :param rx_ext_address: the extended address of the received ISOTP frames 
110      :param bs: block size sent in Flow Control ISOTP frames 
111      :param stmin: minimum desired separation time sent in 
112                    Flow Control ISOTP frames 
113      :param padding: If True, pads sending packets with 0x00 which not 
114                      count to the payload. 
115                      Does not affect receiving packets. 
116      :param listen_only: Does not send Flow Control frames if a First Frame is 
117                          received 
118      :param frame_txtime: Separation time between two CAN frames during send 
119      :param basecls: base class of the packets emitted by this socket 
120      """
121      desc = "read/write packets at a given CAN interface using CAN_ISOTP socket "  # noqa: E501
122      can_isotp_options_fmt = "@2I4B"
123      can_isotp_fc_options_fmt = "@3B"
124      can_isotp_ll_options_fmt = "@3B"
125      sockaddr_can_fmt = "@H3I"
126      auxdata_available = True
127  
128      def __build_can_isotp_options(
129              self,
130              flags=CAN_ISOTP_DEFAULT_FLAGS,
131              frame_txtime=CAN_ISOTP_DEFAULT_FRAME_TXTIME,
132              ext_address=CAN_ISOTP_DEFAULT_EXT_ADDRESS,
133              txpad_content=CAN_ISOTP_DEFAULT_PAD_CONTENT,
134              rxpad_content=CAN_ISOTP_DEFAULT_PAD_CONTENT,
135              rx_ext_address=CAN_ISOTP_DEFAULT_EXT_ADDRESS):
136          # type: (int, int, int, int, int, int) -> bytes
137          return struct.pack(self.can_isotp_options_fmt,
138                             flags,
139                             frame_txtime,
140                             ext_address,
141                             txpad_content,
142                             rxpad_content,
143                             rx_ext_address)
144  
145      # == Must use native not standard types for packing ==
146      # struct can_isotp_options {
147      # __u32 flags;            /* set flags for isotp behaviour.       */
148      #                         /* __u32 value : flags see below        */
149      #
150      # __u32 frame_txtime;     /* frame transmission time (N_As/N_Ar)  */
151      #                       /* __u32 value : time in nano secs      */
152      #
153      # __u8  ext_address;      /* set address for extended addressing  */
154      #                         /* __u8 value : extended address        */
155      #
156      # __u8  txpad_content;    /* set content of padding byte (tx)     */
157      #                         /* __u8 value : content on tx path      */
158      #
159      # __u8  rxpad_content;    /* set content of padding byte (rx)     */
160      #                         /* __u8 value : content on rx path      */
161      #
162      # __u8  rx_ext_address;   /* set address for extended addressing  */
163      #                         /* __u8 value : extended address (rx)   */
164      # };

166      def __build_can_isotp_fc_options(self,
167                                       bs=CAN_ISOTP_DEFAULT_RECV_BS,
168                                       stmin=CAN_ISOTP_DEFAULT_RECV_STMIN,
169                                       wftmax=CAN_ISOTP_DEFAULT_RECV_WFTMAX):
170          # type: (int, int, int) -> bytes
171          return struct.pack(self.can_isotp_fc_options_fmt,
172                             bs,
173                             stmin,
174                             wftmax)

176      # == Must use native not standard types for packing ==
177      # struct can_isotp_fc_options {
178      #
179      # __u8  bs;               /* blocksize provided in FC frame       */
180      #                         /* __u8 value : blocksize. 0 = off      */
181      #
182      # __u8  stmin;            /* separation time provided in FC frame */
183      #                         /* __u8 value :                         */
184      #                         /* 0x00 - 0x7F : 0 - 127 ms             */
185      #                         /* 0x80 - 0xF0 : reserved               */
186      #                         /* 0xF1 - 0xF9 : 100 us - 900 us        */
187      #                         /* 0xFA - 0xFF : reserved               */
188      #
189      # __u8  wftmax;           /* max. number of wait frame transmiss. */
190      #                         /* __u8 value : 0 = omit FC N_PDU WT    */
191      # };

193      def __build_can_isotp_ll_options(self,
194                                       mtu=CAN_ISOTP_DEFAULT_LL_MTU,
195                                       tx_dl=CAN_ISOTP_DEFAULT_LL_TX_DL,
196                                       tx_flags=CAN_ISOTP_DEFAULT_LL_TX_FLAGS
197                                       ):
198          # type: (int, int, int) -> bytes
199          return struct.pack(self.can_isotp_ll_options_fmt,
200                             mtu,
201                             tx_dl,
202                             tx_flags)

204      # == Must use native not standard types for packing ==
205      # struct can_isotp_ll_options {
206      #
207      # __u8  mtu;              /* generated & accepted CAN frame type  */
208      #                         /* __u8 value :                         */
209      #                         /* CAN_MTU   (16) -> standard CAN 2.0   */
210      #                         /* CANFD_MTU (72) -> CAN FD frame       */
211      #
212      # __u8  tx_dl;            /* tx link layer data length in bytes   */
213      #                         /* (configured maximum payload length)  */
214      #                         /* __u8 value : 8,12,16,20,24,32,48,64  */
215      #                         /* => rx path supports all LL_DL values */
216      #
217      # __u8  tx_flags;         /* set into struct canfd_frame.flags    */
218      #                         /* at frame creation: e.g. CANFD_BRS    */
219      #                         /* Obsolete when the BRS flag is fixed  */
220      #                         /* by the CAN netdriver configuration   */
221      # };

223      def __get_sock_ifreq(self, sock, iface):
224          # type: (socket.socket, str) -> ifreq
225          socket_id = ctypes.c_int(sock.fileno())
226          ifr = ifreq()
227          ifr.ifr_name = iface.encode('ascii')
228          ret = LIBC.ioctl(socket_id, SIOCGIFINDEX, ctypes.byref(ifr))
229  
230          if ret < 0:
231              m = u'Failure while getting "{}" interface index.'.format(
232                  iface)
233              raise Scapy_Exception(m)
234          return ifr
235  
236      def __bind_socket(self, sock, iface, tx_id, rx_id):
237          # type: (socket.socket, str, int, int) -> None
238          socket_id = ctypes.c_int(sock.fileno())
239          ifr = self.__get_sock_ifreq(sock, iface)
240  
241          if tx_id > 0x7ff:
242              tx_id = tx_id | socket.CAN_EFF_FLAG
243          if rx_id > 0x7ff:
244              rx_id = rx_id | socket.CAN_EFF_FLAG
245  
246          # select the CAN interface and bind the socket to it
247          addr = sockaddr_can(ctypes.c_uint16(socket.PF_CAN),
248                              ifr.ifr_ifindex,
249                              addr_info(tp(ctypes.c_uint32(rx_id),
250                                           ctypes.c_uint32(tx_id))))
251  
252          error = LIBC.bind(socket_id, ctypes.byref(addr),
253                            ctypes.sizeof(addr))
254  
255          if error < 0:
256              log_isotp.warning("Couldn't bind socket")

258      def __set_option_flags(self,
259                             sock,  # type: socket.socket
260                             extended_addr=None,  # type: Optional[int]
261                             extended_rx_addr=None,  # type: Optional[int]
262                             listen_only=False,  # type: bool
263                             padding=False,  # type: bool
264                             transmit_time=100  # type: int
265                             ):
266          # type: (...) -> None
267          option_flags = CAN_ISOTP_DEFAULT_FLAGS
268          if extended_addr is not None:
269              option_flags = option_flags | CAN_ISOTP_EXTEND_ADDR
270          else:
271              extended_addr = CAN_ISOTP_DEFAULT_EXT_ADDRESS
272  
273          if extended_rx_addr is not None:
274              option_flags = option_flags | CAN_ISOTP_RX_EXT_ADDR
275          else:
276              extended_rx_addr = CAN_ISOTP_DEFAULT_EXT_ADDRESS
277  
278          if listen_only:
279              option_flags = option_flags | CAN_ISOTP_LISTEN_MODE
280  
281          if padding:
282              option_flags = option_flags | CAN_ISOTP_TX_PADDING | CAN_ISOTP_RX_PADDING  # noqa: E501
283  
284          sock.setsockopt(SOL_CAN_ISOTP,
285                          CAN_ISOTP_OPTS,
286                          self.__build_can_isotp_options(
287                              frame_txtime=transmit_time,
288                              flags=option_flags,
289                              ext_address=extended_addr,
290                              rx_ext_address=extended_rx_addr))

292      def __init__(self,
293                   iface=None,  # type: Optional[Union[str, SuperSocket]]
294                   tx_id=0,  # type: int
295                   rx_id=0,  # type: int
296                   ext_address=None,  # type: Optional[int]
297                   rx_ext_address=None,  # type: Optional[int]
298                   bs=CAN_ISOTP_DEFAULT_RECV_BS,  # type: int
299                   stmin=CAN_ISOTP_DEFAULT_RECV_STMIN,  # type: int
300                   padding=False,  # type: bool
301                   listen_only=False,  # type: bool
302                   frame_txtime=CAN_ISOTP_DEFAULT_FRAME_TXTIME,  # type: int
303                   fd=False,  # type: bool
304                   basecls=ISOTP  # type: Type[Packet]
305                   ):
306          # type: (...) -> None
307  
308          if not isinstance(iface, str):
309              # This is for interoperability with ISOTPSoftSockets.
310              # If a NativeCANSocket is provided, the interface name of this
311              # socket is extracted and an ISOTPNativeSocket will be opened
312              # on this interface.
313              iface = cast(SuperSocket, iface)
314              if hasattr(iface, "ins") and hasattr(iface.ins, "getsockname"):
315                  iface = iface.ins.getsockname()
316                  if isinstance(iface, tuple):
317                      iface = cast(str, iface[0])
318              else:
319                  raise Scapy_Exception("Provide a string or a CANSocket "
320                                        "object as iface parameter")
321  
322          self.iface: str = cast(str, iface) or conf.contribs['NativeCANSocket']['iface']  # noqa: E501
323          # store arguments internally
324          self.tx_id = tx_id
325          self.rx_id = rx_id
326          self.ext_address = ext_address
327          self.rx_ext_address = rx_ext_address
328          self.bs = bs
329          self.stmin = stmin
330          self.padding = padding
331          self.listen_only = listen_only
332          self.frame_txtime = frame_txtime
333          self.fd = fd
334          if basecls is None:
335              log_isotp.warning('Provide a basecls ')
336          self.basecls = basecls
337          self._init_socket()
338  
339      def _init_socket(self) -> None:
340          can_socket = socket.socket(socket.PF_CAN, socket.SOCK_DGRAM,
341                                     CAN_ISOTP)
342          self.__set_option_flags(can_socket,
343                                  self.ext_address,
344                                  self.rx_ext_address,
345                                  self.listen_only,
346                                  self.padding,
347                                  self.frame_txtime)
348  
349          can_socket.setsockopt(SOL_CAN_ISOTP,
350                                CAN_ISOTP_RECV_FC,
351                                self.__build_can_isotp_fc_options(
352                                    stmin=self.stmin, bs=self.bs))
353          can_socket.setsockopt(SOL_CAN_ISOTP,
354                                CAN_ISOTP_LL_OPTS,
355                                self.__build_can_isotp_ll_options(
356                                    mtu=CAN_ISOTP_CANFD_MTU if self.fd
357                                    else CAN_ISOTP_DEFAULT_LL_MTU,
358                                    tx_dl=CAN_FD_ISOTP_DEFAULT_LL_TX_DL if self.fd
359                                    else CAN_ISOTP_DEFAULT_LL_TX_DL))
360          can_socket.setsockopt(
361              socket.SOL_SOCKET,
362              SO_TIMESTAMPNS,
363              1
364          )
365  
366          self.__bind_socket(can_socket, self.iface, self.tx_id, self.rx_id)
367          # make sure existing sockets are closed,
368          # required in case of a reconnect.
369          self.closed = False
370          self.close()
371  
372          self.ins = can_socket
373          self.outs = can_socket
374          self.closed = False

376      def recv_raw(self, x=0xffff):
377          # type: (int) -> Tuple[Optional[Type[Packet]], Optional[bytes], Optional[float]]  # noqa: E501
378          """ 
379          Receives a packet, then returns a tuple containing 
380          (cls, pkt_data, time) 
381          """
382          try:
383              pkt, _, ts = self._recv_raw(self.ins, x)
384          except BlockingIOError:  # noqa: F821
385              log_isotp.warning('Captured no data, socket in non-blocking mode.')
386              return None, None, None
387          except socket.timeout:
388              log_isotp.warning('Captured no data, socket read timed out.')
389              return None, None, None
390          except OSError as e:
391              # something bad happened (e.g. the interface went down)
392              log_isotp.warning("Captured no data. %s" % e)
393              if e.errno == 84:
394                  log_isotp.warning("Maybe a consecutive frame was missed. "
395                                    "Increasing `stmin` could solve this problem.")
396              elif e.errno == 110:
397                  log_isotp.warning('Captured no data, socket read timed out.')
398              elif e.errno == 70:
399                  log_isotp.warning(
400                      'Communication error on send. '
401                      'TX path flowcontrol reception timeout.')
402              else:
403                  log_isotp.error(
404                      'Unknown error code received %d. Closing socket!', e.errno)
405                  self.close()
406              return None, None, None
407  
408          if pkt and ts is None:
409              ts = get_last_packet_timestamp(self.ins)
410          return self.basecls, pkt, ts

412      def recv(self, x=0xffff, **kwargs):
413          # type: (int, **Any) -> Optional[Packet]
414          msg = SuperSocket.recv(self, x, **kwargs)
415          if msg is None:
416              return msg
417  
418          if hasattr(msg, "tx_id"):
419              msg.tx_id = self.tx_id
420          if hasattr(msg, "rx_id"):
421              msg.rx_id = self.rx_id
422          if hasattr(msg, "ext_address"):
423              msg.ext_address = self.ext_address
424          if hasattr(msg, "rx_ext_address"):
425              msg.rx_ext_address = self.rx_ext_address
426          return msg
