94   class GMLAN_ServiceEnumerator(GMLAN_Enumerator, StateGeneratingServiceEnumerator):  # noqa: E501
95       """ 
96       This enumerator scans for all services identifiers of GMLAN. During this 
97       scan, corrupted packets might be sent to an ECU and mainly negative 
98       responses will be received. 
99       """
100      _description = "Available services and negative response per state"
101  
102      def _get_initial_requests(self, **kwargs):
103          # type: (Any) -> Iterable[Packet]
104          services = set(x & ~0x40 for x in range(0x100))
105          services.remove(0x10)  # Remove InitiateDiagnosticOperation service
106          services.remove(0x3E)  # Remove TesterPresent service
107          services.remove(0xa5)  # Remove ProgrammingMode service
108          services.remove(0x34)  # Remove RequestDownload
109          return (GMLAN(service=x) for x in services)
110  
111      def _get_table_entry_y(self, tup):
112          # type: (_AutomotiveTestCaseScanResult) -> str
113          return "0x%02x: %s" % (
114              tup[1].service, tup[1].sprintf("%GMLAN.service%"))

117  class GMLAN_TPEnumerator(GMLAN_Enumerator, StateGeneratingServiceEnumerator):
118      """ 
119      Performs a check if TesterPresent is available. If a positive response is 
120      received, a new system state is generated and returned. 
121      """
122      _description = "TesterPresent supported"
123  
124      def _get_initial_requests(self, **kwargs):
125          # type: (Any) -> Iterable[Packet]
126          return [GMLAN(service=0x3E)]
127  
128      @staticmethod
129      def enter(socket,  # type: _SocketUnion
130                configuration,  # type: AutomotiveTestCaseExecutorConfiguration
131                kwargs  # type: Dict[str, Any]
132                ):
133          # type: (...) -> bool
134          if configuration.unittest:
135              configuration["tps"] = None
136              socket.sr1(GMLAN(service=0x3E), timeout=0.1, verbose=False)
137              return True
138  
139          GMLAN_TPEnumerator.cleanup(socket, configuration)
140          configuration["tps"] = GMLAN_TesterPresentSender(
141              cast(SuperSocket, socket))
142          configuration["tps"].start()
143          return True
144  
145      @staticmethod
146      def cleanup(_, configuration):
147          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration) -> bool
148          try:
149              if configuration["tps"]:
150                  configuration["tps"].stop()
151                  configuration["tps"] = None
152          except KeyError:
153              pass
154          return True
155  
156      def get_transition_function(self, socket, edge):
157          # type: (_SocketUnion, _Edge) -> Optional[_TransitionTuple]
158          return self.enter, {"desc": "TP"}, self.cleanup
159  
160      def _get_table_entry_y(self, tup):
161          # type: (_AutomotiveTestCaseScanResult) -> str
162          return "TesterPresent:"

324      def pre_execute(self, socket, state, global_configuration):
325          # type: (_SocketUnion, EcuState, AutomotiveTestCaseExecutorConfiguration) -> None  # noqa: E501
326          if cast(ServiceEnumerator, self)._retry_pkt[state] and \
327                  not global_configuration.unittest:
328              # this is a retry execute. Wait much longer than usual because
329              # a required time delay not expired could have been received
330              # on the previous attempt
331              time.sleep(11)
332  
333      def _evaluate_retry(self,
334                          state,  # type: EcuState
335                          request,  # type: Packet
336                          response,  # type: Packet
337                          **kwargs  # type: Optional[Dict[str, Any]]
338                          ):  # type: (...) -> bool
339  
340          if super(GMLAN_SAEnumerator, self)._evaluate_retry(
341                  state, request, response, **kwargs):
342              return True
343  
344          if response.service == 0x7f and \
345                  self._get_negative_response_code(response) in [0x22, 0x37]:
346              log_automotive.debug(
347                  "Retry %s because requiredTimeDelayNotExpired or "
348                  "requestSequenceError received",
349                  repr(request))
350              return super(GMLAN_SAEnumerator, self)._populate_retry(
351                  state, request)
352          return False
353  
354      def _evaluate_response(self,
355                             state,  # type: EcuState
356                             request,  # type: Packet
357                             response,  # type: Optional[Packet]
358                             **kwargs  # type: Optional[Dict[str, Any]]
359                             ):  # type: (...) -> bool
360          if super(GMLAN_SAEnumerator, self)._evaluate_response(
361                  state, request, response, **kwargs):
362              return True
363  
364          if response is not None and \
365                  response.service == 0x67 and response.subfunction % 2 == 1:
366              log_automotive.debug("Seed received. Leave scan to try a key")
367              return True
368          return False
369  
370      @staticmethod
371      def get_seed_pkt(sock, level=1):
372          # type: (_SocketUnion, int) -> Optional[Packet]
373          req = GMLAN() / GMLAN_SA(subfunction=level)
374          for _ in range(10):
375              seed = sock.sr1(req, timeout=5, verbose=False)
376              if seed is None:
377                  return None
378              elif seed.service == 0x7f and \
379                      GMLAN_Enumerator._get_negative_response_code(seed) != 0x37:
380                  log_automotive.info(
381                      "Security access no seed! NR: %s", repr(seed))
382                  return None
383  
384              elif seed.service == 0x7f and \
385                      GMLAN_Enumerator._get_negative_response_code(seed) == 0x37:
386                  log_automotive.info("Security access retry to get seed")
387                  time.sleep(10)
388                  continue
389              else:
390                  return seed
391          return None
392  
393      @staticmethod
394      def evaluate_security_access_response(res, seed, key):
395          # type: (Optional[Packet], Packet, Optional[Packet]) -> bool
396          if res is None or res.service == 0x7f:
397              log_automotive.debug(repr(seed))
398              log_automotive.debug(repr(key))
399              log_automotive.debug(repr(res))
400              log_automotive.info("Security access error!")
401              return False
402          else:
403              log_automotive.info("Security access granted!")
404              return True
405  
406      @staticmethod
407      def get_key_pkt(seed, keyfunction, level=1):
408          # type: (Packet, Callable[[int], int], int) -> Optional[Packet]
409          try:
410              s = seed.securitySeed
411          except AttributeError:
412              return None
413  
414          return cast(Packet, GMLAN() / GMLAN_SA(subfunction=level + 1,
415                                                 securityKey=keyfunction(s)))
416  
417      @staticmethod
418      def get_security_access(sock, level=1, seed_pkt=None, keyfunction=None):
419          # type: (_SocketUnion, int, Optional[Packet], Optional[Callable[[int], int]]) -> bool  # noqa: E501
420          log_automotive.info(
421              "Try bootloader security access for level %d" % level)
422          if seed_pkt is None:
423              seed_pkt = GMLAN_SAEnumerator.get_seed_pkt(sock, level)
424              if not seed_pkt:
425                  return False
426  
427          if keyfunction is None:
428              return False
429  
430          key_pkt = GMLAN_SAEnumerator.get_key_pkt(seed_pkt, keyfunction, level)
431          if key_pkt is None:
432              return False
433  
434          res = sock.sr1(key_pkt, timeout=5, verbose=False)
435          return GMLAN_SAEnumerator.evaluate_security_access_response(
436              res, seed_pkt, key_pkt)
437  
438      @staticmethod
439      def transition_function(sock, _, kwargs):
440          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration, Dict[str, Any]) -> bool  # noqa: E501
441          return GMLAN_SAEnumerator.get_security_access(
442              sock, level=kwargs["sec_level"], keyfunction=kwargs["keyfunction"])
443  
444      def get_new_edge(self, socket, config):
445          # type: (_SocketUnion, AutomotiveTestCaseExecutorConfiguration) -> Optional[_Edge]  # noqa: E501
446          last_resp = self._results[-1].resp
447          last_state = self._results[-1].state
448  
449          if last_resp is None or last_resp.service == 0x7f:
450              return None
451  
452          try:
453              if last_resp.service != 0x67 or \
454                      last_resp.subfunction % 2 != 1:
455                  return None
456  
457              seed = last_resp
458              sec_lvl = seed.subfunction
459              kf = config[self.__class__.__name__].get("keyfunction", None)
460  
461              if self.get_security_access(socket, level=sec_lvl,
462                                          seed_pkt=seed, keyfunction=kf):
463                  log_automotive.debug("Security Access found.")
464                  # create edge
465                  new_state = copy.copy(last_state)
466                  new_state.security_level = seed.subfunction + 1  # type: ignore  # noqa: E501
467                  if last_state == new_state:
468                      return None
469                  edge = (last_state, new_state)
470                  self._transition_function_args[edge] = (sec_lvl, kf)
471                  return edge
472          except AttributeError:
473              pass
474  
475          return None
476  
477      def get_transition_function(self, socket, edge):
478          # type: (_SocketUnion, _Edge) -> Optional[_TransitionTuple]
479          return self.transition_function, {
480              "sec_level": self._transition_function_args[edge][0],
481              "keyfunction": self._transition_function_args[edge][1],
482              "desc": "SA=%d" % self._transition_function_args[edge][0]}, None

607      def _get_initial_requests(self, **kwargs):
608          # type: (Any) -> Iterable[Packet]
609          self.probe_width = kwargs.pop("probe_width", self.probe_width)
610          self.random_probes_len = \
611              kwargs.pop("random_probes_len", self.random_probes_len)
612          self.sequential_probes_len = \
613              kwargs.pop("sequential_probes_len", self.sequential_probes_len)
614          addresses = random.sample(
615              range(0, self.highest_possible_addr, self.probe_width),
616              self.random_probes_len)
617          scan_range = kwargs.pop("scan_range", addresses)
618          return (GMLAN() / GMLAN_RMBA(memoryAddress=x,
619                                       memorySize=self.probe_width)

622      def post_execute(self, socket, state, global_configuration):
623          # type: (_SocketUnion, EcuState, AutomotiveTestCaseExecutorConfiguration) -> None  # noqa: E501
624          if not self._state_completed[state]:
625              return
626  
627          if not self.random_probe_finished[state]:
628              log_automotive.info("Random memory probing finished")
629              self.random_probe_finished[state] = True
630              for tup in [t for t in self.results_with_positive_response
631                          if t.state == state]:
632                  self.points_of_interest[state].append(
633                      (tup.req.memoryAddress, True))
634                  self.points_of_interest[state].append(
635                      (tup.req.memoryAddress, False))
636  
637          if not len(self.points_of_interest[state]):
638              return
639  
640          log_automotive.info(
641              "Create %d memory points for sequential probing" %
642              len(self.points_of_interest[state]))
643  
644          tested_addrs = [tup.req.memoryAddress for tup in self.results]
645          pos_addrs = [tup.req.memoryAddress for tup in
646                       self.results_with_positive_response if tup.state == state]
647  
648          new_requests = list()
649          new_points_of_interest = list()
650  
651          for poi, upward in self.points_of_interest[state]:
652              if poi not in pos_addrs:
653                  continue
654              temp_new_requests = list()
655              for i in range(
656                      self.probe_width,
657                      self.sequential_probes_len + self.probe_width,
658                      self.probe_width):
659                  if upward:
660                      new_addr = min(poi + i, self.highest_possible_addr)
661                  else:
662                      new_addr = max(poi - i, 0)
663  
664                  if new_addr not in tested_addrs:
665                      pkt = GMLAN() / GMLAN_RMBA(memoryAddress=new_addr,
666                                                 memorySize=self.probe_width)
667                      temp_new_requests.append(pkt)
668  
669              if len(temp_new_requests):
670                  new_points_of_interest.append(
671                      (temp_new_requests[-1].memoryAddress, upward))
672                  new_requests += temp_new_requests
673  
674          self.points_of_interest[state] = list()
675  
676          if len(new_requests):
677              self._state_completed[state] = False
678              self._request_iterators[state] = new_requests
679              self.points_of_interest[state] = new_points_of_interest
680              log_automotive.info(
681                  "Created %d pkts for sequential probing" %
682                  len(new_requests))

684      def show(self, dump=False, filtered=True, verbose=False):
685          # type: (bool, bool, bool) -> Optional[str]
686          s = super(GMLAN_RMBAEnumerator, self).show(dump, filtered, verbose)
687          try:
688              from intelhex import IntelHex
689  
690              ih = IntelHex()
691              for tup in self.results_with_positive_response:
692                  for i, b in enumerate(tup.resp.dataRecord):
693                      ih[tup.req.memoryAddress + i] = orb(b)
694  
695              ih.tofile("RMBA_dump.hex", format="hex")
696          except ImportError:
697              log_automotive.warning(
698                  "Install 'intelhex' to create a hex file of the memory")
699  
700          if dump and s is not None:
701              return s + "\n"
702          else:
703              print(s)
704              return None
