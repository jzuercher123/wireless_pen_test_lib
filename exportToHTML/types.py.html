146  def _registered(typ: Union[Type[T], 'ConstrainedNumberMeta']) -> Union[Type[T], 'ConstrainedNumberMeta']:
147      # In order to generate valid examples of constrained types, Hypothesis needs
148      # to inspect the type object - so we keep a weakref to each contype object
149      # until it can be registered.  When (or if) our Hypothesis plugin is loaded,
150      # it monkeypatches this function.
151      # If Hypothesis is never used, the total effect is to keep a weak reference
152      # which has minimal memory usage and doesn't even affect garbage collection.
153      _DEFINED_TYPES.add(typ)
154      return typ

157  class ConstrainedNumberMeta(type):
158      def __new__(cls, name: str, bases: Any, dct: Dict[str, Any]) -> 'ConstrainedInt':  # type: ignore
159          new_cls = cast('ConstrainedInt', type.__new__(cls, name, bases, dct))
160  
161          if new_cls.gt is not None and new_cls.ge is not None:
162              raise errors.ConfigError('bounds gt and ge cannot be specified at the same time')
163          if new_cls.lt is not None and new_cls.le is not None:
164              raise errors.ConfigError('bounds lt and le cannot be specified at the same time')
165  
166          return _registered(new_cls)  # type: ignore

405  class ConstrainedStr(str):
406      strip_whitespace = False
407      to_upper = False
408      to_lower = False
409      min_length: OptionalInt = None
410      max_length: OptionalInt = None
411      curtail_length: OptionalInt = None
412      regex: Optional[Union[str, Pattern[str]]] = None
413      strict = False
414  
415      @classmethod
416      def __modify_schema__(cls, field_schema: Dict[str, Any]) -> None:
417          update_not_none(
418              field_schema,
419              minLength=cls.min_length,
420              maxLength=cls.max_length,
421              pattern=cls.regex and cls._get_pattern(cls.regex),
422          )
423  
424      @classmethod
425      def __get_validators__(cls) -> 'CallableGenerator':
426          yield strict_str_validator if cls.strict else str_validator
427          yield constr_strip_whitespace
428          yield constr_upper
429          yield constr_lower
430          yield constr_length_validator
431          yield cls.validate
432  
433      @classmethod
434      def validate(cls, value: Union[str]) -> Union[str]:
435          if cls.curtail_length and len(value) > cls.curtail_length:
436              value = value[: cls.curtail_length]
437  
438          if cls.regex:
439              if not re.match(cls.regex, value):
440                  raise errors.StrRegexError(pattern=cls._get_pattern(cls.regex))
441  
442          return value
443  
444      @staticmethod
445      def _get_pattern(regex: Union[str, Pattern[str]]) -> str:
446          return regex if isinstance(regex, str) else regex.pattern

667  class ConstrainedDecimal(Decimal, metaclass=ConstrainedNumberMeta):
668      gt: OptionalIntFloatDecimal = None
669      ge: OptionalIntFloatDecimal = None
670      lt: OptionalIntFloatDecimal = None
671      le: OptionalIntFloatDecimal = None
672      max_digits: OptionalInt = None
673      decimal_places: OptionalInt = None
674      multiple_of: OptionalIntFloatDecimal = None
675  
676      @classmethod
677      def __modify_schema__(cls, field_schema: Dict[str, Any]) -> None:
678          update_not_none(
679              field_schema,
680              exclusiveMinimum=cls.gt,
681              exclusiveMaximum=cls.lt,
682              minimum=cls.ge,
683              maximum=cls.le,
684              multipleOf=cls.multiple_of,
685          )
686  
687      @classmethod
688      def __get_validators__(cls) -> 'CallableGenerator':
689          yield decimal_validator
690          yield number_size_validator
691          yield number_multiple_validator
692          yield cls.validate
693  
694      @classmethod
695      def validate(cls, value: Decimal) -> Decimal:
696          try:
697              normalized_value = value.normalize()
698          except InvalidOperation:
699              normalized_value = value
700          digit_tuple, exponent = normalized_value.as_tuple()[1:]
701          if exponent in {'F', 'n', 'N'}:
702              raise errors.DecimalIsNotFiniteError()
703  
704          if exponent >= 0:
705              # A positive exponent adds that many trailing zeros.
706              digits = len(digit_tuple) + exponent
707              decimals = 0
708          else:
709              # If the absolute value of the negative exponent is larger than the
710              # number of digits, then it's the same as the number of digits,
711              # because it'll consume all of the digits in digit_tuple and then
712              # add abs(exponent) - len(digit_tuple) leading zeros after the
713              # decimal point.
714              if abs(exponent) > len(digit_tuple):
715                  digits = decimals = abs(exponent)
716              else:
717                  digits = len(digit_tuple)
718                  decimals = abs(exponent)
719          whole_digits = digits - decimals
720  
721          if cls.max_digits is not None and digits > cls.max_digits:
722              raise errors.DecimalMaxDigitsError(max_digits=cls.max_digits)
723  
724          if cls.decimal_places is not None and decimals > cls.decimal_places:
725              raise errors.DecimalMaxPlacesError(decimal_places=cls.decimal_places)
726  
727          if cls.max_digits is not None and cls.decimal_places is not None:
728              expected = cls.max_digits - cls.decimal_places
729              if whole_digits > expected:
730                  raise errors.DecimalWholeDigitsError(whole_digits=expected)
731  
732          return value

978  class PaymentCardNumber(str):
979      """ 
980      Based on: https://en.wikipedia.org/wiki/Payment_card_number 
981      """
982  
983      strip_whitespace: ClassVar[bool] = True
984      min_length: ClassVar[int] = 12
985      max_length: ClassVar[int] = 19
986      bin: str
987      last4: str
988      brand: PaymentCardBrand
989  
990      def __init__(self, card_number: str):
991          self.bin = card_number[:6]
992          self.last4 = card_number[-4:]
993          self.brand = self._get_brand(card_number)
994  
995      @classmethod
996      def __get_validators__(cls) -> 'CallableGenerator':
997          yield str_validator
998          yield constr_strip_whitespace
999          yield constr_length_validator
1000         yield cls.validate_digits
1001         yield cls.validate_luhn_check_digit
1002         yield cls
1003         yield cls.validate_length_for_brand
1004 
1005     @property
1006     def masked(self) -> str:
1007         num_masked = len(self) - 10  # len(bin) + len(last4) == 10
1008         return f'{self.bin}{"*" * num_masked}{self.last4}'
1009 
1010     @classmethod
1011     def validate_digits(cls, card_number: str) -> str:
1012         if not card_number.isdigit():
1013             raise errors.NotDigitError
1014         return card_number
1015 
1016     @classmethod
1017     def validate_luhn_check_digit(cls, card_number: str) -> str:
1018         """ 
1019         Based on: https://en.wikipedia.org/wiki/Luhn_algorithm 
1020         """
1021         sum_ = int(card_number[-1])
1022         length = len(card_number)
1023         parity = length % 2
1024         for i in range(length - 1):
1025             digit = int(card_number[i])
1026             if i % 2 == parity:
1027                 digit *= 2
1028             if digit > 9:
1029                 digit -= 9
1030             sum_ += digit
1031         valid = sum_ % 10 == 0
1032         if not valid:
1033             raise errors.LuhnValidationError
1034         return card_number
1035 
1036     @classmethod
1037     def validate_length_for_brand(cls, card_number: 'PaymentCardNumber') -> 'PaymentCardNumber':
1038         """ 
1039         Validate length based on BIN for major brands: 
1040         https://en.wikipedia.org/wiki/Payment_card_number#Issuer_identification_number_(IIN) 
1041         """
1042         required_length: Union[None, int, str] = None
1043         if card_number.brand in PaymentCardBrand.mastercard:
1044             required_length = 16
1045             valid = len(card_number) == required_length
1046         elif card_number.brand == PaymentCardBrand.visa:
1047             required_length = '13, 16 or 19'
1048             valid = len(card_number) in {13, 16, 19}
1049         elif card_number.brand == PaymentCardBrand.amex:
1050             required_length = 15
1051             valid = len(card_number) == required_length
1052         else:
1053             valid = True
1054         if not valid:
1055             raise errors.InvalidLengthForBrand(brand=card_number.brand, required_length=required_length)
1056         return card_number
1057 
1058     @staticmethod
1059     def _get_brand(card_number: str) -> PaymentCardBrand:
1060         if card_number[0] == '4':
1061             brand = PaymentCardBrand.visa
1062         elif 51 <= int(card_number[:2]) <= 55:
1063             brand = PaymentCardBrand.mastercard
1064         elif card_number[:2] in {'34', '37'}:
1065             brand = PaymentCardBrand.amex
1066         else:
1067             brand = PaymentCardBrand.other
1068         return brand

1073 BYTE_SIZES = {
1074     'b': 1,
1075     'kb': 10**3,
1076     'mb': 10**6,
1077     'gb': 10**9,
1078     'tb': 10**12,
1079     'pb': 10**15,
1080     'eb': 10**18,
1081     'kib': 2**10,
1082     'mib': 2**20,
1083     'gib': 2**30,
1084     'tib': 2**40,
1085     'pib': 2**50,
1086     'eib': 2**60,
1087 }
1088 BYTE_SIZES.update({k.lower()[0]: v for k, v in BYTE_SIZES.items() if 'i' not in k})
1089 byte_string_re = re.compile(r'^\s*(\d*\.?\d+)\s*(\w+)?', re.IGNORECASE)
1090 
1091 
1092 class ByteSize(int):
1093     @classmethod
1094     def __get_validators__(cls) -> 'CallableGenerator':
1095         yield cls.validate
1096 
1097     @classmethod
1098     def validate(cls, v: StrIntFloat) -> 'ByteSize':
1099         try:
1100             return cls(int(v))
1101         except ValueError:
1102             pass
1103 
1104         str_match = byte_string_re.match(str(v))
1105         if str_match is None:
1106             raise errors.InvalidByteSize()
1107 
1108         scalar, unit = str_match.groups()
1109         if unit is None:
1110             unit = 'b'
1111 
1112         try:
1113             unit_mult = BYTE_SIZES[unit.lower()]
1114         except KeyError:
1115             raise errors.InvalidByteSizeUnit(unit=unit)
1116 
1117         return cls(int(float(scalar) * unit_mult))
1118 
1119     def human_readable(self, decimal: bool = False) -> str:
1120         if decimal:
1121             divisor = 1000
1122             units = ['B', 'KB', 'MB', 'GB', 'TB', 'PB']
1123             final_unit = 'EB'
1124         else:
1125             divisor = 1024
1126             units = ['B', 'KiB', 'MiB', 'GiB', 'TiB', 'PiB']
1127             final_unit = 'EiB'
1128 
1129         num = float(self)
1130         for unit in units:
1131             if abs(num) < divisor:
1132                 return f'{num:0.1f}{unit}'
1133             num /= divisor
1134 
1135         return f'{num:0.1f}{final_unit}'
1136 
1137     def to(self, unit: str) -> float:
1138         try:
1139             unit_div = BYTE_SIZES[unit.lower()]
1140         except KeyError:
1141             raise errors.InvalidByteSizeUnit(unit=unit)
1142 
1143         return self / unit_div
