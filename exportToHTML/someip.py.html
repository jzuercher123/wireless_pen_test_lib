25   class SOMEIP(Packet):
26       """ SOME/IP Packet."""
27   
28       PROTOCOL_VERSION = 0x01
29       INTERFACE_VERSION = 0x01
30       LEN_OFFSET = 0x08
31       LEN_OFFSET_TP = 0x0c
32       TYPE_REQUEST = 0x00
33       TYPE_REQUEST_NO_RET = 0x01
34       TYPE_NOTIFICATION = 0x02
35       TYPE_REQUEST_ACK = 0x40
36       TYPE_REQUEST_NORET_ACK = 0x41
37       TYPE_NOTIFICATION_ACK = 0x42
38       TYPE_RESPONSE = 0x80
39       TYPE_ERROR = 0x81
40       TYPE_RESPONSE_ACK = 0xc0
41       TYPE_ERROR_ACK = 0xc1
42       TYPE_TP_REQUEST = 0x20
43       TYPE_TP_REQUEST_NO_RET = 0x21
44       TYPE_TP_NOTIFICATION = 0x22
45       TYPE_TP_RESPONSE = 0xa0
46       TYPE_TP_ERROR = 0xa1
47       RET_E_OK = 0x00
48       RET_E_NOT_OK = 0x01
49       RET_E_UNKNOWN_SERVICE = 0x02
50       RET_E_UNKNOWN_METHOD = 0x03

55       RET_E_WRONG_INTERFACE_V = 0x08
56       RET_E_MALFORMED_MSG = 0x09

59       _OVERALL_LEN_NOPAYLOAD = 16

61       name = "SOME/IP"
62   
63       fields_desc = [
64           XShortField("srv_id", 0),
65           MultipleTypeField(
66               [
67                   (XShortField("sub_id", 0),
68                    (lambda pkt: False,
69                     lambda pkt, val: val < 0x8000), "method_id"),
70                   (XShortField("sub_id", 0),
71                    (lambda pkt: False,
72                     lambda pkt, val: val >= 0x8000), "event_id"),
73               ],
74               XShortField("sub_id", 0),

76           IntField("len", None),
77           XShortField("client_id", 0),
78           XShortField("session_id", 0),
79           XByteField("proto_ver", PROTOCOL_VERSION),
80           XByteField("iface_ver", INTERFACE_VERSION),
81           ByteEnumField("msg_type", TYPE_REQUEST, {
82               TYPE_REQUEST: "REQUEST",
83               TYPE_REQUEST_NO_RET: "REQUEST_NO_RETURN",

99               RET_E_OK: "E_OK",
100              RET_E_NOT_OK: "E_NOT_OK",
101              RET_E_UNKNOWN_SERVICE: "E_UNKNOWN_SERVICE",
102              RET_E_UNKNOWN_METHOD: "E_UNKNOWN_METHOD",
103              RET_E_NOT_READY: "E_NOT_READY",
104              RET_E_NOT_REACHABLE: "E_NOT_REACHABLE",
105              RET_E_TIMEOUT: "E_TIMEOUT",
106              RET_E_WRONG_PROTOCOL_V: "E_WRONG_PROTOCOL_VERSION",
107              RET_E_WRONG_INTERFACE_V: "E_WRONG_INTERFACE_VERSION",
108              RET_E_MALFORMED_MSG: "E_MALFORMED_MESSAGE",
109              RET_E_WRONG_MESSAGE_TYPE: "E_WRONG_MESSAGE_TYPE",
110          }),
111          ConditionalField(
112              BitScalingField("offset", 0, 28, scaling=16, unit="bytes"),
113              lambda pkt: SOMEIP._is_tp(pkt)),  # noqa: E501
114          ConditionalField(
115              BitField("res", 0, 3),
116              lambda pkt: SOMEIP._is_tp(pkt)),  # noqa: E501
117          ConditionalField(
118              BitField("more_seg", 0, 1),
119              lambda pkt: SOMEIP._is_tp(pkt)),  # noqa: E501
120          ConditionalField(PacketListField(
121              "data", [Raw()], Raw,
122              length_from=lambda pkt: pkt.len - (SOMEIP.LEN_OFFSET_TP if (SOMEIP._is_tp(pkt) and (pkt.len is None or pkt.len >= SOMEIP.LEN_OFFSET_TP)) else SOMEIP.LEN_OFFSET),  # noqa: E501
123              next_cls_cb=lambda pkt, lst, cur, remain:
124                  SOMEIP.get_payload_cls_by_srv_id(pkt, lst, cur, remain)),
125              lambda pkt: SOMEIP._is_tp(pkt))  # noqa: E501
126      ]

128      payload_cls_by_srv_id = dict()  # To be customized
129  
130      @staticmethod
131      def get_payload_cls_by_srv_id(pkt, lst, cur, remain):
132          return SOMEIP.payload_cls_by_srv_id.get(pkt.srv_id, Raw)
133  
134      def post_build(self, pkt, pay):
135          length = self.len
136          if length is None:
137              if SOMEIP._is_tp(self):
138                  length = SOMEIP.LEN_OFFSET_TP + len(pay)
139              else:
140                  length = SOMEIP.LEN_OFFSET + len(pay)
141  
142              pkt = pkt[:4] + struct.pack("!I", length) + pkt[8:]
143          return pkt + pay
144  
145      def answers(self, other):
146          if isinstance(other, type(self)):
147              if self.msg_type in [SOMEIP.TYPE_REQUEST_NO_RET,
148                                   SOMEIP.TYPE_REQUEST_NORET_ACK,
149                                   SOMEIP.TYPE_NOTIFICATION,
150                                   SOMEIP.TYPE_TP_REQUEST_NO_RET,
151                                   SOMEIP.TYPE_TP_NOTIFICATION]:
152                  return 0
153              return self.payload.answers(other.payload)
154          return 0
155  
156      @staticmethod
157      def _is_tp(pkt):
158          """Returns true if pkt is using SOMEIP-TP, else returns false."""
159          if isinstance(pkt, Packet):
160              return pkt.msg_type & 0x20
161          else:
162              return pkt[15] & 0x20
163  
164      @staticmethod
165      def _is_sd(pkt):
166          """Returns true if pkt is using SOMEIP-SD, else returns false."""
167          if isinstance(pkt, Packet):
168              return pkt.srv_id == 0xffff and pkt.sub_id == 0x8100
169          else:
170              return pkt[:4] == b"\xff\xff\x81\x00"
171  
172      def fragment(self, fragsize=1392):
173          """Fragment SOME/IP-TP"""
174          fnb = 0
175          fl = self
176          lst = list()
177          while fl.underlayer is not None:
178              fnb += 1
179              fl = fl.underlayer
180  
181          for p in fl:
182              s = raw(p[fnb].payload)
183              nb = (len(s) + fragsize) // fragsize
184              for i in range(nb):
185                  q = p.copy()
186                  del q[fnb].payload
187                  q[fnb].len = SOMEIP.LEN_OFFSET_TP + \
188                      len(s[i * fragsize:(i + 1) * fragsize])
189                  q[fnb].more_seg = 1
190                  if i == nb - 1:
191                      q[fnb].more_seg = 0
192                  q[fnb].offset += i * fragsize // 16
193                  r = conf.raw_layer(load=s[i * fragsize:(i + 1) * fragsize])
194                  r.overload_fields = p[fnb].payload.overload_fields.copy()
195                  q.add_payload(r)
196                  lst.append(q)
197  
198          return lst

201  def _bind_someip_layers():
202      bind_top_down(UDP, SOMEIP, sport=30490, dport=30490)
203  
204      for i in range(15):
205          bind_bottom_up(UDP, SOMEIP, sport=30490 + i)
206          bind_bottom_up(TCP, SOMEIP, sport=30490 + i)
207          bind_bottom_up(UDP, SOMEIP, dport=30490 + i)
208          bind_bottom_up(TCP, SOMEIP, dport=30490 + i)

270  def _sdentry_class(payload, **kargs):
271      TYPE_PAYLOAD_I = 0
272      pl_type = orb(payload[TYPE_PAYLOAD_I])
273      cls = None
274  
275      if pl_type in SDENTRY_TYPE_SRV:
276          cls = SDEntry_Service
277      elif pl_type in SDENTRY_TYPE_EVTGRP:
278          cls = SDEntry_EventGroup
279  
280      return cls(payload, **kargs)

343      name = "Config Option"
344      fields_desc = _MAKE_COMMON_SDOPTION_FIELDS_DESC(SDOPTION_CFG_TYPE) + [
345          StrLenField("cfg_str", b"\x00", length_from=lambda pkt: pkt.len - 1)

348      def post_build(self, pkt, pay):
349          if self.len is None:
350              length = len(self.cfg_str) + 1  # res_hdr field takes 1 byte
351              pkt = struct.pack("!H", length) + pkt[2:]
352          return pkt + pay

354      @staticmethod
355      def make_string(data):
356          # Build a valid null-terminated configuration string from a dict or a
357          # list with key-value pairs.
358          #
359          # Example:
360          #    >>> SDOption_Config.make_string({ "hello": "world" })
361          #    b'\x0bhello=world\x00'
362          #
363          #    >>> SDOption_Config.make_string([
364          #    ...     ("x", "y"),
365          #    ...     ("abc", "def"),
366          #    ...     ("123", "456")
367          #    ... ])
368          #    b'\x03x=y\x07abc=def\x07123=456\x00'
369  
370          if isinstance(data, dict):
371              data = data.items()
372  
373          # combine entries
374          data = ("{}={}".format(k, v) for k, v in data)
375          # prepend length
376          data = ("{}{}".format(chr(len(v)), v) for v in data)
377          # concatenate
378          data = "".join(data)
379          data += "\x00"
380  
381          return data.encode("utf8")

451  class SD(_SDPacketBase):
452      """ 
453      SD Packet 
454   
455      NOTE :   when adding 'entries' or 'options', do not use list.append() 
456          method but create a new list 
457      e.g. :  p = SD() 
458              p.option_array = [SDOption_Config(),SDOption_IP6_EndPoint()] 
459      """
460      SOMEIP_MSGID_SRVID = 0xffff
461      SOMEIP_MSGID_SUBID = 0x8100
462      SOMEIP_CLIENT_ID = 0x0000
463      SOMEIP_MINIMUM_SESSION_ID = 0x0001
464      SOMEIP_PROTO_VER = 0x01
465      SOMEIP_IFACE_VER = 0x01
466      SOMEIP_MSG_TYPE = SOMEIP.TYPE_NOTIFICATION
467      SOMEIP_RETCODE = SOMEIP.RET_E_OK
468  
469      name = "SD"
470      fields_desc = [
471          FlagsField("flags", 0, 8, [
472              "res0", "res1", "res2", "res3", "res4",
473              "EXPLICIT_INITIAL_DATA_CONTROL", "UNICAST", "REBOOT"]),
474          X3BytesField("res", 0),
475          FieldLenField("len_entry_array", None,
476                        length_of="entry_array", fmt="!I"),
477          PacketListField("entry_array", None, _sdentry_class,
478                          length_from=lambda pkt: pkt.len_entry_array),
479          FieldLenField("len_option_array", None,
480                        length_of="option_array", fmt="!I"),
481          PacketListField("option_array", None, _sdoption_class,
482                          length_from=lambda pkt: pkt.len_option_array)
483      ]
484  
485      def set_entryArray(self, entry_list):
486          if isinstance(entry_list, list):
487              self.entry_array = entry_list
488          else:
489              self.entry_array = [entry_list]
490  
491      def set_optionArray(self, option_list):
492          if isinstance(option_list, list):
493              self.option_array = option_list
494          else:
495              self.option_array = [option_list]
