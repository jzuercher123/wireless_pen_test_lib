108  class PydanticPlugin(Plugin):
109      def __init__(self, options: Options) -> None:
110          self.plugin_config = PydanticPluginConfig(options)
111          self._plugin_data = self.plugin_config.to_data()
112          super().__init__(options)

114      def get_base_class_hook(self, fullname: str) -> 'Optional[Callable[[ClassDefContext], None]]':
115          sym = self.lookup_fully_qualified(fullname)
116          if sym and isinstance(sym.node, TypeInfo):  # pragma: no branch
117              # No branching may occur if the mypy cache has not been cleared
118              if any(get_fullname(base) == BASEMODEL_FULLNAME for base in sym.node.mro):
119                  return self._pydantic_model_class_maker_callback
120          return None
121  
122      def get_metaclass_hook(self, fullname: str) -> Optional[Callable[[ClassDefContext], None]]:
123          if fullname == MODEL_METACLASS_FULLNAME:
124              return self._pydantic_model_metaclass_marker_callback
125          return None
126  
127      def get_function_hook(self, fullname: str) -> 'Optional[Callable[[FunctionContext], Type]]':
128          sym = self.lookup_fully_qualified(fullname)
129          if sym and sym.fullname == FIELD_FULLNAME:
130              return self._pydantic_field_callback
131          return None
132  
133      def get_method_hook(self, fullname: str) -> Optional[Callable[[MethodContext], Type]]:
134          if fullname.endswith('.from_orm'):
135              return from_orm_callback
136          return None

158      def _pydantic_model_metaclass_marker_callback(self, ctx: ClassDefContext) -> None:
159          """Reset dataclass_transform_spec attribute of ModelMetaclass. 
160   
161          Let the plugin handle it. This behavior can be disabled 
162          if 'debug_dataclass_transform' is set to True', for testing purposes. 
163          """
164          if self.plugin_config.debug_dataclass_transform:
165              return
166          info_metaclass = ctx.cls.info.declared_metaclass
167          assert info_metaclass, "callback not passed from 'get_metaclass_hook'"
168          if getattr(info_metaclass.type, 'dataclass_transform_spec', None):
169              info_metaclass.type.dataclass_transform_spec = None  # type: ignore[attr-defined]

171      def _pydantic_field_callback(self, ctx: FunctionContext) -> 'Type':
172          """ 
173          Extract the type of the `default` argument from the Field function, and use it as the return type. 
174   
175          In particular: 
176          * Check whether the default and default_factory argument is specified. 
177          * Output an error if both are specified. 
178          * Retrieve the type of the argument which is specified, and use it as return type for the function. 
179          """
180          default_any_type = ctx.default_return_type
181  
182          assert ctx.callee_arg_names[0] == 'default', '"default" is no longer first argument in Field()'
183          assert ctx.callee_arg_names[1] == 'default_factory', '"default_factory" is no longer second argument in Field()'
184          default_args = ctx.args[0]
185          default_factory_args = ctx.args[1]
186  
187          if default_args and default_factory_args:
188              error_default_and_default_factory_specified(ctx.api, ctx.context)
189              return default_any_type
190  
191          if default_args:
192              default_type = ctx.arg_types[0][0]
193              default_arg = default_args[0]
194  
195              # Fallback to default Any type if the field is required
196              if not isinstance(default_arg, EllipsisExpr):
197                  return default_type
198  
199          elif default_factory_args:
200              default_factory_type = ctx.arg_types[1][0]
201  
202              # Functions which use `ParamSpec` can be overloaded, exposing the callable's types as a parameter
203              # Pydantic calls the default factory without any argument, so we retrieve the first item
204              if isinstance(default_factory_type, Overloaded):
205                  if MYPY_VERSION_TUPLE > (0, 910):
206                      default_factory_type = default_factory_type.items[0]
207                  else:
208                      # Mypy0.910 exposes the items of overloaded types in a function
209                      default_factory_type = default_factory_type.items()[0]  # type: ignore[operator]
210  
211              if isinstance(default_factory_type, CallableType):
212                  ret_type = default_factory_type.ret_type
213                  # mypy doesn't think `ret_type` has `args`, you'd think mypy should know,
214                  # add this check in case it varies by version
215                  args = getattr(ret_type, 'args', None)
216                  if args:
217                      if all(isinstance(arg, TypeVarType) for arg in args):
218                          # Looks like the default factory is a type like `list` or `dict`, replace all args with `Any`
219                          ret_type.args = tuple(default_any_type for _ in args)  # type: ignore[attr-defined]
220                  return ret_type
221  
222          return default_any_type

225  class PydanticPluginConfig:
226      __slots__ = (
227          'init_forbid_extra',
228          'init_typed',
229          'warn_required_dynamic_aliases',
230          'warn_untyped_fields',
231          'debug_dataclass_transform',
232      )
233      init_forbid_extra: bool
234      init_typed: bool
235      warn_required_dynamic_aliases: bool
236      warn_untyped_fields: bool
237      debug_dataclass_transform: bool  # undocumented
238  
239      def __init__(self, options: Options) -> None:
240          if options.config_file is None:  # pragma: no cover
241              return
242  
243          toml_config = parse_toml(options.config_file)
244          if toml_config is not None:
245              config = toml_config.get('tool', {}).get('pydantic-mypy', {})
246              for key in self.__slots__:
247                  setting = config.get(key, False)
248                  if not isinstance(setting, bool):
249                      raise ValueError(f'Configuration value must be a boolean for key: {key}')
250                  setattr(self, key, setting)
251          else:
252              plugin_config = ConfigParser()
253              plugin_config.read(options.config_file)
254              for key in self.__slots__:
255                  setting = plugin_config.getboolean(CONFIGFILE_KEY, key, fallback=False)
256                  setattr(self, key, setting)
257  
258      def to_data(self) -> Dict[str, Any]:
259          return {key: getattr(self, key) for key in self.__slots__}

262  def from_orm_callback(ctx: MethodContext) -> Type:
263      """ 
264      Raise an error if orm_mode is not enabled 
265      """
266      model_type: Instance
267      ctx_type = ctx.type
268      if isinstance(ctx_type, TypeType):
269          ctx_type = ctx_type.item
270      if isinstance(ctx_type, CallableType) and isinstance(ctx_type.ret_type, Instance):
271          model_type = ctx_type.ret_type  # called on the class
272      elif isinstance(ctx_type, Instance):
273          model_type = ctx_type  # called on an instance (unusual, but still valid)
274      else:  # pragma: no cover
275          detail = f'ctx.type: {ctx_type} (of type {ctx_type.__class__.__name__})'
276          error_unexpected_behavior(detail, ctx.api, ctx.context)
277          return ctx.default_return_type
278      pydantic_metadata = model_type.type.metadata.get(METADATA_KEY)
279      if pydantic_metadata is None:
280          return ctx.default_return_type
281      orm_mode = pydantic_metadata.get('config', {}).get('orm_mode')
282      if orm_mode is not True:
283          error_from_orm(get_name(model_type.type), ctx.api, ctx.context)
284      return ctx.default_return_type

287  class PydanticModelTransformer:
288      tracked_config_fields: Set[str] = {
289          'extra',
290          'allow_mutation',
291          'frozen',
292          'orm_mode',
293          'allow_population_by_field_name',
294          'alias_generator',
295      }
296  
297      def __init__(self, ctx: ClassDefContext, plugin_config: PydanticPluginConfig) -> None:
298          self._ctx = ctx
299          self.plugin_config = plugin_config
300  
301      def transform(self) -> None:
302          """ 
303          Configures the BaseModel subclass according to the plugin settings. 
304   
305          In particular: 
306          * determines the model config and fields, 
307          * adds a fields-aware signature for the initializer and construct methods 
308          * freezes the class if allow_mutation = False or frozen = True 
309          * stores the fields, config, and if the class is settings in the mypy metadata for access by subclasses 
310          """
311          ctx = self._ctx
312          info = ctx.cls.info
313  
314          self.adjust_validator_signatures()
315          config = self.collect_config()
316          fields = self.collect_fields(config)
317          is_settings = any(get_fullname(base) == BASESETTINGS_FULLNAME for base in info.mro[:-1])
318          self.add_initializer(fields, config, is_settings)
319          self.add_construct_method(fields)
320          self.set_frozen(fields, frozen=config.allow_mutation is False or config.frozen is True)
321          info.metadata[METADATA_KEY] = {
322              'fields': {field.name: field.serialize() for field in fields},
323              'config': config.set_values_dict(),
324          }
325  
326      def adjust_validator_signatures(self) -> None:
327          """When we decorate a function `f` with `pydantic.validator(...), mypy sees 
328          `f` as a regular method taking a `self` instance, even though pydantic 
329          internally wraps `f` with `classmethod` if necessary. 
330   
331          Teach mypy this by marking any function whose outermost decorator is a 
332          `validator()` call as a classmethod. 
333          """
334          for name, sym in self._ctx.cls.info.names.items():
335              if isinstance(sym.node, Decorator):
336                  first_dec = sym.node.original_decorators[0]
337                  if (
338                      isinstance(first_dec, CallExpr)
339                      and isinstance(first_dec.callee, NameExpr)
340                      and first_dec.callee.fullname == f'{_NAMESPACE}.class_validators.validator'
341                  ):
342                      sym.node.func.is_class = True
343  
344      def collect_config(self) -> 'ModelConfigData':
345          """ 
346          Collects the values of the config attributes that are used by the plugin, accounting for parent classes. 
347          """
348          ctx = self._ctx
349          cls = ctx.cls
350          config = ModelConfigData()
351          for stmt in cls.defs.body:
352              if not isinstance(stmt, ClassDef):
353                  continue
354              if stmt.name == 'Config':
355                  for substmt in stmt.defs.body:
356                      if not isinstance(substmt, AssignmentStmt):
357                          continue
358                      config.update(self.get_config_update(substmt))
359                  if (
360                      config.has_alias_generator
361                      and not config.allow_population_by_field_name
362                      and self.plugin_config.warn_required_dynamic_aliases
363                  ):
364                      error_required_dynamic_aliases(ctx.api, stmt)
365          for info in cls.info.mro[1:]:  # 0 is the current class
366              if METADATA_KEY not in info.metadata:
367                  continue
368  
369              # Each class depends on the set of fields in its ancestors
370              ctx.api.add_plugin_dependency(make_wildcard_trigger(get_fullname(info)))
371              for name, value in info.metadata[METADATA_KEY]['config'].items():
372                  config.setdefault(name, value)
373          return config
374  
375      def collect_fields(self, model_config: 'ModelConfigData') -> List['PydanticModelField']:
376          """ 
377          Collects the fields for the model, accounting for parent classes 
378          """
379          # First, collect fields belonging to the current class.
380          ctx = self._ctx
381          cls = self._ctx.cls
382          fields = []  # type: List[PydanticModelField]
383          known_fields = set()  # type: Set[str]
384          for stmt in cls.defs.body:
385              if not isinstance(stmt, AssignmentStmt):  # `and stmt.new_syntax` to require annotation
386                  continue
387  
388              lhs = stmt.lvalues[0]
389              if not isinstance(lhs, NameExpr) or not is_valid_field(lhs.name):
390                  continue
391  
392              if not stmt.new_syntax and self.plugin_config.warn_untyped_fields:
393                  error_untyped_fields(ctx.api, stmt)
394  
395              # if lhs.name == '__config__':  # BaseConfig not well handled; I'm not sure why yet
396              #     continue
397  
398              sym = cls.info.names.get(lhs.name)
399              if sym is None:  # pragma: no cover
400                  # This is likely due to a star import (see the dataclasses plugin for a more detailed explanation)
401                  # This is the same logic used in the dataclasses plugin
402                  continue
403  
404              node = sym.node
405              if isinstance(node, PlaceholderNode):  # pragma: no cover
406                  # See the PlaceholderNode docstring for more detail about how this can occur
407                  # Basically, it is an edge case when dealing with complex import logic
408                  # This is the same logic used in the dataclasses plugin
409                  continue
410              if not isinstance(node, Var):  # pragma: no cover
411                  # Don't know if this edge case still happens with the `is_valid_field` check above
412                  # but better safe than sorry
413                  continue
414  
415              # x: ClassVar[int] is ignored by dataclasses.
416              if node.is_classvar:
417                  continue
418  
419              is_required = self.get_is_required(cls, stmt, lhs)
420              alias, has_dynamic_alias = self.get_alias_info(stmt)
421              if (
422                  has_dynamic_alias
423                  and not model_config.allow_population_by_field_name
424                  and self.plugin_config.warn_required_dynamic_aliases
425              ):
426                  error_required_dynamic_aliases(ctx.api, stmt)
427              fields.append(
428                  PydanticModelField(
429                      name=lhs.name,
430                      is_required=is_required,
431                      alias=alias,
432                      has_dynamic_alias=has_dynamic_alias,
433                      line=stmt.line,
434                      column=stmt.column,
435                  )
436              )
437              known_fields.add(lhs.name)
438          all_fields = fields.copy()
439          for info in cls.info.mro[1:]:  # 0 is the current class, -2 is BaseModel, -1 is object
440              if METADATA_KEY not in info.metadata:
441                  continue
442  
443              superclass_fields = []
444              # Each class depends on the set of fields in its ancestors
445              ctx.api.add_plugin_dependency(make_wildcard_trigger(get_fullname(info)))
446  
447              for name, data in info.metadata[METADATA_KEY]['fields'].items():
448                  if name not in known_fields:
449                      field = PydanticModelField.deserialize(info, data)
450                      known_fields.add(name)
451                      superclass_fields.append(field)
452                  else:
453                      (field,) = (a for a in all_fields if a.name == name)
454                      all_fields.remove(field)
455                      superclass_fields.append(field)
456              all_fields = superclass_fields + all_fields
457          return all_fields

459      def add_initializer(self, fields: List['PydanticModelField'], config: 'ModelConfigData', is_settings: bool) -> None:
460          """ 
461          Adds a fields-aware `__init__` method to the class. 
462   
463          The added `__init__` will be annotated with types vs. all `Any` depending on the plugin settings. 
464          """
465          ctx = self._ctx
466          typed = self.plugin_config.init_typed
467          use_alias = config.allow_population_by_field_name is not True
468          force_all_optional = is_settings or bool(
469              config.has_alias_generator and not config.allow_population_by_field_name
470          )
471          init_arguments = self.get_field_arguments(
472              fields, typed=typed, force_all_optional=force_all_optional, use_alias=use_alias
473          )
474          if not self.should_init_forbid_extra(fields, config):
475              var = Var('kwargs')
476              init_arguments.append(Argument(var, AnyType(TypeOfAny.explicit), None, ARG_STAR2))
477  
478          if '__init__' not in ctx.cls.info.names:
479              add_method(ctx, '__init__', init_arguments, NoneType())
480  
481      def add_construct_method(self, fields: List['PydanticModelField']) -> None:
482          """ 
483          Adds a fully typed `construct` classmethod to the class. 
484   
485          Similar to the fields-aware __init__ method, but always uses the field names (not aliases), 
486          and does not treat settings fields as optional. 
487          """
488          ctx = self._ctx
489          set_str = ctx.api.named_type(f'{BUILTINS_NAME}.set', [ctx.api.named_type(f'{BUILTINS_NAME}.str')])
490          optional_set_str = UnionType([set_str, NoneType()])
491          fields_set_argument = Argument(Var('_fields_set', optional_set_str), optional_set_str, None, ARG_OPT)
492          construct_arguments = self.get_field_arguments(fields, typed=True, force_all_optional=False, use_alias=False)
493          construct_arguments = [fields_set_argument] + construct_arguments
494  
495          obj_type = ctx.api.named_type(f'{BUILTINS_NAME}.object')
496          self_tvar_name = '_PydanticBaseModel'  # Make sure it does not conflict with other names in the class
497          tvar_fullname = ctx.cls.fullname + '.' + self_tvar_name
498          if MYPY_VERSION_TUPLE >= (1, 4):
499              tvd = TypeVarType(
500                  self_tvar_name,
501                  tvar_fullname,
502                  (
503                      TypeVarId(-1, namespace=ctx.cls.fullname + '.construct')
504                      if MYPY_VERSION_TUPLE >= (1, 11)
505                      else TypeVarId(-1)
506                  ),
507                  [],
508                  obj_type,
509                  AnyType(TypeOfAny.from_omitted_generics),  # type: ignore[arg-type]
510              )
511              self_tvar_expr = TypeVarExpr(
512                  self_tvar_name,
513                  tvar_fullname,
514                  [],
515                  obj_type,
516                  AnyType(TypeOfAny.from_omitted_generics),  # type: ignore[arg-type]
517              )
518          else:
519              tvd = TypeVarDef(self_tvar_name, tvar_fullname, -1, [], obj_type)
520              self_tvar_expr = TypeVarExpr(self_tvar_name, tvar_fullname, [], obj_type)
521          ctx.cls.info.names[self_tvar_name] = SymbolTableNode(MDEF, self_tvar_expr)
522  
523          # Backward-compatible with TypeVarDef from Mypy 0.910.
524          if isinstance(tvd, TypeVarType):
525              self_type = tvd
526          else:
527              self_type = TypeVarType(tvd)
528  
529          add_method(
530              ctx,
531              'construct',
532              construct_arguments,
533              return_type=self_type,
534              self_type=self_type,
535              tvar_def=tvd,
536              is_classmethod=True,
537          )

539      def set_frozen(self, fields: List['PydanticModelField'], frozen: bool) -> None:
540          """ 
541          Marks all fields as properties so that attempts to set them trigger mypy errors. 
542   
543          This is the same approach used by the attrs and dataclasses plugins. 
544          """
545          ctx = self._ctx
546          info = ctx.cls.info
547          for field in fields:
548              sym_node = info.names.get(field.name)
549              if sym_node is not None:
550                  var = sym_node.node
551                  if isinstance(var, Var):
552                      var.is_property = frozen
553                  elif isinstance(var, PlaceholderNode) and not ctx.api.final_iteration:
554                      # See https://github.com/pydantic/pydantic/issues/5191 to hit this branch for test coverage
555                      ctx.api.defer()
556                  else:  # pragma: no cover
557                      # I don't know whether it's possible to hit this branch, but I've added it for safety
558                      try:
559                          var_str = str(var)
560                      except TypeError:
561                          # This happens for PlaceholderNode; perhaps it will happen for other types in the future..
562                          var_str = repr(var)
563                      detail = f'sym_node.node: {var_str} (of type {var.__class__})'
564                      error_unexpected_behavior(detail, ctx.api, ctx.cls)
565              else:
566                  var = field.to_var(info, use_alias=False)
567                  var.info = info
568                  var.is_property = frozen
569                  var._fullname = get_fullname(info) + '.' + get_name(var)
570                  info.names[get_name(var)] = SymbolTableNode(MDEF, var)

572      def get_config_update(self, substmt: AssignmentStmt) -> Optional['ModelConfigData']:
573          """ 
574          Determines the config update due to a single statement in the Config class definition. 
575   
576          Warns if a tracked config attribute is set to a value the plugin doesn't know how to interpret (e.g., an int) 
577          """
578          lhs = substmt.lvalues[0]
579          if not (isinstance(lhs, NameExpr) and lhs.name in self.tracked_config_fields):
580              return None
581          if lhs.name == 'extra':
582              if isinstance(substmt.rvalue, StrExpr):
583                  forbid_extra = substmt.rvalue.value == 'forbid'
584              elif isinstance(substmt.rvalue, MemberExpr):
585                  forbid_extra = substmt.rvalue.name == 'forbid'
586              else:
587                  error_invalid_config_value(lhs.name, self._ctx.api, substmt)
588                  return None
589              return ModelConfigData(forbid_extra=forbid_extra)
590          if lhs.name == 'alias_generator':
591              has_alias_generator = True
592              if isinstance(substmt.rvalue, NameExpr) and substmt.rvalue.fullname == 'builtins.None':
593                  has_alias_generator = False
594              return ModelConfigData(has_alias_generator=has_alias_generator)
595          if isinstance(substmt.rvalue, NameExpr) and substmt.rvalue.fullname in ('builtins.True', 'builtins.False'):
596              return ModelConfigData(**{lhs.name: substmt.rvalue.fullname == 'builtins.True'})
597          error_invalid_config_value(lhs.name, self._ctx.api, substmt)
598          return None
599  
600      @staticmethod
601      def get_is_required(cls: ClassDef, stmt: AssignmentStmt, lhs: NameExpr) -> bool:
602          """ 
603          Returns a boolean indicating whether the field defined in `stmt` is a required field. 
604          """
605          expr = stmt.rvalue
606          if isinstance(expr, TempNode):
607              # TempNode means annotation-only, so only non-required if Optional
608              value_type = get_proper_type(cls.info[lhs.name].type)
609              return not PydanticModelTransformer.type_has_implicit_default(value_type)
610          if isinstance(expr, CallExpr) and isinstance(expr.callee, RefExpr) and expr.callee.fullname == FIELD_FULLNAME:
611              # The "default value" is a call to `Field`; at this point, the field is
612              # only required if default is Ellipsis (i.e., `field_name: Annotation = Field(...)`) or if default_factory
613              # is specified.
614              for arg, name in zip(expr.args, expr.arg_names):
615                  # If name is None, then this arg is the default because it is the only positional argument.
616                  if name is None or name == 'default':
617                      return arg.__class__ is EllipsisExpr
618                  if name == 'default_factory':
619                      return False
620              # In this case, default and default_factory are not specified, so we need to look at the annotation
621              value_type = get_proper_type(cls.info[lhs.name].type)
622              return not PydanticModelTransformer.type_has_implicit_default(value_type)
623          # Only required if the "default value" is Ellipsis (i.e., `field_name: Annotation = ...`)
624          return isinstance(expr, EllipsisExpr)
625  
626      @staticmethod
627      def type_has_implicit_default(type_: Optional[ProperType]) -> bool:
628          """ 
629          Returns True if the passed type will be given an implicit default value. 
630   
631          In pydantic v1, this is the case for Optional types and Any (with default value None). 
632          """
633          if isinstance(type_, AnyType):
634              # Annotated as Any
635              return True
636          if isinstance(type_, UnionType) and any(
637              isinstance(item, NoneType) or isinstance(item, AnyType) for item in type_.items
638          ):
639              # Annotated as Optional, or otherwise having NoneType or AnyType in the union
640              return True
641          return False
642  
643      @staticmethod
644      def get_alias_info(stmt: AssignmentStmt) -> Tuple[Optional[str], bool]:
645          """ 
646          Returns a pair (alias, has_dynamic_alias), extracted from the declaration of the field defined in `stmt`. 
647   
648          `has_dynamic_alias` is True if and only if an alias is provided, but not as a string literal. 
649          If `has_dynamic_alias` is True, `alias` will be None. 
650          """
651          expr = stmt.rvalue
652          if isinstance(expr, TempNode):
653              # TempNode means annotation-only
654              return None, False
655  
656          if not (
657              isinstance(expr, CallExpr) and isinstance(expr.callee, RefExpr) and expr.callee.fullname == FIELD_FULLNAME
658          ):
659              # Assigned value is not a call to pydantic.fields.Field
660              return None, False
661  
662          for i, arg_name in enumerate(expr.arg_names):
663              if arg_name != 'alias':
664                  continue
665              arg = expr.args[i]
666              if isinstance(arg, StrExpr):
667                  return arg.value, False
668              else:
669                  return None, True
670          return None, False

672      def get_field_arguments(
673          self, fields: List['PydanticModelField'], typed: bool, force_all_optional: bool, use_alias: bool
674      ) -> List[Argument]:
675          """ 
676          Helper function used during the construction of the `__init__` and `construct` method signatures. 
677   
678          Returns a list of mypy Argument instances for use in the generated signatures. 
679          """
680          info = self._ctx.cls.info
681          arguments = [
682              field.to_argument(info, typed=typed, force_optional=force_all_optional, use_alias=use_alias)
683              for field in fields
684              if not (use_alias and field.has_dynamic_alias)
685          ]
686          return arguments
687  
688      def should_init_forbid_extra(self, fields: List['PydanticModelField'], config: 'ModelConfigData') -> bool:
689          """ 
690          Indicates whether the generated `__init__` should get a `**kwargs` at the end of its signature 
691   
692          We disallow arbitrary kwargs if the extra config setting is "forbid", or if the plugin config says to, 
693          *unless* a required dynamic alias is present (since then we can't determine a valid signature). 
694          """
695          if not config.allow_population_by_field_name:
696              if self.is_dynamic_alias_present(fields, bool(config.has_alias_generator)):
697                  return False
698          if config.forbid_extra:
699              return True
700          return self.plugin_config.init_forbid_extra
701  
702      @staticmethod
703      def is_dynamic_alias_present(fields: List['PydanticModelField'], has_alias_generator: bool) -> bool:
704          """ 
705          Returns whether any fields on the model have a "dynamic alias", i.e., an alias that cannot be 
706          determined during static analysis. 
707          """
708          for field in fields:
709              if field.has_dynamic_alias:
710                  return True
711          if has_alias_generator:
712              for field in fields:
713                  if field.alias is None:
714                      return True
715          return False

718  class PydanticModelField:
719      def __init__(
720          self, name: str, is_required: bool, alias: Optional[str], has_dynamic_alias: bool, line: int, column: int
721      ):
722          self.name = name
723          self.is_required = is_required
724          self.alias = alias
725          self.has_dynamic_alias = has_dynamic_alias
726          self.line = line
727          self.column = column
728  
729      def to_var(self, info: TypeInfo, use_alias: bool) -> Var:
730          name = self.name
731          if use_alias and self.alias is not None:
732              name = self.alias
733          return Var(name, info[self.name].type)
734  
735      def to_argument(self, info: TypeInfo, typed: bool, force_optional: bool, use_alias: bool) -> Argument:
736          if typed and info[self.name].type is not None:
737              type_annotation = info[self.name].type
738          else:
739              type_annotation = AnyType(TypeOfAny.explicit)
740          return Argument(
741              variable=self.to_var(info, use_alias),
742              type_annotation=type_annotation,
743              initializer=None,
744              kind=ARG_NAMED_OPT if force_optional or not self.is_required else ARG_NAMED,
745          )
746  
747      def serialize(self) -> JsonDict:
748          return self.__dict__
749  
750      @classmethod
751      def deserialize(cls, info: TypeInfo, data: JsonDict) -> 'PydanticModelField':
752          return cls(**data)

756      def __init__(
757          self,
758          forbid_extra: Optional[bool] = None,
759          allow_mutation: Optional[bool] = None,
760          frozen: Optional[bool] = None,
761          orm_mode: Optional[bool] = None,
762          allow_population_by_field_name: Optional[bool] = None,
763          has_alias_generator: Optional[bool] = None,
764      ):
765          self.forbid_extra = forbid_extra
766          self.allow_mutation = allow_mutation
767          self.frozen = frozen
768          self.orm_mode = orm_mode
769          self.allow_population_by_field_name = allow_population_by_field_name

772      def set_values_dict(self) -> Dict[str, Any]:
773          return {k: v for k, v in self.__dict__.items() if v is not None}
774  
775      def update(self, config: Optional['ModelConfigData']) -> None:
776          if config is None:
777              return
778          for k, v in config.set_values_dict().items():
779              setattr(self, k, v)

781      def setdefault(self, key: str, value: Any) -> None:
782          if getattr(self, key) is None:
783              setattr(self, key, value)

824  def add_method(
825      ctx: ClassDefContext,
826      name: str,
827      args: List[Argument],
828      return_type: Type,
829      self_type: Optional[Type] = None,
830      tvar_def: Optional[TypeVarDef] = None,
831      is_classmethod: bool = False,
832      is_new: bool = False,
833      # is_staticmethod: bool = False,
834  ) -> None:
835      """ 
836      Adds a new method to a class. 
837   
838      This can be dropped if/when https://github.com/python/mypy/issues/7301 is merged 
839      """
840      info = ctx.cls.info
841  
842      # First remove any previously generated methods with the same name
843      # to avoid clashes and problems in the semantic analyzer.
844      if name in info.names:
845          sym = info.names[name]
846          if sym.plugin_generated and isinstance(sym.node, FuncDef):
847              ctx.cls.defs.body.remove(sym.node)  # pragma: no cover
848  
849      self_type = self_type or fill_typevars(info)
850      if is_classmethod or is_new:
851          first = [Argument(Var('_cls'), TypeType.make_normalized(self_type), None, ARG_POS)]
852      # elif is_staticmethod:
853      #     first = []
854      else:
855          self_type = self_type or fill_typevars(info)
856          first = [Argument(Var('__pydantic_self__'), self_type, None, ARG_POS)]
857      args = first + args
858      arg_types, arg_names, arg_kinds = [], [], []
859      for arg in args:
860          assert arg.type_annotation, 'All arguments must be fully typed.'
861          arg_types.append(arg.type_annotation)
862          arg_names.append(get_name(arg.variable))
863          arg_kinds.append(arg.kind)
864  
865      function_type = ctx.api.named_type(f'{BUILTINS_NAME}.function')
866      signature = CallableType(arg_types, arg_kinds, arg_names, return_type, function_type)
867      if tvar_def:
868          signature.variables = [tvar_def]
869  
870      func = FuncDef(name, args, Block([PassStmt()]))
871      func.info = info
872      func.type = set_callable_name(signature, func)
873      func.is_class = is_classmethod
874      # func.is_static = is_staticmethod
875      func._fullname = get_fullname(info) + '.' + name
876      func.line = info.line
877  
878      # NOTE: we would like the plugin generated node to dominate, but we still
879      # need to keep any existing definitions so they get semantically analyzed.
880      if name in info.names:
881          # Get a nice unique name instead.
882          r_name = get_unique_redefinition_name(name, info.names)
883          info.names[r_name] = info.names[name]
884  
885      if is_classmethod:  # or is_staticmethod:
886          func.is_decorated = True
887          v = Var(name, func.type)
888          v.info = info
889          v._fullname = func._fullname
890          # if is_classmethod:
891          v.is_classmethod = True
892          dec = Decorator(func, [NameExpr('classmethod')], v)
893          # else:
894          #     v.is_staticmethod = True
895          #     dec = Decorator(func, [NameExpr('staticmethod')], v)
896  
897          dec.line = info.line
898          sym = SymbolTableNode(MDEF, dec)
899      else:
900          sym = SymbolTableNode(MDEF, func)
901      sym.plugin_generated = True
902  
903      info.names[name] = sym
904      info.defn.defs.body.append(func)

927  def parse_toml(config_file: str) -> Optional[Dict[str, Any]]:
928      if not config_file.endswith('.toml'):
929          return None
930  
931      read_mode = 'rb'
932      if sys.version_info >= (3, 11):
933          import tomllib as toml_
934      else:
935          try:
936              import tomli as toml_
937          except ImportError:
938              # older versions of mypy have toml as a dependency, not tomli
939              read_mode = 'r'
940              try:
941                  import toml as toml_  # type: ignore[no-redef]
942              except ImportError:  # pragma: no cover
943                  import warnings
944  
945                  warnings.warn('No TOML parser installed, cannot read configuration from `pyproject.toml`.')
946                  return None
947  
948      with open(config_file, read_mode) as rf:
949          return toml_.load(rf)  # type: ignore[arg-type]
