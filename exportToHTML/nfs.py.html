59   def loct(x):
60       if isinstance(x, int):
61           return oct(x)
62       if isinstance(x, tuple):
63           return "(%s)" % ", ".join(map(loct, x))
64       if isinstance(x, list):
65           return "[%s]" % ", ".join(map(loct, x))
66       return x

69   class OIntField(IntField):
70       """IntField child with octal representation"""
71       def i2repr(self, pkt, x):
72           return loct(self.i2h(pkt, x))

102  class File_Object(Packet):
103      name = 'File Object'
104      fields_desc = [
105          IntField('length', 0),
106          StrLenField('fh', b'', length_from=lambda pkt: pkt.length),
107          StrLenField('fill', b'', length_from=lambda pkt: (4 - pkt.length) % 4)
108      ]
109  
110      def set(self, new_filehandle, length=None, fill=None):
111          # convert filehandle to bytes if it was passed as a string
112          if new_filehandle.isalnum():
113              new_filehandle = unhexlify(new_filehandle)
114  
115          if length is None:
116              length = len(new_filehandle)
117          if fill is None:
118              fill = b'\x00' * ((4 - length) % 4)
119  
120          self.length = length
121          self.fh = new_filehandle
122          self.fill = fill
123  
124      def extract_padding(self, s):
125          return '', s
