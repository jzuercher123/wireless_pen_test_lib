1    import os
2    import json
3    import logging
4    from typing import Any
5    
6    class DataStorageManager:
7        def __init__(self, report_directory: str = "reports"):
8            self.logger = logging.getLogger(self.__class__.__name__)
9            self.report_directory = report_directory
10           self.ensure_directories()
11   
12       def ensure_directories(self):
13           """ 
14           Ensures that all necessary directories exist. 
15           """
16           data_dirs = [
17               self.report_directory,
18               os.path.join(self.report_directory, 'html'),
19               os.path.join(self.report_directory, 'pdf'),
20               os.path.join(self.report_directory, 'json'),
21               os.path.join(self.report_directory, '..', 'logs')
22           ]
23           for directory in data_dirs:
24               os.makedirs(directory, exist_ok=True)
25               self.logger.debug(f"Ensured directory exists: {directory}")
26   
27       def save_json(self, data: Any, filename: str):
28           """ 
29           Saves data as a JSON file. 
30           """
31           file_path = os.path.join(self.report_directory, 'json', filename)
32           self.logger.info(f"Saving JSON data to {file_path}")
33           try:
34               with open(file_path, 'w') as f:
35                   json.dump(data, f, indent=4)
36               self.logger.debug(f"JSON data saved successfully to {file_path}")
37           except Exception as e:
38               self.logger.error(f"Failed to save JSON data to {file_path}: {e}")
39               raise e
40   
41       def load_json(self, filename: str) -> Any:
42           """ 
43           Loads data from a JSON file. 
44           """
45           file_path = os.path.join(self.report_directory, 'json', filename)
46           self.logger.info(f"Loading JSON data from {file_path}")
47           try:
48               with open(file_path, 'r') as f:
49                   data = json.load(f)
50               self.logger.debug(f"JSON data loaded successfully from {file_path}")
51               return data
52           except Exception as e:
53               self.logger.error(f"Failed to load JSON data from {file_path}: {e}")
54               raise e
55   
56       def save_text(self, data: str, filename: str):
57           """ 
58           Saves data as a text file. 
59           """
60           file_path = os.path.join(self.report_directory, 'txt', filename)
61           os.makedirs(os.path.dirname(file_path), exist_ok=True)
62           self.logger.info(f"Saving text data to {file_path}")
63           try:
64               with open(file_path, 'w') as f:
65                   f.write(data)
66               self.logger.debug(f"Text data saved successfully to {file_path}")
67           except Exception as e:
68               self.logger.error(f"Failed to save text data to {file_path}: {e}")
69               raise e
70   
71       def save_binary(self, data: bytes, filename: str):
72           """ 
73           Saves data as a binary file. 
74           """
75           file_path = os.path.join(self.report_directory, 'binary', filename)
76           os.makedirs(os.path.dirname(file_path), exist_ok=True)
77           self.logger.info(f"Saving binary data to {file_path}")
78           try:
79               with open(file_path, 'wb') as f:
80                   f.write(data)
81               self.logger.debug(f"Binary data saved successfully to {file_path}")
82           except Exception as e:
83               self.logger.error(f"Failed to save binary data to {file_path}: {e}")
84               raise e
85   
86       def get_report_path(self, report_type: str, filename: str) -> str:
87           """ 
88           Returns the full path for a report based on its type. 
89           """
90           valid_types = ['html', 'pdf', 'json', 'txt', 'binary']
91           if report_type not in valid_types:
92               self.logger.error(f"Invalid report type: {report_type}")
93               raise ValueError(f"Report type must be one of {valid_types}")
94           file_path = os.path.join(self.report_directory, report_type, filename)
95           self.logger.debug(f"Report path for {report_type} report: {file_path}")
96           return file_path
