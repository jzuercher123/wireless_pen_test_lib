24                   },
25                   "auth_bypass_scanner": {
26                       "scan_duration": 10

28                   "dos_scanner": {
29                       "scan_duration": 10
30                   }
31               },
32               "exploits": {
33                   "session_hijacking": {
34                       "max_packets": 100
35                   },
36                   "credential_extraction": {

70           os.environ['EXPLOITS_PAYLOAD_DELIVERY_DEFAULT_DURATION'] = '15'

72           # Initialize Core Framework

75       def tearDown(self):
76           # Remove temporary config directory and environment variables
77           shutil.rmtree(self.test_config_dir)
78           del os.environ['GENERAL_LOG_LEVEL']
79           del os.environ['EXPLOITS_PAYLOAD_DELIVERY_DEFAULT_DURATION']
80   
81       def test_core_configuration(self):
82           # Access Core Framework configuration
83           config = self.core.config
84   
85           # Verify general settings
86           self.assertEqual(config.general.interface, "wlan1mon")
87           self.assertEqual(config.general.log_level, "DEBUG")
88           self.assertEqual(config.general.report_directory, "reports")
89   
90           # Verify scanner settings
91           self.assertEqual(config.scanners.encryption_scanner.scan_duration, 20)
92           self.assertEqual(config.scanners.auth_bypass_scanner.scan_duration, 10)
93           self.assertEqual(config.scanners.dos_scanner.scan_duration, 10)
94   
95           # Verify exploit settings
96           self.assertEqual(config.exploits.session_hijacking.max_packets, 100)
97           self.assertEqual(config.exploits.credential_extraction.capture_duration, 20)
98           self.assertEqual(config.exploits.payload_delivery.default_duration, 15)
99           self.assertListEqual(config.exploits.payload_delivery.payload_types, ["reverse_shell", "malicious_script"])
100  
101          # Verify UI settings
102          self.assertEqual(config.ui.theme, "default")
103  
104      def test_invalid_core_configuration(self):

108  
109          with self.assertRaises(Exception):
110              self.core = CoreFramework(modules_path=os.path.join(os.getcwd(), 'protocols'),
111                                        config_path=self.test_config_dir)
