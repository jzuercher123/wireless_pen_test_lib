287  class pfmsghdr(Packet):
288      fields_desc = [
289          Field("rtm_msglen", 0, fmt="=H"),
290          ByteField("rtm_version", 5),
291          ByteEnumField("rtm_type", 0, _RTM_TYPE),
292      ] + (
293          # It begins... the IFs apocalypse
294          [Field("rtm_hdrlen", 0, fmt="=H")]
295          if OPENBSD
296          else []
297      )
298  
299      if OPENBSD:
300  
301          def extract_padding(self, s: bytes) -> Tuple[bytes, Optional[bytes]]:
302              if self.rtm_msglen < 6:
303                  return s, b""
304              return s[: self.rtm_msglen - 6], s[self.rtm_msglen - 6 :]
305  
306      else:
307  
308          def extract_padding(self, s: bytes) -> Tuple[bytes, Optional[bytes]]:
309              if self.rtm_msglen < 4:
310                  return s, b""
311              return s[: self.rtm_msglen - 4], s[self.rtm_msglen - 4 :]

320  class sockaddr(Packet):
321      fields_desc = [
322          # socket.h
323          ByteField("sa_len", 0),
324          ByteEnumField("sa_family", 0, socket.AddressFamily),
325          # sockaddr_in
326          ConditionalField(
327              Field("sin_port", 0, fmt="=H"), lambda pkt: pkt.sa_family == socket.AF_INET
328          ),
329          ConditionalField(
330              IPField("sin_addr", 0), lambda pkt: pkt.sa_family == socket.AF_INET
331          ),
332          ConditionalField(
333              StrFixedLenField("sin_zero", "", length=8),
334              lambda pkt: pkt.sa_family == socket.AF_INET and pkt.sa_len > 7,
335          ),
336          # sockaddr_in6
337          ConditionalField(
338              Field("sin6_port", 0, fmt="=H"),
339              lambda pkt: pkt.sa_family == socket.AF_INET6,
340          ),
341          ConditionalField(
342              Field("sin6_flowinfo", 0, fmt="=I"),
343              lambda pkt: pkt.sa_family == socket.AF_INET6,
344          ),
345          ConditionalField(
346              IP6Field("sin6_addr", "::"), lambda pkt: pkt.sa_family == socket.AF_INET6
347          ),
348          ConditionalField(
349              Field("sin6_scope_id", 0, fmt="=I"),
350              lambda pkt: pkt.sa_family == socket.AF_INET6,
351          ),
352          # sockaddr_dl
353          ConditionalField(
354              Field("sdl_index", 0, fmt="=H"), lambda pkt: pkt.sa_family == socket.AF_LINK
355          ),
356          ConditionalField(
357              Field("sdl_type", 0, fmt="=B"), lambda pkt: pkt.sa_family == socket.AF_LINK
358          ),
359          ConditionalField(
360              Field("sdl_nlen", 0, fmt="=B"), lambda pkt: pkt.sa_family == socket.AF_LINK
361          ),
362          ConditionalField(
363              Field("sdl_alen", 0, fmt="=B"), lambda pkt: pkt.sa_family == socket.AF_LINK
364          ),
365          ConditionalField(
366              Field("sdl_slen", 0, fmt="=B"), lambda pkt: pkt.sa_family == socket.AF_LINK
367          ),
368          ConditionalField(
369              StrLenField("sdl_iface", "", length_from=lambda pkt: pkt.sdl_nlen),
370              lambda pkt: pkt.sa_family == socket.AF_LINK,
371          ),
372          ConditionalField(
373              MultipleTypeField(
374                  [(MACField("sdl_addr", None), lambda pkt: pkt.sdl_alen == 6)],
375                  StrLenField("sdl_addr", "", length_from=lambda pkt: pkt.sdl_alen),
376              ),
377              lambda pkt: pkt.sa_family == socket.AF_LINK,
378          ),
379          ConditionalField(
380              StrLenField("sdl_sel", "", length_from=lambda pkt: pkt.sdl_slen),
381              lambda pkt: pkt.sa_family == socket.AF_LINK,
382          ),
383          ConditionalField(
384              XStrLenField(
385                  "sdl_data",
386                  "",
387                  length_from=lambda pkt: max(
388                      pkt.sa_len - pkt.sdl_nlen - pkt.sdl_alen - pkt.sdl_slen - 8, 0
389                  ),
390              ),
391              lambda pkt: pkt.sa_family == socket.AF_LINK,
392          ),
393          ConditionalField(
394              XStrLenField("sdl_pad", b"", length_from=lambda pkt: 16 - pkt.sa_len),
395              lambda pkt: pkt.sa_len < 16 and pkt.sa_family == socket.AF_LINK,
396          ),
397          # others
398          ConditionalField(
399              XStrLenField(
400                  "sa_data",
401                  "",
402                  length_from=lambda pkt: pkt.sa_len - 2 if pkt.sa_len >= 2 else 0,
403              ),
404              lambda pkt: pkt.sa_family
405              not in [
406                  socket.AF_INET,
407                  socket.AF_INET6,
408                  socket.AF_LINK,
409              ],
410          ),
411      ]

414          return conf.padding_layer

945  def _sr1_bsdsysctl(mib) -> List[Packet]:
946      """ 
947      Send / Receive a BSD sysctl 
948      """
949      # Request routes
950      # 1. estimate needed size
951      oldplen = ctypes.c_size_t()
952      r = LIBC.sysctl(
953          mib,
954          len(mib),
955          None,
956          ctypes.byref(oldplen),
957          None,
958          0,
959      )
960      if r != 0:
961          return None
962      # 2. ask for real
963      oldp = ctypes.create_string_buffer(oldplen.value)
964      r = LIBC.sysctl(
965          mib,
966          len(mib),
967          oldp,
968          ctypes.byref(oldplen),
969          None,
970          0,
971      )
972      if r != 0:
973          return None
974      # Parse response
975      return pfmsghdrs(bytes(oldp))

978  def read_routes():
979      """ 
980      Read the IPv4 routes using PF_ROUTE 
981      """
982      mib = [
983          CTL_NET,
984          socket.PF_ROUTE,
985          0,
986          int(socket.AF_INET),
987          NET_RT_DUMP,
988          0,
989      ]
990      if not NETBSD and not DARWIN:
991          # NetBSD / OSX is missing the fib
992          if OPENBSD:
993              fib = 0  # default table
994          else:  # FreeBSD
995              fib = -1  # means 'all'
996          mib.append(fib)
997      mib = (ctypes.c_int * len(mib))(*mib)
998      resp = _sr1_bsdsysctl(mib)
999      if not resp:
1000         return []
1001     ifaces = _get_if_list()
1002     routes = []
1003     for msg in resp.msgs:
1004         if msg.rtm_type != 0x4 and (not DARWIN or msg.rtm_type != 0x14):  # RTM_GET(2)
1005             continue
1006         # Parse route. addrs contains what addresses are present
1007         flags = msg.rtm_flags
1008         if not flags.RTF_UP:
1009             continue
1010         if DARWIN and flags.RTF_WASCLONED and msg.rtm_parentflags.RTF_PRCLONING:
1011             # OSX needs filtering
1012             continue
1013         addrs = msg.rtm_addrs
1014         net = 0
1015         mask = 0xFFFFFFFF
1016         gw = 0
1017         iface = ""
1018         addr = ""
1019         metric = 1
1020         i = 0
1021         try:
1022             if addrs.RTA_DST:
1023                 net = atol(msg.addrs[i].sin_addr)
1024                 i += 1
1025             if addrs.RTA_GATEWAY:
1026                 if msg.addrs[i].sa_family == socket.AF_LINK:
1027                     gw = "0.0.0.0"
1028                 else:
1029                     gw = msg.addrs[i].sin_addr or "0.0.0.0"
1030                 i += 1
1031             if addrs.RTA_NETMASK:
1032                 nm = msg.addrs[i]
1033                 if nm.sa_family == socket.AF_INET:
1034                     mask = atol(nm.sin_addr)
1035                 elif nm.sa_family in [0x00, 0xFF]:  # NetBSD
1036                     mask = struct.unpack("<I", nm.sa_data[:4].rjust(4, b"\x00"))[0]
1037                 else:
1038                     mask = int.from_bytes(nm.sa_data[:4], "big")
1039                 i += 1
1040             if addrs.RTA_GENMASK:
1041                 i += 1
1042             if addrs.RTA_IFP:
1043                 iface = msg.addrs[i].sdl_iface.decode(errors="ignore")
1044                 i += 1
1045             if addrs.RTA_IFA:
1046                 addr = msg.addrs[i].sin_addr
1047                 i += 1
1048         except Exception:
1049             log_runtime.debug("Failed to read route %s" % repr(msg))
1050             continue
1051         if not iface:
1052             iface = ifaces[msg.rtm_index]["name"]
1053         routes.append((net, mask, gw, iface, addr, metric))
1054     if OPENBSD or DARWIN:
1055         # OpenBSD and OSX include multicast routes
1056         return routes
1057     # Add multicast routes, as those are missing by default
1058     for _iface in ifaces.values():
1059         if _iface["flags"].MULTICAST:
1060             try:
1061                 addr = next(
1062                     x["address"]
1063                     for x in _iface["ips"]
1064                     if x["af_family"] == socket.AF_INET
1065                 )
1066             except StopIteration:
1067                 continue
1068             routes.append(
1069                 (0xE0000000, 0xF0000000, "0.0.0.0", _iface["name"], addr, 250)
1070             )
1071     return routes

1170 def _get_if_list():
1171     # type: () -> Dict[int, Dict[str, Any]]
1172     """ 
1173     Read the interfaces list using a PF_ROUTE socket. 
1174     """
1175     mib = (ctypes.c_int * 6)(
1176         CTL_NET,
1177         socket.PF_ROUTE,
1178         0,
1179         int(socket.AF_UNSPEC),
1180         NET_RT_IFLIST,
1181         0,
1182     )
1183     resp = _sr1_bsdsysctl(mib)
1184     if not resp:
1185         return {}
1186     lifips = {}
1187     for msg in resp.msgs:
1188         if msg.rtm_type not in [0x0C, 0x16]:  # RTM_NEWADDR
1189             continue
1190         if not msg.ifm_addrs.RTA_IFA:
1191             continue
1192         ifindex = msg.ifm_index
1193         addrindex = (
1194             msg.ifm_addrs.RTA_DST
1195             + msg.ifm_addrs.RTA_GATEWAY
1196             + msg.ifm_addrs.RTA_NETMASK
1197             + msg.ifm_addrs.RTA_GENMASK
1198         )
1199         addr = msg.addrs[addrindex]
1200         if addr.sa_family not in [socket.AF_INET, socket.AF_INET6]:
1201             continue
1202         data = {
1203             "af_family": addr.sa_family,
1204             "index": ifindex,
1205             "address": addr.sin_addr,
1206         }
1207         if addr.sa_family == socket.AF_INET:  # ipv4
1208             data["address"] = addr.sin_addr
1209         else:  # ipv6
1210             data.update(
1211                 {
1212                     "address": addr.sin6_addr,
1213                     "scope": in6_getscope(addr.sin6_addr),
1214                 }
1215             )
1216         lifips.setdefault(ifindex, list()).append(data)
1217     interfaces = {}
1218     for msg in resp.msgs:
1219         if msg.rtm_type != 0xE and (not NETBSD or msg.rtm_type != 0x14):  # RTM_IFINFO
1220             continue
1221         ifindex = msg.ifm_index
1222         ifname = None
1223         mac = "00:00:00:00:00:00"
1224         ifflags = msg.ifm_flags
1225         ips = []
1226         for addr in msg.addrs:
1227             if addr.sa_family == socket.AF_LINK:
1228                 ifname = addr.sdl_iface.decode()
1229                 if addr.sdl_addr:
1230                     mac = addr.sdl_addr
1231         if ifname is not None:
1232             if ifindex in lifips:
1233                 ips = lifips[ifindex]
1234             interfaces[ifindex] = {
1235                 "name": ifname,
1236                 "index": ifindex,
1237                 "flags": ifflags,
1238                 "mac": mac,
1239                 "ips": ips,
1240             }
1241     return interfaces
