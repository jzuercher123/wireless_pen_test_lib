157  def check_validator_fields_against_field_name(
158      info: FieldDecoratorInfo,
159      field: str,
160  ) -> bool:
161      """Check if field name is in validator fields. 
162   
163      Args: 
164          info: The field info. 
165          field: The field name to check. 
166   
167      Returns: 
168          `True` if field name is in validator fields, `False` otherwise. 
169      """
170      if '*' in info.fields:
171          return True
172      for v_field_name in info.fields:
173          if v_field_name == field:
174              return True
175      return False

178  def check_decorator_fields_exist(decorators: Iterable[AnyFieldDecorator], fields: Iterable[str]) -> None:
179      """Check if the defined fields in decorators exist in `fields` param. 
180   
181      It ignores the check for a decorator if the decorator has `*` as field or `check_fields=False`. 
182   
183      Args: 
184          decorators: An iterable of decorators. 
185          fields: An iterable of fields name. 
186   
187      Raises: 
188          PydanticUserError: If one of the field names does not exist in `fields` param. 
189      """
190      fields = set(fields)
191      for dec in decorators:
192          if '*' in dec.info.fields:
193              continue
194          if dec.info.check_fields is False:
195              continue
196          for field in dec.info.fields:
197              if field not in fields:
198                  raise PydanticUserError(
199                      f'Decorators defined with incorrect fields: {dec.cls_ref}.{dec.cls_var_name}'
200                      " (use check_fields=False if you're inheriting from the model and intended this)",
201                      code='decorator-missing-field',
202                  )

211  def apply_each_item_validators(
212      schema: core_schema.CoreSchema,
213      each_item_validators: list[Decorator[ValidatorDecoratorInfo]],
214      field_name: str | None,
215  ) -> core_schema.CoreSchema:
216      # This V1 compatibility shim should eventually be removed
217  
218      # fail early if each_item_validators is empty
219      if not each_item_validators:
220          return schema
221  
222      # push down any `each_item=True` validators
223      # note that this won't work for any Annotated types that get wrapped by a function validator
224      # but that's okay because that didn't exist in V1
225      if schema['type'] == 'nullable':
226          schema['schema'] = apply_each_item_validators(schema['schema'], each_item_validators, field_name)
227          return schema
228      elif schema['type'] == 'tuple':
229          if (variadic_item_index := schema.get('variadic_item_index')) is not None:
230              schema['items_schema'][variadic_item_index] = apply_validators(
231                  schema['items_schema'][variadic_item_index],
232                  each_item_validators,
233                  field_name,
234              )
235      elif is_list_like_schema_with_items_schema(schema):
236          inner_schema = schema.get('items_schema', core_schema.any_schema())
237          schema['items_schema'] = apply_validators(inner_schema, each_item_validators, field_name)
238      elif schema['type'] == 'dict':
239          inner_schema = schema.get('values_schema', core_schema.any_schema())
240          schema['values_schema'] = apply_validators(inner_schema, each_item_validators, field_name)
241      else:
242          raise TypeError(
243              f"`@validator(..., each_item=True)` cannot be applied to fields with a schema of {schema['type']}"
244          )
245      return schema

248  def modify_model_json_schema(
249      schema_or_field: CoreSchemaOrField,
250      handler: GetJsonSchemaHandler,
251      *,
252      cls: Any,
253      title: str | None = None,
254  ) -> JsonSchemaValue:
255      """Add title and description for model-like classes' JSON schema. 
256   
257      Args: 
258          schema_or_field: The schema data to generate a JSON schema from. 
259          handler: The `GetCoreSchemaHandler` instance. 
260          cls: The model-like class. 
261          title: The title to set for the model's schema, defaults to the model's name 
262   
263      Returns: 
264          JsonSchemaValue: The updated JSON schema. 
265      """
266      from ..dataclasses import is_pydantic_dataclass
267      from ..root_model import RootModel
268      from ._dataclasses import is_builtin_dataclass
269  
270      BaseModel = import_cached_base_model()
271  
272      json_schema = handler(schema_or_field)
273      original_schema = handler.resolve_ref_schema(json_schema)
274      if title is not None:
275          original_schema['title'] = title
276      elif 'title' not in original_schema:
277          original_schema['title'] = cls.__name__
278      # BaseModel + Dataclass; don't use cls.__doc__ as it will contain the verbose class signature by default
279      docstring = None if cls is BaseModel or is_builtin_dataclass(cls) or is_pydantic_dataclass(cls) else cls.__doc__
280      if docstring and 'description' not in original_schema:
281          original_schema['description'] = inspect.cleandoc(docstring)
282      elif issubclass(cls, RootModel) and cls.model_fields['root'].description:
283          original_schema['description'] = cls.model_fields['root'].description
284      return json_schema

290  def _add_custom_serialization_from_json_encoders(
291      json_encoders: JsonEncoders | None, tp: Any, schema: CoreSchema
292  ) -> CoreSchema:
293      """Iterate over the json_encoders and add the first matching encoder to the schema. 
294   
295      Args: 
296          json_encoders: A dictionary of types and their encoder functions. 
297          tp: The type to check for a matching encoder. 
298          schema: The schema to add the encoder to. 
299      """
300      if not json_encoders:
301          return schema
302      if 'serialization' in schema:
303          return schema
304      # Check the class type and its superclasses for a matching encoder
305      # Decimal.__class__.__mro__ (and probably other cases) doesn't include Decimal itself
306      # if the type is a GenericAlias (e.g. from list[int]) we need to use __class__ instead of .__mro__
307      for base in (tp, *getattr(tp, '__mro__', tp.__class__.__mro__)[:-1]):
308          encoder = json_encoders.get(base)
309          if encoder is None:
310              continue
311  
312          warnings.warn(
313              f'`json_encoders` is deprecated. See https://docs.pydantic.dev/{version_short()}/concepts/serialization/#custom-serializers for alternatives',
314              PydanticDeprecatedSince20,
315          )
316  
317          # TODO: in theory we should check that the schema accepts a serialization key
318          schema['serialization'] = core_schema.plain_serializer_function_ser_schema(encoder, when_used='json')
319          return schema
320  
321      return schema

442      def _enum_schema(self, enum_type: type[Enum]) -> CoreSchema:
443          cases: list[Any] = list(enum_type.__members__.values())
444  
445          enum_ref = get_type_ref(enum_type)
446          description = None if not enum_type.__doc__ else inspect.cleandoc(enum_type.__doc__)
447          if (
448              description == 'An enumeration.'
449          ):  # This is the default value provided by enum.EnumMeta.__new__; don't use it
450              description = None
451          js_updates = {'title': enum_type.__name__, 'description': description}
452          js_updates = {k: v for k, v in js_updates.items() if v is not None}
453  
454          sub_type: Literal['str', 'int', 'float'] | None = None
455          if issubclass(enum_type, int):
456              sub_type = 'int'
457              value_ser_type: core_schema.SerSchema = core_schema.simple_ser_schema('int')
458          elif issubclass(enum_type, str):
459              # this handles `StrEnum` (3.11 only), and also `Foobar(str, Enum)`
460              sub_type = 'str'
461              value_ser_type = core_schema.simple_ser_schema('str')
462          elif issubclass(enum_type, float):
463              sub_type = 'float'
464              value_ser_type = core_schema.simple_ser_schema('float')
465          else:
466              # TODO this is an ugly hack, how do we trigger an Any schema for serialization?
467              value_ser_type = core_schema.plain_serializer_function_ser_schema(lambda x: x)
468  
469          if cases:
470  
471              def get_json_schema(schema: CoreSchema, handler: GetJsonSchemaHandler) -> JsonSchemaValue:
472                  json_schema = handler(schema)
473                  original_schema = handler.resolve_ref_schema(json_schema)
474                  original_schema.update(js_updates)
475                  return json_schema
476  
477              # we don't want to add the missing to the schema if it's the default one
478              default_missing = getattr(enum_type._missing_, '__func__', None) == Enum._missing_.__func__  # type: ignore
479              enum_schema = core_schema.enum_schema(
480                  enum_type,
481                  cases,
482                  sub_type=sub_type,
483                  missing=None if default_missing else enum_type._missing_,
484                  ref=enum_ref,
485                  metadata={'pydantic_js_functions': [get_json_schema]},
486              )
487  
488              if self._config_wrapper.use_enum_values:
489                  enum_schema = core_schema.no_info_after_validator_function(
490                      attrgetter('value'), enum_schema, serialization=value_ser_type
491                  )
492  
493              return enum_schema
494  
495          else:
496  
497              def get_json_schema_no_cases(_, handler: GetJsonSchemaHandler) -> JsonSchemaValue:
498                  json_schema = handler(core_schema.enum_schema(enum_type, cases, sub_type=sub_type, ref=enum_ref))
499                  original_schema = handler.resolve_ref_schema(json_schema)
500                  original_schema.update(js_updates)
501                  return json_schema
502  
503              # Use an isinstance check for enums with no cases.
504              # The most important use case for this is creating TypeVar bounds for generics that should
505              # be restricted to enums. This is more consistent than it might seem at first, since you can only
506              # subclass enum.Enum (or subclasses of enum.Enum) if all parent classes have no cases.
507              # We use the get_json_schema function when an Enum subclass has been declared with no cases
508              # so that we can still generate a valid json schema.
509              return core_schema.is_instance_schema(
510                  enum_type,
511                  metadata={'pydantic_js_functions': [get_json_schema_no_cases]},
512              )

514      def _ip_schema(self, tp: Any) -> CoreSchema:
515          from ._validators import IP_VALIDATOR_LOOKUP
516  
517          ip_type_json_schema_format = {
518              IPv4Address: 'ipv4',
519              IPv4Network: 'ipv4network',
520              IPv4Interface: 'ipv4interface',
521              IPv6Address: 'ipv6',
522              IPv6Network: 'ipv6network',
523              IPv6Interface: 'ipv6interface',
524          }

526          def ser_ip(ip: Any) -> str:
527              if not isinstance(ip, (tp, str)):
528                  raise PydanticSerializationUnexpectedValue(
529                      f"Expected `{tp}` but got `{type(ip)}` with value `'{ip}'` - serialized value may not be as expected."
530                  )
531              return str(ip)
532  
533          return core_schema.lax_or_strict_schema(
534              lax_schema=core_schema.no_info_plain_validator_function(IP_VALIDATOR_LOOKUP[tp]),
535              strict_schema=core_schema.json_or_python_schema(
536                  json_schema=core_schema.no_info_after_validator_function(tp, core_schema.str_schema()),
537                  python_schema=core_schema.is_instance_schema(tp),
538              ),
539              serialization=core_schema.plain_serializer_function_ser_schema(ser_ip),
540              metadata=build_metadata_dict(
541                  js_functions=[lambda _1, _2: {'type': 'string', 'format': ip_type_json_schema_format[tp]}]
542              ),
543          )

545      def _arbitrary_type_schema(self, tp: Any) -> CoreSchema:
546          if not isinstance(tp, type):

589      def clean_schema(self, schema: CoreSchema) -> CoreSchema:
590          schema = self.collect_definitions(schema)
591          schema = simplify_schema_references(schema)
592          if collect_invalid_schemas(schema):
593              raise self.CollectedInvalid()
594          schema = _discriminated_union.apply_discriminators(schema)
595          schema = validate_core_schema(schema)
596          return schema

598      def collect_definitions(self, schema: CoreSchema) -> CoreSchema:
599          ref = cast('str | None', schema.get('ref', None))
600          if ref:
601              self.defs.definitions[ref] = schema
602          if 'ref' in schema:
603              schema = core_schema.definition_reference_schema(schema['ref'])
604          return core_schema.definitions_schema(
605              schema,
606              list(self.defs.definitions.values()),

620      def generate_schema(
621          self,
622          obj: Any,
623          from_dunder_get_core_schema: bool = True,
624      ) -> core_schema.CoreSchema:
625          """Generate core schema. 
626   
627          Args: 
628              obj: The object to generate core schema for. 
629              from_dunder_get_core_schema: Whether to generate schema from either the 
630                  `__get_pydantic_core_schema__` function or `__pydantic_core_schema__` property. 
631   
632          Returns: 
633              The generated core schema. 
634   
635          Raises: 
636              PydanticUndefinedAnnotation: 
637                  If it is not possible to evaluate forward reference. 
638              PydanticSchemaGenerationError: 
639                  If it is not possible to generate pydantic-core schema. 
640              TypeError: 
641                  - If `alias_generator` returns a disallowed type (must be str, AliasPath or AliasChoices). 
642                  - If V1 style validator with `each_item=True` applied on a wrong field. 
643              PydanticUserError: 
644                  - If `typing.TypedDict` is used instead of `typing_extensions.TypedDict` on Python < 3.12. 
645                  - If `__modify_schema__` method is used instead of `__get_pydantic_json_schema__`. 
646          """
647          schema: CoreSchema | None = None
648  
649          if from_dunder_get_core_schema:
650              from_property = self._generate_schema_from_property(obj, obj)
651              if from_property is not None:
652                  schema = from_property
653  
654          if schema is None:
655              schema = self._generate_schema_inner(obj)
656  
657          metadata_js_function = _extract_get_pydantic_json_schema(obj, schema)
658          if metadata_js_function is not None:
659              metadata_schema = resolve_original_schema(schema, self.defs.definitions)
660              if metadata_schema:
661                  self._add_js_function(metadata_schema, metadata_js_function)
662  
663          schema = _add_custom_serialization_from_json_encoders(self._config_wrapper.json_encoders, obj, schema)
664  
665          return schema

667      def _model_schema(self, cls: type[BaseModel]) -> core_schema.CoreSchema:
668          """Generate schema for a Pydantic model."""
669          with self.defs.get_schema_or_ref(cls) as (model_ref, maybe_schema):
670              if maybe_schema is not None:
671                  return maybe_schema
672  
673              fields = cls.model_fields
674              decorators = cls.__pydantic_decorators__
675              computed_fields = decorators.computed_fields
676              check_decorator_fields_exist(
677                  chain(
678                      decorators.field_validators.values(),
679                      decorators.field_serializers.values(),
680                      decorators.validators.values(),
681                  ),
682                  {*fields.keys(), *computed_fields.keys()},
683              )
684              config_wrapper = ConfigWrapper(cls.model_config, check=False)
685              core_config = config_wrapper.core_config(cls)
686              title = self._get_model_title_from_config(cls, config_wrapper)
687              metadata = build_metadata_dict(js_functions=[partial(modify_model_json_schema, cls=cls, title=title)])
688  
689              model_validators = decorators.model_validators.values()
690  
691              with self._config_wrapper_stack.push(config_wrapper), self._types_namespace_stack.push(cls):
692                  self = self._current_generate_schema
693  
694                  extras_schema = None
695                  if core_config.get('extra_fields_behavior') == 'allow':
696                      assert cls.__mro__[0] is cls
697                      assert cls.__mro__[-1] is object
698                      for candidate_cls in cls.__mro__[:-1]:
699                          extras_annotation = getattr(candidate_cls, '__annotations__', {}).get(
700                              '__pydantic_extra__', None
701                          )
702                          if extras_annotation is not None:
703                              if isinstance(extras_annotation, str):
704                                  extras_annotation = _typing_extra.eval_type_backport(
705                                      _typing_extra._make_forward_ref(
706                                          extras_annotation, is_argument=False, is_class=True
707                                      ),
708                                      self._types_namespace,
709                                  )
710                              tp = get_origin(extras_annotation)
711                              if tp not in (Dict, dict):
712                                  raise PydanticSchemaGenerationError(
713                                      'The type annotation for `__pydantic_extra__` must be `Dict[str, ...]`'
714                                  )
715                              extra_items_type = self._get_args_resolving_forward_refs(
716                                  extras_annotation,
717                                  required=True,
718                              )[1]
719                              if extra_items_type is not Any:
720                                  extras_schema = self.generate_schema(extra_items_type)
721                                  break
722  
723                  if cls.__pydantic_root_model__:
724                      root_field = self._common_field_schema('root', fields['root'], decorators)
725                      inner_schema = root_field['schema']
726                      inner_schema = apply_model_validators(inner_schema, model_validators, 'inner')
727                      model_schema = core_schema.model_schema(
728                          cls,
729                          inner_schema,
730                          custom_init=getattr(cls, '__pydantic_custom_init__', None),
731                          root_model=True,
732                          post_init=getattr(cls, '__pydantic_post_init__', None),
733                          config=core_config,
734                          ref=model_ref,
735                          metadata=metadata,
736                      )
737                  else:
738                      fields_schema: core_schema.CoreSchema = core_schema.model_fields_schema(
739                          {k: self._generate_md_field_schema(k, v, decorators) for k, v in fields.items()},
740                          computed_fields=[
741                              self._computed_field_schema(d, decorators.field_serializers)
742                              for d in computed_fields.values()
743                          ],
744                          extras_schema=extras_schema,
745                          model_name=cls.__name__,
746                      )
747                      inner_schema = apply_validators(fields_schema, decorators.root_validators.values(), None)
748                      new_inner_schema = define_expected_missing_refs(inner_schema, recursively_defined_type_refs())
749                      if new_inner_schema is not None:
750                          inner_schema = new_inner_schema
751                      inner_schema = apply_model_validators(inner_schema, model_validators, 'inner')
752  
753                      model_schema = core_schema.model_schema(
754                          cls,
755                          inner_schema,
756                          custom_init=getattr(cls, '__pydantic_custom_init__', None),
757                          root_model=False,
758                          post_init=getattr(cls, '__pydantic_post_init__', None),
759                          config=core_config,
760                          ref=model_ref,
761                          metadata=metadata,
762                      )
763  
764                  schema = self._apply_model_serializers(model_schema, decorators.model_serializers.values())
765                  schema = apply_model_validators(schema, model_validators, 'outer')
766                  self.defs.definitions[model_ref] = schema
767                  return core_schema.definition_reference_schema(model_ref)

789      def _unpack_refs_defs(self, schema: CoreSchema) -> CoreSchema:
790          """Unpack all 'definitions' schemas into `GenerateSchema.defs.definitions` 
791          and return the inner schema. 
792          """
793          if schema['type'] == 'definitions':

800      def _generate_schema_from_property(self, obj: Any, source: Any) -> core_schema.CoreSchema | None:
801          """Try to generate schema from either the `__get_pydantic_core_schema__` function or 
802          `__pydantic_core_schema__` property. 
803   
804          Note: `__get_pydantic_core_schema__` takes priority so it can 
805          decide whether to use a `__pydantic_core_schema__` attribute, or generate a fresh schema. 
806          """
807          # avoid calling `__get_pydantic_core_schema__` if we've already visited this object
808          if is_self_type(obj):
809              obj = self.model_type_stack.get()
810          with self.defs.get_schema_or_ref(obj) as (_, maybe_schema):
811              if maybe_schema is not None:
812                  return maybe_schema
813          if obj is source:
814              ref_mode = 'unpack'
815          else:
816              ref_mode = 'to-def'
817  
818          schema: CoreSchema
819  
820          if (get_schema := getattr(obj, '__get_pydantic_core_schema__', None)) is not None:
821              schema = get_schema(
822                  source, CallbackGetCoreSchemaHandler(self._generate_schema_inner, self, ref_mode=ref_mode)
823              )
824          elif (
825              (existing_schema := getattr(obj, '__pydantic_core_schema__', None)) is not None
826              and not isinstance(existing_schema, MockCoreSchema)
827              and existing_schema.get('cls', None) == obj
828          ):
829              schema = existing_schema
830          elif (validators := getattr(obj, '__get_validators__', None)) is not None:
831              warn(
832                  '`__get_validators__` is deprecated and will be removed, use `__get_pydantic_core_schema__` instead.',
833                  PydanticDeprecatedSince20,
834              )
835              schema = core_schema.chain_schema([core_schema.with_info_plain_validator_function(v) for v in validators()])
836          else:
837              # we have no existing schema information on the property, exit early so that we can go generate a schema
838              return None
839  
840          schema = self._unpack_refs_defs(schema)
841  
842          if is_function_with_inner_schema(schema):
843              ref = schema['schema'].pop('ref', None)  # pyright: ignore[reportCallIssue, reportArgumentType]
844              if ref:
845                  schema['ref'] = ref
846          else:
847              ref = get_ref(schema)
848  
849          if ref:
850              self.defs.definitions[ref] = schema
851              return core_schema.definition_reference_schema(ref)
852  
853          return schema

855      def _resolve_forward_ref(self, obj: Any) -> Any:
856          # we assume that types_namespace has the target of forward references in its scope,
857          # but this could fail, for example, if calling Validator on an imported type which contains
858          # forward references to other types only defined in the module from which it was imported
859          # `Validator(SomeImportedTypeAliasWithAForwardReference)`
860          # or the equivalent for BaseModel
861          # class Model(BaseModel):
862          #   x: SomeImportedTypeAliasWithAForwardReference
863          try:
864              obj = _typing_extra.eval_type_backport(obj, globalns=self._types_namespace)
865          except NameError as e:
866              raise PydanticUndefinedAnnotation.from_name_error(e) from e
867  
868          # if obj is still a ForwardRef, it means we can't evaluate it, raise PydanticUndefinedAnnotation
869          if isinstance(obj, ForwardRef):
870              raise PydanticUndefinedAnnotation(obj.__forward_arg__, f'Unable to evaluate forward reference {obj}')
871  
872          if self._typevars_map:
873              obj = replace_types(obj, self._typevars_map)
874  
875          return obj

883      def _get_args_resolving_forward_refs(self, obj: Any, required: bool = False) -> tuple[Any, ...] | None:
884          args = get_args(obj)
885          if args:
886              args = tuple([self._resolve_forward_ref(a) if isinstance(a, ForwardRef) else a for a in args])
887          elif required:  # pragma: no cover
888              raise TypeError(f'Expected {obj} to have generic parameters but it had none')
889          return args

892          args = self._get_args_resolving_forward_refs(obj)
893          if not args:
894              return Any

898          args = self._get_args_resolving_forward_refs(obj)
899          if not args:
900              return (Any, Any)
901          if len(args) < 2:
902              origin = get_origin(obj)
903              raise TypeError(f'Expected two type arguments for {origin}, got 1')

906      def _generate_schema_inner(self, obj: Any) -> core_schema.CoreSchema:
907          if is_annotated(obj):
908              return self._annotated_schema(obj)
909  
910          if isinstance(obj, dict):
911              # we assume this is already a valid schema
912              return obj  # type: ignore[return-value]
913  
914          if isinstance(obj, str):
915              obj = ForwardRef(obj)
916  
917          if isinstance(obj, ForwardRef):
918              return self.generate_schema(self._resolve_forward_ref(obj))
919  
920          BaseModel = import_cached_base_model()
921  
922          if lenient_issubclass(obj, BaseModel):
923              with self.model_type_stack.push(obj):
924                  return self._model_schema(obj)
925  
926          if isinstance(obj, PydanticRecursiveRef):
927              return core_schema.definition_reference_schema(schema_ref=obj.type_ref)
928  
929          return self.match_type(obj)

931      def match_type(self, obj: Any) -> core_schema.CoreSchema:  # noqa: C901
932          """Main mapping of types to schemas. 
933   
934          The general structure is a series of if statements starting with the simple cases 
935          (non-generic primitive types) and then handling generics and other more complex cases. 
936   
937          Each case either generates a schema directly, calls into a public user-overridable method 
938          (like `GenerateSchema.tuple_variable_schema`) or calls into a private method that handles some 
939          boilerplate before calling into the user-facing method (e.g. `GenerateSchema._tuple_schema`). 
940   
941          The idea is that we'll evolve this into adding more and more user facing methods over time 
942          as they get requested and we figure out what the right API for them is. 
943          """
944          if obj is str:
945              return self.str_schema()
946          elif obj is bytes:
947              return core_schema.bytes_schema()
948          elif obj is int:
949              return core_schema.int_schema()
950          elif obj is float:
951              return core_schema.float_schema()
952          elif obj is bool:
953              return core_schema.bool_schema()
954          elif obj is complex:
955              return core_schema.complex_schema()
956          elif obj is Any or obj is object:
957              return core_schema.any_schema()
958          elif obj is datetime.date:
959              return core_schema.date_schema()
960          elif obj is datetime.datetime:
961              return core_schema.datetime_schema()
962          elif obj is datetime.time:
963              return core_schema.time_schema()
964          elif obj is datetime.timedelta:
965              return core_schema.timedelta_schema()
966          elif obj is Decimal:
967              return core_schema.decimal_schema()
968          elif obj is UUID:
969              return core_schema.uuid_schema()
970          elif obj is Url:
971              return core_schema.url_schema()
972          elif obj is MultiHostUrl:
973              return core_schema.multi_host_url_schema()
974          elif obj is None or obj is _typing_extra.NoneType:
975              return core_schema.none_schema()
976          elif obj in IP_TYPES:
977              return self._ip_schema(obj)
978          elif obj in TUPLE_TYPES:
979              return self._tuple_schema(obj)
980          elif obj in LIST_TYPES:
981              return self._list_schema(Any)
982          elif obj in SET_TYPES:
983              return self._set_schema(Any)
984          elif obj in FROZEN_SET_TYPES:
985              return self._frozenset_schema(Any)
986          elif obj in SEQUENCE_TYPES:
987              return self._sequence_schema(Any)
988          elif obj in DICT_TYPES:
989              return self._dict_schema(Any, Any)
990          elif isinstance(obj, TypeAliasType):
991              return self._type_alias_type_schema(obj)
992          elif obj is type:
993              return self._type_schema()
994          elif _typing_extra.is_callable_type(obj):
995              return core_schema.callable_schema()
996          elif _typing_extra.is_literal_type(obj):
997              return self._literal_schema(obj)
998          elif is_typeddict(obj):
999              return self._typed_dict_schema(obj, None)
1000         elif _typing_extra.is_namedtuple(obj):
1001             return self._namedtuple_schema(obj, None)
1002         elif _typing_extra.is_new_type(obj):
1003             # NewType, can't use isinstance because it fails <3.10
1004             return self.generate_schema(obj.__supertype__)
1005         elif obj == re.Pattern:
1006             return self._pattern_schema(obj)
1007         elif obj is collections.abc.Hashable or obj is typing.Hashable:
1008             return self._hashable_schema()
1009         elif isinstance(obj, typing.TypeVar):
1010             return self._unsubstituted_typevar_schema(obj)
1011         elif is_finalvar(obj):
1012             if obj is Final:
1013                 return core_schema.any_schema()
1014             return self.generate_schema(
1015                 self._get_first_arg_or_any(obj),
1016             )
1017         elif isinstance(obj, (FunctionType, LambdaType, MethodType, partial)):
1018             return self._callable_schema(obj)
1019         elif inspect.isclass(obj) and issubclass(obj, Enum):
1020             return self._enum_schema(obj)
1021         elif is_zoneinfo_type(obj):
1022             return self._zoneinfo_schema()
1023 
1024         if _typing_extra.is_dataclass(obj):
1025             return self._dataclass_schema(obj, None)
1026 
1027         origin = get_origin(obj)
1028         if origin is not None:
1029             return self._match_generic_type(obj, origin)
1030 
1031         res = self._get_prepare_pydantic_annotations_for_known_type(obj, ())
1032         if res is not None:
1033             source_type, annotations = res
1034             return self._apply_annotations(source_type, annotations)
1035 
1036         if self._arbitrary_types:
1037             return self._arbitrary_type_schema(obj)
1038         return self._unknown_type_schema(obj)

1040     def _match_generic_type(self, obj: Any, origin: Any) -> CoreSchema:  # noqa: C901
1041         if isinstance(origin, TypeAliasType):
1042             return self._type_alias_type_schema(obj)
1043 
1044         # Need to handle generic dataclasses before looking for the schema properties because attribute accesses
1045         # on _GenericAlias delegate to the origin type, so lose the information about the concrete parametrization
1046         # As a result, currently, there is no way to cache the schema for generic dataclasses. This may be possible
1047         # to resolve by modifying the value returned by `Generic.__class_getitem__`, but that is a dangerous game.
1048         if _typing_extra.is_dataclass(origin):
1049             return self._dataclass_schema(obj, origin)
1050         if _typing_extra.is_namedtuple(origin):
1051             return self._namedtuple_schema(obj, origin)
1052 
1053         from_property = self._generate_schema_from_property(origin, obj)
1054         if from_property is not None:
1055             return from_property
1056 
1057         if _typing_extra.origin_is_union(origin):
1058             return self._union_schema(obj)
1059         elif origin in TUPLE_TYPES:
1060             return self._tuple_schema(obj)
1061         elif origin in LIST_TYPES:
1062             return self._list_schema(self._get_first_arg_or_any(obj))
1063         elif origin in SET_TYPES:
1064             return self._set_schema(self._get_first_arg_or_any(obj))
1065         elif origin in FROZEN_SET_TYPES:
1066             return self._frozenset_schema(self._get_first_arg_or_any(obj))
1067         elif origin in DICT_TYPES:
1068             return self._dict_schema(*self._get_first_two_args_or_any(obj))
1069         elif is_typeddict(origin):
1070             return self._typed_dict_schema(obj, origin)
1071         elif origin in (typing.Type, type):
1072             return self._subclass_schema(obj)
1073         elif origin in SEQUENCE_TYPES:
1074             return self._sequence_schema(self._get_first_arg_or_any(obj))
1075         elif origin in {typing.Iterable, collections.abc.Iterable, typing.Generator, collections.abc.Generator}:
1076             return self._iterable_schema(obj)
1077         elif origin in (re.Pattern, typing.Pattern):
1078             return self._pattern_schema(obj)
1079 
1080         res = self._get_prepare_pydantic_annotations_for_known_type(obj, ())
1081         if res is not None:
1082             source_type, annotations = res
1083             return self._apply_annotations(source_type, annotations)
1084 
1085         if self._arbitrary_types:
1086             return self._arbitrary_type_schema(origin)
1087         return self._unknown_type_schema(obj)

1146     @staticmethod
1147     def _apply_alias_generator_to_field_info(
1148         alias_generator: Callable[[str], str] | AliasGenerator, field_info: FieldInfo, field_name: str
1149     ) -> None:
1150         """Apply an alias_generator to aliases on a FieldInfo instance if appropriate. 
1151  
1152         Args: 
1153             alias_generator: A callable that takes a string and returns a string, or an AliasGenerator instance. 
1154             field_info: The FieldInfo instance to which the alias_generator is (maybe) applied. 
1155             field_name: The name of the field from which to generate the alias.

1158         # 1. An alias is not specified
