40   # File format (according to p0f.fp) :
41   #
42   # wwww:ttt:D:ss:OOO...:QQ:OS:Details
43   #
44   # wwww    - window size
45   # ttt     - initial TTL
46   # D       - don't fragment bit  (0=unset, 1=set)
47   # ss      - overall SYN packet size
48   # OOO     - option value and order specification
49   # QQ      - quirks list
50   # OS      - OS genre
51   # details - OS description
52   
53   class p0fKnowledgeBase(KnowledgeBase):
54       def __init__(self, filename):
55           KnowledgeBase.__init__(self, filename)
56           # self.ttl_range=[255]
57   
58       def lazy_init(self):
59           try:
60               f = open(self.filename)
61           except IOError:
62               warning("Can't open base %s", self.filename)
63               return
64           try:
65               self.base = []
66               for line in f:
67                   if line[0] in ["#", "\n"]:
68                       continue
69                   line = tuple(line.split(":"))
70                   if len(line) < 8:
71                       continue
72   
73                   def a2i(x):
74                       if x.isdigit():
75                           return int(x)
76                       return x
77                   li = [a2i(e) for e in line[1:4]]
78                   # if li[0] not in self.ttl_range:
79                   #    self.ttl_range.append(li[0])
80                   #    self.ttl_range.sort()
81                   self.base.append((line[0], li[0], li[1], li[2], line[4],
82                                     line[5], line[6], line[7][:-1]))
83           except Exception:
84               warning("Can't parse p0f database (new p0f version ?)")
85               self.base = None
86           f.close()

103  def p0f_selectdb(flags):
104      # tested flags: S, R, A
105      if flags & 0x16 == 0x2:
106          # SYN
107          return p0f_kdb
108      elif flags & 0x16 == 0x12:
109          # SYN/ACK
110          return p0fa_kdb
111      elif flags & 0x16 in [0x4, 0x14]:
112          # RST RST/ACK
113          return p0fr_kdb
114      elif flags & 0x16 == 0x10:
115          # ACK
116          return p0fo_kdb
117      else:
118          return None

121  def packet2p0f(pkt):
122      pkt = pkt.copy()
123      pkt = pkt.__class__(raw(pkt))
124      while pkt.haslayer(IP) and pkt.haslayer(TCP):
125          pkt = pkt.getlayer(IP)
126          if isinstance(pkt.payload, TCP):
127              break
128          pkt = pkt.payload
129  
130      if not isinstance(pkt, IP) or not isinstance(pkt.payload, TCP):
131          raise TypeError("Not a TCP/IP packet")
132      # if pkt.payload.flags & 0x7 != 0x02: #S,!F,!R
133      #    raise TypeError("Not a SYN or SYN/ACK packet")
134  
135      db = p0f_selectdb(pkt.payload.flags)
136  
137      # t = p0f_kdb.ttl_range[:]
138      # t += [pkt.ttl]
139      # t.sort()
140      # ttl=t[t.index(pkt.ttl)+1]
141      ttl = pkt.ttl
142  
143      ss = len(pkt)
144      # from p0f/config.h : PACKET_BIG = 100
145      if ss > 100:
146          if db == p0fr_kdb:
147              # p0fr.fp: "Packet size may be wildcarded. The meaning of
148              #           wildcard is, however, hardcoded as 'size >
149              #           PACKET_BIG'"
150              ss = '*'
151          else:
152              ss = 0
153      if db == p0fo_kdb:
154          # p0fo.fp: "Packet size MUST be wildcarded."
155          ss = '*'
156  
157      ooo = ""
158      mss = -1
159      qqT = False
160      qqP = False
161      # qqBroken = False
162      ilen = (pkt.payload.dataofs << 2) - 20  # from p0f.c
163      for option in pkt.payload.options:
164          ilen -= 1
165          if option[0] == "MSS":
166              ooo += "M" + str(option[1]) + ","
167              mss = option[1]
168              # FIXME: qqBroken
169              ilen -= 3
170          elif option[0] == "WScale":
171              ooo += "W" + str(option[1]) + ","
172              # FIXME: qqBroken
173              ilen -= 2
174          elif option[0] == "Timestamp":
175              if option[1][0] == 0:
176                  ooo += "T0,"
177              else:
178                  ooo += "T,"
179              if option[1][1] != 0:
180                  qqT = True
181              ilen -= 9
182          elif option[0] == "SAckOK":
183              ooo += "S,"
184              ilen -= 1
185          elif option[0] == "NOP":
186              ooo += "N,"
187          elif option[0] == "EOL":
188              ooo += "E,"
189              if ilen > 0:
190                  qqP = True
191          else:
192              if isinstance(option[0], str):
193                  ooo += "?%i," % TCPOptions[1][option[0]]
194              else:
195                  ooo += "?%i," % option[0]
196              # FIXME: ilen
197      ooo = ooo[:-1]
198      if ooo == "":
199          ooo = "."
200  
201      win = pkt.payload.window
202      if mss != -1:
203          if mss != 0 and win % mss == 0:
204              win = "S" + str(win / mss)
205          elif win % (mss + 40) == 0:
206              win = "T" + str(win / (mss + 40))
207      win = str(win)
208  
209      qq = ""
210  
211      if db == p0fr_kdb:
212          if pkt.payload.flags & 0x10 == 0x10:
213              # p0fr.fp: "A new quirk, 'K', is introduced to denote
214              #           RST+ACK packets"
215              qq += "K"
216      # The two next cases should also be only for p0f*r*, but although
217      # it's not documented (or I have not noticed), p0f seems to
218      # support the '0' and 'Q' quirks on any databases (or at the least
219      # "classical" p0f.fp).
220      if pkt.payload.seq == pkt.payload.ack:
221          # p0fr.fp: "A new quirk, 'Q', is used to denote SEQ number
222          #           equal to ACK number."
223          qq += "Q"
224      if pkt.payload.seq == 0:
225          # p0fr.fp: "A new quirk, '0', is used to denote packets
226          #           with SEQ number set to 0."
227          qq += "0"
228      if qqP:
229          qq += "P"
230      if pkt.id == 0:
231          qq += "Z"
232      if pkt.options != []:
233          qq += "I"
234      if pkt.payload.urgptr != 0:
235          qq += "U"
236      if pkt.payload.reserved != 0:
237          qq += "X"
238      if pkt.payload.ack != 0:
239          qq += "A"
240      if qqT:
241          qq += "T"
242      if db == p0fo_kdb:
243          if pkt.payload.flags & 0x20 != 0:
244              # U
245              # p0fo.fp: "PUSH flag is excluded from 'F' quirk checks"
246              qq += "F"
247      else:
248          if pkt.payload.flags & 0x28 != 0:
249              # U or P
250              qq += "F"
251      if db != p0fo_kdb and not isinstance(pkt.payload.payload, NoPayload):
252          # p0fo.fp: "'D' quirk is not checked for."
253          qq += "D"
254      # FIXME : "!" - broken options segment: not handled yet
255  
256      if qq == "":
257          qq = "."
258  
259      return (db, (win, ttl, pkt.flags.DF, ss, ooo, qq))

262  def p0f_correl(x, y):
263      d = 0
264      # wwww can be "*" or "%nn". "Tnn" and "Snn" should work fine with
265      # the x[0] == y[0] test.
266      d += (x[0] == y[0] or y[0] == "*" or (y[0][0] == "%" and x[0].isdigit() and (int(x[0]) % int(y[0][1:])) == 0))  # noqa: E501
267      # ttl
268      d += (y[1] >= x[1] and y[1] - x[1] < 32)
269      for i in [2, 5]:
270          d += (x[i] == y[i] or y[i] == '*')
271      # '*' has a special meaning for ss
272      d += x[3] == y[3]
273      xopt = x[4].split(",")
274      yopt = y[4].split(",")
275      if len(xopt) == len(yopt):
276          same = True
277          for i in range(len(xopt)):
278              if not (xopt[i] == yopt[i] or
279                      (len(yopt[i]) == 2 and len(xopt[i]) > 1 and
280                       yopt[i][1] == "*" and xopt[i][0] == yopt[i][0]) or
281                      (len(yopt[i]) > 2 and len(xopt[i]) > 1 and
282                       yopt[i][1] == "%" and xopt[i][0] == yopt[i][0] and
283                       int(xopt[i][1:]) % int(yopt[i][2:]) == 0)):
284                  same = False
285                  break
286          if same:
287              d += len(xopt)
288      return d

291  @conf.commands.register
292  def p0f(pkt):
293      """Passive OS fingerprinting: which OS emitted this TCP packet ? 
294  p0f(packet) -> accuracy, [list of guesses] 
295  """
296      db, sig = packet2p0f(pkt)
297      if db:
298          pb = db.get_base()
299      else:
300          pb = []
301      if not pb:
302          warning("p0f base empty.")
303          return []
304      # s = len(pb[0][0])
305      r = []
306      max = len(sig[4].split(",")) + 5
307      for b in pb:
308          d = p0f_correl(sig, b)
309          if d == max:
310              r.append((b[6], b[7], b[1] - pkt[IP].ttl))
311      return r

314  def prnp0f(pkt):
315      """Calls p0f and returns a user-friendly output"""
316      # we should print which DB we use
317      try:
318          r = p0f(pkt)
319      except Exception:
320          return
321      if r == []:
322          r = ("UNKNOWN", "[" + ":".join(map(str, packet2p0f(pkt)[1])) + ":?:?]", None)  # noqa: E501
323      else:
324          r = r[0]
325      uptime = None
326      try:
327          uptime = pkt2uptime(pkt)
328      except Exception:
329          pass
330      if uptime == 0:
331          uptime = None
332      res = pkt.sprintf("%IP.src%:%TCP.sport% - " + r[0] + " " + r[1])
333      if uptime is not None:
334          res += pkt.sprintf(" (up: " + str(uptime / 3600) + " hrs)\n  -> %IP.dst%:%TCP.dport% (%TCP.flags%)")  # noqa: E501
335      else:
336          res += pkt.sprintf("\n  -> %IP.dst%:%TCP.dport% (%TCP.flags%)")
337      if r[2] is not None:
338          res += " (distance " + str(r[2]) + ")"
339      print(res)

342  @conf.commands.register
343  def pkt2uptime(pkt, HZ=100):
344      """Calculate the date the machine which emitted the packet booted using TCP timestamp  # noqa: E501 
345  pkt2uptime(pkt, [HZ=100])"""
346      if not isinstance(pkt, Packet):
347          raise TypeError("Not a TCP packet")
348      if isinstance(pkt, NoPayload):
349          raise TypeError("Not a TCP packet")
350      if not isinstance(pkt, TCP):
351          return pkt2uptime(pkt.payload)
352      for opt in pkt.options:
353          if opt[0] == "Timestamp":
354              # t = pkt.time - opt[1][0] * 1.0/HZ
355              # return time.ctime(t)
356              t = opt[1][0] / HZ
357              return t
358      raise TypeError("No timestamp option")

361  def p0f_impersonate(pkt, osgenre=None, osdetails=None, signature=None,
362                      extrahops=0, mtu=1500, uptime=None):
363      """Modifies pkt so that p0f will think it has been sent by a 
364  specific OS.  If osdetails is None, then we randomly pick up a 
365  personality matching osgenre. If osgenre and signature are also None, 
366  we use a local signature (using p0f_getlocalsigs). If signature is 
367  specified (as a tuple), we use the signature. 
368   
369  For now, only TCP Syn packets are supported. 
370  Some specifications of the p0f.fp file are not (yet) implemented."""
371      pkt = pkt.copy()
372      # pkt = pkt.__class__(raw(pkt))
373      while pkt.haslayer(IP) and pkt.haslayer(TCP):
374          pkt = pkt.getlayer(IP)
375          if isinstance(pkt.payload, TCP):
376              break
377          pkt = pkt.payload
378  
379      if not isinstance(pkt, IP) or not isinstance(pkt.payload, TCP):
380          raise TypeError("Not a TCP/IP packet")
381  
382      db = p0f_selectdb(pkt.payload.flags)
383      if osgenre:
384          pb = db.get_base()
385          if pb is None:
386              pb = []
387          pb = [x for x in pb if x[6] == osgenre]
388          if osdetails:
389              pb = [x for x in pb if x[7] == osdetails]
390      elif signature:
391          pb = [signature]
392      else:
393          pb = p0f_getlocalsigs()[db]
394      if db == p0fr_kdb:
395          # 'K' quirk <=> RST+ACK
396          if pkt.payload.flags & 0x4 == 0x4:
397              pb = [x for x in pb if 'K' in x[5]]
398          else:
399              pb = [x for x in pb if 'K' not in x[5]]
400      if not pb:
401          raise Scapy_Exception("No match in the p0f database")
402      pers = pb[random.randint(0, len(pb) - 1)]
403  
404      # options (we start with options because of MSS)
405      # Take the options already set as "hints" to use in the new packet if we
406      # can. MSS, WScale and Timestamp can all be wildcarded in a signature, so
407      # we'll use the already-set values if they're valid integers.
408      orig_opts = dict(pkt.payload.options)
409      int_only = lambda val: val if isinstance(val, int) else None
410      mss_hint = int_only(orig_opts.get('MSS'))
411      wscale_hint = int_only(orig_opts.get('WScale'))
412      ts_hint = [int_only(o) for o in orig_opts.get('Timestamp', (None, None))]
413  
414      options = []
415      if pers[4] != '.':
416          for opt in pers[4].split(','):
417              if opt[0] == 'M':
418                  # MSS might have a maximum size because of window size
419                  # specification
420                  if pers[0][0] == 'S':
421                      maxmss = (2**16 - 1) // int(pers[0][1:])
422                  else:
423                      maxmss = (2**16 - 1)
424                  # disregard hint if out of range
425                  if mss_hint and not 0 <= mss_hint <= maxmss:
426                      mss_hint = None
427                  # If we have to randomly pick up a value, we cannot use
428                  # scapy RandXXX() functions, because the value has to be
429                  # set in case we need it for the window size value. That's
430                  # why we use random.randint()
431                  if opt[1:] == '*':
432                      if mss_hint is not None:
433                          options.append(('MSS', mss_hint))
434                      else:
435                          options.append(('MSS', random.randint(1, maxmss)))
436                  elif opt[1] == '%':
437                      coef = int(opt[2:])
438                      if mss_hint is not None and mss_hint % coef == 0:
439                          options.append(('MSS', mss_hint))
440                      else:
441                          options.append((
442                              'MSS', coef * random.randint(1, maxmss // coef)))
443                  else:
444                      options.append(('MSS', int(opt[1:])))
445              elif opt[0] == 'W':
446                  if wscale_hint and not 0 <= wscale_hint < 2**8:
447                      wscale_hint = None
448                  if opt[1:] == '*':
449                      if wscale_hint is not None:
450                          options.append(('WScale', wscale_hint))
451                      else:
452                          options.append(('WScale', RandByte()))
453                  elif opt[1] == '%':
454                      coef = int(opt[2:])
455                      if wscale_hint is not None and wscale_hint % coef == 0:
456                          options.append(('WScale', wscale_hint))
457                      else:
458                          options.append((
459                              'WScale', coef * RandNum(min=1, max=(2**8 - 1) // coef)))  # noqa: E501
460                  else:
461                      options.append(('WScale', int(opt[1:])))
462              elif opt == 'T0':
463                  options.append(('Timestamp', (0, 0)))
464              elif opt == 'T':
465                  # Determine first timestamp.
466                  if uptime is not None:
467                      ts_a = uptime
468                  elif ts_hint[0] and 0 < ts_hint[0] < 2**32:
469                      # Note: if first ts is 0, p0f registers it as "T0" not "T",
470                      # hence we don't want to use the hint if it was 0.
471                      ts_a = ts_hint[0]
472                  else:
473                      ts_a = random.randint(120, 100 * 60 * 60 * 24 * 365)
474                  # Determine second timestamp.
475                  if 'T' not in pers[5]:
476                      ts_b = 0
477                  elif ts_hint[1] and 0 < ts_hint[1] < 2**32:
478                      ts_b = ts_hint[1]
479                  else:
480                      # FIXME: RandInt() here does not work (bug (?) in
481                      # TCPOptionsField.m2i often raises "OverflowError:
482                      # long int too large to convert to int" in:
483                      #    oval = struct.pack(ofmt, *oval)"
484                      # Actually, this is enough to often raise the error:
485                      #    struct.pack('I', RandInt())
486                      ts_b = random.randint(1, 2**32 - 1)
487                  options.append(('Timestamp', (ts_a, ts_b)))
488              elif opt == 'S':
489                  options.append(('SAckOK', ''))
490              elif opt == 'N':
491                  options.append(('NOP', None))
492              elif opt == 'E':
493                  options.append(('EOL', None))
494              elif opt[0] == '?':
495                  if int(opt[1:]) in TCPOptions[0]:
496                      optname = TCPOptions[0][int(opt[1:])][0]
497                      optstruct = TCPOptions[0][int(opt[1:])][1]
498                      options.append((optname,
499                                      struct.unpack(optstruct,
500                                                    RandString(struct.calcsize(optstruct))._fix())))  # noqa: E501
501                  else:
502                      options.append((int(opt[1:]), ''))
503              # FIXME: qqP not handled
504              else:
505                  warning("unhandled TCP option %s", opt)
506              pkt.payload.options = options
507  
508      # window size
509      if pers[0] == '*':
510          pkt.payload.window = RandShort()
511      elif pers[0].isdigit():
512          pkt.payload.window = int(pers[0])
513      elif pers[0][0] == '%':
514          coef = int(pers[0][1:])
515          pkt.payload.window = coef * RandNum(min=1, max=(2**16 - 1) // coef)
516      elif pers[0][0] == 'T':
517          pkt.payload.window = mtu * int(pers[0][1:])
518      elif pers[0][0] == 'S':
519          # needs MSS set
520          mss = [x for x in options if x[0] == 'MSS']
521          if not mss:
522              raise Scapy_Exception("TCP window value requires MSS, and MSS option not set")  # noqa: E501
523          pkt.payload.window = mss[0][1] * int(pers[0][1:])
524      else:
525          raise Scapy_Exception('Unhandled window size specification')
526  
527      # ttl
528      pkt.ttl = pers[1] - extrahops
529      # DF flag
530      pkt.flags |= (2 * pers[2])
531      # FIXME: ss (packet size) not handled (how ? may be with D quirk
532      # if present)
533      # Quirks
534      if pers[5] != '.':
535          for qq in pers[5]:
536              # FIXME: not handled: P, I, X, !
537              # T handled with the Timestamp option
538              if qq == 'Z':
539                  pkt.id = 0
540              elif qq == 'U':
541                  pkt.payload.urgptr = RandShort()
542              elif qq == 'A':
543                  pkt.payload.ack = RandInt()
544              elif qq == 'F':
545                  if db == p0fo_kdb:
546                      pkt.payload.flags |= 0x20  # U
547                  else:
548                      pkt.payload.flags |= random.choice([8, 32, 40])  # P/U/PU
549              elif qq == 'D' and db != p0fo_kdb:
550                  pkt /= conf.raw_layer(load=RandString(random.randint(1, 10)))  # XXX p0fo.fp  # noqa: E501
551              elif qq == 'Q':
552                  pkt.payload.seq = pkt.payload.ack
553              # elif qq == '0': pkt.payload.seq = 0
554          # if db == p0fr_kdb:
555          # '0' quirk is actually not only for p0fr.fp (see
556          # packet2p0f())
557      if '0' in pers[5]:
558          pkt.payload.seq = 0
559      elif pkt.payload.seq == 0:
560          pkt.payload.seq = RandInt()
561  
562      while pkt.underlayer:
563          pkt = pkt.underlayer
564      return pkt

567  def p0f_getlocalsigs():
568      """This function returns a dictionary of signatures indexed by p0f 
569  db (e.g., p0f_kdb, p0fa_kdb, ...) for the local TCP/IP stack. 
570   
571  You need to have your firewall at least accepting the TCP packets 
572  from/to a high port (30000 <= x <= 40000) on your loopback interface. 
573   
574  Please note that the generated signatures come from the loopback 
575  interface and may (are likely to) be different than those generated on 
576  "normal" interfaces."""
577      pid = os.fork()
578      port = random.randint(30000, 40000)
579      if pid > 0:
580          # parent: sniff
581          result = {}
582  
583          def addresult(res):
584              # TODO: wildcard window size in some cases? and maybe some
585              # other values?
586              if res[0] not in result:
587                  result[res[0]] = [res[1]]
588              else:
589                  if res[1] not in result[res[0]]:
590                      result[res[0]].append(res[1])
591          # XXX could we try with a "normal" interface using other hosts
592          iface = conf.route.route('127.0.0.1')[0]
593          # each packet is seen twice: S + RA, S + SA + A + FA + A
594          # XXX are the packets also seen twice on non Linux systems ?
595          count = 14
596          pl = sniff(iface=iface, filter='tcp and port ' + str(port), count=count, timeout=3)  # noqa: E501
597          for pkt in pl:
598              for elt in packet2p0f(pkt):
599                  addresult(elt)

601      elif pid < 0:

603      else:
604          # child: send
605          # XXX erk

607          s1 = socket.socket(socket.AF_INET, type=socket.SOCK_STREAM)
608          # S & RA
609          try:
610              s1.connect(('127.0.0.1', port))
611          except socket.error:
612              pass
613          # S, SA, A, FA, A
614          s1.bind(('127.0.0.1', port))
615          s1.connect(('127.0.0.1', port))
616          # howto: get an RST w/o ACK packet
617          s1.close()
618          os._exit(0)
