7    """ 
8    KNXNet/IP 
9     
10   This module provides Scapy layers for KNXNet/IP communications over UDP 
11   according to KNX specifications v2.1 / ISO-IEC 14543-3. 
12   Specifications can be downloaded for free here : 
13   https://my.knx.org/en/shop/knx-specifications 
14    
15   Currently, the module (partially) supports the following services : 
16   * SEARCH REQUEST/RESPONSE 
17   * DESCRIPTION REQUEST/RESPONSE 
18   * CONNECT, DISCONNECT, CONNECTION_STATE REQUEST/RESPONSE 
19   * CONFIGURATION REQUEST/RESPONSE 
20   * TUNNELING REQUEST/RESPONSE 
21   """

154  # KNX Standard v2.1 - 03_05_01 p.17
155  class KNXAddressField(ShortField):
156      def i2repr(self, pkt, x):
157          if x is None:
158              return None
159          else:
160              return "%d.%d.%d" % ((x >> 12) & 0xf, (x >> 8) & 0xf, (x & 0xff))
161  
162      def any2i(self, pkt, x):
163          if isinstance(x, str):
164              try:
165                  a, b, c = map(int, x.split("."))
166                  x = (a << 12) | (b << 8) | c
167              except ValueError:
168                  raise ValueError(x)
169          return ShortField.any2i(self, pkt, x)
170  
171  
172  # KNX Standard v2.1 - 03_05_01 p.18
173  class KNXGroupField(ShortField):
174      def i2repr(self, pkt, x):
175          return "%d/%d/%d" % ((x >> 11) & 0x1f, (x >> 8) & 0x7, (x & 0xff))
176  
177      def any2i(self, pkt, x):
178          if isinstance(x, str):
179              try:
180                  a, b, c = map(int, x.split("/"))
181                  x = (a << 11) | (b << 8) | c
182              except ValueError:
183                  raise ValueError(x)
184          return ShortField.any2i(self, pkt, x)

312      name = "L_cEMI"
313      fields_desc = [
314          FieldLenField("additional_information_length", 0, fmt="B",
315                        length_of="additional_information"),
316          StrLenField("additional_information", None,
317                      length_from=lambda pkt: pkt.additional_information_length),
318          # Controlfield 1 (1 byte made of 8*1 bits)
319          BitEnumField("frame_type", 1, 1, {
320              1: "standard"
321          }),
322          BitField("reserved_1", 0, 1),
323          BitField("repeat_on_error", 1, 1),
324          BitEnumField("broadcast_type", 1, 1, {
325              1: "domain"
326          }),
327          BitEnumField("priority", 3, 2, {
328              3: "low"
329          }),
330          BitField("ack_request", 0, 1),
331          BitField("confirmation_error", 0, 1),
332          # Controlfield 2 (1 byte made of 1+3+4 bits)
333          BitEnumField("address_type", 1, 1, {
334              1: "group"
335          }),
336          BitField("hop_count", 6, 3),
337          BitField("extended_frame_format", 0, 4),
338          KNXAddressField("source_address", None),
339          KNXGroupField("destination_address", "1/2/3"),
340          FieldLenField("npdu_length", 0x01, fmt="B", length_of="data"),
341          # TPCI and APCI (2 byte made of 1+1+4+4+6 bits)
342          BitEnumField("packet_type", 0, 1, {
343              0: "data"
344          }),
345          BitEnumField("sequence_type", 0, 1, {
346              0: "unnumbered"
347          }),

560  # we made the choice to define a KNX service as a payload for a KNX Header
561  # it could also be possible to define the body as a conditional PacketField
562  # contained after header
563  
564  class KNX(Packet):
565      name = "KNXnet/IP"
566      fields_desc = [
567          ByteField("header_length", None),
568          XByteField("protocol_version", 0x10),
569          ShortEnumField("service_identifier", None, SERVICE_IDENTIFIER_CODES),
570          ShortField("total_length", None)
571      ]
572  
573      def post_build(self, p, pay):
574          # computes header_length
575          if self.header_length is None:
576              p = struct.pack("!B", len(p)) + p[1:]
577          # computes total_length
578          if self.total_length is None:
579              p = p[:-2] + struct.pack("!H", len(p) + len(pay))
580          return p + pay
