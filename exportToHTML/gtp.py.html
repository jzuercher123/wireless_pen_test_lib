178  class TBCDByteField(StrFixedLenField):
179  
180      def i2h(self, pkt, val):
181          return val
182  
183      def m2i(self, pkt, val):
184          ret = []
185          for v in val:
186              byte = orb(v)
187              left = byte >> 4
188              right = byte & 0xf
189              if left == 0xf:
190                  ret.append(TBCD_TO_ASCII[right:right + 1])
191              else:
192                  ret += [
193                      TBCD_TO_ASCII[right:right + 1],
194                      TBCD_TO_ASCII[left:left + 1]
195                  ]
196          return b"".join(ret)
197  
198      def i2m(self, pkt, val):
199          if not isinstance(val, bytes):
200              val = bytes_encode(val)
201          ret_string = b""
202          for i in range(0, len(val), 2):
203              tmp = val[i:i + 2]
204              if len(tmp) == 2:
205                  ret_string += chb(int(tmp[::-1], 16))
206              else:
207                  ret_string += chb(int(b"F" + tmp[:1], 16))
208          return ret_string

234  class GTPHeader(Packet):
235      # 3GPP TS 29.060 V9.1.0 (2009-12)
236      name = "GTP-C Header"
237      fields_desc = [BitField("version", 1, 3),
238                     BitField("PT", 1, 1),
239                     BitField("reserved", 0, 1),
240                     BitField("E", 0, 1),
241                     BitField("S", 0, 1),
242                     BitField("PN", 0, 1),
243                     ByteEnumField("gtp_type", None, GTPmessageType),
244                     ShortField("length", None),
245                     IntField("teid", 0),
246                     ConditionalField(
247                         XBitField("seq", 0, 16),
248                         lambda pkt:pkt.E == 1 or pkt.S == 1 or pkt.PN == 1),
249                     ConditionalField(
250                         ByteField("npdu", 0),
251                         lambda pkt:pkt.E == 1 or pkt.S == 1 or pkt.PN == 1),
252                     ConditionalField(
253                         ByteEnumField("next_ex", 0, ExtensionHeadersTypes),
254                         lambda pkt:pkt.E == 1 or pkt.S == 1 or pkt.PN == 1),
255                     ]
256  
257      def post_build(self, p, pay):
258          p += pay
259          if self.length is None:
260              # The message length field is calculated different in GTPv1 and GTPv2.  # noqa: E501
261              # For GTPv1 it is defined as the rest of the packet following the mandatory 8-byte GTP header  # noqa: E501
262              # For GTPv2 it is defined as the length of the message in bytes excluding the mandatory part of the GTP-C header (the first 4 bytes)  # noqa: E501
263              tmp_len = len(p) - 4 if self.version == 2 else len(p) - 8
264              p = p[:2] + struct.pack("!H", tmp_len) + p[4:]
265          return p
266  
267      def hashret(self):
268          hsh = struct.pack("B", self.version)
269          if self.seq:
270              hsh += struct.pack("H", self.seq)
271          return hsh + self.payload.hashret()
272  
273      def answers(self, other):
274          return (isinstance(other, GTPHeader) and
275                  self.version == other.version and
276                  (not self.seq or self.seq == other.seq) and
277                  self.payload.answers(other.payload))
278  
279      @classmethod
280      def dispatch_hook(cls, _pkt=None, *args, **kargs):
281          if _pkt and len(_pkt) >= 1:
282              if (orb(_pkt[0]) >> 5) & 0x7 == 2:
283                  from . import gtp_v2
284                  return gtp_v2.GTPHeader
285          if _pkt and len(_pkt) >= 8:
286              _gtp_type = orb(_pkt[1:2])
287              return GTPforcedTypes.get(_gtp_type, GTPHeader)
288          return cls

291  class GTP_U_Header(GTPHeader):
292      # 3GPP TS 29.060 V9.1.0 (2009-12)
293      name = "GTP-U Header"
294      # GTP-U protocol is used to transmit T-PDUs between GSN pairs (or between an SGSN and an RNC in UMTS),  # noqa: E501
295      # encapsulated in G-PDUs. A G-PDU is a packet including a GTP-U header and a T-PDU. The Path Protocol  # noqa: E501
296      # defines the path and the GTP-U header defines the tunnel. Several tunnels may be multiplexed on a single path.  # noqa: E501
297  
298      def guess_payload_class(self, payload):
299          # Snooped from Wireshark
300          # https://github.com/boundary/wireshark/blob/07eade8124fd1d5386161591b52e177ee6ea849f/epan/dissectors/packet-gtp.c#L8195  # noqa: E501
301          if self.E == 1:
302              if self.next_ex == 0x85:
303                  return GTPPDUSessionContainer
304              return GTPHeader.guess_payload_class(self, payload)
305  
306          if self.gtp_type == 255:
307              sub_proto = orb(payload[0])
308              if sub_proto >= 0x45 and sub_proto <= 0x4e:
309                  return IP
310              elif (sub_proto & 0xf0) == 0x60:
311                  return IPv6
312              else:
313                  return PPP
314          return GTPHeader.guess_payload_class(self, payload)

405                         StrLenField("padding", b"", length_from=lambda p: 0),
406                         lambda pkt: pkt.NextExtHdr == 0)]

408      def guess_payload_class(self, payload):
409          if self.NextExtHdr == 0:
410              sub_proto = orb(payload[0])
411              if sub_proto >= 0x45 and sub_proto <= 0x4e:
412                  return IP
413              elif (sub_proto & 0xf0) == 0x60:
414                  return IPv6
415              else:
416                  return PPP
417          return GTPHeader.guess_payload_class(self, payload)
418  
419      def post_dissect(self, s):
420          if self.NextExtHdr == 0:
421              # Padding is handled in this layer
422              length = len(self.original) - len(s)
423              pad_length = (- length) % 4
424              self.padding = s[:pad_length]
425              return s[pad_length:]
426          return s
427  
428      def post_build(self, p, pay):
429          # Length
430          if self.NextExtHdr == 0:
431              p += b"\x00" * ((-len(p)) % 4)
432          else:
433              pay += b"\x00" * ((-len(p + pay)) % 4)
434          if self.ExtHdrLen is None:
435              p = struct.pack("!B", len(p) // 4) + p[1:]
436          return p + pay

444  class IE_Base(Packet):
445      def extract_padding(self, pkt):
446          return "", pkt
447  
448      def post_build(self, p, pay):
449          if self.fields_desc[1].name == "length":
450              if self.length is None:
451                  tmp_len = len(p)
452                  if isinstance(self.payload, conf.padding_layer):
453                      tmp_len += len(self.payload.load)
454                  p = p[:1] + struct.pack("!H", tmp_len - 4) + p[3:]
455          return p + pay

588  class APNStrLenField(StrLenField):
589      # Inspired by DNSStrField
590      def m2i(self, pkt, s):
591          ret_s = b""
592          tmp_s = s
593          while tmp_s:
594              tmp_len = orb(tmp_s[0]) + 1
595              if tmp_len > len(tmp_s):
596                  warning("APN prematured end of character-string (size=%i, remaining bytes=%i)" % (tmp_len, len(tmp_s)))  # noqa: E501
597              ret_s += tmp_s[1:tmp_len]
598              tmp_s = tmp_s[tmp_len:]
599              if len(tmp_s):
600                  ret_s += b"."
601          s = ret_s
602          return s
603  
604      def i2m(self, pkt, s):
605          if not isinstance(s, bytes):
606              s = bytes_encode(s)
607          s = b"".join(chb(len(x)) + x for x in s.split(b"."))
608          return s

935  def IE_Dispatcher(s):
936      """Choose the correct Information Element class."""
937      if len(s) < 1:
938          return Raw(s)
939      # Get the IE type
940      ietype = orb(s[0])
941      cls = ietypecls.get(ietype, Raw)
942  
943      # if ietype greater than 128 are TLVs
944      if cls == Raw and ietype & 128 == 128:
945          cls = IE_NotImplementedTLV
946      return cls(s)
